+// Command fixtureutil loads JSON fixture data into a PostgreSQL database.
+//
+// Usage:
+//
+//	go run ./cmd/fixtureutil --database-url <url> --fixture-dir <dir>
+//
+// The fixture-dir should contain JSON files named "<schema>.<table>.json" or
+// "<schema>.<table>.json.xz". The tool applies migrations first, then loads
+// fixture data in FK-safe order.
+package main
+
+import (
+	"context"
+	"flag"
+	"fmt"
+	"os"
+
+	"github.com/jackc/pgx/v5/pgxpool"
+
+	"github.com/ketang/kapow/api/internal/db"
+	"github.com/ketang/kapow/api/internal/fixture"
+)
+
+func main() {
+	databaseURL := flag.String("database-url", "", "PostgreSQL connection string (falls back to DATABASE_URL env var)")
+	fixtureDir := flag.String("fixture-dir", "", "directory containing JSON fixture files")
+	skipMigrate := flag.Bool("skip-migrate", false, "skip running database migrations")
+	flag.Parse()
+
+	if err := run(*databaseURL, *fixtureDir, *skipMigrate); err != nil {
+		fmt.Fprintf(os.Stderr, "error: %v\n", err)
+		os.Exit(1)
+	}
+}
+
+func run(databaseURL, fixtureDir string, skipMigrate bool) error {
+	dsn := databaseURL
+	if dsn == "" {
+		dsn = os.Getenv("DATABASE_URL")
+	}
+	if dsn == "" {
+		return fmt.Errorf("--database-url flag or DATABASE_URL env var is required")
+	}
+	if fixtureDir == "" {
+		return fmt.Errorf("--fixture-dir is required")
+	}
+
+	ctx := context.Background()
+
+	if !skipMigrate {
+		fmt.Fprintln(os.Stderr, "running migrations...")
+		if err := db.MigrateUp(ctx, dsn); err != nil {
+			return fmt.Errorf("migrate: %w", err)
+		}
+	}
+
+	fmt.Fprintln(os.Stderr, "connecting to database...")
+	pool, err := pgxpool.New(ctx, dsn)
+	if err != nil {
+		return fmt.Errorf("connect: %w", err)
+	}
+	defer pool.Close()
+
+	fmt.Fprintf(os.Stderr, "loading fixtures from %s...\n", fixtureDir)
+	if err := fixture.Load(ctx, pool, os.DirFS(fixtureDir)); err != nil {
+		return fmt.Errorf("load fixtures: %w", err)
+	}
+
+	fmt.Fprintln(os.Stderr, "fixtures loaded successfully")
+	return nil
+}

+// Package fixture loads JSON fixture data into a PostgreSQL database.
+//
+// Fixture files are JSON arrays of objects (one file per table), optionally
+// xz-compressed. Files are named "<schema>.<table>.json" or
+// "<schema>.<table>.json.xz" and stored in a directory that is passed to
+// [Load] as an [io/fs.FS].
+//
+// Tables are loaded in FK-safe order. After all tables are populated, the
+// materialized views dataset_merged and dataset_latest are refreshed.
+package fixture
+
+import (
+	"bytes"
+	"context"
+	"encoding/json"
+	"fmt"
+	"io"
+	"io/fs"
+	"os/exec"
+	"strings"
+
+	"github.com/jackc/pgx/v5"
+	"github.com/jackc/pgx/v5/pgxpool"
+)
+
+// tableSpec defines a fixture table and how to load it.
+type tableSpec struct {
+	Schema string
+	Table  string
+}
+
+// loadOrder is the FK-safe order for loading fixture data. This must match
+// the extraction order in tools/samplesql/extractor.go.
+var loadOrder = []tableSpec{
+	{"location", "state"},
+	{"search_base", "ranking"},
+	{"search_base", "conference"},
+	{"search_base", "cip_data"},
+	{"search_base", "institution"},
+	{"search_base", "dataset"},
+	{"search_base", "annual_dataset"},
+	{"search_base", "institution_alias"},
+	{"location", "institution_normalized_alias"},
+	{"location", "zip_coordinates"},
+}
+
+// materializedViews are refreshed after all data is loaded.
+var materializedViews = []string{
+	"search_base.dataset_merged",
+	"search_base.dataset_latest",
+}
+
+// Load reads JSON fixture files from dataFS and inserts them into the database
+// in FK-safe order. The dataFS should contain files named
+// "<schema>.<table>.json" or "<schema>.<table>.json.xz".
+//
+// The database schema must already exist (migrations applied) before calling
+// Load.
+func Load(ctx context.Context, pool *pgxpool.Pool, dataFS fs.FS) error {
+	// Discover column types for geometry and JSONB detection.
+	colTypes, err := discoverColumnTypes(ctx, pool)
+	if err != nil {
+		return err
+	}
+
+	for _, spec := range loadOrder {
+		data, err := readFixtureFile(dataFS, spec)
+		if err != nil {
+			return err
+		}
+		if data == nil {
+			continue // file not found — skip
+		}
+
+		if err := loadTable(ctx, pool, spec, data, colTypes); err != nil {
+			return err
+		}
+	}
+
+	// Refresh materialized views.
+	for _, view := range materializedViews {
+		if _, err := pool.Exec(ctx, "REFRESH MATERIALIZED VIEW "+view); err != nil {
+			return fmt.Errorf("fixture: refresh %s: %w", view, err)
+		}
+	}
+
+	return nil
+}
+
+// LoadOrder returns the FK-safe table loading order. Useful for tests.
+func LoadOrder() []string {
+	names := make([]string, len(loadOrder))
+	for i, spec := range loadOrder {
+		names[i] = spec.Schema + "." + spec.Table
+	}
+	return names
+}
+
+// readFixtureFile reads and decompresses a fixture file from the FS.
+// Returns nil if the file does not exist.
+func readFixtureFile(dataFS fs.FS, spec tableSpec) ([]byte, error) {
+	filename := fmt.Sprintf("%s.%s.json", spec.Schema, spec.Table)
+	xzFilename := filename + ".xz"
+
+	// Try xz-compressed first.
+	if data, err := fs.ReadFile(dataFS, xzFilename); err == nil {
+		decompressed, err := xzDecompress(data)
+		if err != nil {
+			return nil, fmt.Errorf("fixture: decompress %s: %w", xzFilename, err)
+		}
+		return decompressed, nil
+	}
+
+	// Try uncompressed.
+	data, err := fs.ReadFile(dataFS, filename)
+	if err != nil {
+		return nil, nil //nolint:nilerr // file not found is not an error
+	}
+	return data, nil
+}
+
+// xzDecompress decompresses xz data using the xz command-line tool.
+func xzDecompress(data []byte) ([]byte, error) {
+	cmd := exec.Command("xz", "-dc") //nolint:gosec // G204: xz is a trusted system tool
+	cmd.Stdin = bytes.NewReader(data)
+	out, err := cmd.Output()
+	if err != nil {
+		return nil, fmt.Errorf("xz: %w", err)
+	}
+	return out, nil
+}
+
+// columnInfo holds type metadata for a single column.
+type columnInfo struct {
+	UDTName  string // e.g. "geometry", "geography", "jsonb", "text"
+	DataType string // e.g. "USER-DEFINED", "jsonb", "text"
+}
+
+// discoverColumnTypes queries information_schema to build a map of
+// schema.table.column → columnInfo for all fixture tables.
+func discoverColumnTypes(ctx context.Context, pool *pgxpool.Pool) (map[string]columnInfo, error) {
+	result := make(map[string]columnInfo)
+
+	// Build list of schemas and tables to query.
+	schemas := make(map[string]bool)
+	for _, spec := range loadOrder {
+		schemas[spec.Schema] = true
+	}
+
+	for schema := range schemas {
+		rows, err := pool.Query(ctx,
+			`SELECT table_name, column_name, data_type, udt_name
+			 FROM information_schema.columns
+			 WHERE table_schema = $1
+			 ORDER BY table_name, ordinal_position`, schema)
+		if err != nil {
+			return nil, fmt.Errorf("fixture: discover columns for %s: %w", schema, err)
+		}
+
+		for rows.Next() {
+			var tableName, colName, dataType, udtName string
+			if err := rows.Scan(&tableName, &colName, &dataType, &udtName); err != nil {
+				rows.Close()
+				return nil, fmt.Errorf("fixture: scan column info: %w", err)
+			}
+			key := fmt.Sprintf("%s.%s.%s", schema, tableName, colName)
+			result[key] = columnInfo{UDTName: udtName, DataType: dataType}
+		}
+		rows.Close()
+		if err := rows.Err(); err != nil {
+			return nil, fmt.Errorf("fixture: iterate columns for %s: %w", schema, err)
+		}
+	}
+
+	return result, nil
+}
+
+// loadTable parses JSON data and inserts rows into the target table.
+func loadTable(ctx context.Context, pool *pgxpool.Pool, spec tableSpec, data []byte, colTypes map[string]columnInfo) error {
+	var rows []map[string]any
+	if err := json.Unmarshal(data, &rows); err != nil {
+		return fmt.Errorf("fixture: parse %s.%s: %w", spec.Schema, spec.Table, err)
+	}
+	if len(rows) == 0 {
+		return nil
+	}
+
+	qualifiedName := fmt.Sprintf("%s.%s", spec.Schema, spec.Table)
+
+	// Extract column order from the first row.
+	columns := orderedKeys(rows[0])
+
+	// Check if any columns need special handling (geometry).
+	hasGeo := false
+	geoColumns := make(map[string]bool)
+	for _, col := range columns {
+		key := fmt.Sprintf("%s.%s.%s", spec.Schema, spec.Table, col)
+		info, ok := colTypes[key]
+		if ok && (info.UDTName == "geometry" || info.UDTName == "geography") {
+			geoColumns[col] = true
+			hasGeo = true
+		}
+	}
+
+	if hasGeo {
+		return loadTableWithGeo(ctx, pool, qualifiedName, columns, rows, geoColumns)
+	}
+	return loadTableDirect(ctx, pool, qualifiedName, columns, rows, colTypes, spec)
+}
+
+// loadTableDirect loads data using pgx CopyFrom for tables without geometry.
+func loadTableDirect(ctx context.Context, pool *pgxpool.Pool, qualifiedName string, columns []string, rows []map[string]any, colTypes map[string]columnInfo, spec tableSpec) error {
+	// Build JSONB column set for proper marshaling.
+	jsonbCols := make(map[string]bool)
+	for _, col := range columns {
+		key := fmt.Sprintf("%s.%s.%s", spec.Schema, spec.Table, col)
+		info, ok := colTypes[key]
+		if ok && info.UDTName == "jsonb" {
+			jsonbCols[col] = true
+		}
+	}
+
+	copyRows := make([][]any, len(rows))
+	for i, row := range rows {
+		vals := make([]any, len(columns))
+		for j, col := range columns {
+			vals[j] = convertValue(row[col], jsonbCols[col])
+		}
+		copyRows[i] = vals
+	}
+
+	parts := strings.SplitN(qualifiedName, ".", 2)
+	copyCount, err := pool.CopyFrom(ctx,
+		pgx.Identifier{parts[0], parts[1]},
+		columns,
+		pgx.CopyFromRows(copyRows),
+	)
+	if err != nil {
+		return fmt.Errorf("fixture: copy %s: %w", qualifiedName, err)
+	}
+	if int(copyCount) != len(rows) {
+		return fmt.Errorf("fixture: copy %s: expected %d rows, got %d", qualifiedName, len(rows), copyCount)
+	}
+
+	return nil
+}

+// loadTableWithGeo loads data using a staging table for geometry columns.
+// Geometry values are stored as GeoJSON text in the JSON fixture, so we
+// load into a temp table with text columns, then INSERT with ST_GeomFromGeoJSON.
+func loadTableWithGeo(ctx context.Context, pool *pgxpool.Pool, qualifiedName string, columns []string, rows []map[string]any, geoColumns map[string]bool) error {
+	tx, err := pool.Begin(ctx)
+	if err != nil {
+		return fmt.Errorf("fixture: begin tx for %s: %w", qualifiedName, err)
+	}
+	defer tx.Rollback(ctx) //nolint:errcheck
+
+	// Create temp table mimicking the target but with text for geo columns.
+	tempName := "fixture_staging"
+	createCols := make([]string, len(columns))
+	for i, col := range columns {
+		// All columns are text in the staging table — geo columns get
+		// converted to geometry via ST_GeomFromGeoJSON in the final INSERT.
+		createCols[i] = fmt.Sprintf("%s text", col)
+	}
+
+	dropSQL := fmt.Sprintf("DROP TABLE IF EXISTS %s", tempName)
+	if _, err := tx.Exec(ctx, dropSQL); err != nil {
+		return fmt.Errorf("fixture: drop staging: %w", err)
+	}
+
+	createSQL := fmt.Sprintf("CREATE TEMP TABLE %s (%s)", tempName, strings.Join(createCols, ", "))
+	if _, err := tx.Exec(ctx, createSQL); err != nil {
+		return fmt.Errorf("fixture: create staging for %s: %w", qualifiedName, err)
+	}
+
+	// COPY into staging table with all values as text.
+	copyRows := make([][]any, len(rows))
+	for i, row := range rows {
+		vals := make([]any, len(columns))
+		for j, col := range columns {
+			v := row[col]
+			switch {
+			case v == nil:
+				vals[j] = nil
+			case geoColumns[col]:
+				// GeoJSON object → JSON string
+				b, err := json.Marshal(v)
+				if err != nil {
+					return fmt.Errorf("fixture: marshal geo %s row %d: %w", col, i, err)
+				}
+				vals[j] = string(b)
+			default:
+				vals[j] = toText(v)
+			}
+		}
+		copyRows[i] = vals
+	}
+
+	_, err = tx.CopyFrom(ctx,
+		pgx.Identifier{tempName},
+		columns,
+		pgx.CopyFromRows(copyRows),
+	)
+	if err != nil {
+		return fmt.Errorf("fixture: copy staging %s: %w", qualifiedName, err)
+	}
+
+	// Build INSERT from staging with ST_GeomFromGeoJSON for geo columns.
+	selectCols := make([]string, len(columns))
+	for i, col := range columns {
+		if geoColumns[col] {
+			selectCols[i] = fmt.Sprintf("ST_GeomFromGeoJSON(%s) AS %s", col, col)
+		} else {
+			// Cast text back to original column types.
+			selectCols[i] = col
+		}
+	}
+
+	insertSQL := fmt.Sprintf(
+		"INSERT INTO %s (%s) SELECT %s FROM %s",
+		qualifiedName,
+		strings.Join(columns, ", "),
+		strings.Join(selectCols, ", "),
+		tempName,
+	)
+	if _, err := tx.Exec(ctx, insertSQL); err != nil {
+		return fmt.Errorf("fixture: insert %s from staging: %w", qualifiedName, err)
+	}
+
+	if _, err := tx.Exec(ctx, dropSQL); err != nil {
+		return fmt.Errorf("fixture: cleanup staging: %w", err)
+	}
+
+	if err := tx.Commit(ctx); err != nil {
+		return fmt.Errorf("fixture: commit %s: %w", qualifiedName, err)
+	}
+
+	return nil
+}

+// toText converts a JSON-decoded value to its text representation for the
+// staging table. All staging columns are text, so every value must be a string.
+func toText(v any) string {
+	switch vt := v.(type) {
+	case string:
+		return vt
+	case float64:
+		return fmt.Sprintf("%v", vt)
+	case bool:
+		if vt {
+			return "true"
+		}
+		return "false"
+	default:
+		b, err := json.Marshal(v)
+		if err != nil {
+			return fmt.Sprintf("%v", v)
+		}
+		return string(b)
+	}
+}
+
+// convertValue converts a JSON-decoded value to a pgx-compatible type.
+func convertValue(v any, isJSONB bool) any {
+	if v == nil {
+		return nil
+	}
+	if isJSONB {
+		// JSONB columns: marshal the value back to a JSON byte slice
+		// which pgx will send as JSONB.
+		switch vt := v.(type) {
+		case map[string]any, []any:
+			b, err := json.Marshal(vt)
+			if err != nil {
+				return nil
+			}
+			return b
+		case string:
+			return []byte(vt)
+		default:
+			b, err := json.Marshal(v)
+			if err != nil {
+				return nil
+			}
+			return b
+		}
+	}
+	return v
+}
+
+// orderedKeys returns the keys of a map in a stable order.
+// JSON object key order is not guaranteed, so we sort alphabetically.
+// However, for fixture loading the order doesn't matter as long as it's
+// consistent between the column list and the data rows.
+func orderedKeys(m map[string]any) []string {
+	keys := make([]string, 0, len(m))
+	for k := range m {
+		keys = append(keys, k)
+	}
+	// Sort for determinism.
+	sortStrings(keys)
+	return keys
+}
+
+// sortStrings sorts a string slice in place (insertion sort — fine for small slices).
+func sortStrings(s []string) {
+	for i := 1; i < len(s); i++ {
+		for j := i; j > 0 && s[j] < s[j-1]; j-- {
+			s[j], s[j-1] = s[j-1], s[j]
+		}
+	}
+}
+
+// WriteJSON writes table data as JSON files to the given writer function.
+// This is used by the fixtureutil CLI to generate JSON from database queries.
+func WriteJSON(w io.Writer, rows []map[string]any) error {
+	enc := json.NewEncoder(w)
+	enc.SetIndent("", "  ")
+	return enc.Encode(rows)
+}

+func TestLoadOrder(t *testing.T) {
+	order := LoadOrder()
+
+	// Verify FK-safe ordering: parent tables must come before children.
+	indexOf := make(map[string]int, len(order))
+	for i, name := range order {
+		indexOf[name] = i
+	}
+
+	// location.state must be before search_base.institution (FK on state)
+	if indexOf["location.state"] >= indexOf["search_base.institution"] {
+		t.Error("location.state must come before search_base.institution")
+	}
+
+	// search_base.institution must be before dataset, annual_dataset, institution_alias
+	for _, child := range []string{
+		"search_base.dataset",
+		"search_base.annual_dataset",
+		"search_base.institution_alias",
+		"location.institution_normalized_alias",
+	} {
+		if indexOf["search_base.institution"] >= indexOf[child] {
+			t.Errorf("search_base.institution must come before %s", child)
+		}
+	}
+
+	// Verify expected count
+	if len(order) != 10 {
+		t.Errorf("expected 10 tables in load order, got %d", len(order))
+	}
+}

+// writeJSON writes each table's data as a JSON file in outputDir, one file per
+// table. Each file is a JSON array of objects where keys are column names.
+// After writing, each file is compressed with xz -9e.
+func writeJSON(outputDir string, tables []tableData, compress bool) error {
+	if err := os.MkdirAll(outputDir, 0o755); err != nil {
+		return fmt.Errorf("samplesql: create output dir: %w", err)
+	}
+
+	for _, td := range tables {
+		if len(td.Rows) == 0 {
+			continue
+		}
+
+		rows := make([]map[string]any, 0, len(td.Rows))
+		for _, row := range td.Rows {
+			obj := make(map[string]any, len(td.Columns))
+			for i, col := range td.Columns {
+				obj[col] = formatJSONValue(row[i])
+			}
+			rows = append(rows, obj)
+		}
+
+		filename := fmt.Sprintf("%s.%s.json", td.Schema, td.Table)
+		path := filepath.Join(outputDir, filename)
+
+		data, err := json.MarshalIndent(rows, "", "  ")
+		if err != nil {
+			return fmt.Errorf("samplesql: marshal %s: %w", filename, err)
+		}
+
+		if err := os.WriteFile(path, data, 0o644); err != nil {
+			return fmt.Errorf("samplesql: write %s: %w", filename, err)
+		}
+		fmt.Fprintf(os.Stderr, "  wrote %s (%d rows)\n", filename, len(td.Rows))
+
+		if compress {
+			if err := xzCompress(path); err != nil {
+				return fmt.Errorf("samplesql: compress %s: %w", filename, err)
+			}
+		}
+	}
+
+	return nil
+}

+// formatJSONValue converts a pgx row value into a JSON-safe representation.
+func formatJSONValue(v any) any {
+	if v == nil {
+		return nil
+	}
+
+	switch val := v.(type) {
+	case []byte:
+		// Binary data (e.g. PostGIS geometry WKB) — hex string.
+		// Prefer using ST_AsGeoJSON in the query instead of hitting this path.
+		return fmt.Sprintf("%x", val)
+	case time.Time:
+		return val.Format(time.RFC3339Nano)
+	case map[string]any:
+		// JSONB — keep as nested object (not escaped string)
+		return val
+	case []any:
+		// JSONB array — keep as nested array
+		return val
+	default:
+		// string, bool, int, float — already JSON-compatible
+		return val
+	}
+}

+// xzCompress compresses a file in-place using xz -9e (maximum compression).
+// The original file is replaced by the .xz version.
+func xzCompress(path string) error {
+	cmd := exec.Command("xz", "-9e", path) //nolint:gosec // G204: path is from trusted output dir
+	cmd.Stderr = os.Stderr
+	if err := cmd.Run(); err != nil {
+		return fmt.Errorf("xz: %w", err)
+	}
+	return nil
+}
