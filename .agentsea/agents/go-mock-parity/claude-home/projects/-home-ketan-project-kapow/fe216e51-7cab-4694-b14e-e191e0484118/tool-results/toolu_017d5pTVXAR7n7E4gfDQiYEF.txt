   443→// Table-driven BuildSQL test for every registry field
   444→// ─────────────────────────────────────────────────────────────────────────────
   445→
   446→func TestBuildSQLAllFields(t *testing.T) {
   447→	t.Parallel()
   448→	reg := search.NewRegistry()
   449→
   450→	for _, f := range reg.All() {
   451→		f := f // capture
   452→		t.Run(f.Name(), func(t *testing.T) {
   453→			raw := minimalFilter(f)
   454→			if raw == nil {
   455→				t.Skipf("no minimal filter for kind %s", f.Kind())
   456→			}
   457→
   458→			expr, args, err := f.BuildSQL(raw, 1)
   459→			if err != nil {
   460→				t.Fatalf("BuildSQL error: %v", err)
   461→			}
   462→			if strings.TrimSpace(expr) == "" {
   463→				t.Fatalf("BuildSQL returned empty expression")
   464→			}
   465→
   466→			// Verify placeholder count matches arg count.
   467→			np := countPlaceholders(expr)
   468→			if np != len(args) {
   469→				t.Errorf("placeholder count %d != arg count %d\nexpr: %s\nargs: %v",
   470→					np, len(args), expr, args)
   471→			}
   472→
   473→			// Verify no obvious SQL injection via string interpolation:
   474→			// all variable data must appear only as $N placeholders.
   475→			if strings.Contains(expr, "test_injection") {
   476→				t.Errorf("value leaked into SQL expression: %s", expr)
   477→			}
   478→		})
   479→	}
   480→}

   482→// minimalFilter returns a minimal valid JSON filter for the given field.
   483→func minimalFilter(f search.Field) json.RawMessage {
   484→	desc := f.Descriptor()
   485→	switch f.Kind() {
   486→	case search.KindFullText:
   487→		return mustJSON(search.FullTextFilter{Query: "test_query"})
   488→	case search.KindSubstring:
   489→		return mustJSON(search.SubstringFilter{Value: "test_value"})
   490→	case search.KindExactString:
   491→		return mustJSON(search.ExactStringFilter{Value: "test_value"})
   492→	case search.KindExactNumeric:
   493→		return mustJSON(search.ExactNumericFilter{Value: 1})
   494→	case search.KindNumericRange:
   495→		return mustJSON(search.NumericRangeFilter{Min: ptrF(1)})
   496→	case search.KindSingleEnum:
   497→		if len(desc.Options) > 0 {
   498→			return mustJSON(search.SingleEnumFilter{Value: desc.Options[0].Value})
   499→		}
   500→		return mustJSON(search.SingleEnumFilter{Value: "test"})
   501→	case search.KindMultiEnum:
   502→		if len(desc.Options) > 0 {
   503→			return mustJSON(search.MultiEnumFilter{Values: []string{desc.Options[0].Value}})
   504→		}
   505→		return mustJSON(search.MultiEnumFilter{Values: []string{"52.0101"}})
   506→	case search.KindGeo:
   507→		return mustJSON(search.GeoFilterInput{
   508→			Kind:   search.GeoFilterKindRadius,
   509→			Radius: &search.GeoRadiusFilter{Lat: 40.7, Lng: -74.0, RadiusKm: 50},
   510→		})
   511→	}
   512→	return nil
   513→}

  2114→func TestExportableDescriptors(t *testing.T) {
  2115→	t.Parallel()
  2116→	reg := search.NewRegistry()
  2117→	descs := reg.ExportableDescriptors()
  2118→
  2119→	if len(descs) == 0 {
  2120→		t.Fatal("ExportableDescriptors() returned empty slice")
  2121→	}
  2122→
  2123→	// Every descriptor must have Exportable=true and OutputDefault non-nil.
  2124→	for _, d := range descs {
  2125→		if d.Exportable == nil || !*d.Exportable {
  2126→			t.Errorf("descriptor %q has Exportable != true", d.Name)
  2127→		}
  2128→		if d.OutputDefault == nil {
  2129→			t.Errorf("descriptor %q has nil OutputDefault", d.Name)
  2130→		}
  2131→	}
  2132→
  2133→	// Non-exportable fields must be excluded.
  2134→	nameSet := make(map[string]bool, len(descs))
  2135→	for _, d := range descs {
  2136→		nameSet[d.Name] = true
  2137→	}
  2138→	for _, excluded := range []string{"location", "carnegie_basic", "congressional_district", "fields_of_study"} {
  2139→		if nameSet[excluded] {
  2140→			t.Errorf("non-exportable field %q should not appear in ExportableDescriptors()", excluded)
  2141→		}
  2142→	}
  2143→
  2144→	// Verify ordering: themes should be in AllThemes order.
  2145→	themeIndex := make(map[search.Theme]int, len(search.AllThemes))
  2146→	for i, th := range search.AllThemes {
  2147→		themeIndex[th] = i
  2148→	}
  2149→	for i := 1; i < len(descs); i++ {
  2150→		prevIdx := themeIndex[descs[i-1].Theme]
  2151→		currIdx := themeIndex[descs[i].Theme]
  2152→		if prevIdx > currIdx {
  2153→			t.Errorf("theme ordering violated: %q (%s) before %q (%s)",
  2154→				descs[i-1].Name, descs[i-1].Theme, descs[i].Name, descs[i].Theme)
  2155→		}
  2156→	}
  2157→}
