     2→//!
     3→//! Each input source (literals, boundary seeds, pool, random, Z3 solver, fuzzer,
     4→//! user-provided) implements [`InputStrategy`]. The [`MetaStrategy`] selects
     5→//! among registered strategies using outcome-based adaptive scoring.

    17→// ---------------------------------------------------------------------------
    18→// Strategy context — shared read-only state available to all strategies
    19→// ---------------------------------------------------------------------------
    20→
    21→/// Read-only context passed to strategies on each `next()` call.
    22→///
    23→/// Contains the function's type signature, extracted literals, and frontend
    24→/// capabilities so strategies can generate type-appropriate inputs without
    25→/// holding references to the full analysis.
    26→pub struct StrategyContext {
    27→    /// Parameter names and types for the function under exploration.
    28→    pub params: Vec<ParamInfo>,
    29→    /// Literal constants extracted from the function body by static analysis.
    30→    pub literals: Vec<LiteralValue>,
    31→    /// Frontend capabilities (used to gate complex-type generation).
    32→    pub capabilities: FrontendCapabilities,
    33→}
    34→
    35→impl StrategyContext {
    36→    /// Convenience: extract just the `TypeInfo` for each parameter.
    37→    pub fn param_types(&self) -> Vec<TypeInfo> {
    38→        self.params.iter().map(|p| p.typ.clone()).collect()
    39→    }
    40→}

    42→// ---------------------------------------------------------------------------
    43→// InputStrategy trait
    44→// ---------------------------------------------------------------------------
    45→
    46→/// A composable input generation strategy.
    47→///
    48→/// Strategies produce candidate input vectors and optionally react to execution
    49→/// feedback. The [`MetaStrategy`] polls strategies by adaptive priority and
    50→/// fans out feedback to all registered strategies after each execution.
    51→pub trait InputStrategy: Send {
    52→    /// Produce the next candidate input vector, or `None` if exhausted.
    53→    fn next(&mut self, ctx: &StrategyContext) -> Option<Vec<Value>>;
    54→
    55→    /// Receive feedback from an execution result.
    56→    ///
    57→    /// Strategies that don't use feedback (literals, random, etc.) leave the
    58→    /// default no-op implementation. Reactive strategies (Z3 solver, fuzzer)
    59→    /// override this to record constraints or interesting values.
    60→    fn feedback(&mut self, _inputs: &[Value], _result: &ExecuteResult, _was_new_path: bool) {}
    61→
    62→    /// Human-readable strategy name, used for discovery attribution and config keys.
    63→    fn name(&self) -> &str;
    64→
    65→    /// Total number of candidates this strategy will ever produce, if known.
    66→    ///
    67→    /// Returns `Some(n)` for exhaustible strategies (boundary seeds, literals,
    68→    /// pool, user-provided) and `None` for infinite strategies (random, fuzzer,
    69→    /// Z3 solver).
    70→    fn estimated_size(&self) -> Option<u64> {
    71→        None
    72→    }
    73→}

    79→/// Strategy that yields user-provided candidate inputs in order.
    80→///
    81→/// Exhaustible: returns `None` once all candidates have been yielded.
    82→/// Feedback is ignored — the input list is fixed at construction time.
    83→pub struct UserProvidedStrategy {
    84→    inputs: Vec<Vec<Value>>,
    85→    index: usize,
    86→}

    88→impl UserProvidedStrategy {
    89→    pub fn new(inputs: Vec<Vec<Value>>) -> Self {
    90→        Self { inputs, index: 0 }
    91→    }
    92→}

    94→impl InputStrategy for UserProvidedStrategy {
    95→    fn next(&mut self, _ctx: &StrategyContext) -> Option<Vec<Value>> {
    96→        if self.index < self.inputs.len() {
    97→            let v = self.inputs[self.index].clone();
    98→            self.index += 1;
    99→            Some(v)
   100→        } else {
   101→            None
   102→        }
   103→    }
   104→
   105→    fn name(&self) -> &str {
   106→        "user_provided"
   107→    }
   108→
   109→    fn estimated_size(&self) -> Option<u64> {
   110→        Some(self.inputs.len() as u64)
   111→    }
   112→}

   244→/// Configuration for the adaptive meta-strategy.
   245→#[derive(Debug, Clone)]
   246→pub struct MetaConfig {
   247→    /// Sliding window size for outcome-based scoring.
   248→    pub window_size: usize,
   249→    /// Minimum candidates a strategy must supply before deprioritization.
   250→    /// Clamped to `min(cold_start_threshold, estimated_size)` per strategy.
   251→    pub cold_start_threshold: u64,
   252→    /// Minimum allocation fraction for any non-exhausted strategy (range 0.01–0.05).
   253→    pub floor: f64,
   254→    /// When true, use adaptive scoring. When false (and no static_weights),
   255→    /// use pure round-robin.
   256→    pub adaptive: bool,
   257→    /// Optional static weight distribution. Keys are strategy names, values
   258→    /// are relative weights (normalized internally). Overrides adaptive scoring.
   259→    pub static_weights: Option<Vec<(String, f64)>>,
   260→}
   261→
   262→/// Sensible defaults: adaptive scoring, 100-element window, cold start of 20,
   263→/// 2% floor, no static weights.
   264→impl Default for MetaConfig {
   265→    fn default() -> Self {
   266→        Self {
   267→            window_size: 100,
   268→            cold_start_threshold: 20,
   269→            floor: 0.02,
   270→            adaptive: true,
   271→            static_weights: None,
   272→        }
   273→    }
   274→}

   276→// ---------------------------------------------------------------------------
   277→// MetaStrategy — adaptive selection over registered strategies
   278→// ---------------------------------------------------------------------------
   279→
   280→use rand::rngs::StdRng;
   281→use rand::{Rng, SeedableRng};
   282→use std::collections::VecDeque;
   283→
   284→/// Per-strategy scoring state.
   285→struct StrategyState {
   286→    /// The strategy implementation.
   287→    strategy: Box<dyn InputStrategy>,
   288→    /// Sliding window of recent outcomes (true = new path discovered).
   289→    window: VecDeque<bool>,
   290→    /// Total candidates supplied so far.
   291→    total_supplied: u64,
   292→    /// Whether the strategy has been exhausted (returned None from next()).
   293→    exhausted: bool,
   294→}
   295→
   296→/// Adaptive meta-strategy that selects among registered [`InputStrategy`]
   297→/// implementations using outcome-based scoring.
   298→///
   299→/// Selection algorithm:
   300→/// 1. **Static weights**: if configured, use weighted random sampling among
   301→///    non-exhausted strategies. On exhaustion, redistribute uniformly.
   302→/// 2. **Adaptive**: score = hit rate over sliding window. Cold-start strategies
   303→///    (fewer than threshold candidates) get the average score of graduated
   304→///    strategies. Floor prevents any strategy from dropping below a minimum
   305→///    allocation.
   306→/// 3. **Round-robin**: if adaptive is disabled and no static weights.
   307→pub struct MetaStrategy {
   308→    states: Vec<StrategyState>,
   309→    config: MetaConfig,
   310→    /// Round-robin index (used when adaptive=false and no static_weights).
   311→    rr_index: usize,
   312→}
   313→
   314→impl MetaStrategy {
   315→    pub fn new(strategies: Vec<Box<dyn InputStrategy>>, config: MetaConfig) -> Self {
   316→        let states = strategies
   317→            .into_iter()
   318→            .map(|s| StrategyState {
   319→                strategy: s,
   320→                window: VecDeque::with_capacity(config.window_size),
   321→                total_supplied: 0,
   322→                exhausted: false,
   323→            })
   324→            .collect();
   325→        Self {
   326→            states,
   327→            config,
   328→            rr_index: 0,
   329→        }
   330→    }
   331→
   332→    /// Select and invoke the next strategy, returning the candidate inputs
   333→    /// and the index of the strategy that produced them.
   334→    ///
   335→    /// Returns `None` when all strategies are exhausted.
   336→    pub fn next(&mut self, ctx: &StrategyContext, rng: &mut impl Rng) -> Option<(Vec<Value>, usize)> {
   337→        if self.all_exhausted() {
   338→            return None;
   339→        }
   340→
   341→        if self.config.static_weights.is_some() {
   342→            self.next_static(ctx, rng)
   343→        } else if self.config.adaptive {
   344→            self.next_adaptive(ctx, rng)
   345→        } else {
   346→            self.next_round_robin(ctx)
   347→        }
   348→    }
   349→
   350→    /// Fan out execution feedback to all registered strategies.
   351→    pub fn feedback(&mut self, inputs: &[Value], result: &ExecuteResult, was_new_path: bool) {
   352→        for state in &mut self.states {
   353→            if !state.exhausted {
   354→                state.strategy.feedback(inputs, result, was_new_path);
   355→            }
   356→        }
   357→    }
   358→
   359→    /// Record an outcome for a specific strategy (called after execution).
   360→    pub fn record_outcome(&mut self, strategy_idx: usize, was_new_path: bool) {
   361→        if let Some(state) = self.states.get_mut(strategy_idx) {
   362→            if state.window.len() >= self.config.window_size {
   363→                state.window.pop_front();
   364→            }
   365→            state.window.push_back(was_new_path);
   366→        }
   367→    }
   368→
   369→    /// The name of the strategy at the given index.
   370→    pub fn strategy_name(&self, idx: usize) -> &str {
   371→        self.states[idx].strategy.name()
   372→    }
   373→
   374→    /// Number of registered strategies.
   375→    pub fn len(&self) -> usize {
   376→        self.states.len()
   377→    }
   378→
   379→    /// Whether no strategies are registered.
   380→    pub fn is_empty(&self) -> bool {
   381→        self.states.is_empty()
   382→    }
   383→
   384→    fn all_exhausted(&self) -> bool {
   385→        self.states.iter().all(|s| s.exhausted)
   386→    }
   387→
   388→    /// Try to get the next candidate from a specific strategy, marking it
   389→    /// exhausted if it returns None.
   390→    fn try_next(&mut self, idx: usize, ctx: &StrategyContext) -> Option<Vec<Value>> {
   391→        let state = &mut self.states[idx];
   392→        if state.exhausted {
   393→            return None;
   394→        }
   395→        match state.strategy.next(ctx) {
   396→            Some(inputs) => {
   397→                state.total_supplied += 1;
   398→                Some(inputs)
   399→            }
   400→            None => {
   401→                state.exhausted = true;
   402→                None
   403→            }
   404→        }
   405→    }
   406→
   407→    // --- Selection modes ---
   408→
   409→    fn next_static(&mut self, ctx: &StrategyContext, rng: &mut impl Rng) -> Option<(Vec<Value>, usize)> {
   410→        let weights = self.config.static_weights.clone()?;
   411→
   412→        // Build weight vector for non-exhausted strategies.
   413→        loop {
   414→            let mut candidates: Vec<(usize, f64)> = Vec::new();
   415→            for (idx, state) in self.states.iter().enumerate() {
   416→                if state.exhausted {
   417→                    continue;
   418→                }
   419→                let weight = weights
   420→                    .iter()
   421→                    .find(|(name, _)| name == state.strategy.name())
   422→                    .map(|(_, w)| *w)
   423→                    .unwrap_or(1.0);
   424→                candidates.push((idx, weight));
   425→            }
   426→
   427→            if candidates.is_empty() {
   428→                return None;
   429→            }
   430→
   431→            let idx = weighted_select(&candidates, rng);
   432→            if let Some(inputs) = self.try_next(idx, ctx) {
   433→                return Some((inputs, idx));
   434→            }
   435→            // Strategy exhausted on this call; loop to rebuild candidates.
   436→        }
   437→    }
   438→
   439→    fn next_adaptive(&mut self, ctx: &StrategyContext, rng: &mut impl Rng) -> Option<(Vec<Value>, usize)> {
   440→        loop {
   441→            let scores = self.compute_scores();
   442→            let candidates: Vec<(usize, f64)> = scores
   443→                .into_iter()
   444→                .filter(|(idx, _)| !self.states[*idx].exhausted)
   445→                .collect();
   446→
   447→            if candidates.is_empty() {
   448→                return None;
   449→            }
   450→
   451→            let idx = weighted_select(&candidates, rng);
   452→            if let Some(inputs) = self.try_next(idx, ctx) {
   453→                return Some((inputs, idx));
   454→            }
   455→            // Strategy exhausted; loop to recompute.
   456→        }
   457→    }
   458→
   459→    fn next_round_robin(&mut self, ctx: &StrategyContext) -> Option<(Vec<Value>, usize)> {
   460→        let n = self.states.len();
   461→        for _ in 0..n {
   462→            let idx = self.rr_index % n;
   463→            self.rr_index = self.rr_index.wrapping_add(1);
   464→            if let Some(inputs) = self.try_next(idx, ctx) {
   465→                return Some((inputs, idx));
   466→            }
   467→        }
   468→        None
   469→    }
   470→
   471→    /// Compute adaptive scores for all non-exhausted strategies.
   472→    fn compute_scores(&self) -> Vec<(usize, f64)> {
   473→        // First pass: compute hit rates for graduated strategies to derive average.
   474→        let mut graduated_scores: Vec<f64> = Vec::new();
   475→        for (idx, state) in self.states.iter().enumerate() {
   476→            if state.exhausted {
   477→                continue;
   478→            }
   479→            let threshold = self.effective_cold_start(idx);
   480→            if state.total_supplied >= threshold {
   481→                graduated_scores.push(hit_rate(&state.window));
   482→            }
   483→        }
   484→        let avg_score = if graduated_scores.is_empty() {
   485→            1.0
   486→        } else {
   487→            graduated_scores.iter().sum::<f64>() / graduated_scores.len() as f64
   488→        };
   489→
   490→        // Second pass: assign scores.
   491→        let mut result = Vec::new();
   492→        for (idx, state) in self.states.iter().enumerate() {
   493→            if state.exhausted {
   494→                continue;
   495→            }
   496→            let threshold = self.effective_cold_start(idx);
   497→            let score = if state.total_supplied < threshold {
   498→                avg_score
   499→            } else {
   500→                hit_rate(&state.window)
   501→            };
   502→            result.push((idx, score.max(self.config.floor)));
   503→        }
   504→        result
   505→    }
   506→
   507→    /// Effective cold-start threshold for a strategy: min(configured, estimated_size).
   508→    fn effective_cold_start(&self, idx: usize) -> u64 {
   509→        let base = self.config.cold_start_threshold;
   510→        match self.states[idx].strategy.estimated_size() {
   511→            Some(size) => base.min(size),
   512→            None => base,
   513→        }
   514→    }
   515→}

   517→// ---------------------------------------------------------------------------
   518→// Helpers
   519→// ---------------------------------------------------------------------------

   521→/// Hit rate: fraction of `true` values in the window. Returns 0.0 for empty windows.
   522→fn hit_rate(window: &VecDeque<bool>) -> f64 {
   523→    if window.is_empty() {
   524→        return 0.0;
   525→    }
   526→    let hits = window.iter().filter(|&&b| b).count();
   527→    hits as f64 / window.len() as f64
   528→}
   529→
   530→/// Weighted random selection. Returns the index from `candidates`.
   531→/// Panics if `candidates` is empty.
   532→fn weighted_select(candidates: &[(usize, f64)], rng: &mut impl Rng) -> usize {
   533→    let total: f64 = candidates.iter().map(|(_, w)| w).sum();
   534→    if total <= 0.0 {
   535→        // All weights zero — fall back to uniform.
   536→        let pick = rng.random_range(0..candidates.len());
   537→        return candidates[pick].0;
   538→    }
   539→    let mut roll: f64 = rng.random_range(0.0..total);
   540→    for &(idx, weight) in candidates {
   541→        roll -= weight;
   542→        if roll <= 0.0 {
   543→            return idx;
   544→        }
   545→    }
   546→    // Floating-point rounding: return last candidate.
   547→    candidates.last().unwrap().0
   548→}

   587→// ---------------------------------------------------------------------------
   588→// FuzzerStrategy — infinite, feedback-driven mutation and crossover
   589→// ---------------------------------------------------------------------------
   590→
   591→/// Default per-parameter mutation probability for the fuzzer strategy.
   592→const FUZZER_MUTATION_RATE: f64 = 0.3;
   593→
   594→/// Default crossover probability when two parents are available.
   595→const FUZZER_CROSSOVER_RATE: f64 = 0.7;
   596→
   597→/// Infinite strategy that mutates and crosses over inputs which hit Unknown
   598→/// constraints or discovered new paths.
   599→///
   600→/// Feedback records interesting inputs (those reaching branches with no
   601→/// symbolic constraint, or discovering new coverage). `next()` draws from
   602→/// the interesting pool, generating mutations via `fuzz_inputs()`,
   603→/// `mutate_inputs()`, and `crossover_inputs()`.
   604→pub struct FuzzerStrategy {
   605→    rng: StdRng,
   606→    /// Inputs that hit Unknown constraints or new paths — seeds for mutation.
   607→    interesting: Vec<Vec<Value>>,
   608→    /// Pending mutations generated from a feedback round, drained by `next()`.
   609→    pending: VecDeque<Vec<Value>>,
   610→}
   611→
   612→impl FuzzerStrategy {
   613→    pub fn new(seed: Option<u64>) -> Self {
   614→        let rng = match seed {
   615→            Some(s) => StdRng::seed_from_u64(s),
   616→            None => StdRng::from_os_rng(),
   617→        };
   618→        Self {
   619→            rng,
   620→            interesting: Vec::new(),
   621→            pending: VecDeque::new(),
   622→        }
   623→    }
   624→
   625→    /// Refill `self.pending` by mutating a random interesting input.
   626→    fn refill(&mut self, ctx: &StrategyContext) {
   627→        if self.interesting.is_empty() {
   628→            return;
   629→        }
   630→
   631→        let idx = self.rng.random_range(0..self.interesting.len());
   632→        let base = self.interesting[idx].clone();
   633→
   634→        // 1. Simple value-level mutations (from orchestrator's fuzz_inputs).
   635→        for mutated in fuzz_inputs(&base) {
   636→            self.pending.push_back(mutated);
   637→        }
   638→
   639→        // 2. Type-aware mutation via input_gen.
   640→        let params: Vec<ParamInfo> = ctx.params.clone();
   641→        let mutated = mutate_inputs(&base, &params, FUZZER_MUTATION_RATE, &[], &mut self.rng);
   642→        self.pending.push_back(mutated);
   643→
   644→        // 3. Crossover when at least two interesting inputs exist.
   645→        if self.interesting.len() >= 2 {
   646→            let other_idx = loop {
   647→                let candidate = self.rng.random_range(0..self.interesting.len());
   648→                if candidate != idx || self.interesting.len() == 1 {
   649→                    break candidate;
   650→                }
   651→            };
   652→            let (child_a, _child_b) = crossover_inputs(
   653→                &base,
   654→                &self.interesting[other_idx],
   655→                &params,
   656→                FUZZER_CROSSOVER_RATE,
   657→                &mut self.rng,
   658→            );
   659→            self.pending.push_back(child_a);
   660→        }
   661→    }
   662→}
   663→
   664→impl InputStrategy for FuzzerStrategy {
   665→    fn next(&mut self, ctx: &StrategyContext) -> Option<Vec<Value>> {
   666→        if let Some(candidate) = self.pending.pop_front() {
   667→            return Some(candidate);
   668→        }
   669→
   670→        // Try to refill from interesting pool.
   671→        self.refill(ctx);
   672→        self.pending.pop_front()
   673→    }
   674→
   675→    fn feedback(&mut self, inputs: &[Value], result: &ExecuteResult, was_new_path: bool) {
   676→        // Record inputs that discovered new coverage.
   677→        if was_new_path {
   678→            self.interesting.push(inputs.to_vec());
   679→            return;
   680→        }
   681→
   682→        // Record inputs that hit Unknown constraints (branches the solver
   683→        // cannot handle, so fuzzing is the only way to explore them).
   684→        let has_unknown = result.branch_path.iter().any(|decision| {
   685→            matches!(decision.constraint, SymConstraint::Unknown { .. })
   686→        });
   687→        if has_unknown {
   688→            self.interesting.push(inputs.to_vec());
   689→        }
   690→    }
   691→
   692→    fn name(&self) -> &str {
   693→        "fuzzer"
   694→    }
   695→}
