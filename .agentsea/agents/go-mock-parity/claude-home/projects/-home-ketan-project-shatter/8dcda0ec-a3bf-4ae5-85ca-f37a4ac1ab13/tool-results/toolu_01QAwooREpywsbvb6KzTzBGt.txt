     9→/// Log level for the frontend, controlled by SHATTER_LOG_LEVEL env var.
    10→#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
    11→enum FrontendLogLevel {
    12→    Error = 0,
    13→    Warn = 1,
    14→    Info = 2,
    15→    Debug = 3,
    16→    Trace = 4,
    17→}
    18→
    19→impl FrontendLogLevel {
    20→    fn from_env() -> Self {
    21→        match std::env::var("SHATTER_LOG_LEVEL")
    22→            .unwrap_or_default()
    23→            .to_lowercase()
    24→            .as_str()
    25→        {
    26→            "error" => FrontendLogLevel::Error,
    27→            "warn" => FrontendLogLevel::Warn,
    28→            "debug" => FrontendLogLevel::Debug,
    29→            "trace" => FrontendLogLevel::Trace,
    30→            _ => FrontendLogLevel::Info,
    31→        }
    32→    }
    33→}
    34→
    35→const DEFAULT_EXEC_TIMEOUT_MS: u64 = 5000;
    36→
    37→/// Read the execution timeout from `SHATTER_EXEC_TIMEOUT` env var (seconds).
    38→/// Returns milliseconds. Falls back to `DEFAULT_EXEC_TIMEOUT_MS` if unset, invalid, zero, or negative.
    39→fn exec_timeout_from_env() -> u64 {
    40→    match std::env::var("SHATTER_EXEC_TIMEOUT") {
    41→        Ok(val) => match val.parse::<f64>() {
    42→            Ok(secs) if secs > 0.0 => (secs * 1000.0) as u64,
    43→            _ => DEFAULT_EXEC_TIMEOUT_MS,
    44→        },
    45→        Err(_) => DEFAULT_EXEC_TIMEOUT_MS,
    46→    }
    47→}

    49→/// Write instrumented source to a temp directory and return the output path.
    50→fn write_instrumented_temp(filename: &str, source: &str) -> io::Result<String> {
    51→    let dir = std::env::temp_dir().join(format!(
    52→        "shatter-instrument-{}",
    53→        std::process::id()
    54→    ));
    55→    std::fs::create_dir_all(&dir)?;
    56→    let out_path = dir.join(filename);
    57→    std::fs::write(&out_path, source)?;
    58→    Ok(out_path.to_string_lossy().into_owned())
    59→}

    61→/// Processes protocol requests from stdin and writes responses to stdout.
    62→pub struct Handler<R, W, L> {
    63→    reader: BufReader<R>,
    64→    writer: W,
    65→    log: L,
    66→    log_level: FrontendLogLevel,
    67→    exec_timeout_ms: u64,
    68→    wasm_cache: WasmCache,
    69→    native_registry: Option<NativeRegistry>,
    70→    /// Remembered from the most recent Analyze or Instrument request so Execute
    71→    /// can fall back when the core omits the file field (which it always does).
    72→    last_file: Option<String>,
    73→}
    74→
    75→impl<R: io::Read, W: io::Write, L: io::Write> Handler<R, W, L> {
    76→    pub fn new(reader: R, writer: W, log: L) -> Self {

   123→    /// Process requests until shutdown or EOF. Returns Ok(()) on clean shutdown.
   124→    pub fn run(mut self) -> io::Result<()> {
   125→        self.log_at(FrontendLogLevel::Debug, &format!(
   126→            "Starting Rust frontend (protocol {PROTOCOL_VERSION})"
   127→        ));
   128→
   129→        let mut line = String::new();
   130→        loop {
   131→            line.clear();
   132→            let bytes_read = self.reader.read_line(&mut line)?;
   133→            if bytes_read == 0 {
   134→                // EOF
   135→                self.log_at(FrontendLogLevel::Debug, "Stdin closed, exiting");
   136→                return Ok(());
   137→            }
   138→
   139→            let trimmed = line.trim();
   140→            if trimmed.is_empty() {
   141→                continue;
   142→            }
   143→
   144→            self.logf(&format!("Received: {trimmed}"));
   145→
   146→            let req: Request = match serde_json::from_str(trimmed) {
   147→                Ok(r) => r,
   148→                Err(e) => {
   149→                    self.logf(&format!("Failed to parse request: {e}"));
   150→                    let err_resp = Response {
   151→                        id: 0,
   152→                        status: "error".to_string(),
   153→                        code: Some("invalid_request".to_string()),
   154→                        message: Some(format!("Invalid JSON: {e}")),
   155→                        ..Response::base(0)
   156→                    };
   157→                    self.send(&err_resp)?;
   158→                    continue;
   159→                }
   160→            };
   161→
   162→            let (resp, shutdown) = self.dispatch(&req);
   163→            self.send(&resp)?;
   164→
   165→            if shutdown {
   166→                self.log_at(FrontendLogLevel::Debug, "Shutting down");
   167→                return Ok(());
   168→            }
   169→        }
   170→    }
   171→
   172→    fn dispatch(&mut self, req: &Request) -> (Response, bool) {
   173→        let mut resp = Response::base(req.id);
   174→
   175→        if req.protocol_version != protocol::PROTOCOL_VERSION {
   176→            resp.status = "error".to_string();
   177→            resp.code = Some("version_mismatch".to_string());
   178→            resp.message = Some(format!(
   179→                "unsupported protocol version {:?}, expected {:?}",
   180→                req.protocol_version,
   181→                protocol::PROTOCOL_VERSION,
   182→            ));
   183→            return (resp, false);
   184→        }
   185→
   186→        match req.command.as_str() {
   187→            "handshake" => (self.handle_handshake(resp), false),
   188→            "analyze" => (self.handle_analyze(resp, req), false),
   189→            "instrument" => (self.handle_instrument(resp, req), false),
   190→            "execute" => (self.handle_execute(resp, req), false),
   191→            "setup" => (self.handle_setup(resp, req), false),
   192→            "teardown" => (self.handle_teardown(resp, req), false),
   193→            "generate" => (self.handle_generate(resp, req), false),
   194→            "shutdown" => (self.handle_shutdown(resp), true),
   195→            _ => {
   196→                resp.status = "error".to_string();
   197→                resp.code = Some("invalid_request".to_string());
   198→                resp.message = Some(format!("unknown command: {}", req.command));
   199→                (resp, false)
   200→            }
   201→        }
   202→    }
   203→
   204→    fn handle_handshake(&self, mut resp: Response) -> Response {
   205→        resp.status = "handshake".to_string();
   206→        resp.frontend_version = Some(FRONTEND_VERSION.to_string());
   207→        resp.language = Some(FRONTEND_LANGUAGE.to_string());
   208→        resp.capabilities = Some(vec![
   209→            "analyze".to_string(),
   210→            "execute".to_string(),
   211→            "generate".to_string(),
   212→            "instrument".to_string(),
   213→        ]);
   214→        resp
   215→    }
   216→
   217→    fn handle_analyze(&mut self, mut resp: Response, req: &Request) -> Response {
   218→        let file_path = match &req.file {
   219→            Some(f) => f,
   220→            None => {
   221→                resp.status = "error".to_string();
   222→                resp.code = Some("invalid_request".to_string());
   223→                resp.message = Some("analyze command requires a file path".to_string());
   224→                return resp;
   225→            }
   226→        };
   227→
   228→        self.last_file = Some(file_path.clone());
   229→
   230→        let path = std::path::Path::new(file_path);
   231→        if !path.exists() {
   232→            resp.status = "error".to_string();
   233→            resp.code = Some("file_not_found".to_string());
   234→            resp.message = Some(format!("file not found: {file_path}"));
   235→            return resp;
   236→        }
   237→
   238→        match crate::analyzer::analyze_file(path, req.function.as_deref()) {
   239→            Ok(functions) => {
   240→                resp.status = "analyze".to_string();
   241→                resp.functions = Some(functions);
   242→                resp
   243→            }
   244→            Err(e) => {
   245→                resp.status = "error".to_string();
   246→                resp.code = Some(
   247→                    match &e {
   248→                        crate::analyzer::AnalyzeError::FileNotFound(_) => "file_not_found",
   249→                        crate::analyzer::AnalyzeError::ReadError(_) => "internal_error",
   250→                        crate::analyzer::AnalyzeError::ParseError(_) => "parse_error",
   251→                        crate::analyzer::AnalyzeError::FunctionNotFound(_) => "function_not_found",
   252→                    }
   253→                    .to_string(),
   254→                );
   255→                resp.message = Some(e.to_string());
   256→                resp
   257→            }
   258→        }
   259→    }

   261→    fn handle_instrument(&mut self, mut resp: Response, req: &Request) -> Response {
   262→        let file_path = match &req.file {
   263→            Some(f) => f,
   264→            None => {
   265→                resp.status = "error".to_string();
   266→                resp.code = Some("invalid_request".to_string());
   267→                resp.message = Some("instrument command requires a file path".to_string());
   268→                return resp;
   269→            }
   270→        };
   271→
   272→        self.last_file = Some(file_path.clone());
   273→
   274→        let path = std::path::Path::new(file_path);
   275→        if !path.exists() {
   276→            resp.status = "error".to_string();
   277→            resp.code = Some("file_not_found".to_string());
   278→            resp.message = Some(format!("file not found: {file_path}"));
   279→            return resp;
   280→        }
   281→
   282→        match crate::instrument::instrument_file(path, req.function.as_deref()) {
   283→            Ok(result) => {
   284→                // Write instrumented source to a temp file
   285→                let source_name = path
   286→                    .file_name()
   287→                    .and_then(|n| n.to_str())
   288→                    .unwrap_or("instrumented.rs");
   289→                match write_instrumented_temp(source_name, &result.source) {
   290→                    Ok(output_path) => {
   291→                        resp.status = "instrument".to_string();
   292→                        resp.instrumented = Some(true);
   293→                        resp.output_file = Some(output_path);
   294→                        resp.message = Some(format!(
   295→                            "instrumented {} branch points",
   296→                            result.branch_count
   297→                        ));
   298→                        resp
   299→                    }
   300→                    Err(e) => {
   301→                        resp.status = "error".to_string();
   302→                        resp.code = Some("internal_error".to_string());
   303→                        resp.message = Some(format!("failed to write instrumented output: {e}"));
   304→                        resp
   305→                    }
   306→                }
   307→            }
   308→            Err(e) => {
   309→                resp.status = "error".to_string();
   310→                resp.code = Some(match &e {
   311→                    crate::instrument::InstrumentError::FileNotFound(_) => "file_not_found",
   312→                    crate::instrument::InstrumentError::ReadError(_) => "internal_error",
   313→                    crate::instrument::InstrumentError::ParseError(_) => "parse_error",
   314→                }.to_string());
   315→                resp.message = Some(e.to_string());
   316→                resp
   317→            }
   318→        }
   319→    }

   321→    fn handle_execute(&self, mut resp: Response, req: &Request) -> Response {
   322→        let file_path = match req.file.as_ref().or(self.last_file.as_ref()) {
   323→            Some(f) => f,
   324→            None => {
   325→                resp.status = "error".to_string();
   326→                resp.code = Some("invalid_request".to_string());
   327→                resp.message = Some("execute command requires a file path (none provided and no prior analyze/instrument)".to_string());
   328→                return resp;
   329→            }
   330→        };
   331→        let function_name = match &req.function {
   332→            Some(f) => f,
   333→            None => {
   334→                resp.status = "error".to_string();
   335→                resp.code = Some("invalid_request".to_string());
   336→                resp.message = Some("execute command requires a function name".to_string());
   337→                return resp;
   338→            }
   339→        };
   340→
   341→        if self.log_level >= FrontendLogLevel::Debug {
   342→            eprintln!(
   343→                "[shatter-rust] Executing {function_name} in {file_path} with {} inputs",
   344→                req.inputs.len()
   345→            );
   346→        }
   347→
   348→        match crate::executor::execute_function(
   349→            file_path,
   350→            function_name,
   351→            &req.inputs,
   352→            &req.mocks,
   353→            self.exec_timeout_ms,
   354→        ) {
   355→            Ok(result) => {
   356→                resp.status = "execute".to_string();
   357→                resp.return_value = result.return_value;
   358→                resp.thrown_error = result.thrown_error;
   359→                resp.branch_path = Some(result.branch_path);
   360→                resp.lines_executed = Some(result.lines_executed);
   361→                resp.calls_to_external = Some(result.calls_to_external);
   362→                resp.path_constraints = Some(result.path_constraints);
   363→                resp.side_effects = Some(result.side_effects);
   364→                resp.performance = Some(result.performance);
   365→                resp
   366→            }
   367→            Err(crate::executor::ExecuteError::FileError(msg)) => {
   368→                resp.status = "error".to_string();
   369→                resp.code = Some("file_not_found".to_string());
   370→                resp.message = Some(msg);
   371→                resp
   372→            }
   373→            Err(crate::executor::ExecuteError::CompilationFailed(msg)) => {
   374→                resp.status = "error".to_string();
   375→                resp.code = Some("compilation_error".to_string());
   376→                resp.message = Some(msg);
   377→                resp
   378→            }
   379→            Err(crate::executor::ExecuteError::NonExecutable(msg)) => {
   380→                resp.status = "error".to_string();
   381→                resp.code = Some("non_executable".to_string());
   382→                resp.message = Some(msg);
   383→                resp
   384→            }
   385→            Err(e) => {
   386→                resp.status = "error".to_string();
   387→                resp.code = Some("internal_error".to_string());
   388→                resp.message = Some(e.to_string());
   389→                resp
   390→            }
   391→        }
   392→    }

   485→    fn handle_generate(&mut self, mut resp: Response, req: &Request) -> Response {
   486→        let file_path = match &req.file {
   487→            Some(f) => f,
   488→            None => {
   489→                resp.status = "error".to_string();
   490→                resp.code = Some("invalid_request".to_string());
   491→                resp.message = Some("generate command requires a file path".to_string());
   492→                return resp;
   493→            }
   494→        };
   495→        let func_name = match &req.name {
   496→            Some(n) => n.clone(),
   497→            None => {
   498→                resp.status = "error".to_string();
   499→                resp.code = Some("invalid_request".to_string());
   500→                resp.message = Some("generate command requires a name".to_string());
   501→                return resp;
   502→            }
   503→        };
   504→
   505→        let path = std::path::Path::new(file_path);
   506→        let ext = path.extension().and_then(|e| e.to_str());
   507→
   508→        match ext {
   509→            Some("wasm") => {
   510→                match self.wasm_cache.generate(path, &func_name, req.recipe.as_ref()) {
   511→                    Ok((value, generator_id, recipe)) => {
   512→                        resp.status = "generate".to_string();
   513→                        resp.value = Some(value);
   514→                        resp.generator_id = Some(generator_id);
   515→                        resp.recipe = recipe;
   516→                        resp
   517→                    }
   518→                    Err(e) => {
   519→                        resp.status = "error".to_string();
   520→                        resp.code = Some("internal_error".to_string());
   521→                        resp.message = Some(e);
   522→                        resp
   523→                    }
   524→                }
   525→            }
   526→            Some("rs") => {
   527→                if let Some(ref registry) = self.native_registry {
   528→                    match registry.generate(&func_name, req.recipe.clone()) {
   529→                        Ok((value, generator_id, recipe)) => {
   530→                            resp.status = "generate".to_string();
   531→                            resp.value = Some(value);
   532→                            resp.generator_id = Some(generator_id);
   533→                            resp.recipe = Some(recipe);
   534→                            resp
   535→                        }
   536→                        Err(e) => {
   537→                            resp.status = "error".to_string();
   538→                            resp.code = Some("internal_error".to_string());
   539→                            resp.message = Some(e);
   540→                            resp
   541→                        }
   542→                    }
   543→                } else {
   544→                    resp.status = "error".to_string();
   545→                    resp.code = Some("internal_error".to_string());
   546→                    resp.message = Some(format!(
   547→                        "native generator {func_name:?} requires a custom build (run `shatter build-frontend rust`)"
   548→                    ));
   549→                    resp
   550→                }
   551→            }
   552→            _ => {
   553→                resp.status = "error".to_string();
   554→                resp.code = Some("invalid_request".to_string());
   555→                resp.message = Some(format!(
   556→                    "unsupported generator file type: {}",
   557→                    file_path
   558→                ));
   559→                resp
   560→            }
   561→        }
   562→    }

   564→    fn handle_shutdown(&self, mut resp: Response) -> Response {
   565→        resp.status = "shutdown_ack".to_string();
   566→        resp

   569→    fn send(&mut self, resp: &Response) -> io::Result<()> {
   570→        let data = serde_json::to_string(resp)
   571→            .map_err(io::Error::other)?;
   572→        self.logf(&format!("Sent: {data}"));
   573→        writeln!(self.writer, "{data}")?;
   574→        self.writer.flush()
   575→    }

   577→    /// Log at TRACE level (protocol messages).
   578→    fn logf(&mut self, msg: &str) {
   579→        self.log_at(FrontendLogLevel::Trace, msg);
   580→    }

   582→    /// Log at a specific level.
   583→    fn log_at(&mut self, level: FrontendLogLevel, msg: &str) {
   584→        if self.log_level >= level {
   585→            let _ = writeln!(self.log, "[shatter-rust] {msg}");
