    19→// ---------------------------------------------------------------------------
    20→// Strategy context — shared read-only state available to all strategies
    21→// ---------------------------------------------------------------------------
    22→
    23→/// Read-only context passed to strategies on each `next()` call.
    24→///
    25→/// Contains the function's type signature, extracted literals, and frontend
    26→/// capabilities so strategies can generate type-appropriate inputs without
    27→/// holding references to the full analysis.
    28→pub struct StrategyContext {
    29→    /// Parameter names and types for the function under exploration.
    30→    pub params: Vec<ParamInfo>,
    31→    /// Literal constants extracted from the function body by static analysis.
    32→    pub literals: Vec<LiteralValue>,
    33→    /// Frontend capabilities (used to gate complex-type generation).
    34→    pub capabilities: FrontendCapabilities,
    35→}
    36→
    37→impl StrategyContext {
    38→    /// Convenience: extract just the `TypeInfo` for each parameter.
    39→    pub fn param_types(&self) -> Vec<TypeInfo> {
    40→        self.params.iter().map(|p| p.typ.clone()).collect()
    41→    }
    42→}

    44→// ---------------------------------------------------------------------------
    45→// InputStrategy trait
    46→// ---------------------------------------------------------------------------
    47→
    48→/// A composable input generation strategy.
    49→///
    50→/// Strategies produce candidate input vectors and optionally react to execution
    51→/// feedback. The [`MetaStrategy`] polls strategies by adaptive priority and
    52→/// fans out feedback to all registered strategies after each execution.
    53→pub trait InputStrategy: Send {
    54→    /// Produce the next candidate input vector, or `None` if exhausted.
    55→    fn next(&mut self, ctx: &StrategyContext) -> Option<Vec<Value>>;
    56→
    57→    /// Receive feedback from an execution result.
    58→    ///
    59→    /// Strategies that don't use feedback (literals, random, etc.) leave the
    60→    /// default no-op implementation. Reactive strategies (Z3 solver, fuzzer)
    61→    /// override this to record constraints or interesting values.
    62→    fn feedback(&mut self, _inputs: &[Value], _result: &ExecuteResult, _was_new_path: bool) {}
    63→
    64→    /// Human-readable strategy name, used for discovery attribution and config keys.
    65→    fn name(&self) -> &str;
    66→
    67→    /// Total number of candidates this strategy will ever produce, if known.
    68→    ///
    69→    /// Returns `Some(n)` for exhaustible strategies (boundary seeds, literals,
    70→    /// pool, user-provided) and `None` for infinite strategies (random, fuzzer,
    71→    /// Z3 solver).
    72→    fn estimated_size(&self) -> Option<u64> {
    73→        None
    74→    }
    75→}

    81→/// Strategy that yields user-provided candidate inputs in order.

    93→    }

    98→        if self.index < self.inputs.len() {
    99→            let v = self.inputs[self.index].clone();
   100→            self.index += 1;
   101→            Some(v)
   102→        } else {
   103→            None
   104→        }
   105→    }

   107→    fn name(&self) -> &str {
   108→        "user_provided"
   109→    }

   113→    }
   114→}

   246→/// Configuration for the adaptive meta-strategy.
   247→#[derive(Debug, Clone)]
   248→pub struct MetaConfig {
   249→    /// Sliding window size for outcome-based scoring.
   250→    pub window_size: usize,
   251→    /// Minimum candidates a strategy must supply before deprioritization.
   252→    /// Clamped to `min(cold_start_threshold, estimated_size)` per strategy.
   253→    pub cold_start_threshold: u64,
   254→    /// Minimum allocation fraction for any non-exhausted strategy (range 0.01–0.05).
   255→    pub floor: f64,
   256→    /// When true, use adaptive scoring. When false (and no static_weights),
   257→    /// use pure round-robin.
   258→    pub adaptive: bool,
   259→    /// Optional static weight distribution. Keys are strategy names, values
   260→    /// are relative weights (normalized internally). Overrides adaptive scoring.
   261→    pub static_weights: Option<Vec<(String, f64)>>,
   262→}
   263→
   264→/// Sensible defaults: adaptive scoring, 100-element window, cold start of 20,
   265→/// 2% floor, no static weights.
   266→impl Default for MetaConfig {
   267→    fn default() -> Self {
   268→        Self {
   269→            window_size: 100,
   270→            cold_start_threshold: 20,
   271→            floor: 0.02,
   272→            adaptive: true,
   273→            static_weights: None,
   274→        }
   275→    }
   276→}

   278→// ---------------------------------------------------------------------------
   279→// MetaStrategy — adaptive selection over registered strategies
   280→// ---------------------------------------------------------------------------
   281→
   282→use rand::rngs::StdRng;
   283→use rand::{Rng, SeedableRng};
   284→use std::collections::VecDeque;
   285→
   286→/// Per-strategy scoring state.
   287→struct StrategyState {
   288→    /// The strategy implementation.
   289→    strategy: Box<dyn InputStrategy>,
   290→    /// Sliding window of recent outcomes (true = new path discovered).
   291→    window: VecDeque<bool>,
   292→    /// Total candidates supplied so far.
   293→    total_supplied: u64,
   294→    /// Whether the strategy has been exhausted (returned None from next()).
   295→    exhausted: bool,
   296→}
   297→
   298→/// Adaptive meta-strategy that selects among registered [`InputStrategy`]
   299→/// implementations using outcome-based scoring.
   300→///
   301→/// Selection algorithm:
   302→/// 1. **Static weights**: if configured, use weighted random sampling among
   303→///    non-exhausted strategies. On exhaustion, redistribute uniformly.
   304→/// 2. **Adaptive**: score = hit rate over sliding window. Cold-start strategies
   305→///    (fewer than threshold candidates) get the average score of graduated
   306→///    strategies. Floor prevents any strategy from dropping below a minimum
   307→///    allocation.
   308→/// 3. **Round-robin**: if adaptive is disabled and no static weights.
   309→pub struct MetaStrategy {
   310→    states: Vec<StrategyState>,
   311→    config: MetaConfig,
   312→    /// Round-robin index (used when adaptive=false and no static_weights).
   313→    rr_index: usize,
   314→}
   315→
   316→impl MetaStrategy {
   317→    pub fn new(strategies: Vec<Box<dyn InputStrategy>>, config: MetaConfig) -> Self {
   318→        let states = strategies
   319→            .into_iter()
   320→            .map(|s| StrategyState {
   321→                strategy: s,
   322→                window: VecDeque::with_capacity(config.window_size),
   323→                total_supplied: 0,
   324→                exhausted: false,
   325→            })
   326→            .collect();
   327→        Self {
   328→            states,
   329→            config,
   330→            rr_index: 0,
   331→        }
   332→    }
   333→
   334→    /// Select and invoke the next strategy, returning the candidate inputs
   335→    /// and the index of the strategy that produced them.
   336→    ///
   337→    /// Returns `None` when all strategies are exhausted.
   338→    pub fn next(&mut self, ctx: &StrategyContext, rng: &mut impl Rng) -> Option<(Vec<Value>, usize)> {
   339→        if self.all_exhausted() {
   340→            return None;
   341→        }
   342→
   343→        if self.config.static_weights.is_some() {
   344→            self.next_static(ctx, rng)
   345→        } else if self.config.adaptive {
   346→            self.next_adaptive(ctx, rng)
   347→        } else {
   348→            self.next_round_robin(ctx)
   349→        }
   350→    }
   351→
   352→    /// Fan out execution feedback to all registered strategies.
   353→    pub fn feedback(&mut self, inputs: &[Value], result: &ExecuteResult, was_new_path: bool) {
   354→        for state in &mut self.states {
   355→            if !state.exhausted {
   356→                state.strategy.feedback(inputs, result, was_new_path);
   357→            }
   358→        }
   359→    }
   360→
   361→    /// Record an outcome for a specific strategy (called after execution).
   362→    pub fn record_outcome(&mut self, strategy_idx: usize, was_new_path: bool) {
   363→        if let Some(state) = self.states.get_mut(strategy_idx) {
   364→            if state.window.len() >= self.config.window_size {
   365→                state.window.pop_front();
   366→            }
   367→            state.window.push_back(was_new_path);
   368→        }
   369→    }
   370→
   371→    /// The name of the strategy at the given index.
   372→    pub fn strategy_name(&self, idx: usize) -> &str {
   373→        self.states[idx].strategy.name()
   374→    }
   375→
   376→    /// Number of registered strategies.
   377→    pub fn len(&self) -> usize {
   378→        self.states.len()
   379→    }
   380→
   381→    /// Whether no strategies are registered.
   382→    pub fn is_empty(&self) -> bool {
   383→        self.states.is_empty()
   384→    }
   385→
   386→    fn all_exhausted(&self) -> bool {
   387→        self.states.iter().all(|s| s.exhausted)
   388→    }
   389→
   390→    /// Try to get the next candidate from a specific strategy, marking it
   391→    /// exhausted if it returns None.
   392→    fn try_next(&mut self, idx: usize, ctx: &StrategyContext) -> Option<Vec<Value>> {
   393→        let state = &mut self.states[idx];
   394→        if state.exhausted {
   395→            return None;
   396→        }
   397→        match state.strategy.next(ctx) {
   398→            Some(inputs) => {
   399→                state.total_supplied += 1;
   400→                Some(inputs)
   401→            }
   402→            None => {
   403→                state.exhausted = true;
   404→                None
   405→            }
   406→        }
   407→    }
   408→
   409→    // --- Selection modes ---
   410→
   411→    fn next_static(&mut self, ctx: &StrategyContext, rng: &mut impl Rng) -> Option<(Vec<Value>, usize)> {
   412→        let weights = self.config.static_weights.clone()?;
   413→
   414→        // Build weight vector for non-exhausted strategies.
   415→        loop {
   416→            let mut candidates: Vec<(usize, f64)> = Vec::new();
   417→            for (idx, state) in self.states.iter().enumerate() {
   418→                if state.exhausted {
   419→                    continue;
   420→                }
   421→                let weight = weights
   422→                    .iter()
   423→                    .find(|(name, _)| name == state.strategy.name())
   424→                    .map(|(_, w)| *w)
   425→                    .unwrap_or(1.0);
   426→                candidates.push((idx, weight));
   427→            }
   428→
   429→            if candidates.is_empty() {
   430→                return None;
   431→            }
   432→
   433→            let idx = weighted_select(&candidates, rng);
   434→            if let Some(inputs) = self.try_next(idx, ctx) {
   435→                return Some((inputs, idx));
   436→            }
   437→            // Strategy exhausted on this call; loop to rebuild candidates.
   438→        }
   439→    }
   440→
   441→    fn next_adaptive(&mut self, ctx: &StrategyContext, rng: &mut impl Rng) -> Option<(Vec<Value>, usize)> {
   442→        loop {
   443→            let scores = self.compute_scores();
   444→            let candidates: Vec<(usize, f64)> = scores
   445→                .into_iter()
   446→                .filter(|(idx, _)| !self.states[*idx].exhausted)
   447→                .collect();
   448→
   449→            if candidates.is_empty() {
   450→                return None;
   451→            }
   452→
   453→            let idx = weighted_select(&candidates, rng);
   454→            if let Some(inputs) = self.try_next(idx, ctx) {
   455→                return Some((inputs, idx));
   456→            }
   457→            // Strategy exhausted; loop to recompute.
   458→        }
   459→    }
   460→
   461→    fn next_round_robin(&mut self, ctx: &StrategyContext) -> Option<(Vec<Value>, usize)> {
   462→        let n = self.states.len();
   463→        for _ in 0..n {
   464→            let idx = self.rr_index % n;
   465→            self.rr_index = self.rr_index.wrapping_add(1);
   466→            if let Some(inputs) = self.try_next(idx, ctx) {
   467→                return Some((inputs, idx));
   468→            }
   469→        }
   470→        None
   471→    }
   472→
   473→    /// Compute adaptive scores for all non-exhausted strategies.
   474→    fn compute_scores(&self) -> Vec<(usize, f64)> {
   475→        // First pass: compute hit rates for graduated strategies to derive average.
   476→        let mut graduated_scores: Vec<f64> = Vec::new();
   477→        for (idx, state) in self.states.iter().enumerate() {
   478→            if state.exhausted {
   479→                continue;
   480→            }
   481→            let threshold = self.effective_cold_start(idx);
   482→            if state.total_supplied >= threshold {
   483→                graduated_scores.push(hit_rate(&state.window));
   484→            }
   485→        }
   486→        let avg_score = if graduated_scores.is_empty() {
   487→            1.0
   488→        } else {
   489→            graduated_scores.iter().sum::<f64>() / graduated_scores.len() as f64
   490→        };
   491→
   492→        // Second pass: assign scores.
   493→        let mut result = Vec::new();
   494→        for (idx, state) in self.states.iter().enumerate() {
   495→            if state.exhausted {
   496→                continue;
   497→            }
   498→            let threshold = self.effective_cold_start(idx);
   499→            let score = if state.total_supplied < threshold {
   500→                avg_score
   501→            } else {
   502→                hit_rate(&state.window)
   503→            };
   504→            result.push((idx, score.max(self.config.floor)));
   505→        }
   506→        result
   507→    }
   508→
   509→    /// Effective cold-start threshold for a strategy: min(configured, estimated_size).
   510→    fn effective_cold_start(&self, idx: usize) -> u64 {
   511→        let base = self.config.cold_start_threshold;
   512→        match self.states[idx].strategy.estimated_size() {
   513→            Some(size) => base.min(size),
   514→            None => base,
   515→        }
   516→    }
   517→}

   520→// Helpers
   521→// ---------------------------------------------------------------------------

   523→/// Hit rate: fraction of `true` values in the window. Returns 0.0 for empty windows.
   524→fn hit_rate(window: &VecDeque<bool>) -> f64 {
   525→    if window.is_empty() {
   526→        return 0.0;
   527→    }
   528→    let hits = window.iter().filter(|&&b| b).count();
   529→    hits as f64 / window.len() as f64
   530→}
   531→
   532→/// Weighted random selection. Returns the index from `candidates`.
   533→/// Panics if `candidates` is empty.
   534→fn weighted_select(candidates: &[(usize, f64)], rng: &mut impl Rng) -> usize {
   535→    let total: f64 = candidates.iter().map(|(_, w)| w).sum();
   536→    if total <= 0.0 {
   537→        // All weights zero — fall back to uniform.
   538→        let pick = rng.random_range(0..candidates.len());
   539→        return candidates[pick].0;
   540→    }
   541→    let mut roll: f64 = rng.random_range(0.0..total);
   542→    for &(idx, weight) in candidates {
   543→        roll -= weight;
   544→        if roll <= 0.0 {
   545→            return idx;
   546→        }
   547→    }
   548→    // Floating-point rounding: return last candidate.
   549→    candidates.last().unwrap().0
   550→}

   590→// FuzzerStrategy — infinite, feedback-driven mutation and crossover

   592→
   593→/// Default per-parameter mutation probability for the fuzzer strategy.
   594→const FUZZER_MUTATION_RATE: f64 = 0.3;
   595→
   596→/// Default crossover probability when two parents are available.
   597→const FUZZER_CROSSOVER_RATE: f64 = 0.7;
   598→
   599→/// Infinite strategy that mutates and crosses over inputs which hit Unknown
   600→/// constraints or discovered new paths.
   601→///
   602→/// Feedback records interesting inputs (those reaching branches with no
   603→/// symbolic constraint, or discovering new coverage). `next()` draws from
   604→/// the interesting pool, generating mutations via `fuzz_inputs()`,
   605→/// `mutate_inputs()`, and `crossover_inputs()`.
   606→pub struct FuzzerStrategy {
   607→    rng: StdRng,
   608→    /// Inputs that hit Unknown constraints or new paths — seeds for mutation.
   609→    interesting: Vec<Vec<Value>>,
   610→    /// Pending mutations generated from a feedback round, drained by `next()`.
   611→    pending: VecDeque<Vec<Value>>,
   612→}
   613→
   614→impl FuzzerStrategy {
   615→    pub fn new(seed: Option<u64>) -> Self {
   616→        let rng = match seed {
   617→            Some(s) => StdRng::seed_from_u64(s),
   618→            None => StdRng::from_os_rng(),
   619→        };
   620→        Self {
   621→            rng,
   622→            interesting: Vec::new(),
   623→            pending: VecDeque::new(),
   624→        }
   625→    }
   626→
   627→    /// Refill `self.pending` by mutating a random interesting input.
   628→    fn refill(&mut self, ctx: &StrategyContext) {
   629→        if self.interesting.is_empty() {
   630→            return;
   631→        }
   632→
   633→        let idx = self.rng.random_range(0..self.interesting.len());
   634→        let base = self.interesting[idx].clone();
   635→
   636→        // 1. Simple value-level mutations (from orchestrator's fuzz_inputs).
   637→        for mutated in fuzz_inputs(&base) {
   638→            self.pending.push_back(mutated);
   639→        }
   640→
   641→        // 2. Type-aware mutation via input_gen.
   642→        let params: Vec<ParamInfo> = ctx.params.clone();
   643→        let mutated = mutate_inputs(&base, &params, FUZZER_MUTATION_RATE, &[], &mut self.rng);
   644→        self.pending.push_back(mutated);
   645→
   646→        // 3. Crossover when at least two interesting inputs exist.
   647→        if self.interesting.len() >= 2 {
   648→            let other_idx = loop {
   649→                let candidate = self.rng.random_range(0..self.interesting.len());
   650→                if candidate != idx || self.interesting.len() == 1 {
   651→                    break candidate;
   652→                }
   653→            };
   654→            let (child_a, _child_b) = crossover_inputs(
   655→                &base,
   656→                &self.interesting[other_idx],
   657→                &params,
   658→                FUZZER_CROSSOVER_RATE,
   659→                &mut self.rng,
   660→            );
   661→            self.pending.push_back(child_a);
   662→        }
   663→    }
   664→}
   665→
   666→impl InputStrategy for FuzzerStrategy {
   667→    fn next(&mut self, ctx: &StrategyContext) -> Option<Vec<Value>> {
   668→        if let Some(candidate) = self.pending.pop_front() {
   669→            return Some(candidate);
   670→        }
   671→
   672→        // Try to refill from interesting pool.
   673→        self.refill(ctx);
   674→        self.pending.pop_front()
   675→    }
   676→
   677→    fn feedback(&mut self, inputs: &[Value], result: &ExecuteResult, was_new_path: bool) {
   678→        // Record inputs that discovered new coverage.
   679→        if was_new_path {
   680→            self.interesting.push(inputs.to_vec());
   681→            return;
   682→        }
   683→
   684→        // Record inputs that hit Unknown constraints (branches the solver
   685→        // cannot handle, so fuzzing is the only way to explore them).
   686→        let has_unknown = result.branch_path.iter().any(|decision| {
   687→            matches!(decision.constraint, SymConstraint::Unknown { .. })
   688→        });
   689→        if has_unknown {
   690→            self.interesting.push(inputs.to_vec());
   691→        }
   692→    }
   693→
   694→    fn name(&self) -> &str {
   695→        "fuzzer"
   696→    }
   697→}

   699→// ---------------------------------------------------------------------------
   700→// Z3SolverStrategy — infinite, feedback-driven constraint solving
   701→// ---------------------------------------------------------------------------
   702→
   703→/// Strategy name constant for Z3 solver, used for discovery attribution.
   704→const Z3_SOLVER_STRATEGY_NAME: &str = "z3_solver";
   705→
   706→/// Reactive strategy that uses Z3 constraint solving to generate inputs
   707→/// targeting unexplored branches.
   708→///
   709→/// `feedback()` extracts symbolic constraints from execution results, negates
   710→/// each solvable constraint, solves with Z3, and overlays solutions onto the
   711→/// base inputs. Solved inputs queue in `pending` and drain via `next()`.
   712→///
   713→/// Infinite: produces work while unsolved constraints exist, but the queue
   714→/// may be empty between feedback cycles.
   715→pub struct Z3SolverStrategy {
   716→    solver_timeout_ms: Option<u64>,
   717→    param_infos: Vec<ParamInfo>,
   718→    pending: VecDeque<Vec<Value>>,
   719→}
   720→
   721→impl Z3SolverStrategy {
   722→    pub fn new(solver_timeout_ms: Option<u64>, param_infos: Vec<ParamInfo>) -> Self {
   723→        Self {
   724→            solver_timeout_ms,
   725→            param_infos,
   726→            pending: VecDeque::new(),
   727→        }
   728→    }
   729→}
   730→
   731→impl InputStrategy for Z3SolverStrategy {
   732→    fn next(&mut self, _ctx: &StrategyContext) -> Option<Vec<Value>> {
   733→        self.pending.pop_front()
   734→    }
   735→
   736→    fn feedback(&mut self, inputs: &[Value], result: &ExecuteResult, _was_new_path: bool) {
   737→        let sym_constraints = crate::orchestrator::extract_sym_constraints(result);
   738→
   739→        let solvable: Vec<SymExpr> = sym_constraints
   740→            .iter()
   741→            .filter_map(|c| c.clone())
   742→            .collect();
   743→
   744→        if solvable.is_empty() {
   745→            return;
   746→        }
   747→
   748→        let param_names: Vec<String> = self.param_infos.iter().map(|p| p.name.clone()).collect();
   749→
   750→        for solve_idx in 0..solvable.len() {
   751→            // solve_for_new_path may fail (unsupported expressions, type mismatches,
   752→            // or constraint/param misalignment). Treat all failures as "no solution".
   753→            let solve_result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
   754→                solver::solve_for_new_path(
   755→                    &solvable,
   756→                    solve_idx,
   757→                    self.solver_timeout_ms,
   758→                    &self.param_infos,
   759→                )
   760→            }));
   761→
   762→            match solve_result {
   763→                Ok(Ok(SolveResult::Sat(values))) => {
   764→                    let new_inputs = crate::orchestrator::overlay_solved_values(
   765→                        inputs,
   766→                        &values,
   767→                        &param_names,
   768→                    );
   769→                    self.pending.push_back(new_inputs);
   770→                }
   771→                _ => {
   772→                    // Unsat, solver error, or panic (debug_assert on param mismatch).
   773→                    // Stall tracking is the orchestrator's responsibility.
   774→                }
   775→            }
   776→        }
   777→    }
   778→
   779→    fn name(&self) -> &str {
   780→        Z3_SOLVER_STRATEGY_NAME
   781→    }
   782→}
