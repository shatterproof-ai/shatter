     1→//! JSON report generation for scan results.
     2→//!
     3→//! Produces machine-readable JSON output after a scan completes. Contains
     4→//! per-function data (branch coverage, discovered inputs, behavior clusters,
     5→//! constraint stats) and codebase-level aggregates (total functions, overall
     6→//! coverage, unreachable branches, dependency graph summary).
     7→
     8→use std::fmt::Write;
     9→use std::path::{Path, PathBuf};
    10→
    11→use serde::{Deserialize, Serialize};
    12→
    13→use crate::batch_state::BatchState;
    14→use crate::coverage_metrics::CoverageMetrics;
    15→use crate::scan_orchestrator::{FunctionResult, ParallelScanResult, ScanResult};
    16→
    17→// ---------------------------------------------------------------------------
    18→// Per-function report
    19→// ---------------------------------------------------------------------------
    20→
    21→/// A single discovered input and the path it triggered.
    22→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    23→pub struct DiscoveredInput {
    24→    /// The input values sent to the function.
    25→    pub inputs: Vec<serde_json::Value>,
    26→    /// Return value, if the function returned normally.
    27→    pub return_value: Option<serde_json::Value>,
    28→    /// Error message, if the function threw.
    29→    pub thrown_error: Option<String>,
    30→    /// Lines executed during this call.
    31→    pub lines_executed: Vec<u32>,
    32→}
    33→
    34→/// Constraint solving statistics for a function.
    35→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    36→pub struct ConstraintStats {
    37→    /// Total number of path constraints collected.
    38→    pub total_constraints: usize,
    39→    /// Number of solver-guided inputs generated (currently 0 for random-only).
    40→    pub solver_guided_inputs: usize,
    41→}
    42→
    43→/// A behavior cluster summary for the report.
    44→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    45→pub struct BehaviorClusterSummary {
    46→    /// Cluster identifier.
    47→    pub id: u32,
    48→    /// Representative input args.
    49→    pub representative_inputs: Vec<serde_json::Value>,
    50→    /// Representative return value.
    51→    pub return_value: Option<serde_json::Value>,
    52→    /// Error, if this cluster represents a throwing path.
    53→    pub thrown_error: Option<String>,
    54→}

    56→/// Report data for a single function.
    57→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    58→pub struct FunctionReport {
    59→    /// Name of the function.
    60→    pub function_name: String,
    61→    /// Source file path.
    62→    pub file_path: String,
    63→    /// Total branch points in the function.
    64→    pub branch_count: usize,
    65→    /// Number of branches covered (unique paths discovered).
    66→    pub branches_covered: usize,
    67→    /// Coverage percentage (0.0-100.0).
    68→    pub coverage_pct: f64,
    69→    /// Inputs that discovered new execution paths.
    70→    pub discovered_inputs: Vec<DiscoveredInput>,
    71→    /// Behavior cluster summaries.
    72→    pub behavior_clusters: Vec<BehaviorClusterSummary>,
    73→    /// Constraint solving statistics.
    74→    pub constraint_stats: ConstraintStats,
    75→    /// Total iterations attempted.
    76→    pub iterations: u32,
    77→    /// Number of unique source lines covered.
    78→    pub lines_covered: usize,
    79→    /// Total source lines in the function.
    80→    pub total_lines: u32,
    81→    /// Functions mocked during exploration.
    82→    pub mocks_used: Vec<String>,
    83→}

    96→/// Codebase-level aggregate statistics.

   123→/// Cumulative progress across progressive batch runs.
   124→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   125→pub struct CumulativeReport {
   126→    /// Batch indices that have been completed.
   127→    pub completed_batches: Vec<usize>,
   128→    /// Total functions explored across all batches.
   129→    pub total_functions_explored: usize,
   130→    /// Total functions in the scan scope.
   131→    pub total_scope_functions: usize,
   132→    /// Cumulative coverage metrics merged across all batches.
   133→    pub metrics: CoverageMetrics,
   134→    /// Overall cumulative coverage percentage.
   135→    pub cumulative_coverage_pct: f64,
   136→}

   138→/// The complete JSON scan report.
   139→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   140→pub struct ScanReport {
   141→    /// Schema version for forward compatibility.
   142→    pub version: u32,
   143→    /// Per-function reports.
   144→    pub functions: Vec<FunctionReport>,
   145→    /// Codebase-level aggregates.
   146→    pub codebase: CodebaseReport,
   147→    /// Test order used during the scan.
   148→    pub test_order: Vec<String>,
   149→    /// Cumulative stats across all batches (present only in batch mode).
   150→    #[serde(skip_serializing_if = "Option::is_none", default)]
   151→    pub cumulative: Option<CumulativeReport>,
   152→}

   158→/// Build a [`FunctionReport`] from a scan's [`FunctionResult`].
   159→fn build_function_report(result: &FunctionResult, file_path: &str) -> FunctionReport {
   160→    let exploration = &result.exploration;
   161→
   162→    let discovered_inputs: Vec<DiscoveredInput> = exploration
   163→        .new_path_executions
   164→        .iter()
   165→        .map(|exec| DiscoveredInput {
   166→            inputs: exec.inputs.clone(),
   167→            return_value: exec.return_value.clone(),
   168→            thrown_error: exec.thrown_error.clone(),
   169→            lines_executed: exec.lines_executed.clone(),
   170→        })
   171→        .collect();
   172→
   173→    let behavior_clusters: Vec<BehaviorClusterSummary> = result
   174→        .behavior_map
   175→        .behaviors
   176→        .iter()
   177→        .map(|b| {
   178→            let thrown_error = b.thrown_error.as_ref().map(|e| {
   179→                format!("{}: {}", e.error_type, e.message)
   180→            });
   181→            BehaviorClusterSummary {
   182→                id: b.id,
   183→                representative_inputs: b.input_args.clone(),
   184→                return_value: b.return_value.clone(),
   185→                thrown_error,
   186→            }
   187→        })
   188→        .collect();
   189→
   190→    let total_constraints: usize = exploration
   191→        .raw_results
   192→        .iter()
   193→        .map(|(_, _mocks, r)| r.path_constraints.len())
   194→        .sum();
   195→
   196→    let coverage_pct = if exploration.total_lines > 0 {
   197→        (exploration.lines_covered as f64 / exploration.total_lines as f64 * 100.0).min(100.0)
   198→    } else {
   199→        0.0
   200→    };
   201→
   202→    FunctionReport {
   203→        function_name: result.function_name.clone(),
   204→        file_path: file_path.to_string(),
   205→        branch_count: exploration.unique_paths,
   206→        branches_covered: exploration.unique_paths,
   207→        coverage_pct,
   208→        discovered_inputs,
   209→        behavior_clusters,
   210→        constraint_stats: ConstraintStats {
   211→            total_constraints,
   212→            solver_guided_inputs: 0,
   213→        },
   214→        iterations: exploration.iterations,
   215→        lines_covered: exploration.lines_covered,
   216→        total_lines: exploration.total_lines,
   217→        mocks_used: result.mocks_used.iter().map(|m| m.name.clone()).collect(),
   218→    }
   219→}
   220→
   221→/// Build dependency edges from the function results (caller -> mocked callee).
   222→fn build_dependency_edges(function_results: &[FunctionResult]) -> Vec<DependencyEdge> {
   223→    let mut edges = Vec::new();
   224→    for result in function_results {
   225→        for mock in &result.mocks_used {
   226→            edges.push(DependencyEdge {
   227→                caller: result.function_name.clone(),
   228→                callee: mock.name.clone(),
   229→            });
   230→        }
   231→    }
   232→    edges
   233→}

   235→/// Build a [`CumulativeReport`] from batch state.
   236→fn build_cumulative_report(state: &BatchState) -> CumulativeReport {
   237→    CumulativeReport {
   238→        completed_batches: state.completed_batches(),
   239→        total_functions_explored: state.total_functions_explored(),
   240→        total_scope_functions: state.total_scope_functions,
   241→        metrics: state.cumulative_metrics.clone(),
   242→        cumulative_coverage_pct: state.cumulative_coverage_pct(),
   243→    }
   244→}

   246→/// Generate a [`ScanReport`] from a [`ParallelScanResult`].
   247→///
   248→/// The `file_map` maps function names to their source file paths.
   249→/// When `batch_state` is provided, the report includes cumulative progress
   250→/// across all completed batches.
   251→pub fn generate_report(
   252→    result: &ParallelScanResult,
   253→    file_map: &std::collections::HashMap<String, String>,
   254→    batch_state: Option<&BatchState>,
   255→) -> ScanReport {
   256→    let functions: Vec<FunctionReport> = result
   257→        .function_results
   258→        .iter()
   259→        .map(|fr| {
   260→            let file_path = file_map
   261→                .get(&fr.function_name)
   262→                .map(|s| s.as_str())
   263→                .unwrap_or("");
   264→            build_function_report(fr, file_path)
   265→        })
   266→        .collect();
   267→
   268→    let total_branches: usize = functions.iter().map(|f| f.branch_count).sum();
   269→    let total_covered: usize = functions.iter().map(|f| f.branches_covered).sum();
   270→    let overall_coverage = if total_branches > 0 {
   271→        total_covered as f64 / total_branches as f64 * 100.0
   272→    } else {
   273→        0.0
   274→    };
   275→
   276→    let skipped_functions: Vec<SkippedFunctionReport> = result
   277→        .skipped
   278→        .iter()
   279→        .map(|s| SkippedFunctionReport {
   280→            function_name: s.function_name.clone(),
   281→            reason: s.reason.clone(),
   282→            category: match s.category {
   283→                crate::scan_orchestrator::SkipCategory::Expected => "expected".into(),
   284→                crate::scan_orchestrator::SkipCategory::Error => "error".into(),
   285→            },
   286→        })
   287→        .collect();
   288→
   289→    let dependency_graph = build_dependency_edges(&result.function_results);
   290→
   291→    let cumulative = batch_state.map(build_cumulative_report);
   292→
   293→    ScanReport {
   294→        version: 1,
   295→        functions,
   296→        codebase: CodebaseReport {
   297→            total_functions: result.function_results.len(),
   298→            total_branches,
   299→            overall_coverage,
   300→            skipped_functions,
   301→            dependency_graph,
   302→        },
   303→        test_order: result.test_order.clone(),
   304→        cumulative,
   305→    }
   306→}

   308→/// Generate a [`ScanReport`] from a sequential [`ScanResult`].
   309→///
   310→/// The `file_map` maps function names to their source file paths.
   311→pub fn generate_report_from_scan(
   312→    result: &ScanResult,
   313→    file_map: &std::collections::HashMap<String, String>,
   314→) -> ScanReport {
   315→    let functions: Vec<FunctionReport> = result
   316→        .function_results
   317→        .iter()
   318→        .map(|fr| {
   319→            let file_path = file_map
   320→                .get(&fr.function_name)
   321→                .map(|s| s.as_str())
   322→                .unwrap_or("");
   323→            build_function_report(fr, file_path)
   324→        })
   325→        .collect();
   326→
   327→    let total_branches: usize = functions.iter().map(|f| f.branch_count).sum();
   328→    let total_covered: usize = functions.iter().map(|f| f.branches_covered).sum();
   329→    let overall_coverage = if total_branches > 0 {
   330→        total_covered as f64 / total_branches as f64 * 100.0
   331→    } else {
   332→        0.0
   333→    };
   334→
   335→    let dependency_graph = build_dependency_edges(&result.function_results);
   336→
   337→    let skipped_functions: Vec<SkippedFunctionReport> = result
   338→        .skipped_functions
   339→        .iter()
   340→        .map(|s| SkippedFunctionReport {
   341→            function_name: s.function_name.clone(),
   342→            reason: s.reason.clone(),
   343→            category: match s.category {
   344→                crate::scan_orchestrator::SkipCategory::Expected => "expected".into(),
   345→                crate::scan_orchestrator::SkipCategory::Error => "error".into(),
   346→            },
   347→        })
   348→        .collect();
   349→
   350→    ScanReport {
   351→        version: 1,
   352→        functions,
   353→        codebase: CodebaseReport {
   354→            total_functions: result.function_results.len(),
   355→            total_branches,
   356→            overall_coverage,
   357→            skipped_functions,
   358→            dependency_graph,
   359→        },
   360→        test_order: result.test_order.clone(),
   361→        cumulative: None,
   362→    }
   363→}

   365→/// Write a [`ScanReport`] as pretty-printed JSON to a directory.
   366→///
   367→/// Creates the output directory if it does not exist. Writes to
   368→/// `<output_dir>/scan-report.json`.
   369→pub fn write_report(report: &ScanReport, output_dir: &Path) -> Result<PathBuf, ReportError> {
   370→    std::fs::create_dir_all(output_dir).map_err(|e| ReportError::Io {
   371→        path: output_dir.to_path_buf(),
   372→        source: e,
   373→    })?;
   374→
   375→    let report_path = output_dir.join("scan-report.json");
   376→    let json = serde_json::to_string_pretty(report).map_err(ReportError::Serialize)?;
   377→    std::fs::write(&report_path, json).map_err(|e| ReportError::Io {
   378→        path: report_path.clone(),
   379→        source: e,
   380→    })?;
   381→
   382→    Ok(report_path)
   383→}

   409→/// Format a [`ScanReport`] as a human-readable markdown string.
   410→#[must_use]
   411→pub fn format_markdown_report(report: &ScanReport) -> String {
   412→    let mut out = String::new();
   413→
   414→    write_md_header(&mut out, report);
   415→    write_md_cumulative(&mut out, &report.cumulative);
   416→    write_md_summary_table(&mut out, report);
   417→    write_md_function_details(&mut out, &report.functions);
   418→    write_md_uncovered_branches(&mut out, &report.functions);
   419→    write_md_interesting_inputs(&mut out, &report.functions);
   420→    write_md_skipped_functions(&mut out, &report.codebase.skipped_functions);
   421→
   422→    out
   423→}
   424→
   425→fn write_md_header(out: &mut String, report: &ScanReport) {
   426→    let _ = writeln!(out, "# Shatter Scan Report\n");
   427→
   428→    let total_covered: usize = report.functions.iter().map(|f| f.branches_covered).sum();
   429→    let total_branches = report.codebase.total_branches;
   430→    let coverage = report.codebase.overall_coverage;
   431→
   432→    let _ = writeln!(out, "- **Functions explored:** {}", report.codebase.total_functions);
   433→    let _ = writeln!(out, "- **Total branches:** {total_branches}");
   434→    let _ = writeln!(out, "- **Branches covered:** {total_covered}");
   435→    let _ = writeln!(out, "- **Overall coverage:** {coverage:.1}%");
   436→
   437→    if !report.codebase.skipped_functions.is_empty() {
   438→        let _ = writeln!(
   439→            out,
   440→            "- **Skipped functions:** {}",
   441→            report.codebase.skipped_functions.len()
   442→        );
   443→    }
   444→
   445→    out.push('\n');
   446→}

   448→fn write_md_cumulative(out: &mut String, cumulative: &Option<CumulativeReport>) {
   449→    let Some(cum) = cumulative else {
   450→        return;
   451→    };
   452→
   453→    let _ = writeln!(out, "## Cumulative Progress\n");
   454→    let batches_str = cum
   455→        .completed_batches
   456→        .iter()
   457→        .map(|i| i.to_string())
   458→        .collect::<Vec<_>>()
   459→        .join(", ");
   460→    let _ = writeln!(
   461→        out,
   462→        "- **Batches completed:** {} ({})",
   463→        cum.completed_batches.len(),
   464→        batches_str,
   465→    );
   466→    let _ = writeln!(
   467→        out,
   468→        "- **Functions explored:** {}/{}",
   469→        cum.total_functions_explored, cum.total_scope_functions,
   470→    );
   471→    let covered = cum.metrics.z3_solved + cum.metrics.random_found + cum.metrics.user_provided;
   472→    let _ = writeln!(
   473→        out,
   474→        "- **Branches covered:** {}/{}",
   475→        covered, cum.metrics.total_branches,
   476→    );
   477→    let _ = writeln!(
   478→        out,
   479→        "- **Cumulative coverage:** {:.1}%",
   480→        cum.cumulative_coverage_pct,
   481→    );
   482→    out.push('\n');
   483→}

   485→fn write_md_summary_table(out: &mut String, report: &ScanReport) {
   486→    if report.functions.is_empty() {
   487→        let _ = writeln!(out, "*No functions were explored.*\n");
   488→        return;
   489→    }
   490→
   491→    let _ = writeln!(out, "## Function Summary\n");
   492→    let _ = writeln!(out, "| Status | Function | File | Coverage | Branches | Lines | Iterations |");
   493→    let _ = writeln!(out, "|--------|----------|------|----------|----------|-------|------------|");
   494→
   495→    for func in &report.functions {
   496→        let status = if func.coverage_pct >= 100.0 {
   497→            "PASS"
   498→        } else if func.coverage_pct >= 50.0 {
   499→            "WARN"
   500→        } else {
   501→            "FAIL"
   502→        };
   503→
   504→        let _ = writeln!(
   505→            out,
   506→            "| {status} | `{name}` | {file} | {cov:.1}% | {covered}/{total} | {lc}/{tl} | {iter} |",
   507→            name = func.function_name,
   508→            file = if func.file_path.is_empty() { "-" } else { &func.file_path },
   509→            cov = func.coverage_pct,
   510→            covered = func.branches_covered,
   511→            total = func.branch_count,
   512→            lc = func.lines_covered,
   513→            tl = func.total_lines,
   514→            iter = func.iterations,
   515→        );
   516→    }
   517→
   518→    out.push('\n');
   519→}

   521→fn write_md_function_details(out: &mut String, functions: &[FunctionReport]) {
   522→    if functions.is_empty() {
   523→        return;
   524→    }
   525→
   526→    let _ = writeln!(out, "## Function Details\n");
   527→
   528→    for func in functions {
   529→        let _ = writeln!(out, "### `{}`\n", func.function_name);
   530→
   531→        if !func.file_path.is_empty() {
   532→            let _ = writeln!(out, "- **File:** {}", func.file_path);
   533→        }
   534→        let _ = writeln!(out, "- **Coverage:** {:.1}%", func.coverage_pct);
   535→        let _ = writeln!(
   536→            out,
   537→            "- **Branches:** {}/{}",
   538→            func.branches_covered, func.branch_count
   539→        );
   540→        let _ = writeln!(out, "- **Lines:** {}/{}", func.lines_covered, func.total_lines);
   541→        let _ = writeln!(out, "- **Iterations:** {}", func.iterations);
   542→        let _ = writeln!(
   543→            out,
   544→            "- **Constraints collected:** {}",
   545→            func.constraint_stats.total_constraints
   546→        );
   547→
   548→        if !func.mocks_used.is_empty() {
   549→            let _ = writeln!(out, "- **Mocks:** {}", func.mocks_used.join(", "));
   550→        }
   551→
   552→        if !func.behavior_clusters.is_empty() {
   553→            let _ = writeln!(out, "\n**Behaviors:**\n");
   554→            for cluster in &func.behavior_clusters {
   555→                let outcome = if let Some(ref err) = cluster.thrown_error {
   556→                    format!("throws {err}")
   557→                } else if let Some(ref val) = cluster.return_value {
   558→                    format!("returns {}", format_json_compact(val))
   559→                } else {
   560→                    "returns void".to_string()
   561→                };
   562→                let inputs = format_json_compact_list(&cluster.representative_inputs);
   563→                let _ = writeln!(out, "- Cluster {}: {outcome} (inputs: {inputs})", cluster.id);
   564→            }
   565→        }
   566→
   567→        out.push('\n');
   568→    }
   569→}
   570→
   571→fn write_md_uncovered_branches(out: &mut String, functions: &[FunctionReport]) {
   572→    let low_coverage: Vec<&FunctionReport> = functions
   573→        .iter()
   574→        .filter(|f| f.coverage_pct < 100.0 && f.branch_count > 0)
   575→        .collect();
   576→
   577→    if low_coverage.is_empty() {
   578→        return;
   579→    }
   580→
   581→    let _ = writeln!(out, "## Uncovered Branches\n");
   582→
   583→    for func in &low_coverage {
   584→        let uncovered = func.branch_count.saturating_sub(func.branches_covered);
   585→        let _ = writeln!(
   586→            out,
   587→            "- `{}`: {uncovered} uncovered branch(es) ({:.1}% coverage)",
   588→            func.function_name, func.coverage_pct
   589→        );
   590→    }
   591→
   592→    out.push('\n');
   593→}
   594→
   595→fn write_md_interesting_inputs(out: &mut String, functions: &[FunctionReport]) {
   596→    let has_interesting = functions.iter().any(|f| {
   597→        f.discovered_inputs
   598→            .iter()
   599→            .any(|d| d.thrown_error.is_some() || is_boundary_value(&d.inputs))
   600→    });
   601→
   602→    if !has_interesting {
   603→        return;
   604→    }
   605→
   606→    let _ = writeln!(out, "## Interesting Inputs\n");
   607→
   608→    for func in functions {
   609→        let interesting: Vec<&DiscoveredInput> = func
   610→            .discovered_inputs
   611→            .iter()
   612→            .filter(|d| d.thrown_error.is_some() || is_boundary_value(&d.inputs))
   613→            .collect();
   614→
   615→        if interesting.is_empty() {
   616→            continue;
   617→        }
   618→
   619→        let _ = writeln!(out, "### `{}`\n", func.function_name);
   620→
   621→        for input in &interesting {
   622→            let inputs_str = format_json_compact_list(&input.inputs);
   623→            if let Some(ref err) = input.thrown_error {
   624→                let _ = writeln!(out, "- {inputs_str} -> **error:** {err}");
   625→            } else if let Some(ref val) = input.return_value {
   626→                let _ = writeln!(out, "- {inputs_str} -> {}", format_json_compact(val));
   627→            } else {
   628→                let _ = writeln!(out, "- {inputs_str} -> void");
   629→            }
   630→        }
   631→
   632→        out.push('\n');
   633→    }
   634→}
   635→
   636→fn write_md_skipped_functions(out: &mut String, skipped: &[SkippedFunctionReport]) {
   637→    if skipped.is_empty() {
   638→        return;
   639→    }
   640→
   641→    let expected: Vec<_> = skipped.iter().filter(|s| s.category == "expected").collect();
   642→    let errors: Vec<_> = skipped.iter().filter(|s| s.category == "error").collect();
   643→
   644→    if !expected.is_empty() {
   645→        let _ = writeln!(out, "## Skipped (Expected)\n");
   646→        for s in &expected {
   647→            let _ = writeln!(out, "- `{}`: {}", s.function_name, s.reason);
   648→        }
   649→        out.push('\n');
   650→    }
   651→
   652→    if !errors.is_empty() {
   653→        let _ = writeln!(out, "## Errors\n");
   654→        for s in &errors {
   655→            let _ = writeln!(out, "- `{}`: {}", s.function_name, s.reason);
   656→        }
   657→        out.push('\n');
   658→    }
   659→}
   660→
   661→fn is_boundary_value(inputs: &[serde_json::Value]) -> bool {
   662→    inputs.iter().any(|v| match v {
   663→        serde_json::Value::Number(n) => {
   664→            if let Some(f) = n.as_f64() {
   665→                f == 0.0 || f == -1.0
   666→            } else {
   667→                false
   668→            }
   669→        }
   670→        serde_json::Value::String(s) => s.is_empty(),
   671→        serde_json::Value::Null => true,
   672→        serde_json::Value::Array(a) => a.is_empty(),
   673→        _ => false,
   674→    })
   675→}
   676→
   677→fn format_json_compact(value: &serde_json::Value) -> String {
   678→    let s = value.to_string();
   679→    if s.len() > 60 {
   680→        format!("{}...", &s[..57])
   681→    } else {
   682→        s
   683→    }
   684→}
   685→
   686→fn format_json_compact_list(values: &[serde_json::Value]) -> String {
   687→    let parts: Vec<String> = values.iter().map(format_json_compact).collect();
   688→    parts.join(", ")
   689→}

   695→/// A structured progress event for machine-readable output.
   696→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   697→pub struct ProgressEvent {
   698→    /// Event type — always "progress".
   699→    #[serde(rename = "type")]
   700→    pub event_type: String,
   701→    /// Name of the function currently being processed.
   702→    pub function: String,
   703→    /// 1-based index of the current function.
   704→    pub current: usize,
   705→    /// Total number of functions to process.
   706→    pub total: usize,
   707→    /// Milliseconds elapsed since the scan started.
   708→    pub elapsed_ms: u64,
   709→}
   710→
   711→impl ProgressEvent {
   712→    /// Create a new progress event.
   713→    #[must_use]
   714→    pub fn new(function: &str, current: usize, total: usize, elapsed_ms: u64) -> Self {
   715→        Self {
   716→            event_type: "progress".to_string(),
   717→            function: function.to_string(),
   718→            current,
   719→            total,
   720→            elapsed_ms,
   721→        }
   722→    }
   723→
   724→    /// Serialize this event as a JSON string.
   725→    ///
   726→    /// Returns `None` if serialization fails (should not happen for valid data).
   727→    #[must_use]
   728→    pub fn to_json(&self) -> Option<String> {
   729→        serde_json::to_string(self).ok()
   730→    }
   731→}

   733→/// Report format for scan output.
   734→#[derive(Debug, Clone, Copy, PartialEq, Eq)]
   735→pub enum ReportFormat {
   736→    /// JSON only (default).
   737→    Json,
   738→    /// Markdown only.
   739→    Markdown,
   740→    /// Both JSON and Markdown.
   741→    Both,
   742→}
   743→
   744→impl std::str::FromStr for ReportFormat {
   745→    type Err = String;
   746→
   747→    fn from_str(s: &str) -> Result<Self, Self::Err> {
   748→        match s.to_lowercase().as_str() {
   749→            "json" => Ok(Self::Json),
   750→            "markdown" | "md" => Ok(Self::Markdown),
   751→            "both" => Ok(Self::Both),
   752→            _ => Err(format!(
   753→                "unknown report format '{s}': expected 'json', 'markdown', or 'both'"
   754→            )),
   755→        }
   756→    }
   757→}

   759→/// Errors that can occur during report generation or writing.
   760→#[derive(Debug, thiserror::Error)]
   761→pub enum ReportError {
   762→    #[error("failed to write to {path}: {source}")]
   763→    Io {
   764→        path: PathBuf,
   765→        source: std::io::Error,
   766→    },
   767→    #[error("JSON serialization error: {0}")]
   768→    Serialize(#[from] serde_json::Error),
   769→}
