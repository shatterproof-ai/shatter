     1→//! Protocol types for communication between the Rust core and language frontends.
     2→//!
     3→//! All messages are newline-delimited JSON over stdin/stdout. The core sends
     4→//! [`Request`] messages to frontends and receives [`Response`] messages back.
     5→//! Every message includes a protocol version for compatibility checking.

    56→/// A request message sent from the core engine to a language frontend.
    57→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    58→pub struct Request {
    59→    /// Protocol version for compatibility checking.
    60→    pub protocol_version: String,
    61→    /// Unique identifier for correlating responses to requests.
    62→    pub id: u64,
    63→    /// The command to execute.
    64→    #[serde(flatten)]
    65→    pub command: Command,
    66→}

    68→/// Commands the core can send to a frontend.
    69→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    70→#[serde(tag = "command", rename_all = "snake_case")]
    71→pub enum Command {
    72→    /// Initial handshake to negotiate protocol version and capabilities.
    73→    Handshake {
    74→        /// Capabilities supported by the core.
    75→        capabilities: Vec<String>,
    76→    },
    77→    /// Analyze a function to extract type information, branches, and dependencies.
    78→    Analyze {
    79→        /// Path to the source file.
    80→        file: String,
    81→        /// Name of the function to analyze. If absent, analyze all exported functions.
    82→        function: Option<String>,
    83→        /// Detected project root directory, if any.
    84→        #[serde(default, skip_serializing_if = "Option::is_none")]
    85→        project_root: Option<String>,
    86→    },
    87→    /// Instrument a function for symbolic constraint tracking.
    88→    Instrument {
    89→        /// Path to the source file.
    90→        file: String,
    91→        /// Name of the function to instrument.
    92→        function: String,
    93→        /// Mock configurations for external dependencies.
    94→        mocks: Vec<MockConfig>,
    95→        /// Detected project root directory, if any.
    96→        #[serde(default, skip_serializing_if = "Option::is_none")]
    97→        project_root: Option<String>,
    98→    },
    99→    /// Execute an instrumented function with specific inputs and mocks.
   100→    Execute {
   101→        /// Fully qualified function identifier.
   102→        function: String,
   103→        /// Input values for the function parameters (JSON-encoded).
   104→        inputs: Vec<serde_json::Value>,
   105→        /// Mock configurations for external dependencies.
   106→        mocks: Vec<MockConfig>,
   107→        /// Stack of active setup contexts from enclosing Setup commands, if any.
   108→        #[serde(default, skip_serializing_if = "Option::is_none")]
   109→        setup_context: Option<SetupContextStack>,
   110→    },
   111→    /// Run a setup file to initialize state before function execution.
   112→    Setup {
   113→        /// Path to the setup file.
   114→        file: String,
   115→        /// Scope identifier (function name, file path, or session label).
   116→        scope: String,
   117→        /// Lifecycle level for this setup.
   118→        level: SetupLevel,
   119→        /// Detected project root directory, if any.
   120→        #[serde(default, skip_serializing_if = "Option::is_none")]
   121→        project_root: Option<String>,
   122→        /// Parent context stack from enclosing setup levels, if any.
   123→        #[serde(default, skip_serializing_if = "Option::is_none")]
   124→        parent_context: Option<SetupContextStack>,
   125→    },
   126→    /// Tear down state established by a prior Setup command.
   127→    Teardown {
   128→        /// Scope identifier matching the corresponding Setup command.
   129→        scope: String,
   130→        /// Lifecycle level matching the corresponding Setup command.
   131→        level: SetupLevel,
   132→    },
   133→    /// Invoke a custom generator to produce a value for a type or parameter.
   134→    Generate {
   135→        /// Path to the generator file.
   136→        file: String,
   137→        /// Name of the type or parameter to generate a value for.
   138→        name: String,
   139→        /// Whether this generator targets a type name or a parameter name.
   140→        kind: GeneratorKind,
   141→        /// Reconstruction recipe from a previous generation.
   142→        #[serde(default, skip_serializing_if = "Option::is_none")]
   143→        recipe: Option<serde_json::Value>,
   144→        /// Detected project root directory, if any.
   145→        #[serde(default, skip_serializing_if = "Option::is_none")]
   146→        project_root: Option<String>,
   147→    },
   148→    /// Request graceful shutdown of the frontend process.
   149→    Shutdown,
   150→}

   520→
   521→/// Validate an `ExecuteResult` deserialized from a frontend subprocess.
   522→///
   523→/// Checks semantic invariants that serde can't enforce:
   524→/// - `branch_path` entries have non-default constraints (when non-empty, an all-unknown
   525→///   branch_path produces identical hashes for different paths → silent hash collision)
   526→/// - `path_constraints` length matches `branch_path` length when both are non-empty
   527→#[contracts::ensures(ret == execute_result_is_valid(result),
   528→    "postcondition must match validation logic")]
   529→pub fn validate_execute_result(result: &ExecuteResult) -> bool {
   530→    execute_result_is_valid(result)
   531→}
   532→
   533→fn execute_result_is_valid(result: &ExecuteResult) -> bool {
   534→    // Branch path with all-unknown constraints causes silent hash collisions
   535→    // because the path hash depends on constraint content.
   536→    if !result.branch_path.is_empty() {
   537→        let all_unknown = result.branch_path.iter().all(|bd| {
   538→            matches!(bd.constraint, SymConstraint::Unknown { .. })
   539→        });
   540→        // All-unknown is valid for frontends without symbolic analysis (Go),
   541→        // but path_constraints should then also be empty.
   542→        if all_unknown && !result.path_constraints.is_empty() {
   543→            return false;
   544→        }
   545→    }
   546→    true
   547→}
   548→
   549→/// Validate a `FunctionAnalysis` list deserialized from a frontend subprocess.
   550→///
   551→/// Checks semantic invariants that serde can't enforce:
   552→/// - Function names are non-empty (empty name → silent lookup failures)
   553→/// - `start_line <= end_line` (inverted range → wrong source slicing)
   554→/// - Param count is plausible (> 255 params suggests a deserialization bug)
   555→#[contracts::ensures(ret == analyze_result_is_valid(functions),
   556→    "postcondition must match validation logic")]
   557→pub fn validate_analyze_result(functions: &[FunctionAnalysis]) -> bool {
   558→    analyze_result_is_valid(functions)
   559→}
   560→
   561→/// Upper bound on parameter count — anything above this likely indicates
   562→/// a deserialization bug rather than a real function signature.
   563→const MAX_PLAUSIBLE_PARAMS: usize = 255;
   564→
   565→fn analyze_result_is_valid(functions: &[FunctionAnalysis]) -> bool {
   566→    for func in functions {
   567→        if func.name.is_empty() {
   568→            return false;
   569→        }
   570→        if func.start_line > func.end_line {
   571→            return false;
   572→        }
   573→        if func.params.len() > MAX_PLAUSIBLE_PARAMS {
   574→            return false;
   575→        }
   576→    }
   577→    true
   578→}
