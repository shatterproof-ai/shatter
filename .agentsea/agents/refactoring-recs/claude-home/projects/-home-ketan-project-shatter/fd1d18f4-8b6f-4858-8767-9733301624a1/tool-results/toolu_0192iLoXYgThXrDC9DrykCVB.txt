     1→//! Behavior specification output for explored functions.
     2→//!
     3→//! A [`FunctionSpec`] captures the complete behavioral specification of a function:
     4→//! equivalence classes grouped by branch path, preconditions, postconditions,
     5→//! concrete examples, and provenance (proven vs observed). The spec can be
     6→//! rendered as human-readable markdown, machine-readable JSON, or YAML with
     7→//! property descriptions.

    24→/// Error type for spec bundle I/O operations.

    26→pub enum SpecIoError {
    27→    /// Filesystem I/O error.
    28→    Io(std::io::Error),
    29→    /// JSON serialization or deserialization error.

    34→    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
    35→        match self {
    36→            Self::Io(e) => write!(f, "spec I/O error: {e}"),
    37→            Self::Json(e) => write!(f, "spec JSON error: {e}"),

    39→    }
    40→}

    47→        }
    48→    }

    63→/// Whether a property was proven by constraint solving or merely observed.
    64→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
    65→#[serde(rename_all = "snake_case")]
    66→pub enum Provenance {
    67→    /// Confirmed by Z3 constraint solving — holds for all inputs on this path.
    68→    Proven,
    69→    /// Observed across all sampled inputs but not formally proven.
    70→    Observed,
    71→}

    77→/// High-level category for an invariant property description.
    78→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
    79→#[serde(rename_all = "snake_case")]
    80→pub enum InvariantCategory {
    81→    NumericBound,
    82→    Nullability,
    83→    StringConstraint,
    84→    ReturnTypeInvariant,
    85→    ErrorInvariant,
    86→    InputOutputRelation,
    87→    BooleanConstant,
    88→    ConstantValue,
    89→}

    91→/// Categorical confidence derived from the numeric confidence score.
    92→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
    93→#[serde(rename_all = "snake_case")]
    94→pub enum ConfidenceLevel {
    95→    High,
    96→    Medium,
    97→    Low,
    98→}
    99→
   100→const CONFIDENCE_HIGH_THRESHOLD: f64 = 0.95;
   101→const CONFIDENCE_MEDIUM_THRESHOLD: f64 = 0.75;
   102→
   103→impl ConfidenceLevel {
   104→    fn from_score(score: f64) -> Self {
   105→        if score >= CONFIDENCE_HIGH_THRESHOLD {
   106→            Self::High
   107→        } else if score >= CONFIDENCE_MEDIUM_THRESHOLD {
   108→            Self::Medium
   109→        } else {
   110→            Self::Low
   111→        }
   112→    }
   113→}

   115→/// A human-friendly invariant for YAML spec output.
   116→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   117→pub struct SpecInvariant {
   118→    /// Human-readable property description.
   119→    pub property: String,
   120→    /// High-level invariant category.
   121→    pub kind: InvariantCategory,
   122→    /// Categorical confidence level.
   123→    pub confidence: ConfidenceLevel,
   124→}
   125→
   126→impl From<&ClassifiedInvariant> for SpecInvariant {
   127→    fn from(ci: &ClassifiedInvariant) -> Self {
   128→        Self {
   129→            property: ci.label.clone(),
   130→            kind: categorize_invariant_kind(&ci.invariant.kind),
   131→            confidence: ConfidenceLevel::from_score(ci.confidence),
   132→        }
   133→    }
   134→}
   135→
   136→fn categorize_invariant_kind(kind: &InvariantKind) -> InvariantCategory {
   137→    match kind {
   138→        InvariantKind::NumericComparison { .. } => InvariantCategory::NumericBound,
   139→        InvariantKind::NumericConstant { .. } => InvariantCategory::ConstantValue,
   140→        InvariantKind::NotNull { .. } | InvariantKind::IsNull { .. } => {
   141→            InvariantCategory::Nullability
   142→        }
   143→        InvariantKind::StringNonEmpty { .. } | InvariantKind::StringLength { .. } => {
   144→            InvariantCategory::StringConstraint
   145→        }
   146→        InvariantKind::OutputEqualsInput { .. } => InvariantCategory::InputOutputRelation,
   147→        InvariantKind::AlwaysTrue { .. } | InvariantKind::AlwaysFalse { .. } => {
   148→            InvariantCategory::BooleanConstant
   149→        }
   150→    }
   151→}

   153→/// A postcondition describing what a function does on a given path.
   154→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   155→#[serde(tag = "kind", rename_all = "snake_case")]
   156→pub enum Postcondition {
   157→    /// The function returns a value matching this pattern.
   158→    Returns {
   159→        /// A representative return value (from the canonical example).
   160→        value: serde_json::Value,
   161→    },
   162→    /// The function throws an error.
   163→    Throws {
   164→        /// Error type and message.
   165→        error: ErrorInfo,
   166→    },
   167→    /// The function returns void / undefined / null.
   168→    ReturnsVoid,
   169→}

   182→/// One behavioral equivalence class within a function specification.
   183→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   184→pub struct SpecClass {
   185→    /// Human-readable label for this class (e.g., "returns positive result").
   186→    pub label: String,
   187→    /// The branch path identifying this class.
   188→    pub branch_path: BranchPath,
   189→    /// Preconditions: constraints on inputs that lead to this path.
   190→    pub preconditions: Vec<Precondition>,
   191→    /// Postcondition: what the function does on this path.
   192→    pub postcondition: Postcondition,
   193→    /// Side effects observed on this path.
   194→    pub side_effects: Vec<SideEffect>,
   195→    /// At least one concrete example.
   196→    pub examples: Vec<ConcreteExample>,
   197→    /// How many executions fell into this class.
   198→    pub sample_count: usize,
   199→    /// Whether preconditions are proven or observed.
   200→    pub precondition_provenance: Provenance,
   201→    /// Whether postconditions are proven or observed.
   202→    pub postcondition_provenance: Provenance,
   203→    /// Detected invariants for this equivalence class (empty if invariant detection is disabled).
   204→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   205→    pub invariants: Vec<ClassifiedInvariant>,
   206→}

   220→/// Complete behavioral specification of a function.
   221→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   222→pub struct FunctionSpec {
   223→    /// Fully qualified function name.
   224→    pub function_name: String,
   225→    /// Source file and location (e.g., "src/math.ts:10").
   226→    pub location: Option<String>,
   227→    /// Equivalence classes describing distinct behaviors.
   228→    pub classes: Vec<SpecClass>,
   229→    /// Total iterations used during exploration.
   230→    pub iterations: u32,
   231→    /// Line coverage achieved.
   232→    pub lines_covered: usize,
   233→    /// Total lines in the function.
   234→    pub total_lines: u32,
   235→    /// Function-wide invariants (across all classes). Empty if invariant detection is disabled.
   236→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   237→    pub invariants: Vec<ClassifiedInvariant>,
   238→    /// SHA-256 fingerprint of the function's source, params, and branches.
   239→    /// Used for staleness detection across runs.
   240→    #[serde(default, skip_serializing_if = "Option::is_none")]
   241→    pub fingerprint: Option<String>,
   242→    /// Fields identified as nondeterministic during exploration.
   243→    /// Used by spec diff to exclude nondeterministic fields from postcondition comparison.
   244→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   245→    pub nondeterministic_fields: Vec<crate::nondeterminism::NondeterministicField>,
   246→}

   248→/// Build a [`FunctionSpec`] from an observation output and its equivalence classes.
   249→///
   250→/// Equivalence classes are produced by [`crate::equivalence::group_into_classes`].
   251→/// Branches discovered via Z3 get `Provenance::Proven`; all others get `Observed`.
   252→pub fn build_spec(
   253→    result: &ObservationOutput,
   254→    eq_classes: &[EquivalenceClass],
   255→    location: Option<String>,
   256→    fingerprint: Option<String>,
   257→) -> FunctionSpec {
   258→    let z3_branches = z3_branch_set(&result.discoveries);
   259→    let classes = eq_classes
   260→        .iter()
   261→        .enumerate()
   262→        .map(|(i, ec)| build_spec_class(i, ec, &z3_branches))
   263→        .collect();
   264→
   265→    FunctionSpec {
   266→        function_name: result.function_name.clone(),
   267→        location,
   268→        classes,
   269→        iterations: result.iterations,
   270→        lines_covered: result.lines_covered,
   271→        total_lines: result.total_lines,
   272→        invariants: vec![],
   273→        fingerprint,
   274→        nondeterministic_fields: result.nondeterministic_fields.clone(),
   275→    }
   276→}

   278→/// Build a [`FunctionSpec`] with invariant detection enabled.
   279→///
   280→/// Convenience wrapper: builds the spec via [`build_spec`], then enriches it
   281→/// with invariants via [`detect_spec_invariants`].
   282→pub fn build_spec_with_invariants(
   283→    result: &ObservationOutput,
   284→    eq_classes: &[EquivalenceClass],
   285→    location: Option<String>,
   286→    fingerprint: Option<String>,
   287→) -> FunctionSpec {
   288→    let mut spec = build_spec(result, eq_classes, location, fingerprint);
   289→    detect_spec_invariants(&mut spec, result, eq_classes);
   290→    spec
   291→}
   292→
   293→/// Enrich a [`FunctionSpec`] with Daikon-style invariant detection.
   294→///
   295→/// Detects invariants at both the function-wide level (across all executions)
   296→/// and the per-class level (for classes with at least 2 samples). Mutates
   297→/// `spec` in place so callers can compose observation, spec construction,
   298→/// and invariant detection as separate pipeline stages.
   299→pub fn detect_spec_invariants(
   300→    spec: &mut FunctionSpec,
   301→    result: &ObservationOutput,
   302→    eq_classes: &[EquivalenceClass],
   303→) {
   304→    use crate::invariants::{
   305→        detect_classified_invariants, records_from_raw_results, InvariantTarget,
   306→    };
   307→
   308→    let all_records = records_from_raw_results(&result.function_name, &result.raw_results);
   309→
   310→    // Function-wide invariants (across all executions)
   311→    let mut function_invariants =
   312→        detect_classified_invariants(&all_records, InvariantTarget::Input);
   313→    function_invariants.extend(detect_classified_invariants(
   314→        &all_records,
   315→        InvariantTarget::Output,
   316→    ));
   317→    spec.invariants = function_invariants;
   318→
   319→    // Per-class invariants: match spec classes to equivalence classes by branch path
   320→    for (spec_class, ec) in spec.classes.iter_mut().zip(eq_classes.iter()) {
   321→        let class_records: Vec<_> = all_records
   322→            .iter()
   323→            .filter(|r| ec.all_inputs.contains(&r.parameters))
   324→            .cloned()
   325→            .collect();
   326→
   327→        if class_records.len() >= 2 {
   328→            let mut class_invs =
   329→                detect_classified_invariants(&class_records, InvariantTarget::Input);
   330→            class_invs.extend(detect_classified_invariants(
   331→                &class_records,
   332→                InvariantTarget::Output,
   333→            ));
   334→            spec_class.invariants = class_invs;
   335→        }
   336→    }
   337→}

   339→/// Collect branch IDs discovered via Z3 into a lookup set.
   340→fn z3_branch_set(discoveries: &[(u32, DiscoveryMethod)]) -> HashSet<u32> {
   341→    discoveries
   342→        .iter()
   343→        .filter(|(_, method)| *method == DiscoveryMethod::Z3)
   344→        .map(|(id, _)| *id)
   345→        .collect()
   346→}

   348→/// Build a single [`SpecClass`] from an equivalence class.
   349→///
   350→/// If every branch in the class's path was discovered by Z3, both precondition
   351→/// and postcondition provenance are `Proven`; otherwise `Observed`.
   352→fn build_spec_class(index: usize, ec: &EquivalenceClass, z3_branches: &HashSet<u32>) -> SpecClass {
   353→    let postcondition = if let Some(ref err_msg) = ec.canonical_thrown_error {
   354→        let (error_type, message) = match err_msg.split_once(": ") {
   355→            Some((t, m)) => (t.to_string(), m.to_string()),
   356→            None => ("Error".to_string(), err_msg.clone()),
   357→        };
   358→        Postcondition::Throws {
   359→            error: ErrorInfo {
   360→                error_type,
   361→                message,
   362→                stack: None, error_category: None },
   363→        }
   364→    } else {
   365→        match &ec.canonical_return_value {
   366→            Some(v) if !v.is_null() => Postcondition::Returns { value: v.clone() },
   367→            _ => Postcondition::ReturnsVoid,
   368→        }
   369→    };
   370→
   371→    let label = format_class_label(index, &postcondition);
   372→
   373→    let canonical_example = ConcreteExample {
   374→        inputs: ec.canonical_example.clone(),
   375→        return_value: ec.canonical_return_value.clone(),
   376→        thrown_error: ec.canonical_thrown_error.as_ref().map(|msg| {
   377→            let (error_type, message) = match msg.split_once(": ") {
   378→                Some((t, m)) => (t.to_string(), m.to_string()),
   379→                None => ("Error".to_string(), msg.clone()),
   380→            };
   381→            ErrorInfo {
   382→                error_type,
   383→                message,
   384→                stack: None, error_category: None }
   385→        }),
   386→    };
   387→
   388→    // A class is proven if every branch step in its path was solved by Z3.
   389→    let all_z3 = !ec.branch_path.0.is_empty()
   390→        && ec
   391→            .branch_path
   392→            .0
   393→            .iter()
   394→            .all(|step| z3_branches.contains(&step.branch_id));
   395→    let provenance = if all_z3 {
   396→        Provenance::Proven
   397→    } else {
   398→        Provenance::Observed
   399→    };
   400→
   401→    SpecClass {
   402→        label,
   403→        branch_path: ec.branch_path.clone(),
   404→        preconditions: ec.common_preconditions.clone(),
   405→        postcondition,
   406→        side_effects: vec![],
   407→        examples: vec![canonical_example],
   408→        sample_count: ec.all_inputs.len(),
   409→        precondition_provenance: provenance,
   410→        postcondition_provenance: provenance,
   411→        invariants: vec![],
   412→    }
   413→}
   414→
   415→/// Generate a human-readable label for a spec class.
   416→fn format_class_label(index: usize, postcondition: &Postcondition) -> String {
   417→    let outcome = match postcondition {
   418→        Postcondition::Returns { value } => format!("returns {}", format_value_short(value)),
   419→        Postcondition::Throws { error } => {
   420→            format!("throws {}: {}", error.error_type, error.message)
   421→        }
   422→        Postcondition::ReturnsVoid => "returns void".to_string(),
   423→    };
   424→    format!("Class {} — {}", index + 1, outcome)
   425→}

   427→/// Format the spec as human-readable markdown.
   428→pub fn format_spec_markdown(spec: &FunctionSpec) -> String {
   429→    let mut out = String::new();
   430→
   431→    // Title
   432→    out.push_str(&format!("# Specification: `{}`\n\n", spec.function_name));
   433→
   434→    if let Some(ref loc) = spec.location {
   435→        out.push_str(&format!("**Location:** `{loc}`\n\n"));
   436→    }
   437→
   438→    // Summary
   439→    out.push_str(&format!(
   440→        "**Behavioral classes:** {}  \n",
   441→        spec.classes.len()
   442→    ));
   443→    out.push_str(&format!(
   444→        "**Exploration:** {} iterations, {}/{} lines covered",
   445→        spec.iterations, spec.lines_covered, spec.total_lines,
   446→    ));
   447→    if spec.total_lines > 0 {
   448→        let pct = (spec.lines_covered as f64 / spec.total_lines as f64 * 100.0).min(100.0);
   449→        out.push_str(&format!(" ({pct:.0}%)"));
   450→    }
   451→    out.push_str("\n\n");
   452→
   453→    // Function-wide invariants
   454→    if !spec.invariants.is_empty() {
   455→        out.push_str("**Function invariants:**\n");
   456→        for ci in &spec.invariants {
   457→            out.push_str(&format!(
   458→                "- {} [{}] ({}/{})\n",
   459→                ci.invariant.description,
   460→                ci.confidence,
   461→                ci.satisfied_count,
   462→                ci.total_count,
   463→            ));
   464→        }
   465→        out.push('\n');
   466→    }
   467→
   468→    out.push_str("---\n\n");
   469→
   470→    // Classes
   471→    for (i, class) in spec.classes.iter().enumerate() {
   472→        out.push_str(&format!("## {}\n\n", class.label));
   473→
   474→        let pre_badge = provenance_badge(class.precondition_provenance);
   475→        let post_badge = provenance_badge(class.postcondition_provenance);
   476→
   477→        // Preconditions
   478→        if class.preconditions.is_empty() {
   479→            out.push_str(&format!("**Preconditions** {pre_badge}: _(none derived)_\n\n"));
   480→        } else {
   481→            out.push_str(&format!("**Preconditions** {pre_badge}:\n"));
   482→            for pre in &class.preconditions {
   483→                out.push_str(&format!("- {}\n", format_precondition(pre)));
   484→            }
   485→            out.push('\n');
   486→        }
   487→
   488→        // Postcondition
   489→        out.push_str(&format!(
   490→            "**Postcondition** {post_badge}: {}\n\n",
   491→            format_postcondition(&class.postcondition)
   492→        ));
   493→
   494→        // Side effects
   495→        if !class.side_effects.is_empty() {
   496→            out.push_str("**Side effects:**\n");
   497→            for effect in &class.side_effects {
   498→                out.push_str(&format!("- {}\n", format_side_effect(effect)));
   499→            }
   500→            out.push('\n');
   501→        }
   502→
   503→        // Invariants (per-class)
   504→        if !class.invariants.is_empty() {
   505→            out.push_str("**Invariants:**\n");
   506→            for ci in &class.invariants {
   507→                out.push_str(&format!(
   508→                    "- {} [{}] ({}/{})\n",
   509→                    ci.invariant.description,
   510→                    ci.confidence,
   511→                    ci.satisfied_count,
   512→                    ci.total_count,
   513→                ));
   514→            }
   515→            out.push('\n');
   516→        }
   517→
   518→        // Examples
   519→        out.push_str(&format!(
   520→            "**Example** ({} execution(s) observed):\n",
   521→            class.sample_count
   522→        ));
   523→        for example in &class.examples {
   524→            let inputs_str = example
   525→                .inputs
   526→                .iter()
   527→                .map(format_value_short)
   528→                .collect::<Vec<_>>()
   529→                .join(", ");
   530→
   531→            let outcome = if let Some(ref err) = example.thrown_error {
   532→                format!("throws {}: {}", err.error_type, err.message)
   533→            } else {
   534→                match &example.return_value {
   535→                    Some(v) if !v.is_null() => format!("-> {}", format_value_short(v)),
   536→                    _ => "-> void".to_string(),
   537→                }
   538→            };
   539→
   540→            out.push_str(&format!(
   541→                "```\n{}({inputs_str}) {outcome}\n```\n",
   542→                spec.function_name
   543→            ));
   544→        }
   545→
   546→        if i + 1 < spec.classes.len() {
   547→            out.push_str("\n---\n\n");
   548→        }
   549→    }
   550→
   551→    out
   552→}

   570→/// Intermediate struct for YAML serialization of a spec class.
   571→/// Converts `ClassifiedInvariant` to human-friendly `SpecInvariant`.
   572→#[derive(Serialize)]
   573→struct YamlSpecClass<'a> {
   574→    label: &'a str,
   575→    preconditions: &'a [Precondition],
   576→    postcondition: &'a Postcondition,
   577→    #[serde(skip_serializing_if = "<[SideEffect]>::is_empty")]
   578→    side_effects: &'a [SideEffect],
   579→    #[serde(skip_serializing_if = "Vec::is_empty")]
   580→    invariants: Vec<SpecInvariant>,

   601→/// Format the spec as YAML with invariants expressed as property descriptions.
   602→///
   603→/// Converts internal `ClassifiedInvariant` representations to human-friendly
   604→/// `SpecInvariant` with categorical confidence (high/medium/low) and a
   605→/// high-level `InvariantCategory`.
   606→pub fn format_spec_yaml(spec: &FunctionSpec) -> Result<String, serde_yaml::Error> {
   607→    let yaml_spec = YamlFunctionSpec {
   608→        function: &spec.function_name,
   609→        location: spec.location.as_deref(),

   611→        classes: spec
   612→            .classes
   613→            .iter()
   614→            .map(|c| YamlSpecClass {
   615→                label: &c.label,
   616→                preconditions: &c.preconditions,
   617→                postcondition: &c.postcondition,
   618→                side_effects: &c.side_effects,
   619→                invariants: c.invariants.iter().map(SpecInvariant::from).collect(),
   620→                examples: &c.examples,
   621→                sample_count: c.sample_count,
   622→                precondition_provenance: c.precondition_provenance,
   623→                postcondition_provenance: c.postcondition_provenance,
   624→            })
   625→            .collect(),
   626→        iterations: spec.iterations,
   627→        lines_covered: spec.lines_covered,
   628→        total_lines: spec.total_lines,

   633→/// Write a [`FileSpecBundle`] to disk using atomic write (temp file + rename).
   634→///
   635→/// Creates parent directories if they don't exist. Writes to a `.json.tmp`
   636→/// sibling first, then renames to the final path to avoid partial reads.
   637→pub fn write_file_spec_bundle(bundle: &FileSpecBundle, path: &Path) -> Result<(), SpecIoError> {
   638→    let json = serde_json::to_string_pretty(bundle)?;
   639→    if let Some(parent) = path.parent() {
   640→        fs::create_dir_all(parent)?;
   641→    }
   642→    let tmp_path = path.with_extension("json.tmp");
   643→    fs::write(&tmp_path, &json)?;
   644→    fs::rename(&tmp_path, path)?;
   645→    Ok(())
   646→}

   651→    let bundle = serde_json::from_str(&contents)?;
   652→    Ok(bundle)

   655→/// Result of comparing current function analyses against an existing spec bundle.
   656→///
   657→/// Used by `--output` incremental mode to decide which functions need re-exploration
   658→/// vs which can be carried over from the previous spec.
   659→#[derive(Debug, Clone, PartialEq)]
   660→pub struct IncrementalPlan {
   661→    /// Functions whose fingerprints changed or are new (need re-exploration).
   662→    pub stale: Vec<String>,
   663→    /// Functions whose fingerprints match the existing spec (reuse old spec).
   664→    pub fresh: Vec<String>,
   665→    /// Functions present in old spec but absent from current analysis (deleted).
   666→    pub removed: Vec<String>,
   667→}
   668→
   669→/// Compare current function analyses against an existing spec to determine staleness.
   670→///
   671→/// Computes deep fingerprints (incorporating callee dependencies) for each function
   672→/// in `current_analyses`, then compares against `existing.functions[name].fingerprint`.
   673→/// A function is fresh only if its deep fingerprint matches. This means a caller is
   674→/// marked stale when any of its in-scope callees change, even if the caller's own
   675→/// source is unchanged.
   676→///
   677→/// Functions in the existing bundle not present in current analysis are removed.
   678→///
   679→/// Returns an error if source extraction fails (e.g., file unreadable).
   680→pub fn compute_incremental_plan(
   681→    file_path: &Path,
   682→    current_analyses: &[FunctionAnalysis],
   683→    existing: &FileSpecBundle,
   684→) -> Result<IncrementalPlan, std::io::Error> {
   685→    let existing_by_name: std::collections::HashMap<&str, &FunctionSpec> = existing
   686→        .functions
   687→        .iter()
   688→        .map(|f| (f.function_name.as_str(), f))
   689→        .collect();
   690→
   691→    let deep_fps = compute_deep_fingerprints(file_path, current_analyses)?;
   692→
   693→    let mut stale = Vec::new();
   694→    let mut fresh = Vec::new();
   695→    let mut seen_names: HashSet<String> = HashSet::new();
   696→
   697→    for func in current_analyses {
   698→        seen_names.insert(func.name.clone());
   699→
   700→        let current_fp = deep_fps.get(&func.name);
   701→
   702→        match (existing_by_name.get(func.name.as_str()), current_fp) {
   703→            (Some(spec), Some(fp)) if spec.fingerprint.as_deref() == Some(fp.as_str()) => {
   704→                fresh.push(func.name.clone());
   705→            }
   706→            _ => {
   707→                stale.push(func.name.clone());
   708→            }
   709→        }
   710→    }
   711→
   712→    let removed: Vec<String> = existing
   713→        .functions
   714→        .iter()
   715→        .filter(|f| !seen_names.contains(&f.function_name))
   716→        .map(|f| f.function_name.clone())
   717→        .collect();
   718→
   719→    Ok(IncrementalPlan {
   720→        stale,
   721→        fresh,
   722→        removed,
   723→    })
   724→}
   725→
   726→/// Merge newly explored specs with fresh specs carried over from an existing bundle.
   727→///
   728→/// `new_specs` contains specs for functions that were re-explored (stale).
   729→/// Functions in `current_function_names` that are NOT in `new_specs` are carried
   730→/// over from `existing` (they were fresh and skipped). Functions not in
   731→/// `current_function_names` are dropped (they were removed from the source).
   732→pub fn merge_file_spec_bundles(
   733→    existing: &FileSpecBundle,
   734→    new_specs: &[FunctionSpec],
   735→    current_function_names: &HashSet<String>,
   736→) -> FileSpecBundle {
   737→    let new_names: HashSet<&str> = new_specs.iter().map(|s| s.function_name.as_str()).collect();
   738→
   739→    let mut merged: Vec<FunctionSpec> = new_specs.to_vec();
   740→
   741→    // Carry over fresh specs from existing bundle (not re-explored, still in source)
   742→    for old_spec in &existing.functions {
   743→        if current_function_names.contains(&old_spec.function_name)
   744→            && !new_names.contains(old_spec.function_name.as_str())
   745→        {
   746→            merged.push(old_spec.clone());
   747→        }
   748→    }
   749→
   750→    FileSpecBundle {
   751→        file: existing.file.clone(),
   752→        functions: merged,
   753→    }
   754→}

   764→/// Format a precondition as a human-readable string.
   765→fn format_precondition(pre: &Precondition) -> String {
   766→    match pre {
   767→        Precondition::AllPositive { param_index } => {
   768→            format!("param[{param_index}] > 0")
   769→        }
   770→        Precondition::AllNegative { param_index } => {
   771→            format!("param[{param_index}] < 0")
   772→        }
   773→        Precondition::AllZero { param_index } => {
   774→            format!("param[{param_index}] == 0")
   775→        }
   776→        Precondition::AllEqual { param_index, value } => {
   777→            format!("param[{param_index}] == {}", format_value_short(value))
   778→        }
   779→        Precondition::SameType {
   780→            param_index,
   781→            type_name,
   782→        } => {
   783→            format!("typeof param[{param_index}] == \"{type_name}\"")
   784→        }
   785→    }
   786→}
   787→
   788→/// Format a postcondition as a human-readable string.
   789→fn format_postcondition(post: &Postcondition) -> String {
   790→    match post {
   791→        Postcondition::Returns { value } => {
   792→            format!("returns {}", format_value_short(value))
   793→        }
   794→        Postcondition::Throws { error } => {
   795→            format!("throws {}: {}", error.error_type, error.message)
   796→        }
   797→        Postcondition::ReturnsVoid => "returns void".to_string(),
   798→    }
   799→}

   801→/// Format a side effect as a human-readable string.
   802→fn format_side_effect(effect: &SideEffect) -> String {
   803→    match effect {
   804→        SideEffect::ConsoleOutput { level, message } => {
   805→            format!("console.{level}(\"{message}\")")
   806→        }
   807→        SideEffect::FileWrite { path, .. } => format!("writes to file: {path}"),
   808→        SideEffect::NetworkRequest { method, url, .. } => format!("{method} {url}"),
   809→        SideEffect::EnvironmentRead { variable, value } => {
   810→            let val = value.as_deref().unwrap_or("null");
   811→            format!("reads env: {variable}={val}")
   812→        }
   813→        SideEffect::GlobalMutation { name } => format!("mutates global: {name}"),
   814→        SideEffect::ThrownError {
   815→            error_type,
   816→            message,
   817→            ..
   818→        } => {
   819→            format!("throws {error_type}: {message}")
   820→        }
   821→        SideEffect::GlobalStateChange {
   822→            variable,
   823→            before,
   824→            after,
   825→        } => {
   826→            format!(
   827→                "{variable}: {} -> {}",
   828→                format_value_short(before),
   829→                format_value_short(after)
   830→            )
   831→        }
   832→    }
   833→}

   835→/// Format a JSON value for display, truncating long values.
   836→fn format_value_short(v: &serde_json::Value) -> String {
   837→    let s = v.to_string();
   838→    if s.len() > 40 {
   839→        format!("{}...", &s[..37])
   840→    } else {
   841→        s
   842→    }
   843→}
