   311→struct DiscoveredConfig {

   318→/// Parse a `.shatter/config.yaml` file.
   319→pub fn parse_config(path: &Path) -> Result<ShatterConfig, ConfigError> {
   320→    let contents = std::fs::read_to_string(path).map_err(|e| ConfigError::Io {
   321→        path: path.to_path_buf(),
   322→        source: e,
   323→    })?;
   324→    let config: ShatterConfig =
   325→        serde_yaml::from_str(&contents).map_err(|e| ConfigError::Parse {
   326→            path: path.to_path_buf(),
   327→            source: e,
   328→        })?;
   329→    Ok(config)
   330→}
   331→
   332→/// Parse a candidate inputs JSON file.
   333→pub fn parse_candidate_inputs(path: &Path) -> Result<Vec<CandidateInput>, ConfigError> {
   334→    let contents = std::fs::read_to_string(path).map_err(|e| ConfigError::InputIo {
   335→        path: path.to_path_buf(),
   336→        source: e,
   337→    })?;
   338→    let inputs: Vec<CandidateInput> =
   339→        serde_json::from_str(&contents).map_err(|e| ConfigError::InputParse {
   340→            path: path.to_path_buf(),
   341→            source: e,
   342→        })?;
   343→    Ok(inputs)
   344→}

   346→/// Discover `.shatter/config.yaml` files by walking upward from `start_dir`.
   347→///
   348→/// Returns configs ordered from nearest (highest priority) to farthest (lowest priority).
   349→pub fn discover_configs(start_dir: &Path) -> Result<Vec<ShatterConfig>, ConfigError> {
   350→    let discovered = discover_configs_with_paths(start_dir)?;
   351→    Ok(discovered.into_iter().map(|d| d.config).collect())
   352→}
   353→
   354→/// Internal: discover configs with their `.shatter/` directory paths preserved.
   355→fn discover_configs_with_paths(start_dir: &Path) -> Result<Vec<DiscoveredConfig>, ConfigError> {
   356→    let mut configs = Vec::new();
   357→    let mut current = Some(start_dir.to_path_buf());
   358→
   359→    while let Some(dir) = current {
   360→        let shatter_dir = dir.join(".shatter");
   361→        let config_path = shatter_dir.join("config.yaml");
   362→
   363→        if config_path.is_file() {
   364→            let config = parse_config(&config_path)?;
   365→            configs.push(DiscoveredConfig {
   366→                shatter_dir,
   367→                config,
   368→            });
   369→        }
   370→
   371→        current = dir.parent().map(Path::to_path_buf);
   372→    }
   373→
   374→    Ok(configs)
   375→}
   376→
   377→/// Merge multiple configs where the first config in the slice has highest priority.
   378→///
   379→/// For defaults, the nearest non-None value wins.
   380→/// For function patterns, the nearest matching pattern wins.
   381→pub fn merge_configs(configs: &[ShatterConfig]) -> ShatterConfig {
   382→    if configs.is_empty() {
   383→        return ShatterConfig::default();
   384→    }
   385→    if configs.len() == 1 {
   386→        return configs[0].clone();
   387→    }
   388→
   389→    // Merge defaults: nearest non-None wins.
   390→    let mut max_iterations = None;
   391→    let mut timeout = None;
   392→    let mut setup = None;
   393→    let mut setup_level = None;
   394→    let mut setup_timeout = None;
   395→    let mut session_setup = None;
   396→
   397→    // Merge generators and file_setup maps: start from farthest, overlay nearer.
   398→    // This lets a near config override specific keys while inheriting the rest.
   399→    let mut generators: Option<HashMap<String, String>> = None;
   400→    let mut param_generators: Option<HashMap<String, String>> = None;
   401→    let mut file_setup: Option<HashMap<String, String>> = None;
   402→
   403→    for config in configs.iter().rev() {
   404→        if let Some(ref g) = config.defaults.generators {
   405→            generators.get_or_insert_with(HashMap::new).extend(
   406→                g.iter().map(|(k, v)| (k.clone(), v.clone())),
   407→            );
   408→        }
   409→        if let Some(ref pg) = config.defaults.param_generators {
   410→            param_generators.get_or_insert_with(HashMap::new).extend(
   411→                pg.iter().map(|(k, v)| (k.clone(), v.clone())),
   412→            );
   413→        }
   414→        if let Some(ref fs) = config.defaults.file_setup {
   415→            file_setup.get_or_insert_with(HashMap::new).extend(
   416→                fs.iter().map(|(k, v)| (k.clone(), v.clone())),
   417→            );
   418→        }
   419→    }
   420→
   421→    for config in configs {
   422→        if max_iterations.is_none() {
   423→            max_iterations = config.defaults.max_iterations;
   424→        }
   425→        if timeout.is_none() {
   426→            timeout = config.defaults.timeout;
   427→        }
   428→        if setup.is_none() {
   429→            setup = config.defaults.setup.clone();
   430→        }
   431→        if setup_level.is_none() {
   432→            setup_level = config.defaults.setup_level;
   433→        }
   434→        if setup_timeout.is_none() {
   435→            setup_timeout = config.defaults.setup_timeout;
   436→        }
   437→        if session_setup.is_none() {
   438→            session_setup = config.defaults.session_setup.clone();
   439→        }
   440→    }
   441→
   442→    // Merge function configs: collect all, nearest first (already in order).
   443→    // Later entries only fill in if the pattern hasn't been seen.
   444→    let mut functions: HashMap<String, FunctionConfig> = HashMap::new();
   445→    for config in configs {
   446→        for (pattern, func_config) in &config.functions {
   447→            functions.entry(pattern.clone()).or_insert_with(|| func_config.clone());
   448→        }
   449→    }
   450→
   451→    let mut opaque_types = Vec::new();
   452→    for config in configs {
   453→        for t in &config.opaque_types {
   454→            if !opaque_types.contains(t) {
   455→                opaque_types.push(t.clone());
   456→            }
   457→        }
   458→    }
   459→
   460→    // Merge nondeterminism declarations: union all confirmed/rejected, dedup by (function, path).
   461→    let mut confirmed = Vec::<NondeterminismDeclaration>::new();
   462→    let mut rejected = Vec::<NondeterminismDeclaration>::new();
   463→    let mut has_any = false;
   464→    for config in configs {
   465→        if let Some(ref nd) = config.nondeterminism {
   466→            has_any = true;
   467→            for decl in &nd.confirmed {
   468→                if !confirmed.iter().any(|d| d.function == decl.function && d.path == decl.path) {
   469→                    confirmed.push(decl.clone());
   470→                }
   471→            }
   472→            for decl in &nd.rejected {
   473→                if !rejected.iter().any(|d| d.function == decl.function && d.path == decl.path) {
   474→                    rejected.push(decl.clone());
   475→                }
   476→            }
   477→        }
   478→    }
   479→    let nondeterminism = if has_any {
   480→        Some(NondeterminismConfig { confirmed, rejected })
   481→    } else {
   482→        None
   483→    };
   484→
   485→    ShatterConfig {
   486→        defaults: DefaultsConfig {
   487→            max_iterations,
   488→            timeout,
   489→            setup,
   490→            setup_level,
   491→            setup_timeout,
   492→            session_setup,
   493→            file_setup,
   494→            generators,
   495→            param_generators,
   496→            mocks: None,
   497→            genetic: None,
   498→        },
   499→        functions,
   500→        opaque_types,
   501→        nondeterminism,
   502→    }
   503→}
   504→
   505→/// Resolve the effective configuration for a specific function identifier.
   506→///
   507→/// `function_id` is typically in `"file:function"` format.
   508→/// `configs` should be ordered nearest-first (as returned by `discover_configs`).
   509→/// `cli_max_iterations` and `cli_timeout` are the CLI-provided defaults.
   510→pub fn resolve_function_config(
   511→    function_id: &str,
   512→    configs: &[ShatterConfig],
   513→    cli_max_iterations: u32,
   514→    cli_timeout: u64,
   515→) -> Result<ResolvedFunctionConfig, ConfigError> {
   516→    let merged = merge_configs(configs);
   517→    resolve_from_merged(function_id, &merged, cli_max_iterations, cli_timeout)
   518→}
   519→
   520→/// Resolve config for a function from an already-merged config.
   521→fn resolve_from_merged(
   522→    function_id: &str,
   523→    config: &ShatterConfig,
   524→    cli_max_iterations: u32,
   525→    cli_timeout: u64,
   526→) -> Result<ResolvedFunctionConfig, ConfigError> {
   527→    // Find the first matching function pattern.
   528→    let mut func_config: Option<&FunctionConfig> = None;
   529→
   530→    for (pattern, fc) in &config.functions {
   531→        let matcher = compile_pattern(pattern)?;
   532→        if matcher.is_match(function_id) {
   533→            func_config = Some(fc);
   534→            break;
   535→        }
   536→    }
   537→
   538→    // Resolution order: function config > defaults > CLI flags
   539→    let max_iterations = func_config
   540→        .and_then(|fc| fc.max_iterations)
   541→        .or(config.defaults.max_iterations)
   542→        .unwrap_or(cli_max_iterations);
   543→
   544→    let timeout = func_config
   545→        .and_then(|fc| fc.timeout)
   546→        .or(config.defaults.timeout)
   547→        .unwrap_or(cli_timeout);
   548→
   549→    let skip = func_config
   550→        .and_then(|fc| fc.skip)
   551→        .unwrap_or(false);
   552→
   553→    let setup_level = func_config
   554→        .and_then(|fc| fc.setup_level)
   555→        .or(config.defaults.setup_level)
   556→        .unwrap_or(SetupLevel::Function);
   557→
   558→    let setup_timeout = func_config
   559→        .and_then(|fc| fc.setup_timeout)
   560→        .or(config.defaults.setup_timeout)
   561→        .unwrap_or(DEFAULT_SETUP_TIMEOUT_SECS);
   562→
   563→    // Merge mock overrides: defaults first, then function-level overrides on top.
   564→    let mut mock_overrides: HashMap<String, crate::auto_mock::MockOverride> = config
   565→        .defaults
   566→        .mocks
   567→        .clone()
   568→        .unwrap_or_default();
   569→    if let Some(fc) = func_config
   570→        && let Some(ref func_mocks) = fc.mocks
   571→    {
   572→        for (k, v) in func_mocks {
   573→            mock_overrides.insert(k.clone(), v.clone());
   574→        }
   575→    }
   576→
   577→    Ok(ResolvedFunctionConfig {
   578→        max_iterations,
   579→        timeout,
   580→        skip,
   581→        candidate_inputs: Vec::new(),
   582→        // Setup and generator paths are resolved to absolute paths by
   583→        // resolve_function_config_with_inputs, which has access to .shatter/ dirs.
   584→        setup: None,
   585→        setup_level,
   586→        setup_timeout,
   587→        session_setup: config.defaults.session_setup.clone(),
   588→        file_setup: HashMap::new(),
   589→        generators: HashMap::new(),
   590→        param_generators: HashMap::new(),
   591→        mock_overrides,
   592→    })
   593→}
   594→
   595→/// Resolve config for a function and load candidate inputs if specified.
   596→///
   597→/// `shatter_dir` is the `.shatter/` directory used to resolve relative input paths.
   598→/// If `explicit_inputs` is provided (from `--inputs` CLI flag), it takes precedence.
   599→pub fn resolve_function_config_with_inputs(
   600→    function_id: &str,
   601→    start_dir: &Path,
   602→    explicit_inputs: Option<&Path>,
   603→    cli_max_iterations: u32,
   604→    cli_timeout: u64,
   605→) -> Result<ResolvedFunctionConfig, ConfigError> {
   606→    let discovered = discover_configs_with_paths(start_dir)?;
   607→    let configs: Vec<ShatterConfig> = discovered.iter().map(|d| d.config.clone()).collect();
   608→    let merged = merge_configs(&configs);
   609→
   610→    let mut resolved = resolve_from_merged(function_id, &merged, cli_max_iterations, cli_timeout)?;
   611→
   612→    // Load candidate inputs.
   613→    if let Some(inputs_path) = explicit_inputs {
   614→        resolved.candidate_inputs = parse_candidate_inputs(inputs_path)?;
   615→    } else {
   616→        // Check function config for an inputs path.
   617→        for dc in &discovered {
   618→            for (pattern, fc) in &dc.config.functions {
   619→                let matcher = compile_pattern(pattern)?;
   620→                if matcher.is_match(function_id)
   621→                    && let Some(inputs_rel) = &fc.inputs
   622→                {
   623→                    let inputs_path = dc.shatter_dir.join(inputs_rel);
   624→                    if inputs_path.is_file() {
   625→                        resolved.candidate_inputs =
   626→                            parse_candidate_inputs(&inputs_path)?;
   627→                    }
   628→                    break;
   629→                }
   630→            }
   631→        }
   632→    }
   633→
   634→    // Resolve setup file path: function-level > defaults, nearest config wins.
   635→    for dc in &discovered {
   636→        if resolved.setup.is_some() {
   637→            break;
   638→        }
   639→        for (pattern, fc) in &dc.config.functions {
   640→            let matcher = compile_pattern(pattern)?;
   641→            if matcher.is_match(function_id)
   642→                && let Some(setup_rel) = &fc.setup
   643→            {
   644→                resolved.setup = Some(dc.shatter_dir.join(setup_rel));
   645→                break;
   646→            }
   647→        }
   648→        if resolved.setup.is_none()
   649→            && let Some(setup_rel) = &dc.config.defaults.setup
   650→        {
   651→            resolved.setup = Some(dc.shatter_dir.join(setup_rel));
   652→        }
   653→    }
   654→
   655→    // Resolve file_setup: walk farthest-to-nearest so nearer configs override.
   656→    for dc in discovered.iter().rev() {
   657→        if let Some(ref fs) = dc.config.defaults.file_setup {
   658→            for (glob_pattern, setup_rel) in fs {
   659→                resolved.file_setup.insert(glob_pattern.clone(), dc.shatter_dir.join(setup_rel));
   660→            }
   661→        }
   662→    }
   663→
   664→    // Resolve generators: merge defaults then overlay function-level.
   665→    // Walk farthest-to-nearest so nearer configs override.
   666→    for dc in discovered.iter().rev() {
   667→        if let Some(ref g) = dc.config.defaults.generators {
   668→            for (type_name, gen_rel) in g {
   669→                resolved.generators.insert(type_name.clone(), dc.shatter_dir.join(gen_rel));
   670→            }
   671→        }
   672→        if let Some(ref pg) = dc.config.defaults.param_generators {
   673→            for (param_name, gen_rel) in pg {
   674→                resolved.param_generators.insert(param_name.clone(), dc.shatter_dir.join(gen_rel));
   675→            }
   676→        }
   677→    }
   678→    // Function-level generators overlay defaults (nearest matching function wins).
   679→    for dc in &discovered {
   680→        for (pattern, fc) in &dc.config.functions {
   681→            let matcher = compile_pattern(pattern)?;
   682→            if matcher.is_match(function_id) {
   683→                if let Some(ref g) = fc.generators {
   684→                    for (type_name, gen_rel) in g {
   685→                        resolved.generators.insert(type_name.clone(), dc.shatter_dir.join(gen_rel));
   686→                    }
   687→                }
   688→                if let Some(ref pg) = fc.param_generators {
   689→                    for (param_name, gen_rel) in pg {
   690→                        resolved.param_generators.insert(param_name.clone(), dc.shatter_dir.join(gen_rel));
   691→                    }
   692→                }
   693→                break;
   694→            }
   695→        }
   696→    }
   697→
   698→    Ok(resolved)
   699→}
   700→
   701→fn compile_pattern(pattern: &str) -> Result<GlobMatcher, ConfigError> {
   702→    let glob = Glob::new(pattern).map_err(|e| ConfigError::InvalidPattern {
   703→        pattern: pattern.to_string(),
   704→        source: e,
   705→    })?;
   706→    Ok(glob.compile_matcher())
   707→}
