     9→     9→/// Log level for the frontend, controlled by SHATTER_LOG_LEVEL env var.
    10→    10→#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
    11→    11→enum FrontendLogLevel {
    12→    12→    Error = 0,
    13→    13→    Warn = 1,
    14→    14→    Info = 2,
    15→    15→    Debug = 3,
    16→    16→    Trace = 4,
    17→    17→}
    18→    18→
    19→    19→impl FrontendLogLevel {
    20→    20→    fn from_env() -> Self {
    21→    21→        match std::env::var("SHATTER_LOG_LEVEL")
    22→    22→            .unwrap_or_default()
    23→    23→            .to_lowercase()
    24→    24→            .as_str()
    25→    25→        {
    26→    26→            "error" => FrontendLogLevel::Error,
    27→    27→            "warn" => FrontendLogLevel::Warn,
    28→    28→            "debug" => FrontendLogLevel::Debug,
    29→    29→            "trace" => FrontendLogLevel::Trace,
    30→    30→            _ => FrontendLogLevel::Info,
    31→    31→        }
    32→    32→    }
    33→    33→}

    35→    35→const DEFAULT_EXEC_TIMEOUT_MS: u64 = 5000;
    36→    36→
    37→    37→/// Read the execution timeout from `SHATTER_EXEC_TIMEOUT` env var (seconds).
    38→    38→/// Returns milliseconds. Falls back to `DEFAULT_EXEC_TIMEOUT_MS` if unset, invalid, zero, or negative.
    39→    39→fn exec_timeout_from_env() -> u64 {
    40→    40→    match std::env::var("SHATTER_EXEC_TIMEOUT") {
    41→    41→        Ok(val) => match val.parse::<f64>() {
    42→    42→            Ok(secs) if secs > 0.0 => (secs * 1000.0) as u64,
    43→    43→            _ => DEFAULT_EXEC_TIMEOUT_MS,
    44→    44→        },
    45→    45→        Err(_) => DEFAULT_EXEC_TIMEOUT_MS,
    46→    46→    }
    47→    47→}

    49→    49→/// Write instrumented source to a temp directory and return the output path.
    50→    50→fn write_instrumented_temp(filename: &str, source: &str) -> io::Result<String> {
    51→    51→    let dir = std::env::temp_dir().join(format!(
    52→    52→        "shatter-instrument-{}",
    53→    53→        std::process::id()
    54→    54→    ));
    55→    55→    std::fs::create_dir_all(&dir)?;
    56→    56→    let out_path = dir.join(filename);
    57→    57→    std::fs::write(&out_path, source)?;
    58→    58→    Ok(out_path.to_string_lossy().into_owned())
    59→    59→}

    61→    61→/// Processes protocol requests from stdin and writes responses to stdout.

    65→    65→    log: L,

   123→   123→    /// Process requests until shutdown or EOF. Returns Ok(()) on clean shutdown.
   124→   124→    pub fn run(mut self) -> io::Result<()> {
   125→   125→        self.log_at(FrontendLogLevel::Debug, &format!(
   126→   126→            "Starting Rust frontend (protocol {PROTOCOL_VERSION})"
   127→   127→        ));
   128→   128→
   129→   129→        let mut line = String::new();
   130→   130→        loop {
   131→   131→            line.clear();
   132→   132→            let bytes_read = self.reader.read_line(&mut line)?;
   133→   133→            if bytes_read == 0 {
   134→   134→                // EOF
   135→   135→                self.log_at(FrontendLogLevel::Debug, "Stdin closed, exiting");
   136→   136→                return Ok(());
   137→   137→            }
   138→   138→
   139→   139→            let trimmed = line.trim();
   140→   140→            if trimmed.is_empty() {
   141→   141→                continue;
   142→   142→            }
   143→   143→
   144→   144→            self.logf(&format!("Received: {trimmed}"));
   145→   145→
   146→   146→            let req: Request = match serde_json::from_str(trimmed) {
   147→   147→                Ok(r) => r,
   148→   148→                Err(e) => {
   149→   149→                    self.logf(&format!("Failed to parse request: {e}"));
   150→   150→                    continue;
   151→   151→                }
   152→   152→            };
   153→   153→
   154→   154→            let (resp, shutdown) = self.dispatch(&req);
   155→   155→            self.send(&resp)?;
   156→   156→
   157→   157→            if shutdown {
   158→   158→                self.log_at(FrontendLogLevel::Debug, "Shutting down");
   159→   159→                return Ok(());
   160→   160→            }
   161→   161→        }
   162→   162→    }
   163→   163→
   164→   164→    fn dispatch(&mut self, req: &Request) -> (Response, bool) {
   165→   165→        let mut resp = Response::base(req.id);
   166→   166→
   167→   167→        if req.protocol_version != protocol::PROTOCOL_VERSION {
   168→   168→            resp.status = "error".to_string();
   169→   169→            resp.code = Some("version_mismatch".to_string());
   170→   170→            resp.message = Some(format!(
   171→   171→                "unsupported protocol version {:?}, expected {:?}",
   172→   172→                req.protocol_version,
   173→   173→                protocol::PROTOCOL_VERSION,
   174→   174→            ));
   175→   175→            return (resp, false);
   176→   176→        }
   177→   177→
   178→   178→        match req.command.as_str() {
   179→   179→            "handshake" => (self.handle_handshake(resp), false),
   180→   180→            "analyze" => (self.handle_analyze(resp, req), false),
   181→   181→            "instrument" => (self.handle_instrument(resp, req), false),
   182→   182→            "execute" => (self.handle_execute(resp, req), false),
   183→   183→            "setup" => (self.handle_setup(resp, req), false),
   184→   184→            "teardown" => (self.handle_teardown(resp, req), false),
   185→   185→            "generate" => (self.handle_generate(resp, req), false),
   186→   186→            "shutdown" => (self.handle_shutdown(resp), true),
   187→   187→            _ => {
   188→   188→                resp.status = "error".to_string();
   189→   189→                resp.code = Some("invalid_request".to_string());
   190→   190→                resp.message = Some(format!("unknown command: {}", req.command));
   191→   191→                (resp, false)
   192→   192→            }
   193→   193→        }
   194→   194→    }
   195→   195→

   199→   199→        resp.language = Some(FRONTEND_LANGUAGE.to_string());

   203→   203→            "generate".to_string(),
   204→   204→            "instrument".to_string(),
   205→   205→        ]);
   206→   206→        resp
   207→   207→    }

   209→   209→    fn handle_analyze(&mut self, mut resp: Response, req: &Request) -> Response {
   210→   210→        let file_path = match &req.file {
   211→   211→            Some(f) => f,
   212→   212→            None => {
   213→   213→                resp.status = "error".to_string();
   214→   214→                resp.code = Some("invalid_request".to_string());
   215→   215→                resp.message = Some("analyze command requires a file path".to_string());
   216→   216→                return resp;
   217→   217→            }
   218→   218→        };
   219→   219→
   220→   220→        self.last_file = Some(file_path.clone());
   221→   221→
   222→   222→        let path = std::path::Path::new(file_path);
   223→   223→        if !path.exists() {
   224→   224→            resp.status = "error".to_string();
   225→   225→            resp.code = Some("file_not_found".to_string());
   226→   226→            resp.message = Some(format!("file not found: {file_path}"));
   227→   227→            return resp;
   228→   228→        }
   229→   229→
   230→   230→        match crate::analyzer::analyze_file(path, req.function.as_deref()) {
   231→   231→            Ok(functions) => {
   232→   232→                resp.status = "analyze".to_string();
   233→   233→                resp.functions = Some(functions);
   234→   234→                resp
   235→   235→            }
   236→   236→            Err(e) => {
   237→   237→                resp.status = "error".to_string();
   238→   238→                resp.code = Some(
   239→   239→                    match &e {
   240→   240→                        crate::analyzer::AnalyzeError::FileNotFound(_) => "file_not_found",
   241→   241→                        crate::analyzer::AnalyzeError::ReadError(_) => "internal_error",
   242→   242→                        crate::analyzer::AnalyzeError::ParseError(_) => "parse_error",
   243→   243→                        crate::analyzer::AnalyzeError::FunctionNotFound(_) => "function_not_found",
   244→   244→                    }
   245→   245→                    .to_string(),
   246→   246→                );
   247→   247→                resp.message = Some(e.to_string());
   248→   248→                resp
   249→   249→            }
   250→   250→        }
   251→   251→    }

   313→   313→    fn handle_execute(&self, mut resp: Response, req: &Request) -> Response {
   314→   314→        let file_path = match req.file.as_ref().or(self.last_file.as_ref()) {
   315→   315→            Some(f) => f,
   316→   316→            None => {
   317→   317→                resp.status = "error".to_string();
   318→   318→                resp.code = Some("invalid_request".to_string());
   319→   319→                resp.message = Some("execute command requires a file path (none provided and no prior analyze/instrument)".to_string());
   320→   320→                return resp;
   321→   321→            }
   322→   322→        };
   323→   323→        let function_name = match &req.function {
   324→   324→            Some(f) => f,
   325→   325→            None => {
   326→   326→                resp.status = "error".to_string();
   327→   327→                resp.code = Some("invalid_request".to_string());
   328→   328→                resp.message = Some("execute command requires a function name".to_string());
   329→   329→                return resp;
   330→   330→            }
   331→   331→        };
   332→   332→
   333→   333→        if self.log_level >= FrontendLogLevel::Debug {
   334→   334→            eprintln!(
   335→   335→                "[shatter-rust] Executing {function_name} in {file_path} with {} inputs",
   336→   336→                req.inputs.len()
   337→   337→            );
   338→   338→        }
   339→   339→
   340→   340→        match crate::executor::execute_function(
   341→   341→            file_path,
   342→   342→            function_name,
   343→   343→            &req.inputs,
   344→   344→            &req.mocks,
   345→   345→            self.exec_timeout_ms,
   346→   346→        ) {
   347→   347→            Ok(result) => {
   348→   348→                resp.status = "execute".to_string();
   349→   349→                resp.return_value = result.return_value;
   350→   350→                resp.thrown_error = result.thrown_error;
   351→   351→                resp.branch_path = Some(result.branch_path);
   352→   352→                resp.lines_executed = Some(result.lines_executed);
   353→   353→                resp.calls_to_external = Some(result.calls_to_external);
   354→   354→                resp.path_constraints = Some(result.path_constraints);
   355→   355→                resp.side_effects = Some(result.side_effects);
   356→   356→                resp.performance = Some(result.performance);
   357→   357→                resp
   358→   358→            }
   359→   359→            Err(crate::executor::ExecuteError::FileError(msg)) => {
   360→   360→                resp.status = "error".to_string();
   361→   361→                resp.code = Some("file_not_found".to_string());
   362→   362→                resp.message = Some(msg);
   363→   363→                resp
   364→   364→            }
   365→   365→            Err(crate::executor::ExecuteError::CompilationFailed(msg)) => {
   366→   366→                resp.status = "error".to_string();
   367→   367→                resp.code = Some("compilation_error".to_string());
   368→   368→                resp.message = Some(msg);
   369→   369→                resp
   370→   370→            }
   371→   371→            Err(crate::executor::ExecuteError::NonExecutable(msg)) => {
   372→   372→                resp.status = "error".to_string();
   373→   373→                resp.code = Some("non_executable".to_string());
   374→   374→                resp.message = Some(msg);
   375→   375→                resp
   376→   376→            }
   377→   377→            Err(e) => {
   378→   378→                resp.status = "error".to_string();
   379→   379→                resp.code = Some("internal_error".to_string());
   380→   380→                resp.message = Some(e.to_string());
   381→   381→                resp
   382→   382→            }
   383→   383→        }
   384→   384→    }

   479→   479→            Some(f) => f,
   480→   480→            None => {
   481→   481→                resp.status = "error".to_string();
   482→   482→                resp.code = Some("invalid_request".to_string());
   483→   483→                resp.message = Some("generate command requires a file path".to_string());
   484→   484→                return resp;
   485→   485→            }
   486→   486→        };
   487→   487→        let func_name = match &req.name {
   488→   488→            Some(n) => n.clone(),
   489→   489→            None => {
   490→   490→                resp.status = "error".to_string();
   491→   491→                resp.code = Some("invalid_request".to_string());
   492→   492→                resp.message = Some("generate command requires a name".to_string());
   493→   493→                return resp;
   494→   494→            }
   495→   495→        };
   496→   496→
   497→   497→        let path = std::path::Path::new(file_path);
   498→   498→        let ext = path.extension().and_then(|e| e.to_str());
   499→   499→
   500→   500→        match ext {
   501→   501→            Some("wasm") => {
   502→   502→                match self.wasm_cache.generate(path, &func_name, req.recipe.as_ref()) {
   503→   503→                    Ok((value, generator_id, recipe)) => {
   504→   504→                        resp.status = "generate".to_string();
   505→   505→                        resp.value = Some(value);
   506→   506→                        resp.generator_id = Some(generator_id);
   507→   507→                        resp.recipe = recipe;
   508→   508→                        resp
   509→   509→                    }
   510→   510→                    Err(e) => {
   511→   511→                        resp.status = "error".to_string();
   512→   512→                        resp.code = Some("internal_error".to_string());
   513→   513→                        resp.message = Some(e);
   514→   514→                        resp
   515→   515→                    }
   516→   516→                }
   517→   517→            }
   518→   518→            Some("rs") => {
   519→   519→                if let Some(ref registry) = self.native_registry {
   520→   520→                    match registry.generate(&func_name, req.recipe.clone()) {
   521→   521→                        Ok((value, generator_id, recipe)) => {
   522→   522→                            resp.status = "generate".to_string();
   523→   523→                            resp.value = Some(value);
   524→   524→                            resp.generator_id = Some(generator_id);
   525→   525→                            resp.recipe = Some(recipe);
   526→   526→                            resp
   527→   527→                        }
   528→   528→                        Err(e) => {
   529→   529→                            resp.status = "error".to_string();
   530→   530→                            resp.code = Some("internal_error".to_string());
   531→   531→                            resp.message = Some(e);
   532→   532→                            resp
   533→   533→                        }
   534→   534→                    }
   535→   535→                } else {
   536→   536→                    resp.status = "error".to_string();
   537→   537→                    resp.code = Some("internal_error".to_string());
   538→   538→                    resp.message = Some(format!(
   539→   539→                        "native generator {func_name:?} requires a custom build (run `shatter build-frontend rust`)"
   540→   540→                    ));
   541→   541→                    resp
   542→   542→                }
   543→   543→            }
   544→   544→            _ => {
   545→   545→                resp.status = "error".to_string();
   546→   546→                resp.code = Some("invalid_request".to_string());
   547→   547→                resp.message = Some(format!(
   548→   548→                    "unsupported generator file type: {}",
   549→   549→                    file_path
   550→   550→                ));
   551→   551→                resp
   552→   552→            }
   553→   553→        }
   554→   554→    }
   555→   555→
   556→   556→    fn handle_shutdown(&self, mut resp: Response) -> Response {
   557→   557→        resp.status = "shutdown_ack".to_string();
   558→   558→        resp
   559→   559→    }

   561→   561→    fn send(&mut self, resp: &Response) -> io::Result<()> {
   562→   562→        let data = serde_json::to_string(resp)
   563→   563→            .map_err(io::Error::other)?;
   564→   564→        self.logf(&format!("Sent: {data}"));
   565→   565→        writeln!(self.writer, "{data}")?;
   566→   566→        self.writer.flush()
   567→   567→    }

   569→   569→    /// Log at TRACE level (protocol messages).
   570→   570→    fn logf(&mut self, msg: &str) {
   571→   571→        self.log_at(FrontendLogLevel::Trace, msg);
   572→   572→    }
   573→   573→
   574→   574→    /// Log at a specific level.
   575→   575→    fn log_at(&mut self, level: FrontendLogLevel, msg: &str) {
   576→   576→        if self.log_level >= level {
   577→   577→            let _ = writeln!(self.log, "[shatter-rust] {msg}");
   578→   578→        }
   579→   579→    }
   580→   580→}
