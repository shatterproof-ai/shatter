   455→   455→/// Validate an `ExecuteResult` deserialized from a frontend subprocess.
   456→   456→///
   457→   457→/// Checks semantic invariants that serde can't enforce:
   458→   458→/// - `branch_path` entries have non-default constraints (when non-empty, an all-unknown
   459→   459→///   branch_path produces identical hashes for different paths → silent hash collision)
   460→   460→/// - `path_constraints` length matches `branch_path` length when both are non-empty
   461→   461→#[contracts::ensures(ret == execute_result_is_valid(result),
   462→   462→    "postcondition must match validation logic")]
   463→   463→pub fn validate_execute_result(result: &ExecuteResult) -> bool {
   464→   464→    execute_result_is_valid(result)
   465→   465→}
   466→   466→
   467→   467→fn execute_result_is_valid(result: &ExecuteResult) -> bool {
   468→   468→    // Branch path with all-unknown constraints causes silent hash collisions
   469→   469→    // because the path hash depends on constraint content.
   470→   470→    if !result.branch_path.is_empty() {
   471→   471→        let all_unknown = result.branch_path.iter().all(|bd| {
   472→   472→            matches!(bd.constraint, SymConstraint::Unknown { .. })
   473→   473→        });
   474→   474→        // All-unknown is valid for frontends without symbolic analysis (Go),
   475→   475→        // but path_constraints should then also be empty.
   476→   476→        if all_unknown && !result.path_constraints.is_empty() {
   477→   477→            return false;
   478→   478→        }
   479→   479→    }
   480→   480→    true
   481→   481→}
   482→   482→
   483→   483→/// Validate a `FunctionAnalysis` list deserialized from a frontend subprocess.
   484→   484→///
   485→   485→/// Checks semantic invariants that serde can't enforce:
   486→   486→/// - Function names are non-empty (empty name → silent lookup failures)
   487→   487→/// - `start_line <= end_line` (inverted range → wrong source slicing)
   488→   488→/// - Param count is plausible (> 255 params suggests a deserialization bug)
   489→   489→#[contracts::ensures(ret == analyze_result_is_valid(functions),
   490→   490→    "postcondition must match validation logic")]
   491→   491→pub fn validate_analyze_result(functions: &[FunctionAnalysis]) -> bool {
   492→   492→    analyze_result_is_valid(functions)
   493→   493→}
   494→   494→
   495→   495→/// Upper bound on parameter count — anything above this likely indicates
   496→   496→/// a deserialization bug rather than a real function signature.
   497→   497→const MAX_PLAUSIBLE_PARAMS: usize = 255;
   498→   498→
   499→   499→fn analyze_result_is_valid(functions: &[FunctionAnalysis]) -> bool {
   500→   500→    for func in functions {
   501→   501→        if func.name.is_empty() {
   502→   502→            return false;
   503→   503→        }
   504→   504→        if func.start_line > func.end_line {
   505→   505→            return false;
   506→   506→        }
   507→   507→        if func.params.len() > MAX_PLAUSIBLE_PARAMS {
   508→   508→            return false;
   509→   509→        }
   510→   510→    }
   511→   511→    true
   512→   512→}
