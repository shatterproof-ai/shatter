     1→     1→//! Protocol types for communication between the Rust core and language frontends.
     2→     2→//!
     3→     3→//! All messages are newline-delimited JSON over stdin/stdout. The core sends
     4→     4→//! [`Request`] messages to frontends and receives [`Response`] messages back.
     5→     5→//! Every message includes a protocol version for compatibility checking.

    56→    56→/// A request message sent from the core engine to a language frontend.
    57→    57→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    58→    58→pub struct Request {

    60→    60→    pub protocol_version: String,
    61→    61→    /// Unique identifier for correlating responses to requests.
    62→    62→    pub id: u64,

    64→    64→    #[serde(flatten)]
    65→    65→    pub command: Command,
    66→    66→}

    68→    68→/// Commands the core can send to a frontend.
    69→    69→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    70→    70→#[serde(tag = "command", rename_all = "snake_case")]
    71→    71→pub enum Command {
    72→    72→    /// Initial handshake to negotiate protocol version and capabilities.
    73→    73→    Handshake {
    74→    74→        /// Capabilities supported by the core.
    75→    75→        capabilities: Vec<String>,
    76→    76→    },
    77→    77→    /// Analyze a function to extract type information, branches, and dependencies.
    78→    78→    Analyze {
    79→    79→        /// Path to the source file.
    80→    80→        file: String,
    81→    81→        /// Name of the function to analyze. If absent, analyze all exported functions.
    82→    82→        function: Option<String>,
    83→    83→        /// Detected project root directory, if any.
    84→    84→        #[serde(default, skip_serializing_if = "Option::is_none")]
    85→    85→        project_root: Option<String>,
    86→    86→    },
    87→    87→    /// Instrument a function for symbolic constraint tracking.
    88→    88→    Instrument {
    89→    89→        /// Path to the source file.
    90→    90→        file: String,
    91→    91→        /// Name of the function to instrument.
    92→    92→        function: String,
    93→    93→        /// Mock configurations for external dependencies.
    94→    94→        mocks: Vec<MockConfig>,
    95→    95→        /// Detected project root directory, if any.
    96→    96→        #[serde(default, skip_serializing_if = "Option::is_none")]
    97→    97→        project_root: Option<String>,
    98→    98→    },
    99→    99→    /// Execute an instrumented function with specific inputs and mocks.
   100→   100→    Execute {
   101→   101→        /// Fully qualified function identifier.
   102→   102→        function: String,
   103→   103→        /// Input values for the function parameters (JSON-encoded).
   104→   104→        inputs: Vec<serde_json::Value>,
   105→   105→        /// Mock configurations for external dependencies.
   106→   106→        mocks: Vec<MockConfig>,
   107→   107→        /// Stack of active setup contexts from enclosing Setup commands, if any.
   108→   108→        #[serde(default, skip_serializing_if = "Option::is_none")]
   109→   109→        setup_context: Option<SetupContextStack>,
   110→   110→    },
   111→   111→    /// Run a setup file to initialize state before function execution.
   112→   112→    Setup {
   113→   113→        /// Path to the setup file.
   114→   114→        file: String,
   115→   115→        /// Scope identifier (function name, file path, or session label).
   116→   116→        scope: String,
   117→   117→        /// Lifecycle level for this setup.
   118→   118→        level: SetupLevel,
   119→   119→        /// Detected project root directory, if any.
   120→   120→        #[serde(default, skip_serializing_if = "Option::is_none")]
   121→   121→        project_root: Option<String>,
   122→   122→        /// Parent context stack from enclosing setup levels, if any.
   123→   123→        #[serde(default, skip_serializing_if = "Option::is_none")]
   124→   124→        parent_context: Option<SetupContextStack>,
   125→   125→    },
   126→   126→    /// Tear down state established by a prior Setup command.
   127→   127→    Teardown {
   128→   128→        /// Scope identifier matching the corresponding Setup command.
   129→   129→        scope: String,
   130→   130→        /// Lifecycle level matching the corresponding Setup command.
   131→   131→        level: SetupLevel,
   132→   132→    },
   133→   133→    /// Invoke a custom generator to produce a value for a type or parameter.
   134→   134→    Generate {
   135→   135→        /// Path to the generator file.
   136→   136→        file: String,
   137→   137→        /// Name of the type or parameter to generate a value for.
   138→   138→        name: String,
   139→   139→        /// Whether this generator targets a type name or a parameter name.
   140→   140→        kind: GeneratorKind,
   141→   141→        /// Reconstruction recipe from a previous generation.
   142→   142→        #[serde(default, skip_serializing_if = "Option::is_none")]
   143→   143→        recipe: Option<serde_json::Value>,
   144→   144→        /// Detected project root directory, if any.
   145→   145→        #[serde(default, skip_serializing_if = "Option::is_none")]
   146→   146→        project_root: Option<String>,
   147→   147→    },
   148→   148→    /// Request graceful shutdown of the frontend process.
   149→   149→    Shutdown,
   150→   150→}

   489→   489→/// Validate an `ExecuteResult` deserialized from a frontend subprocess.
   490→   490→///
   491→   491→/// Checks semantic invariants that serde can't enforce:
   492→   492→/// - `branch_path` entries have non-default constraints (when non-empty, an all-unknown
   493→   493→///   branch_path produces identical hashes for different paths → silent hash collision)
   494→   494→/// - `path_constraints` length matches `branch_path` length when both are non-empty
   495→   495→#[contracts::ensures(ret == execute_result_is_valid(result),
   496→   496→    "postcondition must match validation logic")]
   497→   497→pub fn validate_execute_result(result: &ExecuteResult) -> bool {
   498→   498→    execute_result_is_valid(result)
   499→   499→}
   500→   500→
   501→   501→fn execute_result_is_valid(result: &ExecuteResult) -> bool {
   502→   502→    // Branch path with all-unknown constraints causes silent hash collisions
   503→   503→    // because the path hash depends on constraint content.
   504→   504→    if !result.branch_path.is_empty() {
   505→   505→        let all_unknown = result.branch_path.iter().all(|bd| {
   506→   506→            matches!(bd.constraint, SymConstraint::Unknown { .. })
   507→   507→        });
   508→   508→        // All-unknown is valid for frontends without symbolic analysis (Go),
   509→   509→        // but path_constraints should then also be empty.
   510→   510→        if all_unknown && !result.path_constraints.is_empty() {
   511→   511→            return false;
   512→   512→        }
   513→   513→    }
   514→   514→    true
   515→   515→}
   516→   516→
   517→   517→/// Validate a `FunctionAnalysis` list deserialized from a frontend subprocess.
   518→   518→///
   519→   519→/// Checks semantic invariants that serde can't enforce:
   520→   520→/// - Function names are non-empty (empty name → silent lookup failures)
   521→   521→/// - `start_line <= end_line` (inverted range → wrong source slicing)
   522→   522→/// - Param count is plausible (> 255 params suggests a deserialization bug)
   523→   523→#[contracts::ensures(ret == analyze_result_is_valid(functions),
   524→   524→    "postcondition must match validation logic")]
   525→   525→pub fn validate_analyze_result(functions: &[FunctionAnalysis]) -> bool {
   526→   526→    analyze_result_is_valid(functions)
   527→   527→}
   528→   528→
   529→   529→/// Upper bound on parameter count — anything above this likely indicates
   530→   530→/// a deserialization bug rather than a real function signature.
   531→   531→const MAX_PLAUSIBLE_PARAMS: usize = 255;
   532→   532→
   533→   533→fn analyze_result_is_valid(functions: &[FunctionAnalysis]) -> bool {
   534→   534→    for func in functions {
   535→   535→        if func.name.is_empty() {
   536→   536→            return false;
   537→   537→        }
   538→   538→        if func.start_line > func.end_line {
   539→   539→            return false;
   540→   540→        }
   541→   541→        if func.params.len() > MAX_PLAUSIBLE_PARAMS {
   542→   542→            return false;
   543→   543→        }
   544→   544→    }
   545→   545→    true
   546→   546→}
