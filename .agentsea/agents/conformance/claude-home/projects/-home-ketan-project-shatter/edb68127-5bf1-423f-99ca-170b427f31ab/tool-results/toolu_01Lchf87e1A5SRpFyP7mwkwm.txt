    23тЖТ        match std::env::var("SHATTER_LOG_LEVEL")
    24тЖТ            .unwrap_or_default()

    29тЖТ            "warn" => FrontendLogLevel::Warn,
    30тЖТ            "debug" => FrontendLogLevel::Debug,
    31тЖТ            "trace" => FrontendLogLevel::Trace,
    32тЖТ            _ => FrontendLogLevel::Info,
    33тЖТ        }
    34тЖТ    }

    37тЖТconst DEFAULT_EXEC_TIMEOUT_MS: u64 = 5000;
    38тЖТ
    39тЖТ/// Read the execution timeout from `SHATTER_EXEC_TIMEOUT` env var (seconds).
    40тЖТ/// Returns milliseconds. Falls back to `DEFAULT_EXEC_TIMEOUT_MS` if unset, invalid, zero, or negative.
    41тЖТfn exec_timeout_from_env() -> u64 {
    42тЖТ    match std::env::var("SHATTER_EXEC_TIMEOUT") {
    43тЖТ        Ok(val) => match val.parse::<f64>() {
    44тЖТ            Ok(secs) if secs > 0.0 => (secs * 1000.0) as u64,
    45тЖТ            _ => DEFAULT_EXEC_TIMEOUT_MS,
    46тЖТ        },
    47тЖТ        Err(_) => DEFAULT_EXEC_TIMEOUT_MS,
    48тЖТ    }
    49тЖТ}

    51тЖТ/// Write instrumented source to a temp directory and return the output path.
    52тЖТfn write_instrumented_temp(filename: &str, source: &str) -> io::Result<String> {
    53тЖТ    let dir = std::env::temp_dir().join(format!(
    54тЖТ        "shatter-instrument-{}",
    55тЖТ        std::process::id()
    56тЖТ    ));
    57тЖТ    std::fs::create_dir_all(&dir)?;
    58тЖТ    let out_path = dir.join(filename);
    59тЖТ    std::fs::write(&out_path, source)?;
    60тЖТ    Ok(out_path.to_string_lossy().into_owned())
    61тЖТ}

   125тЖТ    /// Process requests until shutdown or EOF. Returns Ok(()) on clean shutdown.
   126тЖТ    pub fn run(mut self) -> io::Result<()> {
   127тЖТ        self.log_at(FrontendLogLevel::Debug, &format!(
   128тЖТ            "Starting Rust frontend (protocol {PROTOCOL_VERSION})"
   129тЖТ        ));
   130тЖТ
   131тЖТ        let mut line = String::new();
   132тЖТ        loop {
   133тЖТ            line.clear();
   134тЖТ            let bytes_read = self.reader.read_line(&mut line)?;
   135тЖТ            if bytes_read == 0 {
   136тЖТ                // EOF
   137тЖТ                self.log_at(FrontendLogLevel::Debug, "Stdin closed, exiting");
   138тЖТ                return Ok(());
   139тЖТ            }
   140тЖТ
   141тЖТ            let trimmed = line.trim();
   142тЖТ            if trimmed.is_empty() {
   143тЖТ                continue;
   144тЖТ            }
   145тЖТ
   146тЖТ            self.logf(&format!("Received: {trimmed}"));
   147тЖТ
   148тЖТ            let req: Request = match serde_json::from_str(trimmed) {
   149тЖТ                Ok(r) => r,
   150тЖТ                Err(e) => {
   151тЖТ                    self.logf(&format!("Failed to parse request: {e}"));
   152тЖТ                    let err_resp = Response {
   153тЖТ                        id: 0,
   154тЖТ                        status: "error".to_string(),
   155тЖТ                        code: Some(ERR_INVALID_REQUEST.to_string()),
   156тЖТ                        message: Some(format!("Invalid JSON: {e}")),
   157тЖТ                        ..Response::base(0)
   158тЖТ                    };
   159тЖТ                    self.send(&err_resp)?;
   160тЖТ                    continue;
   161тЖТ                }
   162тЖТ            };
   163тЖТ
   164тЖТ            let (resp, shutdown) = self.dispatch(&req);
   165тЖТ            self.send(&resp)?;
   166тЖТ
   167тЖТ            if shutdown {
   168тЖТ                self.log_at(FrontendLogLevel::Debug, "Shutting down");
   169тЖТ                return Ok(());
   170тЖТ            }
   171тЖТ        }
   172тЖТ    }
   173тЖТ
   174тЖТ    fn dispatch(&mut self, req: &Request) -> (Response, bool) {
   175тЖТ        let mut resp = Response::base(req.id);
   176тЖТ
   177тЖТ        if req.protocol_version != protocol::PROTOCOL_VERSION {
   178тЖТ            resp.status = "error".to_string();
   179тЖТ            resp.code = Some(ERR_VERSION_MISMATCH.to_string());
   180тЖТ            resp.message = Some(format!(
   181тЖТ                "unsupported protocol version {:?}, expected {:?}",
   182тЖТ                req.protocol_version,
   183тЖТ                protocol::PROTOCOL_VERSION,
   184тЖТ            ));
   185тЖТ            return (resp, false);
   186тЖТ        }
   187тЖТ
   188тЖТ        match req.command.as_str() {
   189тЖТ            "handshake" => (self.handle_handshake(resp), false),
   190тЖТ            "analyze" => (self.handle_analyze(resp, req), false),
   191тЖТ            "instrument" => (self.handle_instrument(resp, req), false),
   192тЖТ            "execute" => (self.handle_execute(resp, req), false),
   193тЖТ            "setup" => (self.handle_setup(resp, req), false),
   194тЖТ            "teardown" => (self.handle_teardown(resp, req), false),
   195тЖТ            "generate" => (self.handle_generate(resp, req), false),
   196тЖТ            "shutdown" => (self.handle_shutdown(resp), true),
   197тЖТ            _ => {
   198тЖТ                resp.status = "error".to_string();
   199тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   200тЖТ                resp.message = Some(format!("unknown command: {}", req.command));
   201тЖТ                (resp, false)
   202тЖТ            }
   203тЖТ        }
   204тЖТ    }

   221тЖТ    fn handle_analyze(&mut self, mut resp: Response, req: &Request) -> Response {
   222тЖТ        let file_path = match &req.file {
   223тЖТ            Some(f) => f,
   224тЖТ            None => {
   225тЖТ                resp.status = "error".to_string();
   226тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   227тЖТ                resp.message = Some("analyze command requires a file path".to_string());
   228тЖТ                return resp;
   229тЖТ            }
   230тЖТ        };
   231тЖТ
   232тЖТ        self.last_file = Some(file_path.clone());
   233тЖТ
   234тЖТ        let path = std::path::Path::new(file_path);
   235тЖТ        if !path.exists() {
   236тЖТ            resp.status = "error".to_string();
   237тЖТ            resp.code = Some(ERR_FILE_NOT_FOUND.to_string());
   238тЖТ            resp.message = Some(format!("file not found: {file_path}"));
   239тЖТ            return resp;
   240тЖТ        }
   241тЖТ
   242тЖТ        match crate::analyzer::analyze_file(path, req.function.as_deref()) {
   243тЖТ            Ok(functions) => {
   244тЖТ                resp.status = "analyze".to_string();
   245тЖТ                resp.functions = Some(functions);
   246тЖТ                resp
   247тЖТ            }
   248тЖТ            Err(e) => {
   249тЖТ                resp.status = "error".to_string();
   250тЖТ                resp.code = Some(
   251тЖТ                    match &e {
   252тЖТ                        crate::analyzer::AnalyzeError::FileNotFound(_) => ERR_FILE_NOT_FOUND,
   253тЖТ                        crate::analyzer::AnalyzeError::ReadError(_) => ERR_INTERNAL_ERROR,
   254тЖТ                        crate::analyzer::AnalyzeError::ParseError(_) => ERR_PARSE_ERROR,
   255тЖТ                        crate::analyzer::AnalyzeError::FunctionNotFound(_) => ERR_FUNCTION_NOT_FOUND,
   256тЖТ                    }
   257тЖТ                    .to_string(),
   258тЖТ                );
   259тЖТ                resp.message = Some(e.to_string());
   260тЖТ                resp
   261тЖТ            }
   262тЖТ        }
   263тЖТ    }

   312тЖТ            Err(e) => {

   315тЖТ                    crate::instrument::InstrumentError::FileNotFound(_) => ERR_FILE_NOT_FOUND,

   317тЖТ                    crate::instrument::InstrumentError::ParseError(_) => ERR_PARSE_ERROR,
   318тЖТ                }.to_string());

   320тЖТ                resp
   321тЖТ            }
   322тЖТ        }
   323тЖТ    }

   325тЖТ    fn handle_execute(&self, mut resp: Response, req: &Request) -> Response {
   326тЖТ        let file_path = match req.file.as_ref().or(self.last_file.as_ref()) {
   327тЖТ            Some(f) => f,
   328тЖТ            None => {
   329тЖТ                resp.status = "error".to_string();
   330тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   331тЖТ                resp.message = Some("execute command requires a file path (none provided and no prior analyze/instrument)".to_string());
   332тЖТ                return resp;
   333тЖТ            }
   334тЖТ        };
   335тЖТ        let function_name = match &req.function {
   336тЖТ            Some(f) => f,
   337тЖТ            None => {
   338тЖТ                resp.status = "error".to_string();
   339тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   340тЖТ                resp.message = Some("execute command requires a function name".to_string());
   341тЖТ                return resp;
   342тЖТ            }
   343тЖТ        };
   344тЖТ
   345тЖТ        if self.log_level >= FrontendLogLevel::Debug {
   346тЖТ            eprintln!(
   347тЖТ                "[shatter-rust] Executing {function_name} in {file_path} with {} inputs",
   348тЖТ                req.inputs.len()
   349тЖТ            );
   350тЖТ        }
   351тЖТ
   352тЖТ        match crate::executor::execute_function(
   353тЖТ            file_path,
   354тЖТ            function_name,
   355тЖТ            &req.inputs,
   356тЖТ            &req.mocks,
   357тЖТ            self.exec_timeout_ms,
   358тЖТ        ) {
   359тЖТ            Ok(result) => {
   360тЖТ                resp.status = "execute".to_string();
   361тЖТ                resp.return_value = result.return_value;
   362тЖТ                resp.thrown_error = result.thrown_error;
   363тЖТ                resp.branch_path = Some(result.branch_path);
   364тЖТ                resp.lines_executed = Some(result.lines_executed);
   365тЖТ                resp.calls_to_external = Some(result.calls_to_external);
   366тЖТ                resp.path_constraints = Some(result.path_constraints);
   367тЖТ                resp.side_effects = Some(result.side_effects);
   368тЖТ                resp.performance = Some(result.performance);
   369тЖТ                resp
   370тЖТ            }
   371тЖТ            Err(crate::executor::ExecuteError::FileError(msg)) => {
   372тЖТ                resp.status = "error".to_string();
   373тЖТ                resp.code = Some(ERR_FILE_NOT_FOUND.to_string());
   374тЖТ                resp.message = Some(msg);
   375тЖТ                resp
   376тЖТ            }
   377тЖТ            Err(crate::executor::ExecuteError::CompilationFailed(msg)) => {
   378тЖТ                resp.status = "error".to_string();
   379тЖТ                resp.code = Some(ERR_COMPILATION_ERROR.to_string());
   380тЖТ                resp.message = Some(msg);
   381тЖТ                resp
   382тЖТ            }
   383тЖТ            Err(crate::executor::ExecuteError::NonExecutable(msg)) => {
   384тЖТ                resp.status = "error".to_string();
   385тЖТ                resp.code = Some(ERR_NOT_SUPPORTED.to_string());
   386тЖТ                resp.message = Some(msg);
   387тЖТ                resp
   388тЖТ            }
   389тЖТ            Err(e) => {
   390тЖТ                resp.status = "error".to_string();
   391тЖТ                resp.code = Some(ERR_INTERNAL_ERROR.to_string());
   392тЖТ                resp.message = Some(e.to_string());
   393тЖТ                resp
   394тЖТ            }
   395тЖТ        }
   396тЖТ    }

   473тЖТ        };
   474тЖТ
   475тЖТ        match crate::setup::run_teardown(level, scope) {
   476тЖТ            Ok(()) => {
   477тЖТ                resp.status = "teardown_ack".to_string();
   478тЖТ                resp
   479тЖТ            }
   480тЖТ            Err(e) => {
   481тЖТ                resp.status = "error".to_string();
   482тЖТ                resp.code = Some(ERR_INTERNAL_ERROR.to_string());
   483тЖТ                resp.message = Some(e.to_string());
   484тЖТ                resp
   485тЖТ            }
   486тЖТ        }
   487тЖТ    }
   488тЖТ
   489тЖТ    fn handle_generate(&mut self, mut resp: Response, req: &Request) -> Response {
   490тЖТ        let file_path = match &req.file {
   491тЖТ            Some(f) => f,
   492тЖТ            None => {
   493тЖТ                resp.status = "error".to_string();
   494тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   495тЖТ                resp.message = Some("generate command requires a file path".to_string());
   496тЖТ                return resp;
   497тЖТ            }
   498тЖТ        };
   499тЖТ        let func_name = match &req.name {
   500тЖТ            Some(n) => n.clone(),
   501тЖТ            None => {
   502тЖТ                resp.status = "error".to_string();
   503тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   504тЖТ                resp.message = Some("generate command requires a name".to_string());
   505тЖТ                return resp;
   506тЖТ            }
   507тЖТ        };
   508тЖТ
   509тЖТ        let path = std::path::Path::new(file_path);
   510тЖТ        let ext = path.extension().and_then(|e| e.to_str());
   511тЖТ
   512тЖТ        match ext {
   513тЖТ            Some("wasm") => {
   514тЖТ                match self.wasm_cache.generate(path, &func_name, req.recipe.as_ref()) {
   515тЖТ                    Ok((value, generator_id, recipe)) => {
   516тЖТ                        resp.status = "generate".to_string();
   517тЖТ                        resp.value = Some(value);
   518тЖТ                        resp.generator_id = Some(generator_id);
   519тЖТ                        resp.recipe = recipe;
   520тЖТ                        resp
   521тЖТ                    }
   522тЖТ                    Err(e) => {
   523тЖТ                        resp.status = "error".to_string();
   524тЖТ                        resp.code = Some(ERR_INTERNAL_ERROR.to_string());
   525тЖТ                        resp.message = Some(e);
   526тЖТ                        resp
   527тЖТ                    }
   528тЖТ                }
   529тЖТ            }
   530тЖТ            Some("rs") => {
   531тЖТ                if let Some(ref registry) = self.native_registry {
   532тЖТ                    match registry.generate(&func_name, req.recipe.clone()) {
   533тЖТ                        Ok((value, generator_id, recipe)) => {
   534тЖТ                            resp.status = "generate".to_string();
   535тЖТ                            resp.value = Some(value);
   536тЖТ                            resp.generator_id = Some(generator_id);
   537тЖТ                            resp.recipe = Some(recipe);
   538тЖТ                            resp
   539тЖТ                        }
   540тЖТ                        Err(e) => {
   541тЖТ                            resp.status = "error".to_string();
   542тЖТ                            resp.code = Some(ERR_INTERNAL_ERROR.to_string());
   543тЖТ                            resp.message = Some(e);
   544тЖТ                            resp
   545тЖТ                        }
   546тЖТ                    }
   547тЖТ                } else {
   548тЖТ                    resp.status = "error".to_string();
   549тЖТ                    resp.code = Some(ERR_INTERNAL_ERROR.to_string());
   550тЖТ                    resp.message = Some(format!(
   551тЖТ                        "native generator {func_name:?} requires a custom build (run `shatter build-frontend rust`)"
   552тЖТ                    ));
   553тЖТ                    resp
   554тЖТ                }
   555тЖТ            }
   556тЖТ            _ => {
   557тЖТ                resp.status = "error".to_string();
   558тЖТ                resp.code = Some(ERR_INVALID_REQUEST.to_string());
   559тЖТ                resp.message = Some(format!(
   560тЖТ                    "unsupported generator file type: {}",
   561тЖТ                    file_path
   562тЖТ                ));
   563тЖТ                resp
   564тЖТ            }
   565тЖТ        }
   566тЖТ    }
   567тЖТ
   568тЖТ    fn handle_shutdown(&self, mut resp: Response) -> Response {
   569тЖТ        resp.status = "shutdown_ack".to_string();
   570тЖТ        resp
   571тЖТ    }

   573тЖТ    fn send(&mut self, resp: &Response) -> io::Result<()> {
   574тЖТ        let data = serde_json::to_string(resp)
   575тЖТ            .map_err(io::Error::other)?;
   576тЖТ        self.logf(&format!("Sent: {data}"));
   577тЖТ        writeln!(self.writer, "{data}")?;
   578тЖТ        self.writer.flush()
   579тЖТ    }

   584тЖТ    }

   590тЖТ        }
   591тЖТ    }
