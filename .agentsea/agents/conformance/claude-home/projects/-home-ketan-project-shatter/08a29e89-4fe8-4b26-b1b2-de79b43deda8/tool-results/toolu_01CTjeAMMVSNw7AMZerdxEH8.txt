    29→/// Spawn a real TypeScript frontend subprocess.
    30→async fn spawn_ts_frontend() -> Frontend {
    31→    let frontend_path = ts_frontend_path();
    32→    assert!(
    33→        frontend_path.exists(),
    34→        "TypeScript frontend not built: {} does not exist. Run `cd shatter-ts && npm run build`.",
    35→        frontend_path.display()
    36→    );
    37→
    38→    let mut config = FrontendConfig::new(PathBuf::from("node"));
    39→    config.args = vec!["--no-warnings".to_string(), frontend_path.to_string_lossy().into_owned()];
    40→    config.request_timeout = DEFAULT_REQUEST_TIMEOUT;
    41→
    42→    Frontend::spawn(&config)
    43→        .await
    44→        .expect("failed to spawn TypeScript frontend")
    45→}

    47→/// Analyze a function from a TypeScript file and return the function analysis.
    48→async fn analyze_function(
    49→    frontend: &mut Frontend,
    50→    file: &str,
    51→    function_name: &str,
    52→) -> shatter_core::protocol::FunctionAnalysis {
    53→    let response = frontend
    54→        .send(ProtoCommand::Analyze {
    55→            file: file.to_string(),
    56→            function: Some(function_name.to_string()),
    57→            project_root: None,
    58→        })
    59→        .await
    60→        .expect("analyze command failed");
    61→
    62→    match response.result {
    63→        ResponseResult::Analyze { functions } => {
    64→            functions
    65→                .into_iter()
    66→                .find(|f| f.name == function_name)
    67→                .unwrap_or_else(|| {
    68→                    panic!("function '{function_name}' not found in analysis results")
    69→                })
    70→        }
    71→        other => panic!("expected Analyze response, got: {other:?}"),
    72→    }
    73→}

    75→/// Instrument a function and assert success.
    76→async fn instrument_function(frontend: &mut Frontend, file: &str, function_name: &str) {
    77→    instrument_function_with_mocks(frontend, file, function_name, vec![]).await;
    78→}

    80→/// Instrument a function with mock configurations and assert success.
    81→async fn instrument_function_with_mocks(
    82→    frontend: &mut Frontend,
    83→    file: &str,
    84→    function_name: &str,
    85→    mocks: Vec<shatter_core::protocol::MockConfig>,
    86→) {
    87→    let response = frontend
    88→        .send(ProtoCommand::Instrument {
    89→            file: file.to_string(),
    90→            function: function_name.to_string(),
    91→            mocks,
    92→            project_root: None,
    93→        })
    94→        .await
    95→        .expect("instrument command failed");
    96→
    97→    match response.result {
    98→        ResponseResult::Instrument { instrumented, .. } => {
    99→            assert!(instrumented, "instrumentation returned false");
   100→        }
   101→        ResponseResult::Error { code, message, .. } => {
   102→            panic!("instrument error ({code:?}): {message}");
   103→        }
   104→        other => panic!("expected Instrument response, got: {other:?}"),
   105→    }
   106→}

   108→/// Collect the set of distinct return values from an exploration result.
   109→fn return_value_set(result: &ExploreResult) -> HashSet<String> {
   110→    result
   111→        .executions
   112→        .iter()
   113→        .map(|exec| {
   114→            if let Some(ref err) = exec.thrown_error {
   115→                format!("ERROR:{}", err.message)
   116→            } else {
   117→                match &exec.return_value {
   118→                    Some(v) => v.to_string(),
   119→                    None => "null".to_string(),
   120→                }
   121→            }
   122→        })
   123→        .collect()
   124→}

   126→/// Test that concolic exploration of classifyNumber discovers all 4 branches.
   127→///
   128→/// classifyNumber(n: number) has 4 paths:
   129→///   1. n < 0        -> "negative"
   130→///   2. n === 0      -> "zero"
   131→///   3. n > 0, even  -> "positive-even"
   132→///   4. n > 0, odd   -> "positive-odd"
   133→///
   134→/// The "zero" branch requires exactly n=0, which random number generation
   135→/// is extremely unlikely to hit. Z3 solving should find it by negating the
   136→/// n < 0 constraint to get n >= 0, then negating n === 0 to get n = 0.
   137→#[tokio::test]
   138→async fn concolic_classifynumber_discovers_all_branches() {
   139→    let file = examples_dir().join("01-arithmetic.ts");
   140→    let file_str = file.to_string_lossy().to_string();
   141→
   142→    let mut frontend = spawn_ts_frontend().await;
   143→
   144→    // Step 1: Analyze the function to get its type signature.
   145→    let analysis = analyze_function(&mut frontend, &file_str, "classifyNumber").await;
   146→    assert_eq!(analysis.params.len(), 1, "classifyNumber takes 1 param");
   147→
   148→    // Step 2: Instrument the function for branch tracking.
   149→    instrument_function(&mut frontend, &file_str, "classifyNumber").await;
   150→
   151→    // Step 3: Run concolic exploration with the orchestrator.
   152→    let config = ExploreConfig {
   153→        max_iterations: 20,
   154→        max_executions: 100,
   155→        plateau_threshold: 15,
   156→        ..Default::default()
   157→    };
   158→
   159→    // Seed with a few diverse values to start exploration.
   160→    let seed_inputs = vec![
   161→        vec![serde_json::json!(5)],
   162→        vec![serde_json::json!(-3)],
   163→    ];
   164→
   165→    let result = orchestrator::explore(
   166→        &mut frontend,
   167→        "classifyNumber",
   168→        seed_inputs,
   169→        vec![], // no user-provided inputs
   170→        &analysis.params,
   171→        &config,
   172→        None,
   173→    )
   174→    .await
   175→    .expect("concolic exploration failed");
   176→
   177→    // Step 4: Verify results.
   178→    let return_values = return_value_set(&result);
   179→
   180→    // The concolic pipeline should discover all 4 branches.
   181→    assert!(
   182→        return_values.contains("\"negative\""),
   183→        "should discover 'negative' branch; found: {return_values:?}"
   184→    );
   185→    assert!(
   186→        return_values.contains("\"zero\""),
   187→        "should discover 'zero' branch (requires Z3 to solve n === 0); found: {return_values:?}"
   188→    );
   189→    assert!(
   190→        return_values.contains("\"positive-even\""),
   191→        "should discover 'positive-even' branch; found: {return_values:?}"
   192→    );
   193→    assert!(
   194→        return_values.contains("\"positive-odd\""),
   195→        "should discover 'positive-odd' branch; found: {return_values:?}"
   196→    );
   197→
   198→    // The orchestrator should have used Z3 to generate at least some inputs.
   199→    assert!(
   200→        result.z3_generated > 0,
   201→        "Z3 should have generated at least one input; got z3_generated={}",
   202→        result.z3_generated
   203→    );
   204→
   205→    assert!(
   206→        result.unique_paths >= 4,
   207→        "should have at least 4 unique paths; got {}",
   208→        result.unique_paths
   209→    );
   210→
   211→    frontend.shutdown().await.expect("frontend shutdown failed");
   212→}
