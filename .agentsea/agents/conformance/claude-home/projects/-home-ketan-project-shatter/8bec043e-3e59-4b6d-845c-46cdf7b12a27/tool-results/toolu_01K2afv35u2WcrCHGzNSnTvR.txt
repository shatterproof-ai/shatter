     1→//! Behavior maps, call graphs, and compositional testing logic.
     2→//!
     3→//! When the concolic engine tests function A that calls function B, behavior maps
     4→//! let us reuse prior knowledge about B. A [`BehaviorMap`] records B's observed
     5→//! input→output mappings so that when testing A, B is mocked using its known
     6→//! behaviors. [`CallGraph`] orders functions for testing (leaves first), and
     7→//! [`BehaviorCoverage`] tracks which of B's behaviors A actually exercises.
     8→

    14→    BranchDecision, ErrorInfo, ExternalCall, ExecutionRecord, SideEffect,
    15→};
    16→use crate::protocol::{ExecuteResult, MockBehavior, MockConfig};

    22→/// A single call to an external dependency, captured with ordering information.
    23→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    24→pub struct TracedCall {
    25→    /// Name of the called function/symbol.
    26→    pub function_name: String,
    27→    /// Arguments passed to the call.
    28→    pub arguments: Vec<serde_json::Value>,
    29→    /// Return value from the call.
    30→    pub return_value: serde_json::Value,
    31→    /// Zero-based index indicating the order of this call relative to all
    32→    /// dependency interactions (calls and side effects combined).
    33→    pub call_index: u32,
    34→}
    35→
    36→/// Categorization of a side effect.
    37→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    38→#[serde(rename_all = "snake_case")]
    39→pub enum SideEffectKind {
    40→    ConsoleOutput,
    41→    FileWrite,
    42→    NetworkRequest,
    43→    EnvironmentRead,
    44→    GlobalMutation,
    45→    ThrownError,
    46→    GlobalStateChange,
    47→}
    48→
    49→/// A side effect observed during execution, captured with ordering information.
    50→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    51→pub struct TracedSideEffect {
    52→    /// What category of side effect this is.
    53→    pub kind: SideEffectKind,
    54→    /// Human-readable description of the side effect.
    55→    pub description: String,
    56→    /// Zero-based index indicating the order of this side effect relative to
    57→    /// all dependency interactions (calls and side effects combined).
    58→    pub call_index: u32,
    59→}
    60→
    61→/// Full dependency interaction trace for a single execution.
    62→///
    63→/// Captures every external call and side effect in order, enabling
    64→/// compositional reasoning about how a function interacts with its
    65→/// dependencies.
    66→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    67→pub struct DependencyTrace {
    68→    /// External calls made during execution, in order.
    69→    pub external_calls: Vec<TracedCall>,
    70→    /// Side effects observed during execution, in order.
    71→    pub side_effects: Vec<TracedSideEffect>,
    72→    /// Total number of dependency interactions (calls + side effects),
    73→    /// useful for verifying ordering completeness.
    74→    pub call_ordering: u32,
    75→}
    76→
    77→/// Classify a [`SideEffect`] into a [`SideEffectKind`].
    78→fn side_effect_kind(effect: &SideEffect) -> SideEffectKind {
    79→    match effect {
    80→        SideEffect::ConsoleOutput { .. } => SideEffectKind::ConsoleOutput,
    81→        SideEffect::FileWrite { .. } => SideEffectKind::FileWrite,
    82→        SideEffect::NetworkRequest { .. } => SideEffectKind::NetworkRequest,
    83→        SideEffect::EnvironmentRead { .. } => SideEffectKind::EnvironmentRead,
    84→        SideEffect::GlobalMutation { .. } => SideEffectKind::GlobalMutation,
    85→        SideEffect::ThrownError { .. } => SideEffectKind::ThrownError,
    86→        SideEffect::GlobalStateChange { .. } => SideEffectKind::GlobalStateChange,
    87→    }
    88→}
    89→
    90→/// Produce a human-readable description of a [`SideEffect`].
    91→fn side_effect_description(effect: &SideEffect) -> String {
    92→    match effect {
    93→        SideEffect::ConsoleOutput { level, message } => {
    94→            format!("console.{level}: {message}")
    95→        }
    96→        SideEffect::FileWrite { path, .. } => {
    97→            format!("file write: {path}")
    98→        }
    99→        SideEffect::NetworkRequest { method, url, .. } => {
   100→            format!("{method} {url}")
   101→        }
   102→        SideEffect::EnvironmentRead { variable, value } => {
   103→            let val = value.as_deref().unwrap_or("null");
   104→            format!("env read: {variable}={val}")
   105→        }
   106→        SideEffect::GlobalMutation { name } => {
   107→            format!("mutated global: {name}")
   108→        }
   109→        SideEffect::ThrownError {
   110→            error_type,
   111→            message,
   112→            ..
   113→        } => {
   114→            format!("{error_type}: {message}")
   115→        }
   116→        SideEffect::GlobalStateChange {
   117→            variable,
   118→            before,
   119→            after,
   120→        } => {
   121→            format!("{variable}: {before} -> {after}")
   122→        }
   123→    }
   124→}
   125→
   126→/// Build [`TracedCall`]s from a slice of [`ExternalCall`]s, starting at the
   127→/// given call index offset.
   128→fn build_traced_calls(calls: &[ExternalCall], start_index: u32) -> Vec<TracedCall> {
   129→    calls
   130→        .iter()
   131→        .enumerate()
   132→        .map(|(i, call)| TracedCall {
   133→            function_name: call.symbol.clone(),
   134→            arguments: call.args.clone(),
   135→            return_value: call.return_value.clone(),
   136→            call_index: start_index + i as u32,
   137→        })
   138→        .collect()
   139→}
   140→
   141→/// Build [`TracedSideEffect`]s from a slice of [`SideEffect`]s, starting at
   142→/// the given call index offset.
   143→fn build_traced_side_effects(
   144→    effects: &[SideEffect],
   145→    start_index: u32,
   146→) -> Vec<TracedSideEffect> {
   147→    effects
   148→        .iter()
   149→        .enumerate()
   150→        .map(|(i, effect)| TracedSideEffect {
   151→            kind: side_effect_kind(effect),
   152→            description: side_effect_description(effect),
   153→            call_index: start_index + i as u32,
   154→        })
   155→        .collect()
   156→}
   157→
   158→/// Build a [`DependencyTrace`] from an [`ExecuteResult`].
   159→///
   160→/// External calls are indexed first, then side effects.
   161→pub fn build_dependency_trace(result: &ExecuteResult) -> DependencyTrace {
   162→    let external_calls = build_traced_calls(&result.calls_to_external, 0);
   163→    let se_start = external_calls.len() as u32;
   164→    let side_effects = build_traced_side_effects(&result.side_effects, se_start);
   165→    let call_ordering = se_start + side_effects.len() as u32;
   166→
   167→    DependencyTrace {
   168→        external_calls,
   169→        side_effects,
   170→        call_ordering,
   171→    }
   172→}
   173→
   174→/// Build a [`DependencyTrace`] from an [`ExecutionRecord`].
   175→///
   176→/// External calls are indexed first, then side effects.
   177→pub fn build_dependency_trace_from_record(record: &ExecutionRecord) -> DependencyTrace {
   178→    let external_calls = build_traced_calls(&record.calls_to_external, 0);
   179→    let se_start = external_calls.len() as u32;
   180→    let side_effects = build_traced_side_effects(&record.side_effects, se_start);
   181→    let call_ordering = se_start + side_effects.len() as u32;
   182→
   183→    DependencyTrace {
   184→        external_calls,
   185→        side_effects,
   186→        call_ordering,
   187→    }
   188→}

   194→/// A single observed behavior: specific inputs produced specific outputs.
   195→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   196→pub struct Behavior {
   197→    pub id: u32,
   198→    pub input_args: Vec<serde_json::Value>,
   199→    pub return_value: Option<serde_json::Value>,
   200→    pub thrown_error: Option<ErrorInfo>,
   201→    pub branch_path: Vec<BranchDecision>,
   202→    pub side_effects: Vec<SideEffect>,
   203→    /// Full dependency interaction trace, present when the execution involved
   204→    /// external calls or side effects.
   205→    #[serde(default, skip_serializing_if = "Option::is_none")]
   206→    pub dependency_trace: Option<DependencyTrace>,
   207→}

   209→/// All observed behaviors for a function, built from execution records.
   210→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   211→pub struct BehaviorMap {
   212→    pub function_id: String,
   213→    pub behaviors: Vec<Behavior>,
   214→    /// SHA-256 fingerprint of the function's source, params, and branches.
   215→    /// Used for staleness detection: if the fingerprint matches a cached value,
   216→    /// the function is unchanged and can be skipped during re-exploration.
   217→    #[serde(default, skip_serializing_if = "Option::is_none")]
   218→    pub fingerprint: Option<String>,
   219→    /// Fields identified as nondeterministic during exploration.
   220→    /// Populated from the nondeterminism detection report when available.
   221→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   222→    pub nondeterministic_fields: Vec<crate::nondeterminism::NondeterministicField>,
   223→}
   224→
   225→impl BehaviorMap {
   226→    /// Build a behavior map from execution records, deduplicating by `input_hash`.
   227→    pub fn from_records(function_id: impl Into<String>, records: &[ExecutionRecord]) -> Self {
   228→        let mut seen_hashes = HashSet::new();
   229→        let mut behaviors = Vec::new();
   230→        let mut next_id: u32 = 0;
   231→
   232→        for record in records {
   233→            if !seen_hashes.insert(record.input_hash) {
   234→                continue;
   235→            }
   236→            let dependency_trace =
   237→                if record.calls_to_external.is_empty() && record.side_effects.is_empty() {
   238→                    None
   239→                } else {
   240→                    Some(build_dependency_trace_from_record(record))
   241→                };
   242→            behaviors.push(Behavior {
   243→                id: next_id,
   244→                input_args: record.parameters.clone(),
   245→                return_value: record.return_value.clone(),
   246→                thrown_error: record.thrown_error.clone(),
   247→                branch_path: record.branch_path.clone(),
   248→                side_effects: record.side_effects.clone(),
   249→                dependency_trace,
   250→            });
   251→            next_id += 1;
   252→        }
   253→
   254→        Self {
   255→            function_id: function_id.into(),
   256→            behaviors,
   257→            fingerprint: None,
   258→            nondeterministic_fields: vec![],
   259→        }
   260→    }
   261→
   262→    /// Build a behavior map from an [`ObservationOutput`](crate::explorer::ObservationOutput).
   263→    ///
   264→    /// Converts each [`ExecutionSummary`](crate::explorer::ExecutionSummary) that discovered
   265→    /// a new path into a [`Behavior`] entry. When raw results are available,
   266→    /// dependency traces are populated from the matching [`ExecuteResult`].
   267→    pub fn from_exploration_result(
   268→        function_id: impl Into<String>,
   269→        result: &crate::explorer::ObservationOutput,
   270→    ) -> Self {
   271→        let behaviors = result
   272→            .new_path_executions
   273→            .iter()
   274→            .enumerate()
   275→            .map(|(i, exec)| {
   276→                let thrown_error = exec.thrown_error.as_ref().map(|msg| ErrorInfo {
   277→                    error_type: "Error".to_string(),
   278→                    message: msg.clone(),
   279→                    stack: None, error_category: None });
   280→                // Try to find the matching raw result for this execution to
   281→                // extract the dependency trace.
   282→                let dependency_trace = result
   283→                    .raw_results
   284→                    .iter()
   285→                    .find(|(inputs, _)| *inputs == exec.inputs)
   286→                    .filter(|(_, res)| {
   287→                        !res.calls_to_external.is_empty() || !res.side_effects.is_empty()
   288→                    })
   289→                    .map(|(_, res)| build_dependency_trace(res));
   290→                Behavior {
   291→                    id: i as u32,
   292→                    input_args: exec.inputs.clone(),
   293→                    return_value: exec.return_value.clone(),
   294→                    thrown_error,
   295→                    branch_path: vec![],
   296→                    side_effects: vec![],
   297→                    dependency_trace,
   298→                }
   299→            })
   300→            .collect();
   301→
   302→        Self {
   303→            function_id: function_id.into(),
   304→            behaviors,
   305→            fingerprint: None,
   306→            nondeterministic_fields: result.nondeterministic_fields.clone(),
   307→        }
   308→    }
   309→
   310→    /// Convert this behavior map into a [`MockConfig`] for use when testing callers.
   311→    ///
   312→    /// Each behavior's return value becomes an entry in `return_values`. Behaviors
   313→    /// that threw an error (and have no return value) are skipped.
   314→    pub fn to_mock_config(&self) -> MockConfig {
   315→        let return_values: Vec<serde_json::Value> = self
   316→            .behaviors
   317→            .iter()
   318→            .filter_map(|b| b.return_value.clone())
   319→            .collect();
   320→
   321→        MockConfig {
   322→            symbol: self.function_id.clone(),
   323→            return_values,
   324→            should_track_calls: true,
   325→            default_behavior: MockBehavior::RepeatLast,
   326→        }
   327→    }
   328→}

   334→/// Error type for call graph operations.
   335→#[derive(Debug, Clone, PartialEq, thiserror::Error)]
   336→pub enum CallGraphError {
   337→    /// A cycle was detected during topological sort.
   338→    #[error("cycle detected involving: {0}")]
   339→    Cycle(String),
   340→}
   341→
   342→/// An entry in the test order, representing either a single function or a
   343→/// mutually-recursive group that must be tested together.
   344→#[derive(Debug, Clone, PartialEq)]
   345→pub enum TestOrderEntry {
   346→    /// A single function (possibly self-recursive).
   347→    Single {
   348→        function_id: String,
   349→        is_self_recursive: bool,
   350→    },
   351→    /// A group of mutually-recursive functions that must be tested together.
   352→    MutualGroup {
   353→        /// Function IDs in the group, sorted for determinism.
   354→        function_ids: Vec<String>,
   355→    },
   356→}
   357→
   358→/// Dependency graph built from [`FunctionAnalysis`] results.
   359→///
   360→/// Used to compute test ordering so leaf functions are tested first.
   361→/// Self-recursive functions (functions that call themselves) are detected
   362→/// and annotated rather than rejected as cycles.
   363→pub struct CallGraph {
   364→    /// function_id → set of function_ids it calls (excluding self-calls).
   365→    edges: HashMap<String, HashSet<String>>,
   366→    /// Functions that call themselves.
   367→    self_recursive: HashSet<String>,
   368→}
   369→
   370→/// Internal state for Tarjan's SCC algorithm.
   371→#[derive(Default)]
   372→struct TarjanState<'a> {
   373→    index_counter: u32,
   374→    stack: Vec<&'a str>,
   375→    on_stack: HashSet<&'a str>,
   376→    indices: HashMap<&'a str, u32>,
   377→    lowlinks: HashMap<&'a str, u32>,
   378→    result: Vec<Vec<String>>,
   379→}
   380→
   381→impl CallGraph {
   382→    /// Build a call graph from function analyses.
   383→    ///
   384→    /// Matches each function's `ExternalDependency.symbol` against the set of
   385→    /// known function names to build edges. Self-calls are detected and stored
   386→    /// separately rather than as regular edges.
   387→    pub fn from_analyses(analyses: &[crate::protocol::FunctionAnalysis]) -> Self {
   388→        let known_names: HashSet<&str> = analyses.iter().map(|a| a.name.as_str()).collect();
   389→        let mut edges = HashMap::new();
   390→        let mut self_recursive = HashSet::new();
   391→
   392→        for analysis in analyses {
   393→            let mut callees: HashSet<String> = HashSet::new();
   394→            for dep in &analysis.dependencies {
   395→                if !known_names.contains(dep.symbol.as_str()) {
   396→                    continue;
   397→                }
   398→                if dep.symbol == analysis.name {
   399→                    self_recursive.insert(analysis.name.clone());
   400→                } else {
   401→                    callees.insert(dep.symbol.clone());
   402→                }
   403→            }
   404→            edges.insert(analysis.name.clone(), callees);
   405→        }
   406→
   407→        Self {
   408→            edges,
   409→            self_recursive,
   410→        }
   411→    }
   412→
   413→    /// Compute strongly connected components using Tarjan's algorithm.
   414→    ///
   415→    /// Returns SCCs in reverse topological order (leaves first).
   416→    fn strongly_connected_components(&self) -> Vec<Vec<String>> {
   417→        // Collect all nodes (sorted for determinism)
   418→        let mut all_nodes: Vec<&str> = self.edges.keys().map(|s| s.as_str()).collect();
   419→        for callees in self.edges.values() {
   420→            for callee in callees {
   421→                all_nodes.push(callee.as_str());
   422→            }
   423→        }
   424→        all_nodes.sort();
   425→        all_nodes.dedup();
   426→
   427→        let mut state = TarjanState::default();
   428→
   429→        for &node in &all_nodes {
   430→            if !state.indices.contains_key(node) {
   431→                self.tarjan_visit(node, &mut state);
   432→            }
   433→        }
   434→
   435→        state.result
   436→    }
   437→
   438→    fn tarjan_visit<'a>(&'a self, node: &'a str, state: &mut TarjanState<'a>) {
   439→        state.indices.insert(node, state.index_counter);
   440→        state.lowlinks.insert(node, state.index_counter);
   441→        state.index_counter += 1;
   442→        state.stack.push(node);
   443→        state.on_stack.insert(node);
   444→
   445→        // Visit successors (sorted for determinism)
   446→        if let Some(callees) = self.edges.get(node) {
   447→            let mut sorted_callees: Vec<&str> = callees.iter().map(|s| s.as_str()).collect();
   448→            sorted_callees.sort();
   449→            for callee in sorted_callees {
   450→                if !state.indices.contains_key(callee) {
   451→                    self.tarjan_visit(callee, state);
   452→                    let callee_low = state.lowlinks[callee];
   453→                    let node_low = state.lowlinks.get_mut(node).expect("node in lowlinks");
   454→                    *node_low = (*node_low).min(callee_low);
   455→                } else if state.on_stack.contains(callee) {
   456→                    let callee_idx = state.indices[callee];
   457→                    let node_low = state.lowlinks.get_mut(node).expect("node in lowlinks");
   458→                    *node_low = (*node_low).min(callee_idx);
   459→                }
   460→            }
   461→        }
   462→
   463→        // If node is a root of an SCC
   464→        if state.lowlinks[node] == state.indices[node] {
   465→            let mut component = Vec::new();
   466→            loop {
   467→                let w = state.stack.pop().expect("stack not empty");
   468→                state.on_stack.remove(w);
   469→                component.push(w.to_string());
   470→                if w == node {
   471→                    break;
   472→                }
   473→            }
   474→            component.sort(); // Deterministic ordering within SCC
   475→            state.result.push(component);
   476→        }
   477→    }
   478→
   479→    /// Topological sort returning leaf functions first.
   480→    ///
   481→    /// Self-recursive functions are returned as `Single` entries with
   482→    /// `is_self_recursive: true`. Mutually-recursive function groups are
   483→    /// returned as `MutualGroup` entries. Non-recursive functions are
   484→    /// returned as `Single` entries.
   485→    pub fn test_order(&self) -> Result<Vec<TestOrderEntry>, CallGraphError> {
   486→        // Tarjan's algorithm returns SCCs in reverse topological order,
   487→        // which is exactly what we want: leaves first.
   488→        let sccs = self.strongly_connected_components();
   489→
   490→        let mut result = Vec::new();
   491→        for scc in sccs {
   492→            if scc.len() == 1 {
   493→                let id = &scc[0];
   494→                result.push(TestOrderEntry::Single {
   495→                    function_id: id.clone(),
   496→                    is_self_recursive: self.self_recursive.contains(id),
   497→                });
   498→            } else {
   499→                result.push(TestOrderEntry::MutualGroup {
   500→                    function_ids: scc,
   501→                });
   502→            }
   503→        }
   504→
   505→        Ok(result)
   506→    }
   507→
   508→    /// Direct callees of a function (excluding self-calls).
   509→    pub fn callees(&self, function_id: &str) -> HashSet<String> {
   510→        self.edges
   511→            .get(function_id)
   512→            .cloned()
   513→            .unwrap_or_default()
   514→    }
   515→
   516→    /// Whether a function calls itself.
   517→    pub fn is_self_recursive(&self, function_id: &str) -> bool {
   518→        self.self_recursive.contains(function_id)
   519→    }
   520→
   521→    /// All functions detected as self-recursive.
   522→    pub fn self_recursive_functions(&self) -> &HashSet<String> {
   523→        &self.self_recursive
   524→    }
   525→
   526→    /// All nodes in the graph.
   527→    pub fn nodes(&self) -> HashSet<&str> {
   528→        let mut nodes: HashSet<&str> = self.edges.keys().map(|s| s.as_str()).collect();
   529→        for callees in self.edges.values() {
   530→            for callee in callees {
   531→                nodes.insert(callee.as_str());
   532→            }
   533→        }
   534→        nodes
   535→    }
   536→}

   542→/// Tracks which of a callee's behaviors were exercised when testing a caller.
   543→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   544→pub struct BehaviorCoverage {
   545→    pub caller: String,
   546→    pub callee: String,
   547→    /// Indices into the callee's `BehaviorMap.behaviors` that were matched.
   548→    pub exercised_behavior_ids: Vec<u32>,
   549→    pub total_behaviors: u32,
   550→}
   551→
   552→impl BehaviorCoverage {
   553→    /// Determine which of the callee's behaviors were exercised by the caller.
   554→    ///
   555→    /// Matches [`ExternalCall`] return values from the caller's records against
   556→    /// the callee's behavior return values.
   557→    pub fn compute(
   558→        caller_id: &str,
   559→        caller_records: &[ExecutionRecord],
   560→        callee_map: &BehaviorMap,
   561→    ) -> Self {
   562→        let mut exercised: HashSet<u32> = HashSet::new();
   563→
   564→        // Collect all external call return values targeting the callee.
   565→        let call_returns: Vec<&serde_json::Value> = caller_records
   566→            .iter()
   567→            .flat_map(|r| &r.calls_to_external)
   568→            .filter(|call| call.symbol == callee_map.function_id)
   569→            .map(|call| &call.return_value)
   570→            .collect();
   571→
   572→        for behavior in &callee_map.behaviors {
   573→            if let Some(ref ret) = behavior.return_value
   574→                && call_returns.contains(&ret)
   575→            {
   576→                exercised.insert(behavior.id);
   577→            }
   578→        }
   579→
   580→        let mut exercised_ids: Vec<u32> = exercised.into_iter().collect();
   581→        exercised_ids.sort();
   582→
   583→        Self {
   584→            caller: caller_id.to_string(),
   585→            callee: callee_map.function_id.clone(),
   586→            exercised_behavior_ids: exercised_ids,
