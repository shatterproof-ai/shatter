     1→     1→//! Protocol types for communication between the Rust core and language frontends.
     2→     2→//!
     3→     3→//! All messages are newline-delimited JSON over stdin/stdout. The core sends
     4→     4→//! [`Request`] messages to frontends and receives [`Response`] messages back.
     5→     5→//! Every message includes a protocol version for compatibility checking.

    26→    26→/// Granularity level for setup/teardown lifecycle management.
    27→    27→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
    28→    28→#[serde(rename_all = "snake_case")]
    29→    29→pub enum SetupLevel {
    30→    30→    Session,
    31→    31→    File,
    32→    32→    Function,
    33→    33→    Execution,
    34→    34→}
    35→    35→
    36→    36→/// A single entry in a setup context stack, associating a lifecycle level
    37→    37→/// with the opaque context value returned by its Setup command.
    38→    38→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    39→    39→pub struct SetupContextEntry {
    40→    40→    pub level: SetupLevel,
    41→    41→    pub context: serde_json::Value,
    42→    42→}
    43→    43→
    44→    44→/// Stack of active setup contexts, ordered from outermost (session) to
    45→    45→/// innermost (execution). Passed to Execute so frontends can restore
    46→    46→/// all active setup state.
    47→    47→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    48→    48→pub struct SetupContextStack {
    49→    49→    pub contexts: Vec<SetupContextEntry>,
    50→    50→}
    51→    51→
    52→    52→impl From<crate::config::SetupMode> for SetupLevel {
    53→    53→    fn from(mode: crate::config::SetupMode) -> Self {
    54→    54→        match mode {
    55→    55→            crate::config::SetupMode::PerFunction => SetupLevel::Function,
    56→    56→            crate::config::SetupMode::PerExecution => SetupLevel::Execution,
    57→    57→        }
    58→    58→    }
    59→    59→}

    65→    65→/// A request message sent from the core engine to a language frontend.
    66→    66→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    67→    67→pub struct Request {
    68→    68→    /// Protocol version for compatibility checking.
    69→    69→    pub protocol_version: String,
    70→    70→    /// Unique identifier for correlating responses to requests.
    71→    71→    pub id: u64,
    72→    72→    /// The command to execute.
    73→    73→    #[serde(flatten)]
    74→    74→    pub command: Command,
    75→    75→}

    77→    77→/// Commands the core can send to a frontend.
    78→    78→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    79→    79→#[serde(tag = "command", rename_all = "snake_case")]
    80→    80→pub enum Command {
    81→    81→    /// Initial handshake to negotiate protocol version and capabilities.
    82→    82→    Handshake {
    83→    83→        /// Capabilities supported by the core.
    84→    84→        capabilities: Vec<String>,
    85→    85→    },
    86→    86→    /// Analyze a function to extract type information, branches, and dependencies.
    87→    87→    Analyze {
    88→    88→        /// Path to the source file.
    89→    89→        file: String,
    90→    90→        /// Name of the function to analyze. If absent, analyze all exported functions.
    91→    91→        function: Option<String>,
    92→    92→        /// Detected project root directory, if any.
    93→    93→        #[serde(default, skip_serializing_if = "Option::is_none")]
    94→    94→        project_root: Option<String>,
    95→    95→    },
    96→    96→    /// Instrument a function for symbolic constraint tracking.
    97→    97→    Instrument {
    98→    98→        /// Path to the source file.
    99→    99→        file: String,
   100→   100→        /// Name of the function to instrument.
   101→   101→        function: String,
   102→   102→        /// Mock configurations for external dependencies.
   103→   103→        mocks: Vec<MockConfig>,
   104→   104→        /// Detected project root directory, if any.
   105→   105→        #[serde(default, skip_serializing_if = "Option::is_none")]
   106→   106→        project_root: Option<String>,
   107→   107→    },
   108→   108→    /// Execute an instrumented function with specific inputs and mocks.
   109→   109→    Execute {
   110→   110→        /// Fully qualified function identifier.
   111→   111→        function: String,
   112→   112→        /// Input values for the function parameters (JSON-encoded).
   113→   113→        inputs: Vec<serde_json::Value>,
   114→   114→        /// Mock configurations for external dependencies.
   115→   115→        mocks: Vec<MockConfig>,
   116→   116→        /// Stack of active setup contexts from enclosing Setup commands, if any.
   117→   117→        #[serde(default, skip_serializing_if = "Option::is_none")]
   118→   118→        setup_context: Option<SetupContextStack>,
   119→   119→    },
   120→   120→    /// Run a setup file to initialize state before function execution.
   121→   121→    Setup {
   122→   122→        /// Path to the setup file.
   123→   123→        file: String,
   124→   124→        /// Scope identifier (function name, file path, or session label).
   125→   125→        scope: String,
   126→   126→        /// Lifecycle level for this setup.
   127→   127→        level: SetupLevel,
   128→   128→        /// Detected project root directory, if any.
   129→   129→        #[serde(default, skip_serializing_if = "Option::is_none")]
   130→   130→        project_root: Option<String>,
   131→   131→        /// Parent context stack from enclosing setup levels, if any.
   132→   132→        #[serde(default, skip_serializing_if = "Option::is_none")]
   133→   133→        parent_context: Option<SetupContextStack>,
   134→   134→    },
   135→   135→    /// Tear down state established by a prior Setup command.
   136→   136→    Teardown {
   137→   137→        /// Scope identifier matching the corresponding Setup command.
   138→   138→        scope: String,
   139→   139→        /// Lifecycle level matching the corresponding Setup command.
   140→   140→        level: SetupLevel,
   141→   141→    },
   142→   142→    /// Invoke a custom generator to produce a value for a type or parameter.
   143→   143→    Generate {
   144→   144→        /// Path to the generator file.
   145→   145→        file: String,
   146→   146→        /// Name of the type or parameter to generate a value for.
   147→   147→        name: String,
   148→   148→        /// Whether this generator targets a type name or a parameter name.
   149→   149→        kind: GeneratorKind,
   150→   150→        /// Reconstruction recipe from a previous generation.
   151→   151→        #[serde(default, skip_serializing_if = "Option::is_none")]
   152→   152→        recipe: Option<serde_json::Value>,
   153→   153→        /// Detected project root directory, if any.
   154→   154→        #[serde(default, skip_serializing_if = "Option::is_none")]
   155→   155→        project_root: Option<String>,
   156→   156→    },
   157→   157→    /// Request graceful shutdown of the frontend process.
   158→   158→    Shutdown,
   159→   159→}
   160→   160→
   161→   161→/// Whether a generator targets a type name or a parameter name.
   162→   162→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   163→   163→#[serde(rename_all = "snake_case")]
   164→   164→pub enum GeneratorKind {
   165→   165→    /// Generator for a named type (e.g., "User").
   166→   166→    TypeName,
   167→   167→    /// Generator for a named parameter (e.g., "authToken").
   168→   168→    ParamName,
   169→   169→}

   171→   171→/// Configuration for mocking an external dependency during execution.
   172→   172→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   173→   173→pub struct MockConfig {
   174→   174→    /// Fully qualified symbol name of the dependency to mock.
   175→   175→    pub symbol: String,
   176→   176→    /// Pre-configured return values to cycle through.
   177→   177→    pub return_values: Vec<serde_json::Value>,
   178→   178→    /// Whether to record arguments passed to the mock.
   179→   179→    pub should_track_calls: bool,
   180→   180→    /// Default behavior when return_values are exhausted.
   181→   181→    pub default_behavior: MockBehavior,
   182→   182→}
   183→   183→
   184→   184→/// How a mock should behave when its pre-configured return values are exhausted.
   185→   185→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   186→   186→#[serde(rename_all = "snake_case")]
   187→   187→pub enum MockBehavior {
   188→   188→    /// Return a generated value based on the return type.
   189→   189→    ReturnGenerated,
   190→   190→    /// Repeat the last return value.
   191→   191→    RepeatLast,
   192→   192→    /// Throw an error.
   193→   193→    ThrowError,
   194→   194→    /// Call through to the real implementation.
   195→   195→    Passthrough,
   196→   196→}

   202→   202→/// A response message sent from a language frontend to the core engine.

   214→   214→/// Response payloads from a frontend.
   215→   215→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   216→   216→#[serde(tag = "status", rename_all = "snake_case")]
   217→   217→pub enum ResponseResult {
   218→   218→    /// Successful handshake response.
   219→   219→    Handshake {
   220→   220→        /// Protocol version the frontend supports.
   221→   221→        frontend_version: String,
   222→   222→        /// Language this frontend handles (e.g., "typescript", "go").
   223→   223→        language: String,
   224→   224→        /// Capabilities supported by the frontend.
   225→   225→        capabilities: Vec<String>,
   226→   226→    },
   227→   227→    /// Successful analysis result.
   228→   228→    Analyze {
   229→   229→        /// Functions found and analyzed.
   230→   230→        #[serde(default)]
   231→   231→        functions: Vec<FunctionAnalysis>,
   232→   232→    },
   233→   233→    /// Successful instrumentation result.
   234→   234→    Instrument {
   235→   235→        /// Whether instrumentation succeeded.
   236→   236→        instrumented: bool,
   237→   237→        /// Path to the instrumented output file, if applicable.
   238→   238→        output_file: Option<String>,
   239→   239→        /// Number of executable statement lines the instrumentor inserted record calls for.
   240→   240→        /// Used as the denominator for line coverage instead of raw source span.
   241→   241→        #[serde(default, skip_serializing_if = "Option::is_none")]
   242→   242→        instrumentable_line_count: Option<u32>,
   243→   243→    },
   244→   244→    /// Successful execution result (boxed to reduce enum size).
   245→   245→    Execute(Box<ExecuteResult>),
   246→   246→    /// Successful setup result.
   247→   247→    Setup {
   248→   248→        /// Opaque context to pass to subsequent Execute commands.
   249→   249→        setup_context: serde_json::Value,
   250→   250→    },
   251→   251→    /// Acknowledgment that teardown completed.
   252→   252→    TeardownAck,
   253→   253→    /// Result of invoking a custom generator.
   254→   254→    Generate {
   255→   255→        /// The generated value.
   256→   256→        value: serde_json::Value,
   257→   257→        /// Human-readable label for this generated value.
   258→   258→        generator_id: String,
   259→   259→        /// Serializable recipe for replaying this generation.
   260→   260→        #[serde(default, skip_serializing_if = "Option::is_none")]
   261→   261→        recipe: Option<serde_json::Value>,
   262→   262→    },
   263→   263→    /// Acknowledgment of shutdown request.
   264→   264→    ShutdownAck,
   265→   265→    /// Error response for any command.
   266→   266→    Error {
   267→   267→        /// Machine-readable error code.
   268→   268→        code: ErrorCode,
   269→   269→        /// Human-readable error message.
   270→   270→        message: String,
   271→   271→        /// Additional error details (e.g., stack trace, source location).
   272→   272→        details: Option<serde_json::Value>,
   273→   273→    },
   274→   274→}

   498→   498→/// Validate an `ExecuteResult` deserialized from a frontend subprocess.
   499→   499→///
   500→   500→/// Checks semantic invariants that serde can't enforce:
   501→   501→/// - `branch_path` entries have non-default constraints (when non-empty, an all-unknown
   502→   502→///   branch_path produces identical hashes for different paths → silent hash collision)
   503→   503→/// - `path_constraints` length matches `branch_path` length when both are non-empty
   504→   504→#[contracts::ensures(ret == execute_result_is_valid(result),
   505→   505→    "postcondition must match validation logic")]
   506→   506→pub fn validate_execute_result(result: &ExecuteResult) -> bool {
   507→   507→    execute_result_is_valid(result)
   508→   508→}
   509→   509→
   510→   510→fn execute_result_is_valid(result: &ExecuteResult) -> bool {
   511→   511→    // Branch path with all-unknown constraints causes silent hash collisions
   512→   512→    // because the path hash depends on constraint content.
   513→   513→    if !result.branch_path.is_empty() {
   514→   514→        let all_unknown = result.branch_path.iter().all(|bd| {
   515→   515→            matches!(bd.constraint, SymConstraint::Unknown { .. })
   516→   516→        });
   517→   517→        // All-unknown is valid for frontends without symbolic analysis (Go),
   518→   518→        // but path_constraints should then also be empty.
   519→   519→        if all_unknown && !result.path_constraints.is_empty() {
   520→   520→            return false;
   521→   521→        }
   522→   522→    }
   523→   523→    true
   524→   524→}
   525→   525→
   526→   526→/// Validate a `FunctionAnalysis` list deserialized from a frontend subprocess.
   527→   527→///
   528→   528→/// Checks semantic invariants that serde can't enforce:
   529→   529→/// - Function names are non-empty (empty name → silent lookup failures)
   530→   530→/// - `start_line <= end_line` (inverted range → wrong source slicing)
   531→   531→/// - Param count is plausible (> 255 params suggests a deserialization bug)
   532→   532→#[contracts::ensures(ret == analyze_result_is_valid(functions),
   533→   533→    "postcondition must match validation logic")]
   534→   534→pub fn validate_analyze_result(functions: &[FunctionAnalysis]) -> bool {
   535→   535→    analyze_result_is_valid(functions)
   536→   536→}
   537→   537→
   538→   538→/// Upper bound on parameter count — anything above this likely indicates
   539→   539→/// a deserialization bug rather than a real function signature.
   540→   540→const MAX_PLAUSIBLE_PARAMS: usize = 255;
   541→   541→
   542→   542→fn analyze_result_is_valid(functions: &[FunctionAnalysis]) -> bool {
   543→   543→    for func in functions {
   544→   544→        if func.name.is_empty() {
   545→   545→            return false;
   546→   546→        }
   547→   547→        if func.start_line > func.end_line {
   548→   548→            return false;
   549→   549→        }
   550→   550→        if func.params.len() > MAX_PLAUSIBLE_PARAMS {
   551→   551→            return false;
   552→   552→        }
   553→   553→    }
   554→   554→    true
   555→   555→}
