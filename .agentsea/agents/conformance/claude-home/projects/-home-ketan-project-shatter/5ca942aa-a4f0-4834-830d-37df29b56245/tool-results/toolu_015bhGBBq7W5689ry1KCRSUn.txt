     1→     1→//! Shatter configuration: `.shatter/config.yaml` parsing, hierarchical discovery, and merging.
     2→     2→//!
     3→     3→//! Users can place `.shatter/` directories at any level of their project tree.
     4→     4→//! Each directory can contain a `config.yaml` with per-function settings and
     5→     5→//! an `inputs/` subdirectory with candidate input files.
     6→     6→//!
     7→     7→//! Config resolution walks upward from each target file to the filesystem root,
     8→     8→//! collecting all `.shatter/config.yaml` files. The nearest config wins on
     9→     9→//! conflicts (closest to the target file takes precedence).
    10→    10→
    11→    11→use std::collections::HashMap;
    12→    12→use std::path::{Path, PathBuf};
    13→    13→
    14→    14→use globset::{Glob, GlobMatcher};
    15→    15→use serde::{Deserialize, Serialize};
    16→    16→
    17→    17→/// When to run the setup file relative to function executions.
    18→    18→#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize, Serialize)]
    19→    19→#[serde(rename_all = "snake_case")]
    20→    20→pub enum SetupMode {
    21→    21→    /// Run setup once before all executions of a function (default).
    22→    22→    PerFunction,
    23→    23→    /// Run setup before each individual execution.
    24→    24→    PerExecution,
    25→    25→}
    26→    26→
    27→    27→/// Error type for config operations.
    28→    28→#[derive(Debug, thiserror::Error)]
    29→    29→pub enum ConfigError {
    30→    30→    #[error("failed to read config file '{path}': {source}")]
    31→    31→    Io {
    32→    32→        path: PathBuf,
    33→    33→        source: std::io::Error,
    34→    34→    },
    35→    35→
    36→    36→    #[error("failed to parse config YAML '{path}': {source}")]
    37→    37→    Parse {
    38→    38→        path: PathBuf,
    39→    39→        source: serde_yaml::Error,
    40→    40→    },
    41→    41→
    42→    42→    #[error("failed to read candidate inputs '{path}': {source}")]
    43→    43→    InputIo {
    44→    44→        path: PathBuf,
    45→    45→        source: std::io::Error,
    46→    46→    },
    47→    47→
    48→    48→    #[error("failed to parse candidate inputs '{path}': {source}")]
    49→    49→    InputParse {
    50→    50→        path: PathBuf,
    51→    51→        source: serde_json::Error,
    52→    52→    },
    53→    53→
    54→    54→    #[error("invalid glob pattern '{pattern}': {source}")]
    55→    55→    InvalidPattern {
    56→    56→        pattern: String,
    57→    57→        source: globset::Error,
    58→    58→    },
    59→    59→}

    79→    79→}

   277→   277→/// A config file found during hierarchical discovery, paired with its directory.
   278→   278→#[derive(Debug, Clone)]
   279→   279→struct DiscoveredConfig {
   280→   280→    /// The `.shatter/` directory containing this config.
   281→   281→    shatter_dir: PathBuf,
   282→   282→    /// The parsed config.
   283→   283→    config: ShatterConfig,
   284→   284→}
   285→   285→
   286→   286→/// Parse a `.shatter/config.yaml` file.
   287→   287→pub fn parse_config(path: &Path) -> Result<ShatterConfig, ConfigError> {
   288→   288→    let contents = std::fs::read_to_string(path).map_err(|e| ConfigError::Io {
   289→   289→        path: path.to_path_buf(),
   290→   290→        source: e,
   291→   291→    })?;
   292→   292→    let config: ShatterConfig =
   293→   293→        serde_yaml::from_str(&contents).map_err(|e| ConfigError::Parse {
   294→   294→            path: path.to_path_buf(),
   295→   295→            source: e,
   296→   296→        })?;
   297→   297→    Ok(config)
   298→   298→}
   299→   299→
   300→   300→/// Parse a candidate inputs JSON file.
   301→   301→pub fn parse_candidate_inputs(path: &Path) -> Result<Vec<CandidateInput>, ConfigError> {
   302→   302→    let contents = std::fs::read_to_string(path).map_err(|e| ConfigError::InputIo {
   303→   303→        path: path.to_path_buf(),
   304→   304→        source: e,
   305→   305→    })?;
   306→   306→    let inputs: Vec<CandidateInput> =
   307→   307→        serde_json::from_str(&contents).map_err(|e| ConfigError::InputParse {
   308→   308→            path: path.to_path_buf(),
   309→   309→            source: e,
   310→   310→        })?;
   311→   311→    Ok(inputs)
   312→   312→}
   313→   313→
   314→   314→/// Discover `.shatter/config.yaml` files by walking upward from `start_dir`.
   315→   315→///
   316→   316→/// Returns configs ordered from nearest (highest priority) to farthest (lowest priority).
   317→   317→pub fn discover_configs(start_dir: &Path) -> Result<Vec<ShatterConfig>, ConfigError> {
   318→   318→    let discovered = discover_configs_with_paths(start_dir)?;
   319→   319→    Ok(discovered.into_iter().map(|d| d.config).collect())
   320→   320→}
   321→   321→
   322→   322→/// Internal: discover configs with their `.shatter/` directory paths preserved.
   323→   323→fn discover_configs_with_paths(start_dir: &Path) -> Result<Vec<DiscoveredConfig>, ConfigError> {
   324→   324→    let mut configs = Vec::new();
   325→   325→    let mut current = Some(start_dir.to_path_buf());
   326→   326→
   327→   327→    while let Some(dir) = current {
   328→   328→        let shatter_dir = dir.join(".shatter");
   329→   329→        let config_path = shatter_dir.join("config.yaml");
   330→   330→
   331→   331→        if config_path.is_file() {
   332→   332→            let config = parse_config(&config_path)?;
   333→   333→            configs.push(DiscoveredConfig {
   334→   334→                shatter_dir,
   335→   335→                config,
   336→   336→            });
   337→   337→        }
   338→   338→
   339→   339→        current = dir.parent().map(Path::to_path_buf);
   340→   340→    }
   341→   341→
   342→   342→    Ok(configs)
   343→   343→}
   344→   344→
   345→   345→/// Merge multiple configs where the first config in the slice has highest priority.
   346→   346→///
   347→   347→/// For defaults, the nearest non-None value wins.
   348→   348→/// For function patterns, the nearest matching pattern wins.
   349→   349→pub fn merge_configs(configs: &[ShatterConfig]) -> ShatterConfig {
   350→   350→    if configs.is_empty() {
   351→   351→        return ShatterConfig::default();
   352→   352→    }
   353→   353→    if configs.len() == 1 {
   354→   354→        return configs[0].clone();
   355→   355→    }
   356→   356→
   357→   357→    // Merge defaults: nearest non-None wins.
   358→   358→    let mut max_iterations = None;
   359→   359→    let mut timeout = None;
   360→   360→    let mut setup = None;
   361→   361→    let mut setup_mode = None;
   362→   362→
   363→   363→    // Merge generators maps: start from farthest, overlay nearer.
   364→   364→    // This lets a near config override specific keys while inheriting the rest.
   365→   365→    let mut generators: Option<HashMap<String, String>> = None;
   366→   366→    let mut param_generators: Option<HashMap<String, String>> = None;
   367→   367→
   368→   368→    for config in configs.iter().rev() {
   369→   369→        if let Some(ref g) = config.defaults.generators {
   370→   370→            generators.get_or_insert_with(HashMap::new).extend(
   371→   371→                g.iter().map(|(k, v)| (k.clone(), v.clone())),
   372→   372→            );
   373→   373→        }
   374→   374→        if let Some(ref pg) = config.defaults.param_generators {
   375→   375→            param_generators.get_or_insert_with(HashMap::new).extend(
   376→   376→                pg.iter().map(|(k, v)| (k.clone(), v.clone())),
   377→   377→            );
   378→   378→        }
   379→   379→    }
   380→   380→
   381→   381→    for config in configs {
   382→   382→        if max_iterations.is_none() {
   383→   383→            max_iterations = config.defaults.max_iterations;
   384→   384→        }
   385→   385→        if timeout.is_none() {
   386→   386→            timeout = config.defaults.timeout;
   387→   387→        }
   388→   388→        if setup.is_none() {
   389→   389→            setup = config.defaults.setup.clone();
   390→   390→        }
   391→   391→        if setup_mode.is_none() {
   392→   392→            setup_mode = config.defaults.setup_mode;
   393→   393→        }
   394→   394→    }
   395→   395→
   396→   396→    // Merge function configs: collect all, nearest first (already in order).
   397→   397→    // Later entries only fill in if the pattern hasn't been seen.
   398→   398→    let mut functions: HashMap<String, FunctionConfig> = HashMap::new();
   399→   399→    for config in configs {
   400→   400→        for (pattern, func_config) in &config.functions {
   401→   401→            functions.entry(pattern.clone()).or_insert_with(|| func_config.clone());
   402→   402→        }
   403→   403→    }
   404→   404→
   405→   405→    let mut opaque_types = Vec::new();
   406→   406→    for config in configs {
   407→   407→        for t in &config.opaque_types {
   408→   408→            if !opaque_types.contains(t) {
   409→   409→                opaque_types.push(t.clone());
   410→   410→            }
   411→   411→        }
   412→   412→    }
   413→   413→
   414→   414→    // Merge nondeterminism declarations: union all confirmed/rejected, dedup by (function, path).
   415→   415→    let mut confirmed = Vec::<NondeterminismDeclaration>::new();
   416→   416→    let mut rejected = Vec::<NondeterminismDeclaration>::new();
   417→   417→    let mut has_any = false;
   418→   418→    for config in configs {
   419→   419→        if let Some(ref nd) = config.nondeterminism {
   420→   420→            has_any = true;
   421→   421→            for decl in &nd.confirmed {
   422→   422→                if !confirmed.iter().any(|d| d.function == decl.function && d.path == decl.path) {
   423→   423→                    confirmed.push(decl.clone());
   424→   424→                }
   425→   425→            }
   426→   426→            for decl in &nd.rejected {
   427→   427→                if !rejected.iter().any(|d| d.function == decl.function && d.path == decl.path) {
   428→   428→                    rejected.push(decl.clone());
   429→   429→                }
   430→   430→            }
   431→   431→        }
   432→   432→    }
   433→   433→    let nondeterminism = if has_any {
   434→   434→        Some(NondeterminismConfig { confirmed, rejected })
   435→   435→    } else {
   436→   436→        None
   437→   437→    };
   438→   438→
   439→   439→    ShatterConfig {
   440→   440→        defaults: DefaultsConfig {
   441→   441→            max_iterations,
   442→   442→            timeout,
   443→   443→            setup,
   444→   444→            setup_mode,
   445→   445→            generators,
   446→   446→            param_generators,
   447→   447→            mocks: None,
   448→   448→            genetic: None,
   449→   449→        },
   450→   450→        functions,
   451→   451→        opaque_types,
   452→   452→        nondeterminism,
   453→   453→    }
   454→   454→}
   455→   455→
   456→   456→/// Resolve the effective configuration for a specific function identifier.
   457→   457→///
   458→   458→/// `function_id` is typically in `"file:function"` format.
   459→   459→/// `configs` should be ordered nearest-first (as returned by `discover_configs`).
   460→   460→/// `cli_max_iterations` and `cli_timeout` are the CLI-provided defaults.
   461→   461→pub fn resolve_function_config(
   462→   462→    function_id: &str,
   463→   463→    configs: &[ShatterConfig],
   464→   464→    cli_max_iterations: u32,
   465→   465→    cli_timeout: u64,
   466→   466→) -> Result<ResolvedFunctionConfig, ConfigError> {
   467→   467→    let merged = merge_configs(configs);
   468→   468→    resolve_from_merged(function_id, &merged, cli_max_iterations, cli_timeout)
   469→   469→}
   470→   470→
   471→   471→/// Resolve config for a function from an already-merged config.
   472→   472→fn resolve_from_merged(
   473→   473→    function_id: &str,
   474→   474→    config: &ShatterConfig,
   475→   475→    cli_max_iterations: u32,
   476→   476→    cli_timeout: u64,
   477→   477→) -> Result<ResolvedFunctionConfig, ConfigError> {
   478→   478→    // Find the first matching function pattern.
   479→   479→    let mut func_config: Option<&FunctionConfig> = None;
   480→   480→
   481→   481→    for (pattern, fc) in &config.functions {
   482→   482→        let matcher = compile_pattern(pattern)?;
   483→   483→        if matcher.is_match(function_id) {
   484→   484→            func_config = Some(fc);
   485→   485→            break;
   486→   486→        }
   487→   487→    }
   488→   488→
   489→   489→    // Resolution order: function config > defaults > CLI flags
   490→   490→    let max_iterations = func_config
   491→   491→        .and_then(|fc| fc.max_iterations)
   492→   492→        .or(config.defaults.max_iterations)
   493→   493→        .unwrap_or(cli_max_iterations);
   494→   494→
   495→   495→    let timeout = func_config
   496→   496→        .and_then(|fc| fc.timeout)
   497→   497→        .or(config.defaults.timeout)
   498→   498→        .unwrap_or(cli_timeout);
   499→   499→
   500→   500→    let skip = func_config
   501→   501→        .and_then(|fc| fc.skip)
   502→   502→        .unwrap_or(false);
   503→   503→
   504→   504→    let setup_mode = func_config
   505→   505→        .and_then(|fc| fc.setup_mode)
   506→   506→        .or(config.defaults.setup_mode)
   507→   507→        .unwrap_or(SetupMode::PerFunction);
   508→   508→
   509→   509→    // Merge mock overrides: defaults first, then function-level overrides on top.
   510→   510→    let mut mock_overrides: HashMap<String, crate::auto_mock::MockOverride> = config
   511→   511→        .defaults
   512→   512→        .mocks
   513→   513→        .clone()
   514→   514→        .unwrap_or_default();
   515→   515→    if let Some(fc) = func_config
   516→   516→        && let Some(ref func_mocks) = fc.mocks
   517→   517→    {
   518→   518→        for (k, v) in func_mocks {
   519→   519→            mock_overrides.insert(k.clone(), v.clone());
   520→   520→        }
   521→   521→    }
   522→   522→
   523→   523→    Ok(ResolvedFunctionConfig {
   524→   524→        max_iterations,
   525→   525→        timeout,
   526→   526→        skip,
   527→   527→        candidate_inputs: Vec::new(),
   528→   528→        // Setup and generator paths are resolved to absolute paths by
   529→   529→        // resolve_function_config_with_inputs, which has access to .shatter/ dirs.
   530→   530→        setup: None,
   531→   531→        setup_mode,
   532→   532→        generators: HashMap::new(),
   533→   533→        param_generators: HashMap::new(),
   534→   534→        mock_overrides,
   535→   535→    })
   536→   536→}
   537→   537→
   538→   538→/// Resolve config for a function and load candidate inputs if specified.
   539→   539→///
   540→   540→/// `shatter_dir` is the `.shatter/` directory used to resolve relative input paths.
   541→   541→/// If `explicit_inputs` is provided (from `--inputs` CLI flag), it takes precedence.
   542→   542→pub fn resolve_function_config_with_inputs(
   543→   543→    function_id: &str,
   544→   544→    start_dir: &Path,
   545→   545→    explicit_inputs: Option<&Path>,
   546→   546→    cli_max_iterations: u32,
   547→   547→    cli_timeout: u64,
   548→   548→) -> Result<ResolvedFunctionConfig, ConfigError> {
   549→   549→    let discovered = discover_configs_with_paths(start_dir)?;
   550→   550→    let configs: Vec<ShatterConfig> = discovered.iter().map(|d| d.config.clone()).collect();
   551→   551→    let merged = merge_configs(&configs);
   552→   552→
   553→   553→    let mut resolved = resolve_from_merged(function_id, &merged, cli_max_iterations, cli_timeout)?;
   554→   554→
   555→   555→    // Load candidate inputs.
   556→   556→    if let Some(inputs_path) = explicit_inputs {
   557→   557→        resolved.candidate_inputs = parse_candidate_inputs(inputs_path)?;
   558→   558→    } else {
   559→   559→        // Check function config for an inputs path.
   560→   560→        for dc in &discovered {
   561→   561→            for (pattern, fc) in &dc.config.functions {
   562→   562→                let matcher = compile_pattern(pattern)?;
   563→   563→                if matcher.is_match(function_id)
   564→   564→                    && let Some(inputs_rel) = &fc.inputs
   565→   565→                {
   566→   566→                    let inputs_path = dc.shatter_dir.join(inputs_rel);
   567→   567→                    if inputs_path.is_file() {
   568→   568→                        resolved.candidate_inputs =
   569→   569→                            parse_candidate_inputs(&inputs_path)?;
   570→   570→                    }
   571→   571→                    break;
   572→   572→                }
   573→   573→            }
   574→   574→        }
   575→   575→    }
   576→   576→
   577→   577→    // Resolve setup file path: function-level > defaults, nearest config wins.
   578→   578→    for dc in &discovered {
   579→   579→        if resolved.setup.is_some() {
   580→   580→            break;
   581→   581→        }
   582→   582→        for (pattern, fc) in &dc.config.functions {
   583→   583→            let matcher = compile_pattern(pattern)?;
   584→   584→            if matcher.is_match(function_id)
   585→   585→                && let Some(setup_rel) = &fc.setup
   586→   586→            {
   587→   587→                resolved.setup = Some(dc.shatter_dir.join(setup_rel));
   588→   588→                break;
   589→   589→            }
   590→   590→        }
   591→   591→        if resolved.setup.is_none()
   592→   592→            && let Some(setup_rel) = &dc.config.defaults.setup
   593→   593→        {
   594→   594→            resolved.setup = Some(dc.shatter_dir.join(setup_rel));
   595→   595→        }
   596→   596→    }
   597→   597→
   598→   598→    // Resolve generators: merge defaults then overlay function-level.
   599→   599→    // Walk farthest-to-nearest so nearer configs override.
   600→   600→    for dc in discovered.iter().rev() {
   601→   601→        if let Some(ref g) = dc.config.defaults.generators {
   602→   602→            for (type_name, gen_rel) in g {
   603→   603→                resolved.generators.insert(type_name.clone(), dc.shatter_dir.join(gen_rel));
   604→   604→            }
   605→   605→        }
   606→   606→        if let Some(ref pg) = dc.config.defaults.param_generators {
   607→   607→            for (param_name, gen_rel) in pg {
   608→   608→                resolved.param_generators.insert(param_name.clone(), dc.shatter_dir.join(gen_rel));
   609→   609→            }
   610→   610→        }
   611→   611→    }
   612→   612→    // Function-level generators overlay defaults (nearest matching function wins).
   613→   613→    for dc in &discovered {
   614→   614→        for (pattern, fc) in &dc.config.functions {
   615→   615→            let matcher = compile_pattern(pattern)?;
   616→   616→            if matcher.is_match(function_id) {
   617→   617→                if let Some(ref g) = fc.generators {
   618→   618→                    for (type_name, gen_rel) in g {
   619→   619→                        resolved.generators.insert(type_name.clone(), dc.shatter_dir.join(gen_rel));
   620→   620→                    }
   621→   621→                }
   622→   622→                if let Some(ref pg) = fc.param_generators {
   623→   623→                    for (param_name, gen_rel) in pg {
   624→   624→                        resolved.param_generators.insert(param_name.clone(), dc.shatter_dir.join(gen_rel));
   625→   625→                    }
   626→   626→                }
   627→   627→                break;
   628→   628→            }
   629→   629→        }
   630→   630→    }
   631→   631→
   632→   632→    Ok(resolved)
   633→   633→}
   634→   634→
   635→   635→fn compile_pattern(pattern: &str) -> Result<GlobMatcher, ConfigError> {
   636→   636→    let glob = Glob::new(pattern).map_err(|e| ConfigError::InvalidPattern {
   637→   637→        pattern: pattern.to_string(),
   638→   638→        source: e,
   639→   639→    })?;
   640→   640→    Ok(glob.compile_matcher())
   641→   641→}
