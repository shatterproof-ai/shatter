   439→// Table-driven BuildSQL test for every registry field
   440→// ─────────────────────────────────────────────────────────────────────────────
   441→
   442→func TestBuildSQLAllFields(t *testing.T) {
   443→	t.Parallel()
   444→	reg := search.NewRegistry()
   445→
   446→	for _, f := range reg.All() {
   447→		f := f // capture
   448→		t.Run(f.Name(), func(t *testing.T) {
   449→			raw := minimalFilter(f)
   450→			if raw == nil {
   451→				t.Skipf("no minimal filter for kind %s", f.Kind())
   452→			}
   453→
   454→			expr, args, err := f.BuildSQL(raw, 1)
   455→			if err != nil {
   456→				t.Fatalf("BuildSQL error: %v", err)
   457→			}
   458→			if strings.TrimSpace(expr) == "" {
   459→				t.Fatalf("BuildSQL returned empty expression")
   460→			}
   461→
   462→			// Verify placeholder count matches arg count.
   463→			np := countPlaceholders(expr)
   464→			if np != len(args) {
   465→				t.Errorf("placeholder count %d != arg count %d\nexpr: %s\nargs: %v",
   466→					np, len(args), expr, args)
   467→			}
   468→
   469→			// Verify no obvious SQL injection via string interpolation:
   470→			// all variable data must appear only as $N placeholders.
   471→			if strings.Contains(expr, "test_injection") {
   472→				t.Errorf("value leaked into SQL expression: %s", expr)
   473→			}
   474→		})
   475→	}
   476→}

   478→// minimalFilter returns a minimal valid JSON filter for the given field.
   479→func minimalFilter(f search.Field) json.RawMessage {
   480→	desc := f.Descriptor()
   481→	switch f.Kind() {
   482→	case search.KindFullText:
   483→		return mustJSON(search.FullTextFilter{Query: "test_query"})
   484→	case search.KindSubstring:
   485→		return mustJSON(search.SubstringFilter{Value: "test_value"})
   486→	case search.KindExactString:
   487→		return mustJSON(search.ExactStringFilter{Value: "test_value"})
   488→	case search.KindExactNumeric:
   489→		return mustJSON(search.ExactNumericFilter{Value: 1})
   490→	case search.KindNumericRange:
   491→		return mustJSON(search.NumericRangeFilter{Min: ptrF(1)})
   492→	case search.KindSingleEnum:
   493→		if len(desc.Options) > 0 {
   494→			return mustJSON(search.SingleEnumFilter{Value: desc.Options[0].Value})
   495→		}
   496→		return mustJSON(search.SingleEnumFilter{Value: "test"})
   497→	case search.KindMultiEnum:
   498→		if len(desc.Options) > 0 {
   499→			return mustJSON(search.MultiEnumFilter{Values: []string{desc.Options[0].Value}})
   500→		}
   501→		return mustJSON(search.MultiEnumFilter{Values: []string{"52.0101"}})
   502→	case search.KindGeo:
   503→		return mustJSON(search.GeoFilterInput{
   504→			Kind:   search.GeoFilterKindRadius,
   505→			Radius: &search.GeoRadiusFilter{Lat: 40.7, Lng: -74.0, RadiusKm: 50},
   506→		})
   507→	}
   508→	return nil
   509→}

   515→func TestDescriptors(t *testing.T) {
   516→	t.Parallel()
   517→	reg := search.NewRegistry()
   518→	descs := reg.Descriptors()
   519→
   520→	if len(descs) == 0 {

   523→
   524→	for _, d := range descs {
   525→		if d.Name == "" {
   526→			t.Errorf("descriptor has empty Name: %+v", d)
   527→		}
   528→		if d.Label == "" {
