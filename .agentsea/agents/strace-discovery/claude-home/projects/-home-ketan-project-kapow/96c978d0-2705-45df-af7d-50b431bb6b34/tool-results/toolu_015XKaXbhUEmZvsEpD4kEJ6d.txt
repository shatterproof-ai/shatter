   108→+func run(databaseURL, fixtureDir string, skipMigrate bool) error {
   109→+	dsn := databaseURL
   110→+	if dsn == "" {
   111→+		dsn = os.Getenv("DATABASE_URL")
   112→+	}
   113→+	if dsn == "" {
   114→+		return fmt.Errorf("--database-url flag or DATABASE_URL env var is required")
   115→+	}
   116→+	if fixtureDir == "" {
   117→+		return fmt.Errorf("--fixture-dir is required")

   121→+

   124→+		if err := db.MigrateUp(ctx, dsn); err != nil {
   125→+			return fmt.Errorf("migrate: %w", err)
   126→+		}
   127→+	}
   128→+

   130→+	pool, err := pgxpool.New(ctx, dsn)
   131→+	if err != nil {

   133→+	}

   136→+	fmt.Fprintf(os.Stderr, "loading fixtures from %s...\n", fixtureDir)

   138→+		return fmt.Errorf("load fixtures: %w", err)

   150→+// Package fixture loads JSON fixture data into a PostgreSQL database.
   151→+//
   152→+// Fixture files are JSON arrays of objects (one file per table), optionally
   153→+// xz-compressed. Files are named "<schema>.<table>.json" or
   154→+// "<schema>.<table>.json.xz" and stored in a directory that is passed to
   155→+// [Load] as an [io/fs.FS].
   156→+//
   157→+// Tables are loaded in FK-safe order. After all tables are populated, the
   158→+// materialized views dataset_merged and dataset_latest are refreshed.
   159→+package fixture
   160→+

   162→+	"bytes"
   163→+	"context"
   164→+	"encoding/json"
   165→+	"fmt"
   166→+	"io"

   168→+	"os/exec"
   169→+	"strings"

   171→+	"github.com/jackc/pgx/v5"
   172→+	"github.com/jackc/pgx/v5/pgxpool"

   176→+type tableSpec struct {
   177→+	Schema string
   178→+	Table  string
   179→+}

   182→+// the extraction order in tools/samplesql/extractor.go.
   183→+var loadOrder = []tableSpec{
   184→+	{"location", "state"},
   185→+	{"search_base", "ranking"},
   186→+	{"search_base", "conference"},
   187→+	{"search_base", "cip_data"},

   192→+	{"location", "institution_normalized_alias"},
   193→+	{"location", "zip_coordinates"},

   198→+	"search_base.dataset_merged",

   202→+// Load reads JSON fixture files from dataFS and inserts them into the database
   203→+// in FK-safe order. The dataFS should contain files named
   204→+// "<schema>.<table>.json" or "<schema>.<table>.json.xz".
   205→+//
   206→+// The database schema must already exist (migrations applied) before calling
   207→+// Load.
   208→+func Load(ctx context.Context, pool *pgxpool.Pool, dataFS fs.FS) error {
   209→+	// Discover column types for geometry and JSONB detection.
   210→+	colTypes, err := discoverColumnTypes(ctx, pool)
   211→+	if err != nil {
   212→+		return err
   213→+	}
   214→+
   215→+	for _, spec := range loadOrder {
   216→+		data, err := readFixtureFile(dataFS, spec)
   217→+		if err != nil {
   218→+			return err
   219→+		}
   220→+		if data == nil {
   221→+			continue // file not found — skip
   222→+		}
   223→+
   224→+		if err := loadTable(ctx, pool, spec, data, colTypes); err != nil {
   225→+			return err
   226→+		}
   227→+	}
   228→+
   229→+	// Refresh materialized views.
   230→+	for _, view := range materializedViews {
   231→+		if _, err := pool.Exec(ctx, "REFRESH MATERIALIZED VIEW "+view); err != nil {
   232→+			return fmt.Errorf("fixture: refresh %s: %w", view, err)
   233→+		}
   234→+	}
   235→+
   236→+	return nil
   237→+}

   239→+// LoadOrder returns the FK-safe table loading order. Useful for tests.
   240→+func LoadOrder() []string {
   241→+	names := make([]string, len(loadOrder))
   242→+	for i, spec := range loadOrder {
   243→+		names[i] = spec.Schema + "." + spec.Table

   248→+// readFixtureFile reads and decompresses a fixture file from the FS.
   249→+// Returns nil if the file does not exist.
   250→+func readFixtureFile(dataFS fs.FS, spec tableSpec) ([]byte, error) {
   251→+	filename := fmt.Sprintf("%s.%s.json", spec.Schema, spec.Table)
   252→+	xzFilename := filename + ".xz"
   253→+
   254→+	// Try xz-compressed first.
   255→+	if data, err := fs.ReadFile(dataFS, xzFilename); err == nil {
   256→+		decompressed, err := xzDecompress(data)
   257→+		if err != nil {
   258→+			return nil, fmt.Errorf("fixture: decompress %s: %w", xzFilename, err)
   259→+		}
   260→+		return decompressed, nil
   261→+	}
   262→+
   263→+	// Try uncompressed.
   264→+	data, err := fs.ReadFile(dataFS, filename)
   265→+	if err != nil {
   266→+		return nil, nil //nolint:nilerr // file not found is not an error
   267→+	}
   268→+	return data, nil
   269→+}
   270→+
   271→+// xzDecompress decompresses xz data using the xz command-line tool.
   272→+func xzDecompress(data []byte) ([]byte, error) {
   273→+	cmd := exec.Command("xz", "-dc") //nolint:gosec // G204: xz is a trusted system tool
   274→+	cmd.Stdin = bytes.NewReader(data)
   275→+	out, err := cmd.Output()
   276→+	if err != nil {
   277→+		return nil, fmt.Errorf("xz: %w", err)
   278→+	}
   279→+	return out, nil
   280→+}

   282→+// columnInfo holds type metadata for a single column.
   283→+type columnInfo struct {
   284→+	UDTName  string // e.g. "geometry", "geography", "jsonb", "text"
   285→+	DataType string // e.g. "USER-DEFINED", "jsonb", "text"
   286→+}

   288→+// discoverColumnTypes queries information_schema to build a map of
   289→+// schema.table.column → columnInfo for all fixture tables.
   290→+func discoverColumnTypes(ctx context.Context, pool *pgxpool.Pool) (map[string]columnInfo, error) {
   291→+	result := make(map[string]columnInfo)
   292→+
   293→+	// Build list of schemas and tables to query.
   294→+	schemas := make(map[string]bool)
   295→+	for _, spec := range loadOrder {
   296→+		schemas[spec.Schema] = true
   297→+	}
   298→+
   299→+	for schema := range schemas {
   300→+		rows, err := pool.Query(ctx,
   301→+			`SELECT table_name, column_name, data_type, udt_name
   302→+			 FROM information_schema.columns
   303→+			 WHERE table_schema = $1
   304→+			 ORDER BY table_name, ordinal_position`, schema)
   305→+		if err != nil {
   306→+			return nil, fmt.Errorf("fixture: discover columns for %s: %w", schema, err)
   307→+		}
   308→+
   309→+		for rows.Next() {
   310→+			var tableName, colName, dataType, udtName string
   311→+			if err := rows.Scan(&tableName, &colName, &dataType, &udtName); err != nil {
   312→+				rows.Close()
   313→+				return nil, fmt.Errorf("fixture: scan column info: %w", err)
   314→+			}
   315→+			key := fmt.Sprintf("%s.%s.%s", schema, tableName, colName)
   316→+			result[key] = columnInfo{UDTName: udtName, DataType: dataType}
   317→+		}
   318→+		rows.Close()
   319→+		if err := rows.Err(); err != nil {
   320→+			return nil, fmt.Errorf("fixture: iterate columns for %s: %w", schema, err)
   321→+		}
   322→+	}
   323→+
   324→+	return result, nil
   325→+}

   327→+// loadTable parses JSON data and inserts rows into the target table.
   328→+func loadTable(ctx context.Context, pool *pgxpool.Pool, spec tableSpec, data []byte, colTypes map[string]columnInfo) error {
   329→+	var rows []map[string]any
   330→+	if err := json.Unmarshal(data, &rows); err != nil {
   331→+		return fmt.Errorf("fixture: parse %s.%s: %w", spec.Schema, spec.Table, err)
   332→+	}
   333→+	if len(rows) == 0 {
   334→+		return nil
   335→+	}
   336→+
   337→+	qualifiedName := fmt.Sprintf("%s.%s", spec.Schema, spec.Table)
   338→+
   339→+	// Extract column order from the first row.
   340→+	columns := orderedKeys(rows[0])
   341→+
   342→+	// Check if any columns need special handling (geometry).
   343→+	hasGeo := false
   344→+	geoColumns := make(map[string]bool)
   345→+	for _, col := range columns {
   346→+		key := fmt.Sprintf("%s.%s.%s", spec.Schema, spec.Table, col)
   347→+		info, ok := colTypes[key]
   348→+		if ok && (info.UDTName == "geometry" || info.UDTName == "geography") {
   349→+			geoColumns[col] = true
   350→+			hasGeo = true
   351→+		}
   352→+	}
   353→+
   354→+	if hasGeo {
   355→+		return loadTableWithGeo(ctx, pool, qualifiedName, columns, rows, geoColumns)
   356→+	}
   357→+	return loadTableDirect(ctx, pool, qualifiedName, columns, rows, colTypes, spec)
   358→+}

   360→+// loadTableDirect loads data using pgx CopyFrom for tables without geometry.
   361→+func loadTableDirect(ctx context.Context, pool *pgxpool.Pool, qualifiedName string, columns []string, rows []map[string]any, colTypes map[string]columnInfo, spec tableSpec) error {
   362→+	// Build JSONB column set for proper marshaling.
   363→+	jsonbCols := make(map[string]bool)
   364→+	for _, col := range columns {
   365→+		key := fmt.Sprintf("%s.%s.%s", spec.Schema, spec.Table, col)
   366→+		info, ok := colTypes[key]
   367→+		if ok && info.UDTName == "jsonb" {
   368→+			jsonbCols[col] = true
   369→+		}
   370→+	}
   371→+
   372→+	copyRows := make([][]any, len(rows))
   373→+	for i, row := range rows {
   374→+		vals := make([]any, len(columns))
   375→+		for j, col := range columns {
   376→+			vals[j] = convertValue(row[col], jsonbCols[col])
   377→+		}
   378→+		copyRows[i] = vals
   379→+	}
   380→+
   381→+	parts := strings.SplitN(qualifiedName, ".", 2)
   382→+	copyCount, err := pool.CopyFrom(ctx,
   383→+		pgx.Identifier{parts[0], parts[1]},
   384→+		columns,
   385→+		pgx.CopyFromRows(copyRows),
   386→+	)
   387→+	if err != nil {
   388→+		return fmt.Errorf("fixture: copy %s: %w", qualifiedName, err)
   389→+	}
   390→+	if int(copyCount) != len(rows) {
   391→+		return fmt.Errorf("fixture: copy %s: expected %d rows, got %d", qualifiedName, len(rows), copyCount)
   392→+	}
   393→+
   394→+	return nil
   395→+}

   397→+// loadTableWithGeo loads data using a staging table for geometry columns.
   398→+// Geometry values are stored as GeoJSON text in the JSON fixture, so we
   399→+// load into a temp table with text columns, then INSERT with ST_GeomFromGeoJSON.
   400→+func loadTableWithGeo(ctx context.Context, pool *pgxpool.Pool, qualifiedName string, columns []string, rows []map[string]any, geoColumns map[string]bool) error {
   401→+	tx, err := pool.Begin(ctx)
   402→+	if err != nil {
   403→+		return fmt.Errorf("fixture: begin tx for %s: %w", qualifiedName, err)
   404→+	}
   405→+	defer tx.Rollback(ctx) //nolint:errcheck
   406→+
   407→+	// Create temp table mimicking the target but with text for geo columns.
   408→+	tempName := "fixture_staging"
   409→+	createCols := make([]string, len(columns))
   410→+	for i, col := range columns {
   411→+		// All columns are text in the staging table — geo columns get
   412→+		// converted to geometry via ST_GeomFromGeoJSON in the final INSERT.
   413→+		createCols[i] = fmt.Sprintf("%s text", col)
   414→+	}
   415→+
   416→+	dropSQL := fmt.Sprintf("DROP TABLE IF EXISTS %s", tempName)
   417→+	if _, err := tx.Exec(ctx, dropSQL); err != nil {
   418→+		return fmt.Errorf("fixture: drop staging: %w", err)
   419→+	}
   420→+
   421→+	createSQL := fmt.Sprintf("CREATE TEMP TABLE %s (%s)", tempName, strings.Join(createCols, ", "))
   422→+	if _, err := tx.Exec(ctx, createSQL); err != nil {
   423→+		return fmt.Errorf("fixture: create staging for %s: %w", qualifiedName, err)
   424→+	}
   425→+
   426→+	// COPY into staging table with all values as text.
   427→+	copyRows := make([][]any, len(rows))
   428→+	for i, row := range rows {
   429→+		vals := make([]any, len(columns))
   430→+		for j, col := range columns {
   431→+			v := row[col]
   432→+			switch {
   433→+			case v == nil:
   434→+				vals[j] = nil
   435→+			case geoColumns[col]:
   436→+				// GeoJSON object → JSON string
   437→+				b, err := json.Marshal(v)
   438→+				if err != nil {
   439→+					return fmt.Errorf("fixture: marshal geo %s row %d: %w", col, i, err)
   440→+				}
   441→+				vals[j] = string(b)
   442→+			default:
   443→+				vals[j] = toText(v)
   444→+			}
   445→+		}
   446→+		copyRows[i] = vals
   447→+	}
   448→+
   449→+	_, err = tx.CopyFrom(ctx,
   450→+		pgx.Identifier{tempName},
   451→+		columns,
   452→+		pgx.CopyFromRows(copyRows),
   453→+	)
   454→+	if err != nil {
   455→+		return fmt.Errorf("fixture: copy staging %s: %w", qualifiedName, err)
   456→+	}
   457→+
   458→+	// Build INSERT from staging with ST_GeomFromGeoJSON for geo columns.
   459→+	selectCols := make([]string, len(columns))
   460→+	for i, col := range columns {
   461→+		if geoColumns[col] {
   462→+			selectCols[i] = fmt.Sprintf("ST_GeomFromGeoJSON(%s) AS %s", col, col)
   463→+		} else {
   464→+			// Cast text back to original column types.
   465→+			selectCols[i] = col
   466→+		}
   467→+	}
   468→+
   469→+	insertSQL := fmt.Sprintf(
   470→+		"INSERT INTO %s (%s) SELECT %s FROM %s",
   471→+		qualifiedName,
   472→+		strings.Join(columns, ", "),
   473→+		strings.Join(selectCols, ", "),
   474→+		tempName,
   475→+	)
   476→+	if _, err := tx.Exec(ctx, insertSQL); err != nil {
   477→+		return fmt.Errorf("fixture: insert %s from staging: %w", qualifiedName, err)
   478→+	}
   479→+
   480→+	if _, err := tx.Exec(ctx, dropSQL); err != nil {
   481→+		return fmt.Errorf("fixture: cleanup staging: %w", err)
   482→+	}
   483→+
   484→+	if err := tx.Commit(ctx); err != nil {
   485→+		return fmt.Errorf("fixture: commit %s: %w", qualifiedName, err)
   486→+	}
   487→+
   488→+	return nil
   489→+}

   491→+// toText converts a JSON-decoded value to its text representation for the
   492→+// staging table. All staging columns are text, so every value must be a string.
   493→+func toText(v any) string {
   494→+	switch vt := v.(type) {
   495→+	case string:
   496→+		return vt
   497→+	case float64:
   498→+		return fmt.Sprintf("%v", vt)
   499→+	case bool:
   500→+		if vt {
   501→+			return "true"
   502→+		}
   503→+		return "false"
   504→+	default:
   505→+		b, err := json.Marshal(v)
   506→+		if err != nil {
   507→+			return fmt.Sprintf("%v", v)
   508→+		}
   509→+		return string(b)
   510→+	}
   511→+}

   513→+// convertValue converts a JSON-decoded value to a pgx-compatible type.
   514→+func convertValue(v any, isJSONB bool) any {
   515→+	if v == nil {
   516→+		return nil
   517→+	}
   518→+	if isJSONB {
   519→+		// JSONB columns: marshal the value back to a JSON byte slice
   520→+		// which pgx will send as JSONB.
   521→+		switch vt := v.(type) {
   522→+		case map[string]any, []any:
   523→+			b, err := json.Marshal(vt)
   524→+			if err != nil {
   525→+				return nil
   526→+			}
   527→+			return b
   528→+		case string:
   529→+			return []byte(vt)
   530→+		default:

   541→+// orderedKeys returns the keys of a map in a stable order.
   542→+// JSON object key order is not guaranteed, so we sort alphabetically.
   543→+// However, for fixture loading the order doesn't matter as long as it's
   544→+// consistent between the column list and the data rows.
   545→+func orderedKeys(m map[string]any) []string {
   546→+	keys := make([]string, 0, len(m))
   547→+	for k := range m {
   548→+		keys = append(keys, k)
   549→+	}
   550→+	// Sort for determinism.
   551→+	sortStrings(keys)
   552→+	return keys
   553→+}

   555→+// sortStrings sorts a string slice in place (insertion sort — fine for small slices).
   556→+func sortStrings(s []string) {
   557→+	for i := 1; i < len(s); i++ {
   558→+		for j := i; j > 0 && s[j] < s[j-1]; j-- {
   559→+			s[j], s[j-1] = s[j-1], s[j]
   560→+		}
   561→+	}
   562→+}
