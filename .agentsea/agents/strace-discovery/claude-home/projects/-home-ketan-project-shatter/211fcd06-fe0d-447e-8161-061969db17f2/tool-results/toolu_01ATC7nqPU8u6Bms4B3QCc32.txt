     1→//! Shatter configuration: `.shatter/config.yaml` parsing, hierarchical discovery, and merging.
     2→//!
     3→//! Users can place `.shatter/` directories at any level of their project tree.
     4→//! Each directory can contain a `config.yaml` with per-function settings and
     5→//! an `inputs/` subdirectory with candidate input files.
     6→//!
     7→//! Config resolution walks upward from each target file to the filesystem root,
     8→//! collecting all `.shatter/config.yaml` files. The nearest config wins on
     9→//! conflicts (closest to the target file takes precedence).
    10→
    11→use std::collections::HashMap;
    12→use std::path::{Path, PathBuf};
    13→
    14→use globset::{Glob, GlobMatcher};
    15→use serde::{Deserialize, Serialize};
    16→
    17→/// When to run the setup file relative to function executions.
    18→#[derive(Debug, Clone, Copy, PartialEq, Eq, Deserialize, Serialize)]
    19→#[serde(rename_all = "snake_case")]
    20→pub enum SetupMode {
    21→    /// Run setup once before all executions of a function (default).
    22→    PerFunction,
    23→    /// Run setup before each individual execution.
    24→    PerExecution,
    25→}
    26→
    27→/// Error type for config operations.
    28→#[derive(Debug, thiserror::Error)]
    29→pub enum ConfigError {
    30→    #[error("failed to read config file '{path}': {source}")]
    31→    Io {
    32→        path: PathBuf,
    33→        source: std::io::Error,
    34→    },
    35→
    36→    #[error("failed to parse config YAML '{path}': {source}")]
    37→    Parse {
    38→        path: PathBuf,
    39→        source: serde_yaml::Error,
    40→    },
    41→
    42→    #[error("failed to read candidate inputs '{path}': {source}")]
    43→    InputIo {
    44→        path: PathBuf,
    45→        source: std::io::Error,
    46→    },
    47→
    48→    #[error("failed to parse candidate inputs '{path}': {source}")]
    49→    InputParse {
    50→        path: PathBuf,
    51→        source: serde_json::Error,
    52→    },
    53→
    54→    #[error("invalid glob pattern '{pattern}': {source}")]
    55→    InvalidPattern {
    56→        pattern: String,
    57→        source: globset::Error,
    58→    },
    59→}

    61→/// Top-level `.shatter/config.yaml` structure.
    62→#[derive(Debug, Clone, Default, Deserialize, Serialize, PartialEq)]

    64→    /// Default settings applied to all functions unless overridden.
    65→    #[serde(default)]

    71→
    72→    /// Additional type names to treat as opaque (unexecutable).
    73→    #[serde(default)]

    75→
    76→    /// User-declared nondeterminism: confirmed and rejected field declarations.
    77→    #[serde(default)]
    78→    pub nondeterminism: Option<NondeterminismConfig>,
    79→}

   277→/// A config file found during hierarchical discovery, paired with its directory.
   278→#[derive(Debug, Clone)]
   279→struct DiscoveredConfig {
   280→    /// The `.shatter/` directory containing this config.
   281→    shatter_dir: PathBuf,
   282→    /// The parsed config.
   283→    config: ShatterConfig,
   284→}
   285→
   286→/// Parse a `.shatter/config.yaml` file.
   287→pub fn parse_config(path: &Path) -> Result<ShatterConfig, ConfigError> {
   288→    let contents = std::fs::read_to_string(path).map_err(|e| ConfigError::Io {
   289→        path: path.to_path_buf(),
   290→        source: e,
   291→    })?;
   292→    let config: ShatterConfig =
   293→        serde_yaml::from_str(&contents).map_err(|e| ConfigError::Parse {
   294→            path: path.to_path_buf(),
   295→            source: e,
   296→        })?;
   297→    Ok(config)
   298→}
   299→
   300→/// Parse a candidate inputs JSON file.
   301→pub fn parse_candidate_inputs(path: &Path) -> Result<Vec<CandidateInput>, ConfigError> {
   302→    let contents = std::fs::read_to_string(path).map_err(|e| ConfigError::InputIo {
   303→        path: path.to_path_buf(),
   304→        source: e,
   305→    })?;
   306→    let inputs: Vec<CandidateInput> =
   307→        serde_json::from_str(&contents).map_err(|e| ConfigError::InputParse {
   308→            path: path.to_path_buf(),
   309→            source: e,
   310→        })?;
   311→    Ok(inputs)
   312→}
   313→
   314→/// Discover `.shatter/config.yaml` files by walking upward from `start_dir`.
   315→///
   316→/// Returns configs ordered from nearest (highest priority) to farthest (lowest priority).
   317→pub fn discover_configs(start_dir: &Path) -> Result<Vec<ShatterConfig>, ConfigError> {
   318→    let discovered = discover_configs_with_paths(start_dir)?;
   319→    Ok(discovered.into_iter().map(|d| d.config).collect())
   320→}
   321→
   322→/// Internal: discover configs with their `.shatter/` directory paths preserved.
   323→fn discover_configs_with_paths(start_dir: &Path) -> Result<Vec<DiscoveredConfig>, ConfigError> {
   324→    let mut configs = Vec::new();
   325→    let mut current = Some(start_dir.to_path_buf());
   326→
   327→    while let Some(dir) = current {
   328→        let shatter_dir = dir.join(".shatter");
   329→        let config_path = shatter_dir.join("config.yaml");
   330→
   331→        if config_path.is_file() {
   332→            let config = parse_config(&config_path)?;
   333→            configs.push(DiscoveredConfig {
   334→                shatter_dir,
   335→                config,
   336→            });
   337→        }
   338→
   339→        current = dir.parent().map(Path::to_path_buf);
   340→    }
   341→
   342→    Ok(configs)
   343→}

   345→/// Merge multiple configs where the first config in the slice has highest priority.
   346→///
   347→/// For defaults, the nearest non-None value wins.
   348→/// For function patterns, the nearest matching pattern wins.
   349→pub fn merge_configs(configs: &[ShatterConfig]) -> ShatterConfig {
   350→    if configs.is_empty() {
   351→        return ShatterConfig::default();
   352→    }
   353→    if configs.len() == 1 {
   354→        return configs[0].clone();
   355→    }
   356→
   357→    // Merge defaults: nearest non-None wins.
   358→    let mut max_iterations = None;
   359→    let mut timeout = None;
   360→    let mut setup = None;
   361→    let mut setup_mode = None;
   362→
   363→    // Merge generators maps: start from farthest, overlay nearer.
   364→    // This lets a near config override specific keys while inheriting the rest.
   365→    let mut generators: Option<HashMap<String, String>> = None;
   366→    let mut param_generators: Option<HashMap<String, String>> = None;
   367→
   368→    for config in configs.iter().rev() {
   369→        if let Some(ref g) = config.defaults.generators {
   370→            generators.get_or_insert_with(HashMap::new).extend(
   371→                g.iter().map(|(k, v)| (k.clone(), v.clone())),
   372→            );
   373→        }
   374→        if let Some(ref pg) = config.defaults.param_generators {
   375→            param_generators.get_or_insert_with(HashMap::new).extend(
   376→                pg.iter().map(|(k, v)| (k.clone(), v.clone())),
   377→            );
   378→        }
   379→    }
   380→
   381→    for config in configs {
   382→        if max_iterations.is_none() {
   383→            max_iterations = config.defaults.max_iterations;
   384→        }
   385→        if timeout.is_none() {
   386→            timeout = config.defaults.timeout;
   387→        }
   388→        if setup.is_none() {
   389→            setup = config.defaults.setup.clone();
   390→        }
   391→        if setup_mode.is_none() {
   392→            setup_mode = config.defaults.setup_mode;
   393→        }
   394→    }
   395→
   396→    // Merge function configs: collect all, nearest first (already in order).
   397→    // Later entries only fill in if the pattern hasn't been seen.
   398→    let mut functions: HashMap<String, FunctionConfig> = HashMap::new();
   399→    for config in configs {
   400→        for (pattern, func_config) in &config.functions {
   401→            functions.entry(pattern.clone()).or_insert_with(|| func_config.clone());
   402→        }
   403→    }
   404→
   405→    let mut opaque_types = Vec::new();
   406→    for config in configs {
   407→        for t in &config.opaque_types {
   408→            if !opaque_types.contains(t) {
   409→                opaque_types.push(t.clone());
   410→            }
   411→        }
   412→    }
   413→
   414→    // Merge nondeterminism declarations: union all confirmed/rejected, dedup by (function, path).
   415→    let mut confirmed = Vec::<NondeterminismDeclaration>::new();
   416→    let mut rejected = Vec::<NondeterminismDeclaration>::new();
   417→    let mut has_any = false;
   418→    for config in configs {
   419→        if let Some(ref nd) = config.nondeterminism {
   420→            has_any = true;
   421→            for decl in &nd.confirmed {
   422→                if !confirmed.iter().any(|d| d.function == decl.function && d.path == decl.path) {
   423→                    confirmed.push(decl.clone());
   424→                }
   425→            }
   426→            for decl in &nd.rejected {
   427→                if !rejected.iter().any(|d| d.function == decl.function && d.path == decl.path) {
   428→                    rejected.push(decl.clone());
   429→                }
   430→            }
   431→        }
   432→    }
   433→    let nondeterminism = if has_any {
   434→        Some(NondeterminismConfig { confirmed, rejected })
   435→    } else {
   436→        None
   437→    };
   438→
   439→    ShatterConfig {
   440→        defaults: DefaultsConfig {
   441→            max_iterations,
   442→            timeout,
   443→            setup,
   444→            setup_mode,
   445→            generators,
   446→            param_generators,
   447→            mocks: None,
   448→            genetic: None,
   449→        },
   450→        functions,
   451→        opaque_types,
   452→        nondeterminism,
   453→    }
   454→}

   456→/// Resolve the effective configuration for a specific function identifier.
   457→///
   458→/// `function_id` is typically in `"file:function"` format.
   459→/// `configs` should be ordered nearest-first (as returned by `discover_configs`).
   460→/// `cli_max_iterations` and `cli_timeout` are the CLI-provided defaults.
   461→pub fn resolve_function_config(
   462→    function_id: &str,
   463→    configs: &[ShatterConfig],
   464→    cli_max_iterations: u32,
   465→    cli_timeout: u64,
   466→) -> Result<ResolvedFunctionConfig, ConfigError> {
   467→    let merged = merge_configs(configs);
   468→    resolve_from_merged(function_id, &merged, cli_max_iterations, cli_timeout)
   469→}
   470→
   471→/// Resolve config for a function from an already-merged config.
   472→fn resolve_from_merged(
   473→    function_id: &str,
   474→    config: &ShatterConfig,
   475→    cli_max_iterations: u32,
   476→    cli_timeout: u64,
   477→) -> Result<ResolvedFunctionConfig, ConfigError> {
   478→    // Find the first matching function pattern.
   479→    let mut func_config: Option<&FunctionConfig> = None;
   480→
   481→    for (pattern, fc) in &config.functions {
   482→        let matcher = compile_pattern(pattern)?;
   483→        if matcher.is_match(function_id) {
   484→            func_config = Some(fc);
   485→            break;
   486→        }
   487→    }
   488→
   489→    // Resolution order: function config > defaults > CLI flags
   490→    let max_iterations = func_config
   491→        .and_then(|fc| fc.max_iterations)
   492→        .or(config.defaults.max_iterations)
   493→        .unwrap_or(cli_max_iterations);
   494→
   495→    let timeout = func_config
   496→        .and_then(|fc| fc.timeout)
   497→        .or(config.defaults.timeout)
   498→        .unwrap_or(cli_timeout);
   499→
   500→    let skip = func_config
   501→        .and_then(|fc| fc.skip)
   502→        .unwrap_or(false);
   503→
   504→    let setup_mode = func_config
   505→        .and_then(|fc| fc.setup_mode)
   506→        .or(config.defaults.setup_mode)
   507→        .unwrap_or(SetupMode::PerFunction);
   508→
   509→    // Merge mock overrides: defaults first, then function-level overrides on top.
   510→    let mut mock_overrides: HashMap<String, crate::auto_mock::MockOverride> = config
   511→        .defaults
   512→        .mocks
   513→        .clone()
   514→        .unwrap_or_default();
   515→    if let Some(fc) = func_config
   516→        && let Some(ref func_mocks) = fc.mocks
   517→    {
   518→        for (k, v) in func_mocks {
   519→            mock_overrides.insert(k.clone(), v.clone());
   520→        }
   521→    }
   522→
   523→    Ok(ResolvedFunctionConfig {
   524→        max_iterations,
   525→        timeout,
   526→        skip,
   527→        candidate_inputs: Vec::new(),
   528→        // Setup and generator paths are resolved to absolute paths by
   529→        // resolve_function_config_with_inputs, which has access to .shatter/ dirs.
   530→        setup: None,
   531→        setup_mode,
   532→        generators: HashMap::new(),
   533→        param_generators: HashMap::new(),
   534→        mock_overrides,
   535→    })
   536→}

   538→/// Resolve config for a function and load candidate inputs if specified.
   539→///
   540→/// `shatter_dir` is the `.shatter/` directory used to resolve relative input paths.
   541→/// If `explicit_inputs` is provided (from `--inputs` CLI flag), it takes precedence.
   542→pub fn resolve_function_config_with_inputs(
   543→    function_id: &str,
   544→    start_dir: &Path,
   545→    explicit_inputs: Option<&Path>,
   546→    cli_max_iterations: u32,
   547→    cli_timeout: u64,
   548→) -> Result<ResolvedFunctionConfig, ConfigError> {
   549→    let discovered = discover_configs_with_paths(start_dir)?;
   550→    let configs: Vec<ShatterConfig> = discovered.iter().map(|d| d.config.clone()).collect();
   551→    let merged = merge_configs(&configs);
   552→
   553→    let mut resolved = resolve_from_merged(function_id, &merged, cli_max_iterations, cli_timeout)?;
   554→
   555→    // Load candidate inputs.
   556→    if let Some(inputs_path) = explicit_inputs {
   557→        resolved.candidate_inputs = parse_candidate_inputs(inputs_path)?;
   558→    } else {
   559→        // Check function config for an inputs path.
   560→        for dc in &discovered {
   561→            for (pattern, fc) in &dc.config.functions {
   562→                let matcher = compile_pattern(pattern)?;
   563→                if matcher.is_match(function_id)
   564→                    && let Some(inputs_rel) = &fc.inputs
   565→                {
   566→                    let inputs_path = dc.shatter_dir.join(inputs_rel);
   567→                    if inputs_path.is_file() {
   568→                        resolved.candidate_inputs =
   569→                            parse_candidate_inputs(&inputs_path)?;
   570→                    }
   571→                    break;
   572→                }
   573→            }
   574→        }
   575→    }
   576→
   577→    // Resolve setup file path: function-level > defaults, nearest config wins.
   578→    for dc in &discovered {
   579→        if resolved.setup.is_some() {
   580→            break;
   581→        }
   582→        for (pattern, fc) in &dc.config.functions {
   583→            let matcher = compile_pattern(pattern)?;
   584→            if matcher.is_match(function_id)
   585→                && let Some(setup_rel) = &fc.setup
   586→            {
   587→                resolved.setup = Some(dc.shatter_dir.join(setup_rel));
   588→                break;
   589→            }
   590→        }
   591→        if resolved.setup.is_none()
   592→            && let Some(setup_rel) = &dc.config.defaults.setup
   593→        {
   594→            resolved.setup = Some(dc.shatter_dir.join(setup_rel));
   595→        }
   596→    }
   597→
   598→    // Resolve generators: merge defaults then overlay function-level.
   599→    // Walk farthest-to-nearest so nearer configs override.
   600→    for dc in discovered.iter().rev() {
   601→        if let Some(ref g) = dc.config.defaults.generators {
   602→            for (type_name, gen_rel) in g {
   603→                resolved.generators.insert(type_name.clone(), dc.shatter_dir.join(gen_rel));
   604→            }
   605→        }
   606→        if let Some(ref pg) = dc.config.defaults.param_generators {
   607→            for (param_name, gen_rel) in pg {
   608→                resolved.param_generators.insert(param_name.clone(), dc.shatter_dir.join(gen_rel));
   609→            }
   610→        }
   611→    }
   612→    // Function-level generators overlay defaults (nearest matching function wins).
   613→    for dc in &discovered {
   614→        for (pattern, fc) in &dc.config.functions {
   615→            let matcher = compile_pattern(pattern)?;
   616→            if matcher.is_match(function_id) {
   617→                if let Some(ref g) = fc.generators {
   618→                    for (type_name, gen_rel) in g {
   619→                        resolved.generators.insert(type_name.clone(), dc.shatter_dir.join(gen_rel));
   620→                    }
   621→                }
   622→                if let Some(ref pg) = fc.param_generators {
   623→                    for (param_name, gen_rel) in pg {
   624→                        resolved.param_generators.insert(param_name.clone(), dc.shatter_dir.join(gen_rel));
   625→                    }
   626→                }
   627→                break;
   628→            }
   629→        }
   630→    }
   631→
   632→    Ok(resolved)
   633→}
   634→
   635→fn compile_pattern(pattern: &str) -> Result<GlobMatcher, ConfigError> {
   636→    let glob = Glob::new(pattern).map_err(|e| ConfigError::InvalidPattern {
   637→        pattern: pattern.to_string(),
   638→        source: e,
   639→    })?;
   640→    Ok(glob.compile_matcher())
   641→}
