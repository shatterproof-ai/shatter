     9→/// Log level for the frontend, controlled by SHATTER_LOG_LEVEL env var.
    10→#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
    11→enum FrontendLogLevel {
    12→    Error = 0,
    13→    Warn = 1,
    14→    Info = 2,
    15→    Debug = 3,
    16→    Trace = 4,
    17→}
    18→
    19→impl FrontendLogLevel {
    20→    fn from_env() -> Self {
    21→        match std::env::var("SHATTER_LOG_LEVEL")
    22→            .unwrap_or_default()
    23→            .to_lowercase()
    24→            .as_str()
    25→        {
    26→            "error" => FrontendLogLevel::Error,
    27→            "warn" => FrontendLogLevel::Warn,
    28→            "debug" => FrontendLogLevel::Debug,
    29→            "trace" => FrontendLogLevel::Trace,
    30→            _ => FrontendLogLevel::Info,
    31→        }
    32→    }
    33→}

    35→const DEFAULT_EXEC_TIMEOUT_MS: u64 = 5000;
    36→
    37→/// Read the execution timeout from `SHATTER_EXEC_TIMEOUT` env var (seconds).
    38→/// Returns milliseconds. Falls back to `DEFAULT_EXEC_TIMEOUT_MS` if unset, invalid, zero, or negative.
    39→fn exec_timeout_from_env() -> u64 {
    40→    match std::env::var("SHATTER_EXEC_TIMEOUT") {
    41→        Ok(val) => match val.parse::<f64>() {
    42→            Ok(secs) if secs > 0.0 => (secs * 1000.0) as u64,
    43→            _ => DEFAULT_EXEC_TIMEOUT_MS,
    44→        },
    45→        Err(_) => DEFAULT_EXEC_TIMEOUT_MS,
    46→    }
    47→}

    49→/// Write instrumented source to a temp directory and return the output path.
    50→fn write_instrumented_temp(filename: &str, source: &str) -> io::Result<String> {
    51→    let dir = std::env::temp_dir().join(format!(
    52→        "shatter-instrument-{}",
    53→        std::process::id()
    54→    ));
    55→    std::fs::create_dir_all(&dir)?;
    56→    let out_path = dir.join(filename);
    57→    std::fs::write(&out_path, source)?;
    58→    Ok(out_path.to_string_lossy().into_owned())
    59→}

    61→/// Processes protocol requests from stdin and writes responses to stdout.
    62→pub struct Handler<R, W, L> {
    63→    reader: BufReader<R>,
    64→    writer: W,
    65→    log: L,
    66→    log_level: FrontendLogLevel,
    67→    exec_timeout_ms: u64,
    68→    wasm_cache: WasmCache,
    69→    native_registry: Option<NativeRegistry>,
    70→    /// Remembered from the most recent Analyze or Instrument request so Execute
    71→    /// can fall back when the core omits the file field (which it always does).
    72→    last_file: Option<String>,
    73→}
    74→
    75→impl<R: io::Read, W: io::Write, L: io::Write> Handler<R, W, L> {
    76→    pub fn new(reader: R, writer: W, log: L) -> Self {
    77→        Self {
    78→            reader: BufReader::new(reader),
    79→            writer,

    82→            exec_timeout_ms: exec_timeout_from_env(),
    83→            wasm_cache: WasmCache::new(),
    84→            native_registry: None,
    85→            last_file: None,
    86→        }
    87→    }

   123→    /// Process requests until shutdown or EOF. Returns Ok(()) on clean shutdown.
   124→    pub fn run(mut self) -> io::Result<()> {
   125→        self.log_at(FrontendLogLevel::Debug, &format!(
   126→            "Starting Rust frontend (protocol {PROTOCOL_VERSION})"
   127→        ));
   128→
   129→        let mut line = String::new();
   130→        loop {
   131→            line.clear();
   132→            let bytes_read = self.reader.read_line(&mut line)?;
   133→            if bytes_read == 0 {
   134→                // EOF
   135→                self.log_at(FrontendLogLevel::Debug, "Stdin closed, exiting");
   136→                return Ok(());
   137→            }
   138→
   139→            let trimmed = line.trim();
   140→            if trimmed.is_empty() {
   141→                continue;
   142→            }
   143→
   144→            self.logf(&format!("Received: {trimmed}"));
   145→
   146→            let req: Request = match serde_json::from_str(trimmed) {
   147→                Ok(r) => r,
   148→                Err(e) => {
   149→                    self.logf(&format!("Failed to parse request: {e}"));
   150→                    continue;
   151→                }
   152→            };
   153→
   154→            let (resp, shutdown) = self.dispatch(&req);
   155→            self.send(&resp)?;
   156→
   157→            if shutdown {
   158→                self.log_at(FrontendLogLevel::Debug, "Shutting down");
   159→                return Ok(());
   160→            }
   161→        }
   162→    }
   163→
   164→    fn dispatch(&mut self, req: &Request) -> (Response, bool) {
   165→        let mut resp = Response::base(req.id);
   166→
   167→        if req.protocol_version != protocol::PROTOCOL_VERSION {
   168→            resp.status = "error".to_string();
   169→            resp.code = Some("version_mismatch".to_string());
   170→            resp.message = Some(format!(
   171→                "unsupported protocol version {:?}, expected {:?}",
   172→                req.protocol_version,
   173→                protocol::PROTOCOL_VERSION,
   174→            ));
   175→            return (resp, false);
   176→        }
   177→
   178→        match req.command.as_str() {
   179→            "handshake" => (self.handle_handshake(resp), false),
   180→            "analyze" => (self.handle_analyze(resp, req), false),
   181→            "instrument" => (self.handle_instrument(resp, req), false),
   182→            "execute" => (self.handle_execute(resp, req), false),
   183→            "setup" => (self.handle_setup(resp, req), false),
   184→            "teardown" => (self.handle_teardown(resp, req), false),
   185→            "generate" => (self.handle_generate(resp, req), false),
   186→            "shutdown" => (self.handle_shutdown(resp), true),
   187→            _ => {
   188→                resp.status = "error".to_string();
   189→                resp.code = Some("invalid_request".to_string());
   190→                resp.message = Some(format!("unknown command: {}", req.command));
   191→                (resp, false)
   192→            }
   193→        }
   194→    }
   195→
   196→    fn handle_handshake(&self, mut resp: Response) -> Response {
   197→        resp.status = "handshake".to_string();
   198→        resp.frontend_version = Some(FRONTEND_VERSION.to_string());
   199→        resp.language = Some(FRONTEND_LANGUAGE.to_string());
   200→        resp.capabilities = Some(vec![
   201→            "analyze".to_string(),
   202→            "execute".to_string(),
   203→            "generate".to_string(),
   204→            "instrument".to_string(),
   205→        ]);
   206→        resp
   207→    }
   208→
   209→    fn handle_analyze(&mut self, mut resp: Response, req: &Request) -> Response {
   210→        let file_path = match &req.file {
   211→            Some(f) => f,
   212→            None => {
   213→                resp.status = "error".to_string();
   214→                resp.code = Some("invalid_request".to_string());
   215→                resp.message = Some("analyze command requires a file path".to_string());
   216→                return resp;
   217→            }
   218→        };
   219→
   220→        self.last_file = Some(file_path.clone());
   221→
   222→        let path = std::path::Path::new(file_path);
   223→        if !path.exists() {
   224→            resp.status = "error".to_string();
   225→            resp.code = Some("file_not_found".to_string());
   226→            resp.message = Some(format!("file not found: {file_path}"));
   227→            return resp;
   228→        }
   229→
   230→        match crate::analyzer::analyze_file(path, req.function.as_deref()) {
   231→            Ok(functions) => {
   232→                resp.status = "analyze".to_string();
   233→                resp.functions = Some(functions);
   234→                resp
   235→            }
   236→            Err(e) => {
   237→                resp.status = "error".to_string();
   238→                resp.code = Some(
   239→                    match &e {
   240→                        crate::analyzer::AnalyzeError::FileNotFound(_) => "file_not_found",
   241→                        crate::analyzer::AnalyzeError::ReadError(_) => "internal_error",
   242→                        crate::analyzer::AnalyzeError::ParseError(_) => "parse_error",
   243→                        crate::analyzer::AnalyzeError::FunctionNotFound(_) => "function_not_found",
   244→                    }
   245→                    .to_string(),
   246→                );
   247→                resp.message = Some(e.to_string());
   248→                resp
   249→            }
   250→        }
   251→    }

   253→    fn handle_instrument(&mut self, mut resp: Response, req: &Request) -> Response {
   254→        let file_path = match &req.file {
   255→            Some(f) => f,
   256→            None => {
   257→                resp.status = "error".to_string();
   258→                resp.code = Some("invalid_request".to_string());
   259→                resp.message = Some("instrument command requires a file path".to_string());
   260→                return resp;
   261→            }
   262→        };
   263→
   264→        self.last_file = Some(file_path.clone());
   265→
   266→        let path = std::path::Path::new(file_path);
   267→        if !path.exists() {
   268→            resp.status = "error".to_string();
   269→            resp.code = Some("file_not_found".to_string());
   270→            resp.message = Some(format!("file not found: {file_path}"));
   271→            return resp;
   272→        }
   273→
   274→        match crate::instrument::instrument_file(path, req.function.as_deref()) {
   275→            Ok(result) => {
   276→                // Write instrumented source to a temp file
   277→                let source_name = path
   278→                    .file_name()
   279→                    .and_then(|n| n.to_str())
   280→                    .unwrap_or("instrumented.rs");
   281→                match write_instrumented_temp(source_name, &result.source) {
   282→                    Ok(output_path) => {
   283→                        resp.status = "instrument".to_string();
   284→                        resp.instrumented = Some(true);
   285→                        resp.output_file = Some(output_path);
   286→                        resp.message = Some(format!(
   287→                            "instrumented {} branch points",
   288→                            result.branch_count
   289→                        ));
   290→                        resp
   291→                    }
   292→                    Err(e) => {
   293→                        resp.status = "error".to_string();
   294→                        resp.code = Some("internal_error".to_string());
   295→                        resp.message = Some(format!("failed to write instrumented output: {e}"));
   296→                        resp
   297→                    }
   298→                }
   299→            }
   300→            Err(e) => {
   301→                resp.status = "error".to_string();
   302→                resp.code = Some(match &e {
   303→                    crate::instrument::InstrumentError::FileNotFound(_) => "file_not_found",
   304→                    crate::instrument::InstrumentError::ReadError(_) => "internal_error",
   305→                    crate::instrument::InstrumentError::ParseError(_) => "parse_error",
   306→                }.to_string());
   307→                resp.message = Some(e.to_string());
   308→                resp
   309→            }
   310→        }
   311→    }

   313→    fn handle_execute(&self, mut resp: Response, req: &Request) -> Response {
   314→        let file_path = match req.file.as_ref().or(self.last_file.as_ref()) {
   315→            Some(f) => f,
   316→            None => {
   317→                resp.status = "error".to_string();
   318→                resp.code = Some("invalid_request".to_string());
   319→                resp.message = Some("execute command requires a file path (none provided and no prior analyze/instrument)".to_string());
   320→                return resp;
   321→            }
   322→        };
   323→        let function_name = match &req.function {
   324→            Some(f) => f,
   325→            None => {
   326→                resp.status = "error".to_string();
   327→                resp.code = Some("invalid_request".to_string());
   328→                resp.message = Some("execute command requires a function name".to_string());
   329→                return resp;
   330→            }
   331→        };
   332→
   333→        if self.log_level >= FrontendLogLevel::Debug {
   334→            eprintln!(
   335→                "[shatter-rust] Executing {function_name} in {file_path} with {} inputs",
   336→                req.inputs.len()
   337→            );
   338→        }
   339→
   340→        match crate::executor::execute_function(
   341→            file_path,
   342→            function_name,
   343→            &req.inputs,
   344→            &req.mocks,
   345→            self.exec_timeout_ms,
   346→        ) {
   347→            Ok(result) => {
   348→                resp.status = "execute".to_string();
   349→                resp.return_value = result.return_value;
   350→                resp.thrown_error = result.thrown_error;
   351→                resp.branch_path = Some(result.branch_path);
   352→                resp.lines_executed = Some(result.lines_executed);
   353→                resp.calls_to_external = Some(result.calls_to_external);
   354→                resp.path_constraints = Some(result.path_constraints);
   355→                resp.side_effects = Some(result.side_effects);
   356→                resp.performance = Some(result.performance);
   357→                resp
   358→            }
   359→            Err(crate::executor::ExecuteError::FileError(msg)) => {
   360→                resp.status = "error".to_string();
   361→                resp.code = Some("file_not_found".to_string());
   362→                resp.message = Some(msg);
   363→                resp
   364→            }
   365→            Err(crate::executor::ExecuteError::CompilationFailed(msg)) => {
   366→                resp.status = "error".to_string();
   367→                resp.code = Some("compilation_error".to_string());
   368→                resp.message = Some(msg);
   369→                resp
   370→            }
   371→            Err(crate::executor::ExecuteError::NonExecutable(msg)) => {
   372→                resp.status = "error".to_string();
   373→                resp.code = Some("non_executable".to_string());
   374→                resp.message = Some(msg);
   375→                resp
   376→            }
   377→            Err(e) => {
   378→                resp.status = "error".to_string();
   379→                resp.code = Some("internal_error".to_string());
   380→                resp.message = Some(e.to_string());
   381→                resp
   382→            }
   383→        }
   384→    }

   477→    fn handle_generate(&mut self, mut resp: Response, req: &Request) -> Response {
   478→        let file_path = match &req.file {
   479→            Some(f) => f,
   480→            None => {
   481→                resp.status = "error".to_string();
   482→                resp.code = Some("invalid_request".to_string());
   483→                resp.message = Some("generate command requires a file path".to_string());
   484→                return resp;
   485→            }
   486→        };
   487→        let func_name = match &req.name {
   488→            Some(n) => n.clone(),
   489→            None => {
   490→                resp.status = "error".to_string();
   491→                resp.code = Some("invalid_request".to_string());
   492→                resp.message = Some("generate command requires a name".to_string());
   493→                return resp;
   494→            }
   495→        };
   496→
   497→        let path = std::path::Path::new(file_path);
   498→        let ext = path.extension().and_then(|e| e.to_str());
   499→
   500→        match ext {
   501→            Some("wasm") => {
   502→                match self.wasm_cache.generate(path, &func_name, req.recipe.as_ref()) {
   503→                    Ok((value, generator_id, recipe)) => {
   504→                        resp.status = "generate".to_string();
   505→                        resp.value = Some(value);
   506→                        resp.generator_id = Some(generator_id);
   507→                        resp.recipe = recipe;
   508→                        resp
   509→                    }
   510→                    Err(e) => {
   511→                        resp.status = "error".to_string();
   512→                        resp.code = Some("internal_error".to_string());
   513→                        resp.message = Some(e);
   514→                        resp
   515→                    }
   516→                }
   517→            }
   518→            Some("rs") => {
   519→                if let Some(ref registry) = self.native_registry {
   520→                    match registry.generate(&func_name, req.recipe.clone()) {
   521→                        Ok((value, generator_id, recipe)) => {
   522→                            resp.status = "generate".to_string();
   523→                            resp.value = Some(value);
   524→                            resp.generator_id = Some(generator_id);
   525→                            resp.recipe = Some(recipe);
   526→                            resp
   527→                        }
   528→                        Err(e) => {
   529→                            resp.status = "error".to_string();
   530→                            resp.code = Some("internal_error".to_string());
   531→                            resp.message = Some(e);
   532→                            resp
   533→                        }
   534→                    }
   535→                } else {
   536→                    resp.status = "error".to_string();
   537→                    resp.code = Some("internal_error".to_string());
   538→                    resp.message = Some(format!(
   539→                        "native generator {func_name:?} requires a custom build (run `shatter build-frontend rust`)"
   540→                    ));
   541→                    resp
   542→                }
   543→            }
   544→            _ => {
   545→                resp.status = "error".to_string();
   546→                resp.code = Some("invalid_request".to_string());
   547→                resp.message = Some(format!(
   548→                    "unsupported generator file type: {}",
   549→                    file_path
   550→                ));
   551→                resp
   552→            }
   553→        }
   554→    }
   555→
   556→    fn handle_shutdown(&self, mut resp: Response) -> Response {
   557→        resp.status = "shutdown_ack".to_string();
   558→        resp
   559→    }
   560→
   561→    fn send(&mut self, resp: &Response) -> io::Result<()> {
   562→        let data = serde_json::to_string(resp)
   563→            .map_err(io::Error::other)?;
   564→        self.logf(&format!("Sent: {data}"));
   565→        writeln!(self.writer, "{data}")?;
   566→        self.writer.flush()
   567→    }
   568→
   569→    /// Log at TRACE level (protocol messages).
   570→    fn logf(&mut self, msg: &str) {
   571→        self.log_at(FrontendLogLevel::Trace, msg);
   572→    }
   573→
   574→    /// Log at a specific level.
   575→    fn log_at(&mut self, level: FrontendLogLevel, msg: &str) {
   576→        if self.log_level >= level {
   577→            let _ = writeln!(self.log, "[shatter-rust] {msg}");
   578→        }
   579→    }
   580→}

   587→    fn send_recv(req_json: &str) -> Response {
   588→        let input = format!("{req_json}\n");
   589→        let mut output = Vec::new();
   590→        let handler = Handler::new(
   591→            input.as_bytes(),
   592→            &mut output,
   593→            io::sink(),
   594→        );
   595→        handler.run().expect("handler.run");
   596→        let output_str = String::from_utf8(output).expect("valid utf8");
   597→        let first_line = output_str.lines().next().expect("at least one response line");
   598→        serde_json::from_str(first_line).expect("valid JSON response")
   599→    }
   600→
   601→    /// Send multiple requests and return all responses.
   602→    fn conversation(requests: &[&str]) -> Vec<Response> {
   603→        let input = requests.join("\n") + "\n";
   604→        let mut output = Vec::new();
   605→        let handler = Handler::new(
   606→            input.as_bytes(),
   607→            &mut output,
   608→            io::sink(),
   609→        );
   610→        handler.run().expect("handler.run");
   611→        let output_str = String::from_utf8(output).expect("valid utf8");
   612→        output_str
   613→            .lines()
   614→            .filter(|l| !l.is_empty())
   615→            .map(|l| serde_json::from_str(l).expect("valid JSON response"))
   616→            .collect()
   617→    }
