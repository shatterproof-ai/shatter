     1→//! Concolic execution loop: worklist-driven path exploration with Z3 solving.
     2→//!
     3→//! The orchestrator drives the concolic testing cycle in two-phase rounds:
     4→//!
     5→//! **Observe phase** — drain the worklist, execute all pending inputs via the
     6→//! frontend, classify each execution as new-path or duplicate.
     7→//!
     8→//! **Solve/Generate phase** — for each new-path observation, extract symbolic
     9→//! constraints, negate branches with Z3, fuzz unknown constraints, and drill
    10→//! stalled frontiers. The resulting candidate inputs feed the next round's
    11→//! worklist.
    12→//!
    13→//! The outer loop iterates: Observe → Solve/Generate → feed candidates → next
    14→//! Observe round, until a termination condition fires (budget, plateau, or
    15→//! worklist exhaustion).
    16→

    40→/// Parsed frontend capabilities from the handshake response.
    41→///
    42→/// During handshake, frontends declare which commands they support and which
    43→/// complex types they can reconstruct. The core uses this to avoid generating
    44→/// complex-typed inputs the frontend can't handle.
    45→#[derive(Debug, Clone, Default)]
    46→pub struct FrontendCapabilities {
    47→    /// Standard commands the frontend supports ("analyze", "execute", etc.).
    48→    pub commands: HashSet<String>,
    49→    /// Complex types the frontend can reconstruct from `__complex_type` JSON.
    50→    pub complex_types: HashSet<ComplexKind>,
    51→}
    52→
    53→impl FrontendCapabilities {
    54→    /// Parse raw capability strings from a handshake response.
    55→    ///
    56→    /// Strings prefixed with `"complex_type:"` are parsed as `ComplexKind` values.
    57→    /// All other strings are treated as command capabilities.
    58→    /// Unknown complex type names are silently ignored.
    59→    pub fn from_raw(capabilities: &[String]) -> Self {
    60→        let mut commands = HashSet::new();
    61→        let mut complex_types = HashSet::new();
    62→        for cap in capabilities {
    63→            if let Some(kind_str) = cap.strip_prefix("complex_type:") {
    64→                // ComplexKind uses serde rename_all = "snake_case", so we
    65→                // deserialize the bare string as a JSON string value.
    66→                if let Ok(kind) = serde_json::from_value::<ComplexKind>(
    67→                    serde_json::Value::String(kind_str.to_string()),
    68→                ) {
    69→                    complex_types.insert(kind);
    70→                }
    71→                // Silently ignore unknown complex type names
    72→            } else {
    73→                commands.insert(cap.clone());
    74→            }
    75→        }
    76→        Self {
    77→            commands,
    78→            complex_types,
    79→        }
    80→    }
    81→
    82→    /// Check whether the frontend declared support for a specific complex type.
    83→    pub fn supports_complex(&self, kind: ComplexKind) -> bool {
    84→        self.complex_types.contains(&kind)
    85→    }
    86→}

   132→/// How an input was generated — determines worklist priority.
   133→#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
   134→pub enum InputSource {
   135→    /// Least priority: initial seed values.
   136→    Seed = 0,
   137→    /// Low priority: fuzzed from concrete values of unknown constraints.
   138→    Fuzzed = 1,
   139→    /// Between fuzz and drill: interpolated between true/false witnesses.
   140→    BoundarySearch = 2,
   141→    /// Medium priority: targeted mutation of blocking params on a stalled frontier.
   142→    Drilled = 3,
   143→    /// High priority: Z3-solved inputs targeting a specific branch.
   144→    Z3Solved = 4,
   145→    /// Highest priority: user-provided candidate inputs from `.shatter/` config.
   146→    UserProvided = 5,
   147→}

   149→/// An entry in the exploration worklist.
   150→#[derive(Debug, Clone)]
   151→pub struct WorklistEntry {
   152→    /// Input values to pass to the function.
   153→    pub inputs: Vec<serde_json::Value>,
   154→    /// How these inputs were generated.
   155→    pub source: InputSource,
   156→    /// Optional fitness score (0.0–1.0) from genetic scoring.
   157→    ///
   158→    /// When present, fitness is the primary ordering key for the worklist's
   159→    /// BinaryHeap. When absent (`None`), the entry falls back to source-based
   160→    /// ordering, which preserves backward compatibility with the pre-genetic
   161→    /// pipeline.
   162→    pub fitness: Option<f64>,
   163→}
   164→
   165→impl Eq for WorklistEntry {}
   166→
   167→impl PartialEq for WorklistEntry {
   168→    fn eq(&self, other: &Self) -> bool {
   169→        self.source == other.source && self.fitness_key() == other.fitness_key()
   170→    }
   171→}
   172→
   173→impl PartialOrd for WorklistEntry {
   174→    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
   175→        Some(self.cmp(other))
   176→    }
   177→}
   178→
   179→impl Ord for WorklistEntry {
   180→    /// Primary ordering: fitness score (higher is better). Entries with a
   181→    /// fitness score always outrank entries without one. Among entries without
   182→    /// fitness, the original source-based priority applies.
   183→    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
   184→        match (self.fitness, other.fitness) {
   185→            (Some(_), None) => std::cmp::Ordering::Greater,
   186→            (None, Some(_)) => std::cmp::Ordering::Less,
   187→            (Some(_), Some(_)) => self
   188→                .fitness_key()
   189→                .cmp(&other.fitness_key())
   190→                .then_with(|| self.source.cmp(&other.source)),
   191→            (None, None) => self.source.cmp(&other.source),
   192→        }
   193→    }
   194→}
   195→
   196→impl WorklistEntry {
   197→    /// Convert fitness f64 to an integer key for total ordering.
   198→    ///
   199→    /// Multiplies by 1_000_000 and truncates to i64 so that BinaryHeap
   200→    /// (which requires Ord) can rank by fitness without floating-point
   201→    /// comparison issues.
   202→    fn fitness_key(&self) -> i64 {
   203→        self.fitness.map_or(0, |f| (f * 1_000_000.0) as i64)
   204→    }
   205→}

   298→/// Compute a hash of the branch path (branch_id + taken pairs) to identify unique paths.
   299→pub fn hash_branch_path(branch_path: &[crate::execution_record::BranchDecision]) -> u64 {
   300→    let mut hasher = std::hash::DefaultHasher::new();
   301→    for decision in branch_path {
   302→        decision.branch_id.hash(&mut hasher);
   303→        decision.taken.hash(&mut hasher);
   304→    }
   305→    hasher.finish()
   306→}

   308→/// Extract the list of solvable `SymExpr` path conditions from an execution result's branch path.
   309→///
   310→/// Each constraint is adjusted based on the `taken` field so that negating it
   311→/// in the solver produces inputs that flip the branch:
   312→/// - `taken=true`: the path condition is the constraint itself
   313→/// - `taken=false`: the path condition is `NOT(constraint)`, because the branch
   314→///   condition evaluated to false
   315→///
   316→/// Returns `None` for branches with `Unknown` constraints; those are skipped
   317→/// by the solver but may be targeted by fuzzing.
   318→pub(crate) fn extract_sym_constraints(result: &ExecuteResult) -> Vec<Option<SymExpr>> {
   319→    result
   320→        .branch_path
   321→        .iter()
   322→        .map(|decision| match &decision.constraint {
   323→            SymConstraint::Expr { expr } => {
   324→                if decision.taken {
   325→                    Some(expr.clone())
   326→                } else {
   327→                    // The branch was not taken, so the actual path condition
   328→                    // is NOT(constraint). Wrapping it here ensures that when
   329→                    // the solver negates this entry, it produces the raw
   330→                    // constraint — i.e., the condition needed to take the branch.
   331→                    Some(SymExpr::UnOp {
   332→                        op: crate::sym_expr::UnOpKind::Not,
   333→                        operand: Box::new(expr.clone()),
   334→                    })
   335→                }
   336→            }
   337→            SymConstraint::Unknown { .. } => None,
   338→        })
   339→        .collect()
   340→}

   342→/// Convert Z3 `ConcreteValue`s back into JSON values suitable for the Execute protocol.
   343→///
   344→/// For `Complex` values, produces a `__complex_type` tagged JSON object.
   345→/// The solver unwraps complex types to their repr for solving, but when
   346→/// converting back to JSON we need to re-wrap with the type tag so the
   347→/// frontend can reconstruct the native value.
   348→pub(crate) fn concrete_to_json(value: &ConcreteValue) -> serde_json::Value {
   349→    match value {
   350→        ConcreteValue::Int(i) => serde_json::json!(*i),
   351→        ConcreteValue::Float(f) => serde_json::json!(*f),
   352→        ConcreteValue::Str(s) => serde_json::json!(s),
   353→        ConcreteValue::Bool(b) => serde_json::json!(*b),
   354→        ConcreteValue::Complex { kind, repr } => {
   355→            // Serialize the kind to its snake_case name for the wire format
   356→            let kind_str = serde_json::to_value(kind)
   357→                .ok()
   358→                .and_then(|v| v.as_str().map(|s| s.to_string()))
   359→                .unwrap_or_else(|| format!("{kind:?}").to_lowercase());
   360→            let repr_json = concrete_to_json(repr);
   361→            // Build tagged wire format: {"__complex_type": "<kind>", "value": <repr>}
   362→            serde_json::json!({
   363→                "__complex_type": kind_str,
   364→                "value": repr_json,
   365→            })
   366→        }
   367→    }
   368→}

   370→/// Build a new input vector by overlaying Z3-solved values onto existing inputs.
   371→///
   372→/// The solver returns variable names like "x", "config.timeout" etc. We match
   373→/// these against parameter names in `base_inputs` (positionally). For now we
   374→/// support simple flat parameters — if the variable name matches the parameter
   375→/// index convention (param_0, param_1, …) or the base is a single param, we
   376→/// update it directly.
   377→#[requires(base_inputs.len() == param_names.len(), "base_inputs and param_names must be positionally aligned")]
   378→#[contracts::ensures(ret.len() == base_inputs.len(), "overlay must preserve input vector length")]
   379→pub(crate) fn overlay_solved_values(
   380→    base_inputs: &[serde_json::Value],
   381→    solved: &std::collections::HashMap<String, ConcreteValue>,
   382→    param_names: &[String],
   383→) -> Vec<serde_json::Value> {
   384→    let mut result = base_inputs.to_vec();
   385→
   386→    for (var_name, value) in solved {
   387→        // Try to match variable name to a parameter by name.
   388→        if let Some(idx) = param_names.iter().position(|n| n == var_name) {
   389→            if idx < result.len() {
   390→                result[idx] = concrete_to_json(value);
   391→            }
   392→        } else if param_names.len() == 1 && base_inputs.len() == 1 && !var_name.contains('.') {
   393→            // Single-param function with a simple (non-derived) variable name:
   394→            // the solver variable likely refers to the param. Skip derived names
   395→            // like "email.length" which are internal Z3 variables, not params.
   396→            result[0] = concrete_to_json(value);
   397→        }
   398→    }
   399→
   400→    result
   401→}

   403→/// Result of trying to observe a single worklist entry.
   404→enum ObserveOneResult {
   405→    /// Entry was executed and produced an observation.
   406→    Observed(Box<Observation>),
   407→    /// Entry was skipped by triage prediction.
   408→    TriageSkipped,
   409→    /// Frontend returned an error or unexpected response — entry skipped.
   410→    FrontendSkipped,
   411→    /// A termination budget was hit before executing.
   412→    Terminated(TerminationReason),
   413→}
   414→
   415→/// Execute a single worklist entry and classify the result.
   416→///
   417→/// Returns an `Observation` with path classification, or a skip/termination
   418→/// indicator. The caller is responsible for updating coverage state afterward.
   419→async fn observe_one(
   420→    entry: &WorklistEntry,
   421→    frontend: &mut Frontend,
   422→    function_name: &str,
   423→    config: &ExploreConfig,
   424→    covered_paths: &mut HashSet<u64>,
   425→    triage_state: &mut TriageState,
   426→    budget: &ExploreBudget,
   427→) -> Result<ObserveOneResult, ExploreError> {
   428→    // Check termination budgets.
   429→    if budget.unique_paths >= config.max_iterations {
   430→        return Ok(ObserveOneResult::Terminated(TerminationReason::MaxIterations));
   431→    }
   432→    if budget.total_executions >= config.max_executions {
   433→        return Ok(ObserveOneResult::Terminated(TerminationReason::MaxExecutions));
   434→    }
   435→    if let Some(timeout) = config.timeout_explore
   436→        && budget.explore_start.elapsed() >= timeout
   437→    {
   438→        return Ok(ObserveOneResult::Terminated(TerminationReason::TimeoutExplore));
   439→    }
   440→    if config.plateau_threshold > 0 && budget.plateau_counter >= config.plateau_threshold {
   441→        return Ok(ObserveOneResult::Terminated(TerminationReason::CoveragePlateau));
   442→    }
   443→
   444→    // Triage: predict whether this input will produce a novel path.
   445→    let is_sampled_skip = if entry.source != InputSource::UserProvided {
   446→        let verdict = triage_state.triage_candidate(&entry.inputs, covered_paths);
   447→        triage_state.record_verdict(&verdict);
   448→        if verdict == TriageVerdict::Skip {
   449→            if triage_state.should_sample() {
   450→                true
   451→            } else {
   452→                return Ok(ObserveOneResult::TriageSkipped);
   453→            }
   454→        } else {
   455→            false
   456→        }
   457→    } else {
   458→        false
   459→    };
   460→
   461→    // Execute concretely via the frontend.
   462→    let response = frontend
   463→        .send(Command::Execute {
   464→            function: function_name.to_string(),
   465→            inputs: entry.inputs.clone(),
   466→            mocks: config.mocks.clone(),
   467→            setup_context: None,
   468→        })
   469→        .await?;
   470→
   471→    let exec_result = match response.result {
   472→        ResponseResult::Execute(result) => *result,
   473→        ResponseResult::Error { message, .. } => {
   474→            log::warn!("frontend error during execute: {message}");
   475→            return Ok(ObserveOneResult::FrontendSkipped);
   476→        }
   477→        _ => return Ok(ObserveOneResult::FrontendSkipped),
   478→    };
   479→
   480→    let path_id = hash_branch_path(&exec_result.branch_path);
   481→
   482→    // Validate sampled skip prediction before modifying covered_paths.
   483→    if is_sampled_skip {
   484→        let already_covered = covered_paths.contains(&path_id);
   485→        triage_state.record_sample(0, if already_covered { 0 } else { 1 });
   486→    }
   487→
   488→    let is_new_path = covered_paths.insert(path_id);
   489→    if is_new_path {
   490→        triage_state.update(&exec_result.branch_path);
   491→    }
   492→
   493→    Ok(ObserveOneResult::Observed(Box::new(Observation {
   494→        inputs: entry.inputs.clone(),
   495→        result: exec_result,
   496→        source: entry.source,
   497→        path_id,
   498→        is_new_path,
   499→        is_sampled_skip,
   500→    })))
   501→}

   503→/// Budget counters checked by `observe_one` to enforce termination limits.
   504→struct ExploreBudget {
   505→    unique_paths: usize,
   506→    total_executions: usize,
   507→    plateau_counter: usize,
   508→    explore_start: Instant,

   511→/// Solve/Generate phase: process new-path observations and produce candidate inputs.
   512→///
   513→/// For each new-path observation:
   514→/// - Extract symbolic constraints and negate each with Z3
   515→/// - Try boundary search for Unknown constraints (interpolate between witnesses)
   516→/// - Fall back to blind fuzzing for Unknown constraints without witnesses
   517→/// - Drill stalled frontiers with targeted mutations
   518→#[allow(clippy::too_many_arguments)] // boundary search + fitness scoring need broad context
   519→fn solve_and_generate(
   520→    observations: &[Observation],
   521→    frontier_set: &mut FrontierSet,
   522→    param_infos: &[ParamInfo],
   523→    param_names: &[String],
   524→    raw_results: &[(Vec<serde_json::Value>, ExecuteResult)],
   525→    seen_branch_sides: &std::collections::HashSet<(u32, bool)>,
   526→    config: &ExploreConfig,
   527→    rng: &mut StdRng,
   528→    target_branches: &HashSet<u32>,
   529→    fitness_context: &mut FitnessContext,
   530→    fitness_weights: &FitnessWeights,
   531→) -> SolveOutput {
   532→    let mut output = SolveOutput::default();
   533→
   534→    for obs in observations.iter().filter(|o| o.is_new_path) {
   535→        // Extract symbolic constraints from the branch path.
   536→        let sym_constraints = extract_sym_constraints(&obs.result);
   537→
   538→        // Collect solvable constraints with their original branch_path indices.
   539→        let solvable_with_idx: Vec<(usize, SymExpr)> = sym_constraints
   540→            .iter()
   541→            .enumerate()
   542→            .filter_map(|(i, c)| c.as_ref().map(|e| (i, e.clone())))
   543→            .collect();
   544→        let solvable: Vec<SymExpr> = solvable_with_idx.iter().map(|(_, e)| e.clone()).collect();
   545→
   546→        // Try to negate each branch constraint with Z3.
   547→        if !solvable.is_empty() {
   548→            for (solve_idx, &(branch_idx, _)) in solvable_with_idx.iter().enumerate() {
   549→                match solver::solve_for_new_path(&solvable, solve_idx, config.solver_timeout_ms, param_infos) {
   550→                    Ok(SolveResult::Sat(values)) => {
   551→                        let new_inputs =
   552→                            overlay_solved_values(&obs.inputs, &values, param_names);
   553→                        output.candidates.push(WorklistEntry {
   554→                            inputs: new_inputs,
   555→                            source: InputSource::Z3Solved,
   556→                            fitness: None,
   557→                        });
   558→                        output.z3_count += 1;
   559→                    }
   560→                    Ok(SolveResult::Unsat) => {
   561→                        if let Some(bd) = obs.result.branch_path.get(branch_idx) {
   562→                            frontier_set.increment_stall(bd.branch_id);
   563→                        }
   564→                    }
   565→                    Err(_) => {
   566→                        if let Some(bd) = obs.result.branch_path.get(branch_idx) {
   567→                            frontier_set.increment_stall(bd.branch_id);
   568→                        }
   569→                    }
   570→                }
   571→            }
   572→        }
   573→
   574→        // For Unknown constraints, try boundary search first (interpolate between
   575→        // true/false witnesses), then fall back to blind fuzzing.
   576→        let mut boundary_attempted = false;
   577→        let mut boundary_branches = 0usize;
   578→        for (i, constraint_opt) in sym_constraints.iter().enumerate() {
   579→            if constraint_opt.is_none() && i < obs.result.branch_path.len() {
   580→                let bd = &obs.result.branch_path[i];
   581→                let opposite_seen =
   582→                    seen_branch_sides.contains(&(bd.branch_id, !bd.taken));
   583→
   584→                if opposite_seen
   585→                    && let Some((tw, fw)) =
   586→                        boundary_search::find_witness_pair(raw_results, bd.branch_id)
   587→                {
   588→                    let candidates = boundary_search::interpolate_inputs(
   589→                        &tw,
   590→                        &fw,
   591→                        param_infos,
   592→                        boundary_search::MAX_BOUNDARY_STEPS,
   593→                    );
   594→                    for interp in candidates {
   595→                        output.candidates.push(WorklistEntry {
   596→                            inputs: interp,
   597→                            source: InputSource::BoundarySearch,
   598→                            fitness: None,
   599→                        });
   600→                        output.boundary_count += 1;
   601→                    }
   602→                    boundary_attempted = true;
   603→                    boundary_branches += 1;
   604→                    if boundary_branches
   605→                        >= boundary_search::MAX_BOUNDARY_BRANCHES_PER_ROUND
   606→                    {
   607→                        break;
   608→                    }
   609→                }
   610→            }
   611→        }
   612→
   613→        // Fall back to type-aware mutation if boundary search wasn't applicable.
   614→        //
   615→        // Uses input_gen::mutate_inputs (type-aware, respects param types) for
   616→        // several rounds, plus the legacy fuzz_inputs as a cheap supplement.
   617→        if !boundary_attempted {
   618→            for (i, constraint_opt) in sym_constraints.iter().enumerate() {
   619→                if constraint_opt.is_none() && i < obs.result.branch_path.len() {
   620→                    // Type-aware mutations (3 rounds at full mutation rate).
   621→                    for _ in 0..MUTATE_ROUNDS_PER_UNKNOWN {
   622→                        let mutated = input_gen::mutate_inputs(
   623→                            &obs.inputs,
   624→                            param_infos,
   625→                            MUTATE_RATE_UNKNOWN,
   626→                            &[],
   627→                            rng,
   628→                        );
   629→                        output.candidates.push(WorklistEntry {
   630→                            inputs: mutated,
   631→                            source: InputSource::Fuzzed,
   632→                            fitness: None,
   633→                        });
   634→                        output.fuzz_count += 1;
   635→                    }
   636→                    // Legacy deterministic fuzz as supplement.
   637→                    for fuzzed in fuzz_inputs(&obs.inputs) {
   638→                        output.candidates.push(WorklistEntry {
   639→                            inputs: fuzzed,
   640→                            source: InputSource::Fuzzed,
   641→                            fitness: None,
   642→                        });
   643→                        output.fuzz_count += 1;
   644→                    }
   645→                    // Only fuzz once per observation (avoid exponential blowup).
   646→                    break;
   647→                }
   648→            }
   649→        }
   650→    }
   651→
   652→    // Parameter drilling: for stalled frontiers, generate targeted mutations.
   653→    {
   654→        let mut stalled: Vec<Frontier> = frontier_set
   655→            .iter()
   656→            .filter(|f| {
   657→                f.stall_count >= drilling::DRILL_STALL_THRESHOLD
   658→                    && f.stall_count < crate::frontier::DEFAULT_MAX_STALL
   659→            })
   660→            .cloned()
   661→            .collect();
   662→        stalled.sort_by(|a, b| a.depth.cmp(&b.depth).then(a.stall_count.cmp(&b.stall_count)));
   663→        stalled.truncate(drilling::MAX_FRONTIERS_PER_ROUND);
   664→
   665→        for frontier in &stalled {
   666→            let count =
   667→                drilling::DRILL_MUTATIONS_PER_PARAM * frontier.blocking_params.len().max(1);
   668→            let drilled = drilling::generate_drilled_inputs(
   669→                &frontier.best_prefix,
   670→                &frontier.blocking_params,
   671→                param_infos,
   672→                count,
   673→                rng,
   674→            );
   675→            for inputs in drilled {
   676→                output.candidates.push(WorklistEntry {
   677→                    inputs,
   678→                    source: InputSource::Drilled,
   679→                    fitness: None,
   680→                });
   681→                output.drill_count += 1;
   682→            }
   683→            frontier_set.increment_stall(frontier.branch_id);
   684→        }
   685→    }
   686→
   687→    // Score each candidate using genetic fitness. The parent observation's
   688→    // branch path gives us approximate fitness context: candidates derived
   689→    // from high-fitness executions should be explored first.
   690→    if !target_branches.is_empty() {
   691→        for candidate in &mut output.candidates {
   692→            // Create a synthetic ExecuteResult from the parent observation's
   693→            // branch path to estimate fitness. This gives candidates a
   694→            // relative ranking even before execution.
   695→            if let Some(parent_obs) = observations.iter().find(|o| o.is_new_path) {
   696→                let breakdown = crate::genetic_fitness::score(
   697→                    &parent_obs.result,
   698→                    target_branches,
   699→                    fitness_context,
   700→                    fitness_weights,
   701→                );
   702→                candidate.fitness = Some(breakdown.total);
   703→            }
   704→        }
   705→    }
   706→
   707→    output
   708→}

   710→/// Run the concolic exploration loop on a function via a frontend subprocess.
   711→///
   712→/// The loop alternates between two phases per round:
   713→/// 1. **Observe** — drain the worklist, execute all inputs, classify paths
   714→/// 2. **Solve/Generate** — extract constraints from new paths, produce candidates
   715→///
   716→/// `function_name` is the fully-qualified name of the function to explore.
   717→/// `seed_inputs` provides initial input sets to begin exploration.
   718→/// `user_inputs` provides user-provided candidate inputs (highest priority).
   719→/// `param_infos` provides parameter metadata including names and types.
   720→pub async fn explore(
   721→    frontend: &mut Frontend,
   722→    function_name: &str,
   723→    seed_inputs: Vec<Vec<serde_json::Value>>,
   724→    user_inputs: Vec<Vec<serde_json::Value>>,
   725→    param_infos: &[ParamInfo],
   726→    config: &ExploreConfig,
   727→) -> Result<ExploreResult, ExploreError> {
   728→    let param_names: Vec<String> = param_infos.iter().map(|p| p.name.clone()).collect();
   729→    let mut worklist = BinaryHeap::new();
   730→    let mut covered_paths: HashSet<u64> = HashSet::new();
   731→    let mut executions = Vec::new();
   732→    let mut raw_results: Vec<(Vec<serde_json::Value>, ExecuteResult)> = Vec::new();
   733→    let mut seen_branch_ids: HashSet<u32> = HashSet::new();
   734→    let mut discoveries: Vec<(u32, DiscoveryMethod)> = Vec::new();
   735→    let mut total_executions: usize = 0;
   736→    let mut z3_generated: usize = 0;
   737→    let mut fuzz_generated: usize = 0;
   738→    let mut boundary_generated: usize = 0;
   739→    let mut drill_generated: usize = 0;
   740→    let mut termination_reason = TerminationReason::WorklistExhausted;
   741→    let mut seen_branch_sides: HashSet<(u32, bool)> = HashSet::new();
   742→    let mut frontier_set = FrontierSet::new();
   743→    let mut rng = StdRng::from_os_rng();
   744→    let mut triage_state = TriageState::new(param_names.clone());
   745→    let mut triage_skipped: usize = 0;
   746→    let mut triage_mispredictions: usize = 0;
   747→
   748→    // Fitness context shares novelty state with covered_paths — the
   749→    // orchestrator marks paths as seen in both sets whenever a new path
   750→    // is discovered, keeping FitnessContext's novelty scoring in sync.
   751→    let mut fitness_context = FitnessContext::new();
   752→    let fitness_weights = FitnessWeights::default();
   753→
   754→    // Target branches: branch IDs seen so far on only one side (not yet
   755→    // covered on the opposite). Updated after each new-path observation.
   756→    // Used by fitness scoring to compute proximity/coverage scores.
   757→    let mut target_branches: HashSet<u32> = HashSet::new();
   758→
   759→    // MetaStrategy integration point: strategies will be registered here
   760→    // by a follow-up issue that replaces the hardcoded seed/fuzz/drill
   761→    // generation with adaptive strategy selection.
   762→    let _meta_strategy: MetaStrategy = MetaStrategy::new(vec![], Default::default());
   763→
   764→    // Add user-provided candidates with highest priority.
   765→    for inputs in user_inputs {
   766→        worklist.push(WorklistEntry {
   767→            inputs,
   768→            source: InputSource::UserProvided,
   769→            fitness: None,
   770→        });
   771→    }
   772→
   773→    // Seed the worklist.
   774→    for inputs in seed_inputs {
   775→        worklist.push(WorklistEntry {
   776→            inputs,
   777→            source: InputSource::Seed,
   778→            fitness: None,
   779→        });
   780→    }
   781→
   782→    // --- Float probe phase ---
   783→    let float_indices = crate::float_probe::float_param_indices(param_infos);
   784→    let mut float_probe_results: Vec<crate::float_probe::FloatProbeResult> = Vec::new();
   785→    if !float_indices.is_empty() {
   786→        for &idx in &float_indices {
   787→            let pairs = crate::float_probe::generate_probe_pairs(
   788→                param_infos,
   789→                idx,
   790→                crate::float_probe::PROBE_COUNT,
   791→                &mut rng,
   792→            );
   793→            let mut agreements = 0usize;
   794→            let mut total_probes = 0usize;
   795→            let mut divergent_values = Vec::new();
   796→
   797→            for (float_inputs, floor_inputs) in pairs {
   798→                let float_resp = frontend
   799→                    .send(Command::Execute {
   800→                        function: function_name.to_string(),
   801→                        inputs: float_inputs.clone(),
   802→                        mocks: config.mocks.clone(),
   803→                        setup_context: None,
   804→                    })
   805→                    .await?;
   806→
   807→                let floor_resp = frontend
   808→                    .send(Command::Execute {
   809→                        function: function_name.to_string(),
   810→                        inputs: floor_inputs,
   811→                        mocks: config.mocks.clone(),
   812→                        setup_context: None,
   813→                    })
   814→                    .await?;
   815→
   816→                if let (
   817→                    ResponseResult::Execute(float_result),
   818→                    ResponseResult::Execute(floor_result),
   819→                ) = (&float_resp.result, &floor_resp.result)
   820→                {
   821→                    total_probes += 1;
   822→
   823→                    if crate::float_probe::executions_agree(float_result, floor_result) {
   824→                        agreements += 1;
   825→                    } else if let Some(v) = float_inputs.get(idx).and_then(|v| v.as_f64()) {
   826→                        divergent_values.push(v);
   827→                    }
   828→                }
   829→            }
   830→
   831→            let classification = crate::float_probe::classify(
   832→                agreements,
   833→                total_probes,
   834→                crate::float_probe::AGREEMENT_THRESHOLD,
   835→            );
   836→            float_probe_results.push(crate::float_probe::FloatProbeResult {
   837→                param_index: idx,
   838→                param_name: param_infos[idx].name.clone(),
   839→                classification,
   840→                agreements,
   841→                total_probes,
   842→                divergent_values,
   843→            });
   844→        }
   845→    }
   846→
   847→    let explore_start = Instant::now();
   848→    let mut plateau_counter: usize = 0;
   849→
   850→    // --- Two-phase exploration loop: Observe → Solve/Generate ---
   851→    //
   852→    // Each iteration processes one worklist entry:
   853→    //   1. Observe — execute the entry, classify the path
   854→    //   2. Solve/Generate — if new path, extract constraints and produce candidates
   855→    //
   856→    // Solving after each observation (not batched) preserves the original
   857→    // convergence behavior: Z3 candidates from observation N are available
   858→    // for execution at iteration N+1.
   859→    while let Some(entry) = worklist.pop() {
   860→        // Phase 1: Observe — execute and classify one worklist entry.
   861→        let budget = ExploreBudget {
   862→            unique_paths: executions.len(),
   863→            total_executions,
   864→            plateau_counter,
   865→            explore_start,
   866→        };
   867→
   868→        let observe_result = observe_one(
   869→            &entry,
   870→            frontend,
   871→            function_name,
   872→            config,
   873→            &mut covered_paths,
   874→            &mut triage_state,
   875→            &budget,
   876→        )
   877→        .await?;
   878→
   879→        let obs = match observe_result {
   880→            ObserveOneResult::Observed(obs) => *obs,
   881→            ObserveOneResult::TriageSkipped => {
   882→                triage_skipped += 1;
   883→                continue;
   884→            }
   885→            ObserveOneResult::FrontendSkipped => {
   886→                total_executions += 1;
   887→                continue;
   888→            }
   889→            ObserveOneResult::Terminated(reason) => {
   890→                termination_reason = reason;
   891→                break;
   892→            }
   893→        };
   894→
   895→        total_executions += 1;
   896→        if obs.is_sampled_skip && !obs.is_new_path {
   897→            // Prediction was correct (duplicate path) — no misprediction.
   898→        } else if obs.is_sampled_skip && obs.is_new_path {
   899→            triage_mispredictions += 1;
   900→        }
   901→
   902→        // Record raw result for pipeline composability.
   903→        raw_results.push((obs.inputs.clone(), obs.result.clone()));
   904→
   905→        if !obs.is_new_path {
   906→            plateau_counter += 1;
   907→            continue;
   908→        }
   909→
   910→        // New path discovered — reset plateau counter.
   911→        plateau_counter = 0;
   912→
   913→        // Track per-branch discovery attribution.
   914→        let method = match obs.source {
   915→            InputSource::Z3Solved => DiscoveryMethod::Z3,
   916→            InputSource::UserProvided => DiscoveryMethod::UserProvided,
   917→            InputSource::Drilled => DiscoveryMethod::Drilled,
   918→            InputSource::BoundarySearch => DiscoveryMethod::BoundarySearch,
   919→            InputSource::Seed | InputSource::Fuzzed => DiscoveryMethod::Random,
   920→        };
   921→        for decision in &obs.result.branch_path {
   922→            if seen_branch_ids.insert(decision.branch_id) {
   923→                discoveries.push((decision.branch_id, method));
   924→            }
   925→        }
   926→
   927→        // Update frontier set and target branches: track which branch sides
   928→        // have been seen. Branches seen on only one side are targets for
   929→        // fitness scoring.
   930→        for decision in &obs.result.branch_path {
   931→            seen_branch_sides.insert((decision.branch_id, decision.taken));
   932→            let opposite_seen =
   933→                seen_branch_sides.contains(&(decision.branch_id, !decision.taken));
   934→            if opposite_seen {
   935→                frontier_set.remove(decision.branch_id);
   936→                target_branches.remove(&decision.branch_id);
   937→            } else {
   938→                target_branches.insert(decision.branch_id);
   939→                let prev_stall = frontier_set
   940→                    .iter()
   941→                    .find(|f| f.branch_id == decision.branch_id)
   942→                    .map_or(0, |f| f.stall_count);
   943→                let blocking =
   944→                    drilling::identify_blocking_params(&decision.constraint, param_infos);
   945→                let depth =
   946→                    drilling::branch_depth(&obs.result.branch_path, decision.branch_id);
   947→                frontier_set.insert(Frontier {
   948→                    branch_id: decision.branch_id,
   949→                    depth,
   950→                    blocking_params: blocking,
   951→                    best_prefix: obs.inputs.clone(),
   952→                    stall_count: prev_stall,
   953→                });
   954→            }
   955→        }
   956→
   957→        // Sync fitness context: mark this path as seen so future fitness
   958→        // scoring correctly identifies repeat paths as non-novel.
   959→        fitness_context.mark_seen(obs.path_id);
   960→
   961→        executions.push(obs.result.clone());
   962→
   963→        // Phase 2: Solve/Generate — produce new candidates from this observation.
   964→        let solve_output = solve_and_generate(
   965→            &[obs],
   966→            &mut frontier_set,
   967→            param_infos,
   968→            &param_names,
   969→            &raw_results,
   970→            &seen_branch_sides,
   971→            config,
   972→            &mut rng,
   973→            &target_branches,
   974→            &mut fitness_context,
   975→            &fitness_weights,
   976→        );
   977→
   978→        z3_generated += solve_output.z3_count;
   979→        fuzz_generated += solve_output.fuzz_count;
   980→        drill_generated += solve_output.drill_count;
   981→        boundary_generated += solve_output.boundary_count;
   982→
   983→        for candidate in solve_output.candidates {
   984→            worklist.push(candidate);
   985→        }
   986→    }
   987→
   988→    let unique_paths = covered_paths.len();
   989→    Ok(ExploreResult {
   990→        function_name: function_name.to_string(),
   991→        total_lines: 0, // Caller must set from FunctionAnalysis (end_line - start_line + 1)
   992→        executions,
   993→        unique_paths,
   994→        total_executions,
   995→        z3_generated,
   996→        fuzz_generated,
   997→        boundary_generated,
   998→        drill_generated,
   999→        termination_reason,
  1000→        raw_results,
  1001→        discoveries,
  1002→        triage_skipped,
  1003→        triage_mispredictions,
  1004→        nondeterministic_fields: vec![],
  1005→        float_probe_results,
  1006→    })
  1007→}

  1009→/// Simple input fuzzing: produce a handful of variations on the base inputs.
  1010→///
  1011→/// For each JSON value in the input list, generate mutations:
  1012→/// - Numbers: ±1, ×-1, 0, boundary values
  1013→/// - Booleans: flip
  1014→/// - Strings: empty string, "a"
  1015→pub(crate) fn fuzz_inputs(base: &[serde_json::Value]) -> Vec<Vec<serde_json::Value>> {
  1016→    let mut results = Vec::new();
  1017→
  1018→    for (idx, val) in base.iter().enumerate() {
  1019→        let mutations = fuzz_single_value(val);
  1020→        for mutated in mutations {
  1021→            let mut new_inputs = base.to_vec();
  1022→            new_inputs[idx] = mutated;
  1023→            results.push(new_inputs);
  1024→        }
  1025→    }
  1026→
  1027→    results
  1028→}
  1029→
  1030→pub(crate) fn fuzz_single_value(val: &serde_json::Value) -> Vec<serde_json::Value> {
  1031→    match val {
  1032→        serde_json::Value::Number(n) => {
  1033→            let mut mutations = Vec::new();
  1034→            if let Some(i) = n.as_i64() {
  1035→                let candidates = [i + 1, i - 1, 0, -i];
  1036→                for c in candidates {
  1037→                    let json_val = serde_json::json!(c);
  1038→                    if c != i && !mutations.contains(&json_val) {
  1039→                        mutations.push(json_val);
  1040→                    }
  1041→                }
  1042→            } else if let Some(f) = n.as_f64() {
  1043→                let candidates = [f + 1.0, f - 1.0, 0.0];
  1044→                for c in candidates {
  1045→                    let json_val = serde_json::json!(c);
  1046→                    if (c - f).abs() > f64::EPSILON && !mutations.contains(&json_val) {
  1047→                        mutations.push(json_val);
  1048→                    }
  1049→                }
  1050→            }
  1051→            mutations
  1052→        }
  1053→        serde_json::Value::Bool(b) => vec![serde_json::json!(!b)],
  1054→        serde_json::Value::String(_) => {
  1055→            vec![serde_json::json!(""), serde_json::json!("a")]
  1056→        }
  1057→        _ => vec![],
  1058→    }
  1059→}

  1317→    // -- Integration test: concolic loop finds x=42 via Z3 --
  1318→
  1319→    /// This test simulates the concolic loop without a real frontend by directly
  1320→    /// testing the solver-driven input generation for f(x) { if (x === 42) ... }.
  1321→    ///
  1322→    /// The acceptance criteria require that Z3 solving can find x=42 for an
  1323→    /// exact-equality branch that random exploration cannot feasibly discover.
  1324→    #[test]
  1325→    fn z3_finds_exact_equality_input() {
  1326→        // Simulate: we executed f(0) and observed the branch `x == 42` taken=false.
  1327→        // The constraint for the branch is (x == 42), and we negate it → we need x != 42.
  1328→        // Wait — the branch was NOT taken, so the path constraint recorded is actually
  1329→        // the negation of the condition: NOT(x == 42). To explore the true branch,
  1330→        // we negate that: x == 42.
  1331→        //
  1332→        // In our protocol, the constraint is recorded as `x == 42` with `taken: false`.
  1333→        // The solver receives the constraint as-is and negates it to explore the other path.
  1334→        // Since taken=false means the constraint evaluated to false, the original path has
  1335→        // NOT(x == 42). Negating that yields x == 42 — exactly what we want.
  1336→
  1337→        let x_eq_42 = SymExpr::BinOp {
  1338→            op: BinOpKind::Eq,
  1339→            left: Box::new(SymExpr::Param {
  1340→                name: "x".into(),
  1341→                path: vec![],
  1342→            }),
  1343→            right: Box::new(SymExpr::Const(ConstValue::Int(42))),
  1344→        };
  1345→
  1346→        // The path has one constraint: x == 42 (which evaluated to false).
  1347→        // solve_for_new_path negates constraint[0], giving us x == 42 → NOT(x == 42)?
  1348→        // No — solve_for_new_path keeps the prefix and negates the target.
  1349→        // With index 0 and only one constraint: it negates constraint[0].
  1350→        // The constraint is (x == 42). Negating it gives (x != 42) which is SAT for many values.
  1351→        //
  1352→        // Actually, to find x=42, we need to SOLVE the constraint x==42 directly.
  1353→        // In the real concolic loop, when a branch is not taken, the constraint
  1354→        // represents the condition, and the path records that it was false.
  1355→        // To flip the branch, we want the condition to be true: x == 42.
  1356→        // This means we should solve the constraint directly, not negate it.
  1357→        //
  1358→        // Our solver API `solve_for_new_path` negates constraint[negate_index].
  1359→        // So if we pass the constraint as-is (x == 42) and negate it, we get x != 42.
  1360→        // But we want x == 42!
  1361→        //
  1362→        // The trick: the frontend records the *evaluated* constraint. When taken=false,
  1363→        // it means the condition was false. So the path constraint is NOT(x == 42).
  1364→        // To represent this, we'd store NOT(x == 42) in the constraint list.
  1365→        // Then negating it gives x == 42. ✓
  1366→        //
  1367→        // For this test, let's just use solve_constraints directly to verify Z3 can find x=42.
  1368→        let result = solver::solve_constraints(&[x_eq_42], None, &[]).expect("solver should not error");
  1369→
  1370→        match result {
  1371→            SolveResult::Sat(values) => {
  1372→                let x = match values.get("x") {
  1373→                    Some(ConcreteValue::Int(v)) => *v,
  1374→                    other => panic!("expected Int for x, got {other:?}"),
  1375→                };
  1376→                assert_eq!(x, 42, "Z3 should find x = 42");
  1377→            }
  1378→            SolveResult::Unsat => panic!("expected sat — Z3 should be able to find x = 42"),
  1379→        }
  1380→    }
