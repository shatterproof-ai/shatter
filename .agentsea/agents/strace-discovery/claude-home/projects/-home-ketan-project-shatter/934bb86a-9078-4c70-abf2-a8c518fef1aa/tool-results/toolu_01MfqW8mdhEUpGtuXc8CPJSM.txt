     9→/// Log level for the frontend, controlled by SHATTER_LOG_LEVEL env var.
    10→#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
    11→enum FrontendLogLevel {
    12→    Error = 0,
    13→    Warn = 1,
    14→    Info = 2,
    15→    Debug = 3,
    16→    Trace = 4,
    17→}
    18→
    19→impl FrontendLogLevel {
    20→    fn from_env() -> Self {
    21→        match std::env::var("SHATTER_LOG_LEVEL")
    22→            .unwrap_or_default()
    23→            .to_lowercase()
    24→            .as_str()
    25→        {
    26→            "error" => FrontendLogLevel::Error,
    27→            "warn" => FrontendLogLevel::Warn,
    28→            "debug" => FrontendLogLevel::Debug,
    29→            "trace" => FrontendLogLevel::Trace,
    30→            _ => FrontendLogLevel::Info,
    31→        }
    32→    }
    33→}
    34→
    35→const DEFAULT_EXEC_TIMEOUT_MS: u64 = 5000;
    36→
    37→/// Read the execution timeout from `SHATTER_EXEC_TIMEOUT` env var (seconds).
    38→/// Returns milliseconds. Falls back to `DEFAULT_EXEC_TIMEOUT_MS` if unset, invalid, zero, or negative.
    39→fn exec_timeout_from_env() -> u64 {
    40→    match std::env::var("SHATTER_EXEC_TIMEOUT") {
    41→        Ok(val) => match val.parse::<f64>() {
    42→            Ok(secs) if secs > 0.0 => (secs * 1000.0) as u64,
    43→            _ => DEFAULT_EXEC_TIMEOUT_MS,
    44→        },
    45→        Err(_) => DEFAULT_EXEC_TIMEOUT_MS,
    46→    }
    47→}

    49→/// Write instrumented source to a temp directory and return the output path.
    50→fn write_instrumented_temp(filename: &str, source: &str) -> io::Result<String> {
    51→    let dir = std::env::temp_dir().join(format!(
    52→        "shatter-instrument-{}",
    53→        std::process::id()
    54→    ));
    55→    std::fs::create_dir_all(&dir)?;
    56→    let out_path = dir.join(filename);
    57→    std::fs::write(&out_path, source)?;
    58→    Ok(out_path.to_string_lossy().into_owned())
    59→}

    61→/// Processes protocol requests from stdin and writes responses to stdout.
    62→pub struct Handler<R, W, L> {
    63→    reader: BufReader<R>,
    64→    writer: W,
    65→    log: L,
    66→    log_level: FrontendLogLevel,
    67→    exec_timeout_ms: u64,
    68→    wasm_cache: WasmCache,
    69→    native_registry: Option<NativeRegistry>,
    70→    /// Remembered from the most recent Analyze or Instrument request so Execute
    71→    /// can fall back when the core omits the file field (which it always does).
    72→    last_file: Option<String>,
    73→}
    74→
    75→impl<R: io::Read, W: io::Write, L: io::Write> Handler<R, W, L> {

   123→    /// Process requests until shutdown or EOF. Returns Ok(()) on clean shutdown.
   124→    pub fn run(mut self) -> io::Result<()> {
   125→        self.log_at(FrontendLogLevel::Debug, &format!(
   126→            "Starting Rust frontend (protocol {PROTOCOL_VERSION})"
   127→        ));
   128→
   129→        let mut line = String::new();
   130→        loop {
   131→            line.clear();
   132→            let bytes_read = self.reader.read_line(&mut line)?;
   133→            if bytes_read == 0 {
   134→                // EOF
   135→                self.log_at(FrontendLogLevel::Debug, "Stdin closed, exiting");
   136→                return Ok(());
   137→            }
   138→
   139→            let trimmed = line.trim();
   140→            if trimmed.is_empty() {
   141→                continue;
   142→            }
   143→
   144→            self.logf(&format!("Received: {trimmed}"));
   145→
   146→            let req: Request = match serde_json::from_str(trimmed) {
   147→                Ok(r) => r,
   148→                Err(e) => {
   149→                    self.logf(&format!("Failed to parse request: {e}"));
   150→                    let err_resp = Response {
   151→                        id: 0,
   152→                        status: "error".to_string(),
   153→                        code: Some("invalid_request".to_string()),
   154→                        message: Some(format!("Invalid JSON: {e}")),
   155→                        ..Response::base(0)
   156→                    };
   157→                    self.send(&err_resp)?;
   158→                    continue;
   159→                }
   160→            };
   161→
   162→            let (resp, shutdown) = self.dispatch(&req);
   163→            self.send(&resp)?;
   164→
   165→            if shutdown {
   166→                self.log_at(FrontendLogLevel::Debug, "Shutting down");
   167→                return Ok(());
   168→            }
   169→        }
   170→    }
   171→
   172→    fn dispatch(&mut self, req: &Request) -> (Response, bool) {
   173→        let mut resp = Response::base(req.id);
   174→
   175→        if req.protocol_version != protocol::PROTOCOL_VERSION {
   176→            resp.status = "error".to_string();
   177→            resp.code = Some("version_mismatch".to_string());
   178→            resp.message = Some(format!(
   179→                "unsupported protocol version {:?}, expected {:?}",
   180→                req.protocol_version,
   181→                protocol::PROTOCOL_VERSION,
   182→            ));
   183→            return (resp, false);
   184→        }
   185→
   186→        match req.command.as_str() {
   187→            "handshake" => (self.handle_handshake(resp), false),
   188→            "analyze" => (self.handle_analyze(resp, req), false),
   189→            "instrument" => (self.handle_instrument(resp, req), false),
   190→            "execute" => (self.handle_execute(resp, req), false),
   191→            "setup" => (self.handle_setup(resp, req), false),
   192→            "teardown" => (self.handle_teardown(resp, req), false),
   193→            "generate" => (self.handle_generate(resp, req), false),
   194→            "shutdown" => (self.handle_shutdown(resp), true),
   195→            _ => {
   196→                resp.status = "error".to_string();
   197→                resp.code = Some("invalid_request".to_string());
   198→                resp.message = Some(format!("unknown command: {}", req.command));
   199→                (resp, false)
   200→            }
   201→        }
   202→    }
   203→
   204→    fn handle_handshake(&self, mut resp: Response) -> Response {
   205→        resp.status = "handshake".to_string();
   206→        resp.frontend_version = Some(FRONTEND_VERSION.to_string());
   207→        resp.language = Some(FRONTEND_LANGUAGE.to_string());
   208→        resp.capabilities = Some(vec![
   209→            "analyze".to_string(),
   210→            "execute".to_string(),
   211→            "generate".to_string(),
   212→            "instrument".to_string(),
   213→            "setup".to_string(),
   214→            "teardown".to_string(),
   215→        ]);
   216→        resp
   217→    }

   219→    fn handle_analyze(&mut self, mut resp: Response, req: &Request) -> Response {
   220→        let file_path = match &req.file {
   221→            Some(f) => f,
   222→            None => {
   223→                resp.status = "error".to_string();
   224→                resp.code = Some("invalid_request".to_string());
   225→                resp.message = Some("analyze command requires a file path".to_string());
   226→                return resp;
   227→            }
   228→        };
   229→
   230→        self.last_file = Some(file_path.clone());
   231→
   232→        let path = std::path::Path::new(file_path);
   233→        if !path.exists() {
   234→            resp.status = "error".to_string();
   235→            resp.code = Some("file_not_found".to_string());
   236→            resp.message = Some(format!("file not found: {file_path}"));
   237→            return resp;
   238→        }
   239→
   240→        match crate::analyzer::analyze_file(path, req.function.as_deref()) {
   241→            Ok(functions) => {
   242→                resp.status = "analyze".to_string();
   243→                resp.functions = Some(functions);
   244→                resp
   245→            }
   246→            Err(e) => {
   247→                resp.status = "error".to_string();
   248→                resp.code = Some(
   249→                    match &e {
   250→                        crate::analyzer::AnalyzeError::FileNotFound(_) => "file_not_found",
   251→                        crate::analyzer::AnalyzeError::ReadError(_) => "internal_error",
   252→                        crate::analyzer::AnalyzeError::ParseError(_) => "parse_error",
   253→                        crate::analyzer::AnalyzeError::FunctionNotFound(_) => "function_not_found",
   254→                    }
   255→                    .to_string(),
   256→                );
   257→                resp.message = Some(e.to_string());
   258→                resp
   259→            }
   260→        }
   261→    }

   263→    fn handle_instrument(&mut self, mut resp: Response, req: &Request) -> Response {
   264→        let file_path = match &req.file {
   265→            Some(f) => f,
   266→            None => {
   267→                resp.status = "error".to_string();
   268→                resp.code = Some("invalid_request".to_string());
   269→                resp.message = Some("instrument command requires a file path".to_string());
   270→                return resp;
   271→            }
   272→        };
   273→
   274→        self.last_file = Some(file_path.clone());
   275→
   276→        let path = std::path::Path::new(file_path);
   277→        if !path.exists() {
   278→            resp.status = "error".to_string();
   279→            resp.code = Some("file_not_found".to_string());
   280→            resp.message = Some(format!("file not found: {file_path}"));
   281→            return resp;
   282→        }
   283→
   284→        match crate::instrument::instrument_file(path, req.function.as_deref()) {
   285→            Ok(result) => {
   286→                // Write instrumented source to a temp file
   287→                let source_name = path
   288→                    .file_name()
   289→                    .and_then(|n| n.to_str())
   290→                    .unwrap_or("instrumented.rs");
   291→                match write_instrumented_temp(source_name, &result.source) {
   292→                    Ok(output_path) => {
   293→                        resp.status = "instrument".to_string();
   294→                        resp.instrumented = Some(true);
   295→                        resp.output_file = Some(output_path);
   296→                        resp.message = Some(format!(
   297→                            "instrumented {} branch points",
   298→                            result.branch_count
   299→                        ));
   300→                        resp
   301→                    }
   302→                    Err(e) => {
   303→                        resp.status = "error".to_string();
   304→                        resp.code = Some("internal_error".to_string());
   305→                        resp.message = Some(format!("failed to write instrumented output: {e}"));
   306→                        resp
   307→                    }
   308→                }
   309→            }
   310→            Err(e) => {
   311→                resp.status = "error".to_string();
   312→                resp.code = Some(match &e {
   313→                    crate::instrument::InstrumentError::FileNotFound(_) => "file_not_found",
   314→                    crate::instrument::InstrumentError::ReadError(_) => "internal_error",
   315→                    crate::instrument::InstrumentError::ParseError(_) => "parse_error",
   316→                }.to_string());
   317→                resp.message = Some(e.to_string());
   318→                resp
   319→            }
   320→        }
   321→    }

   323→    fn handle_execute(&self, mut resp: Response, req: &Request) -> Response {
   324→        let file_path = match req.file.as_ref().or(self.last_file.as_ref()) {
   325→            Some(f) => f,
   326→            None => {
   327→                resp.status = "error".to_string();
   328→                resp.code = Some("invalid_request".to_string());
   329→                resp.message = Some("execute command requires a file path (none provided and no prior analyze/instrument)".to_string());
   330→                return resp;
   331→            }
   332→        };
   333→        let function_name = match &req.function {
   334→            Some(f) => f,
   335→            None => {
   336→                resp.status = "error".to_string();
   337→                resp.code = Some("invalid_request".to_string());
   338→                resp.message = Some("execute command requires a function name".to_string());
   339→                return resp;
   340→            }
   341→        };
   342→
   343→        if self.log_level >= FrontendLogLevel::Debug {
   344→            eprintln!(
   345→                "[shatter-rust] Executing {function_name} in {file_path} with {} inputs",
   346→                req.inputs.len()
   347→            );
   348→        }
   349→
   350→        match crate::executor::execute_function(
   351→            file_path,
   352→            function_name,
   353→            &req.inputs,
   354→            &req.mocks,
   355→            self.exec_timeout_ms,
   356→        ) {
   357→            Ok(result) => {
   358→                resp.status = "execute".to_string();
   359→                resp.return_value = result.return_value;
   360→                resp.thrown_error = result.thrown_error;
   361→                resp.branch_path = Some(result.branch_path);
   362→                resp.lines_executed = Some(result.lines_executed);
   363→                resp.calls_to_external = Some(result.calls_to_external);
   364→                resp.path_constraints = Some(result.path_constraints);
   365→                resp.side_effects = Some(result.side_effects);
   366→                resp.performance = Some(result.performance);
   367→                resp
   368→            }
   369→            Err(crate::executor::ExecuteError::FileError(msg)) => {
   370→                resp.status = "error".to_string();
   371→                resp.code = Some("file_not_found".to_string());
   372→                resp.message = Some(msg);
   373→                resp
   374→            }
   375→            Err(crate::executor::ExecuteError::CompilationFailed(msg)) => {
   376→                resp.status = "error".to_string();
   377→                resp.code = Some("compilation_error".to_string());
   378→                resp.message = Some(msg);
   379→                resp
   380→            }
   381→            Err(crate::executor::ExecuteError::NonExecutable(msg)) => {
   382→                resp.status = "error".to_string();
   383→                resp.code = Some("non_executable".to_string());
   384→                resp.message = Some(msg);
   385→                resp
   386→            }
   387→            Err(e) => {
   388→                resp.status = "error".to_string();
   389→                resp.code = Some("internal_error".to_string());
   390→                resp.message = Some(e.to_string());
   391→                resp
   392→            }
   393→        }
   394→    }

   487→    fn handle_generate(&mut self, mut resp: Response, req: &Request) -> Response {
   488→        let file_path = match &req.file {
   489→            Some(f) => f,
   490→            None => {
   491→                resp.status = "error".to_string();
   492→                resp.code = Some("invalid_request".to_string());
   493→                resp.message = Some("generate command requires a file path".to_string());
   494→                return resp;
   495→            }
   496→        };
   497→        let func_name = match &req.name {
   498→            Some(n) => n.clone(),
   499→            None => {
   500→                resp.status = "error".to_string();
   501→                resp.code = Some("invalid_request".to_string());
   502→                resp.message = Some("generate command requires a name".to_string());
   503→                return resp;
   504→            }
   505→        };
   506→
   507→        let path = std::path::Path::new(file_path);
   508→        let ext = path.extension().and_then(|e| e.to_str());
   509→
   510→        match ext {
   511→            Some("wasm") => {
   512→                match self.wasm_cache.generate(path, &func_name, req.recipe.as_ref()) {
   513→                    Ok((value, generator_id, recipe)) => {
   514→                        resp.status = "generate".to_string();
   515→                        resp.value = Some(value);
   516→                        resp.generator_id = Some(generator_id);
   517→                        resp.recipe = recipe;
   518→                        resp
   519→                    }
   520→                    Err(e) => {
   521→                        resp.status = "error".to_string();
   522→                        resp.code = Some("internal_error".to_string());
   523→                        resp.message = Some(e);
   524→                        resp
   525→                    }
   526→                }
   527→            }
   528→            Some("rs") => {
   529→                if let Some(ref registry) = self.native_registry {
   530→                    match registry.generate(&func_name, req.recipe.clone()) {
   531→                        Ok((value, generator_id, recipe)) => {
   532→                            resp.status = "generate".to_string();
   533→                            resp.value = Some(value);
   534→                            resp.generator_id = Some(generator_id);
   535→                            resp.recipe = Some(recipe);
   536→                            resp
   537→                        }
   538→                        Err(e) => {
   539→                            resp.status = "error".to_string();
   540→                            resp.code = Some("internal_error".to_string());
   541→                            resp.message = Some(e);
   542→                            resp
   543→                        }
   544→                    }
   545→                } else {
   546→                    resp.status = "error".to_string();
   547→                    resp.code = Some("internal_error".to_string());
   548→                    resp.message = Some(format!(
   549→                        "native generator {func_name:?} requires a custom build (run `shatter build-frontend rust`)"
   550→                    ));
   551→                    resp
   552→                }
   553→            }
   554→            _ => {
   555→                resp.status = "error".to_string();
   556→                resp.code = Some("invalid_request".to_string());
   557→                resp.message = Some(format!(
   558→                    "unsupported generator file type: {}",
   559→                    file_path
   560→                ));
   561→                resp
   562→            }
   563→        }
   564→    }

   566→    fn handle_shutdown(&self, mut resp: Response) -> Response {
   567→        resp.status = "shutdown_ack".to_string();
   568→        resp
   569→    }

   571→    fn send(&mut self, resp: &Response) -> io::Result<()> {
   572→        let data = serde_json::to_string(resp)
   573→            .map_err(io::Error::other)?;
   574→        self.logf(&format!("Sent: {data}"));
   575→        writeln!(self.writer, "{data}")?;
   576→        self.writer.flush()
   577→    }

   579→    /// Log at TRACE level (protocol messages).
   580→    fn logf(&mut self, msg: &str) {
   581→        self.log_at(FrontendLogLevel::Trace, msg);
   582→    }

   584→    /// Log at a specific level.
   585→    fn log_at(&mut self, level: FrontendLogLevel, msg: &str) {
   586→        if self.log_level >= level {
   587→            let _ = writeln!(self.log, "[shatter-rust] {msg}");
   588→        }
   589→    }

   598→        let input = format!("{req_json}\n");
   599→        let mut output = Vec::new();

   607→        let first_line = output_str.lines().next().expect("at least one response line");
   608→        serde_json::from_str(first_line).expect("valid JSON response")
   609→    }

   623→            .lines()
   624→            .filter(|l| !l.is_empty())
   625→            .map(|l| serde_json::from_str(l).expect("valid JSON response"))

   627→    }
