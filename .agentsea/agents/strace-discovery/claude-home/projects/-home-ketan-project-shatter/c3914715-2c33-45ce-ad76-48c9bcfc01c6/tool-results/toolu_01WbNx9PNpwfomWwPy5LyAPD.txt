     1→//! Protocol types for communication between the Rust core and language frontends.
     2→//!
     3→//! All messages are newline-delimited JSON over stdin/stdout. The core sends
     4→//! [`Request`] messages to frontends and receives [`Response`] messages back.
     5→//! Every message includes a protocol version for compatibility checking.
     6→

    26→/// Granularity level for setup/teardown lifecycle management.
    27→#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
    28→#[serde(rename_all = "snake_case")]
    29→pub enum SetupLevel {
    30→    Session,
    31→    File,
    32→    Function,
    33→    Execution,
    34→}
    35→
    36→/// A single entry in a setup context stack, associating a lifecycle level
    37→/// with the opaque context value returned by its Setup command.
    38→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    39→pub struct SetupContextEntry {
    40→    pub level: SetupLevel,
    41→    pub context: serde_json::Value,
    42→}
    43→
    44→/// Stack of active setup contexts, ordered from outermost (session) to
    45→/// innermost (execution). Passed to Execute so frontends can restore
    46→/// all active setup state.
    47→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    48→pub struct SetupContextStack {
    49→    pub contexts: Vec<SetupContextEntry>,
    50→}
    51→
    52→impl From<crate::config::SetupMode> for SetupLevel {
    53→    fn from(mode: crate::config::SetupMode) -> Self {
    54→        match mode {
    55→            crate::config::SetupMode::PerFunction => SetupLevel::Function,
    56→            crate::config::SetupMode::PerExecution => SetupLevel::Execution,
    57→        }
    58→    }
    59→}

    65→/// A request message sent from the core engine to a language frontend.
    66→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    67→pub struct Request {
    68→    /// Protocol version for compatibility checking.
    69→    pub protocol_version: String,
    70→    /// Unique identifier for correlating responses to requests.
    71→    pub id: u64,
    72→    /// The command to execute.
    73→    #[serde(flatten)]
    74→    pub command: Command,
    75→}

    77→/// Commands the core can send to a frontend.
    78→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    79→#[serde(tag = "command", rename_all = "snake_case")]
    80→pub enum Command {
    81→    /// Initial handshake to negotiate protocol version and capabilities.
    82→    Handshake {
    83→        /// Capabilities supported by the core.
    84→        capabilities: Vec<String>,
    85→    },
    86→    /// Analyze a function to extract type information, branches, and dependencies.
    87→    Analyze {
    88→        /// Path to the source file.
    89→        file: String,
    90→        /// Name of the function to analyze. If absent, analyze all exported functions.
    91→        function: Option<String>,
    92→        /// Detected project root directory, if any.
    93→        #[serde(default, skip_serializing_if = "Option::is_none")]
    94→        project_root: Option<String>,
    95→    },
    96→    /// Instrument a function for symbolic constraint tracking.
    97→    Instrument {
    98→        /// Path to the source file.
    99→        file: String,
   100→        /// Name of the function to instrument.
   101→        function: String,
   102→        /// Mock configurations for external dependencies.
   103→        mocks: Vec<MockConfig>,
   104→        /// Detected project root directory, if any.
   105→        #[serde(default, skip_serializing_if = "Option::is_none")]
   106→        project_root: Option<String>,
   107→    },
   108→    /// Execute an instrumented function with specific inputs and mocks.
   109→    Execute {
   110→        /// Fully qualified function identifier.
   111→        function: String,
   112→        /// Input values for the function parameters (JSON-encoded).
   113→        inputs: Vec<serde_json::Value>,
   114→        /// Mock configurations for external dependencies.
   115→        mocks: Vec<MockConfig>,
   116→        /// Stack of active setup contexts from enclosing Setup commands, if any.
   117→        #[serde(default, skip_serializing_if = "Option::is_none")]
   118→        setup_context: Option<SetupContextStack>,
   119→    },
   120→    /// Run a setup file to initialize state before function execution.
   121→    Setup {
   122→        /// Path to the setup file.
   123→        file: String,
   124→        /// Scope identifier (function name, file path, or session label).
   125→        scope: String,
   126→        /// Lifecycle level for this setup.
   127→        level: SetupLevel,
   128→        /// Detected project root directory, if any.
   129→        #[serde(default, skip_serializing_if = "Option::is_none")]
   130→        project_root: Option<String>,
   131→        /// Parent context stack from enclosing setup levels, if any.
   132→        #[serde(default, skip_serializing_if = "Option::is_none")]
   133→        parent_context: Option<SetupContextStack>,
   134→    },
   135→    /// Tear down state established by a prior Setup command.
   136→    Teardown {
   137→        /// Scope identifier matching the corresponding Setup command.
   138→        scope: String,
   139→        /// Lifecycle level matching the corresponding Setup command.
   140→        level: SetupLevel,
   141→    },
   142→    /// Invoke a custom generator to produce a value for a type or parameter.
   143→    Generate {
   144→        /// Path to the generator file.
   145→        file: String,
   146→        /// Name of the type or parameter to generate a value for.
   147→        name: String,
   148→        /// Whether this generator targets a type name or a parameter name.
   149→        kind: GeneratorKind,
   150→        /// Reconstruction recipe from a previous generation.
   151→        #[serde(default, skip_serializing_if = "Option::is_none")]
   152→        recipe: Option<serde_json::Value>,
   153→        /// Detected project root directory, if any.
   154→        #[serde(default, skip_serializing_if = "Option::is_none")]
   155→        project_root: Option<String>,
   156→    },
   157→    /// Request graceful shutdown of the frontend process.
   158→    Shutdown,
   159→}

   161→/// Whether a generator targets a type name or a parameter name.
   162→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]

   167→    /// Generator for a named parameter (e.g., "authToken").

   498→/// Validate an `ExecuteResult` deserialized from a frontend subprocess.
   499→///
   500→/// Checks semantic invariants that serde can't enforce:
   501→/// - `branch_path` entries have non-default constraints (when non-empty, an all-unknown
   502→///   branch_path produces identical hashes for different paths → silent hash collision)
   503→/// - `path_constraints` length matches `branch_path` length when both are non-empty
   504→#[contracts::ensures(ret == execute_result_is_valid(result),
   505→    "postcondition must match validation logic")]
   506→pub fn validate_execute_result(result: &ExecuteResult) -> bool {
   507→    execute_result_is_valid(result)
   508→}
   509→
   510→fn execute_result_is_valid(result: &ExecuteResult) -> bool {
   511→    // Branch path with all-unknown constraints causes silent hash collisions
   512→    // because the path hash depends on constraint content.
   513→    if !result.branch_path.is_empty() {
   514→        let all_unknown = result.branch_path.iter().all(|bd| {
   515→            matches!(bd.constraint, SymConstraint::Unknown { .. })
   516→        });
   517→        // All-unknown is valid for frontends without symbolic analysis (Go),
   518→        // but path_constraints should then also be empty.
   519→        if all_unknown && !result.path_constraints.is_empty() {
   520→            return false;
   521→        }
   522→    }
   523→    true
   524→}
   525→
   526→/// Validate a `FunctionAnalysis` list deserialized from a frontend subprocess.
   527→///
   528→/// Checks semantic invariants that serde can't enforce:
   529→/// - Function names are non-empty (empty name → silent lookup failures)
   530→/// - `start_line <= end_line` (inverted range → wrong source slicing)
   531→/// - Param count is plausible (> 255 params suggests a deserialization bug)
   532→#[contracts::ensures(ret == analyze_result_is_valid(functions),
   533→    "postcondition must match validation logic")]
   534→pub fn validate_analyze_result(functions: &[FunctionAnalysis]) -> bool {
   535→    analyze_result_is_valid(functions)
   536→}
   537→
   538→/// Upper bound on parameter count — anything above this likely indicates
   539→/// a deserialization bug rather than a real function signature.
   540→const MAX_PLAUSIBLE_PARAMS: usize = 255;
   541→
   542→fn analyze_result_is_valid(functions: &[FunctionAnalysis]) -> bool {
   543→    for func in functions {
   544→        if func.name.is_empty() {
   545→            return false;
   546→        }
   547→        if func.start_line > func.end_line {
   548→            return false;
   549→        }
   550→        if func.params.len() > MAX_PLAUSIBLE_PARAMS {
   551→            return false;
   552→        }
   553→    }
   554→    true
   555→}

   563→        value: &T,
