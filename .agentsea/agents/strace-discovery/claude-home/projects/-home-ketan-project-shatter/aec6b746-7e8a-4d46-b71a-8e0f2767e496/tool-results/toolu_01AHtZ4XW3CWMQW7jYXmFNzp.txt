     1→//! Protocol types for communication between the Rust core and language frontends.
     2→//!
     3→//! All messages are newline-delimited JSON over stdin/stdout. The core sends
     4→//! [`Request`] messages to frontends and receives [`Response`] messages back.
     5→//! Every message includes a protocol version for compatibility checking.
     6→
     7→use std::collections::HashMap;
     8→
     9→use serde::{Deserialize, Serialize};
    10→
    11→use crate::config::SetupMode;
    12→use crate::crypto_registry::{CryptoDirection, OutputSemantics, ParamRole};
    13→use crate::execution_record::{
    14→    BranchDecision, ErrorInfo, ExternalCall, SideEffect, SymConstraint, TraceEvent,
    15→    TruncationInfo,
    16→};
    17→use crate::sym_expr::SymExpr;
    18→use crate::types::{ParamInfo, TypeInfo};
    19→
    20→/// Current protocol version.
    21→pub const PROTOCOL_VERSION: &str = "0.1.0";
    22→
    23→// ---------------------------------------------------------------------------
    24→// Request: Core → Frontend
    25→// ---------------------------------------------------------------------------
    26→
    27→/// A request message sent from the core engine to a language frontend.
    28→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    29→pub struct Request {
    30→    /// Protocol version for compatibility checking.
    31→    pub protocol_version: String,
    32→    /// Unique identifier for correlating responses to requests.
    33→    pub id: u64,
    34→    /// The command to execute.
    35→    #[serde(flatten)]
    36→    pub command: Command,
    37→}
    38→
    39→/// Commands the core can send to a frontend.
    40→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    41→#[serde(tag = "command", rename_all = "snake_case")]
    42→pub enum Command {
    43→    /// Initial handshake to negotiate protocol version and capabilities.
    44→    Handshake {
    45→        /// Capabilities supported by the core.
    46→        capabilities: Vec<String>,
    47→    },
    48→    /// Analyze a function to extract type information, branches, and dependencies.
    49→    Analyze {
    50→        /// Path to the source file.
    51→        file: String,
    52→        /// Name of the function to analyze. If absent, analyze all exported functions.
    53→        function: Option<String>,
    54→        /// Detected project root directory, if any.
    55→        #[serde(default, skip_serializing_if = "Option::is_none")]
    56→        project_root: Option<String>,
    57→    },
    58→    /// Instrument a function for symbolic constraint tracking.
    59→    Instrument {
    60→        /// Path to the source file.
    61→        file: String,
    62→        /// Name of the function to instrument.
    63→        function: String,
    64→        /// Mock configurations for external dependencies.
    65→        mocks: Vec<MockConfig>,
    66→        /// Detected project root directory, if any.
    67→        #[serde(default, skip_serializing_if = "Option::is_none")]
    68→        project_root: Option<String>,
    69→    },
    70→    /// Execute an instrumented function with specific inputs and mocks.
    71→    Execute {
    72→        /// Fully qualified function identifier.
    73→        function: String,
    74→        /// Input values for the function parameters (JSON-encoded).
    75→        inputs: Vec<serde_json::Value>,
    76→        /// Mock configurations for external dependencies.
    77→        mocks: Vec<MockConfig>,
    78→        /// Opaque context returned by a prior Setup command, if any.
    79→        #[serde(default, skip_serializing_if = "Option::is_none")]
    80→        setup_context: Option<serde_json::Value>,
    81→    },
    82→    /// Run a setup file to initialize state before function execution.
    83→    Setup {
    84→        /// Path to the setup file.
    85→        file: String,
    86→        /// Name of the function this setup is associated with.
    87→        function: String,
    88→        /// When to run setup relative to executions.
    89→        mode: SetupMode,
    90→        /// Detected project root directory, if any.
    91→        #[serde(default, skip_serializing_if = "Option::is_none")]
    92→        project_root: Option<String>,
    93→    },
    94→    /// Tear down state established by a prior Setup command.
    95→    Teardown {
    96→        /// Name of the function whose setup should be torn down.
    97→        function: String,
    98→    },
    99→    /// Invoke a custom generator to produce a value for a type or parameter.
   100→    Generate {
   101→        /// Path to the generator file.
   102→        file: String,
   103→        /// Name of the type or parameter to generate a value for.
   104→        name: String,
   105→        /// Whether this generator targets a type name or a parameter name.
   106→        kind: GeneratorKind,
   107→        /// Reconstruction recipe from a previous generation.
   108→        #[serde(default, skip_serializing_if = "Option::is_none")]
   109→        recipe: Option<serde_json::Value>,
   110→        /// Detected project root directory, if any.
   111→        #[serde(default, skip_serializing_if = "Option::is_none")]
   112→        project_root: Option<String>,
   113→    },
   114→    /// Request graceful shutdown of the frontend process.
   115→    Shutdown,
   116→}
   117→
   118→/// Whether a generator targets a type name or a parameter name.
   119→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   120→#[serde(rename_all = "snake_case")]
   121→pub enum GeneratorKind {
   122→    /// Generator for a named type (e.g., "User").
   123→    TypeName,
   124→    /// Generator for a named parameter (e.g., "authToken").
   125→    ParamName,
   126→}

   128→/// Configuration for mocking an external dependency during execution.
   129→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   130→pub struct MockConfig {
   131→    /// Fully qualified symbol name of the dependency to mock.
   132→    pub symbol: String,
   133→    /// Pre-configured return values to cycle through.
   134→    pub return_values: Vec<serde_json::Value>,
   135→    /// Whether to record arguments passed to the mock.
   136→    pub should_track_calls: bool,
   137→    /// Default behavior when return_values are exhausted.
   138→    pub default_behavior: MockBehavior,
   139→}
   140→
   141→/// How a mock should behave when its pre-configured return values are exhausted.
   142→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   143→#[serde(rename_all = "snake_case")]
   144→pub enum MockBehavior {
   145→    /// Return a generated value based on the return type.
   146→    ReturnGenerated,
   147→    /// Repeat the last return value.
   148→    RepeatLast,
   149→    /// Throw an error.
   150→    ThrowError,
   151→    /// Call through to the real implementation.
   152→    Passthrough,
   153→}

   159→/// A response message sent from a language frontend to the core engine.
   160→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   161→pub struct Response {
   162→    /// Protocol version for compatibility checking.
   163→    pub protocol_version: String,
   164→    /// Request ID this response corresponds to.
   165→    pub id: u64,
   166→    /// The response payload.
   167→    #[serde(flatten)]
   168→    pub result: ResponseResult,
   169→}
   170→
   171→/// Response payloads from a frontend.
   172→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   173→#[serde(tag = "status", rename_all = "snake_case")]
   174→pub enum ResponseResult {
   175→    /// Successful handshake response.
   176→    Handshake {
   177→        /// Protocol version the frontend supports.
   178→        frontend_version: String,
   179→        /// Language this frontend handles (e.g., "typescript", "go").
   180→        language: String,
   181→        /// Capabilities supported by the frontend.
   182→        capabilities: Vec<String>,
   183→    },
   184→    /// Successful analysis result.
   185→    Analyze {
   186→        /// Functions found and analyzed.
   187→        #[serde(default)]
   188→        functions: Vec<FunctionAnalysis>,
   189→    },
   190→    /// Successful instrumentation result.
   191→    Instrument {
   192→        /// Whether instrumentation succeeded.
   193→        instrumented: bool,
   194→        /// Path to the instrumented output file, if applicable.
   195→        output_file: Option<String>,
   196→        /// Number of executable statement lines the instrumentor inserted record calls for.
   197→        /// Used as the denominator for line coverage instead of raw source span.
   198→        #[serde(default, skip_serializing_if = "Option::is_none")]
   199→        instrumentable_line_count: Option<u32>,
   200→    },
   201→    /// Successful execution result (boxed to reduce enum size).
   202→    Execute(Box<ExecuteResult>),
   203→    /// Successful setup result.
   204→    Setup {
   205→        /// Opaque context to pass to subsequent Execute commands.
   206→        setup_context: serde_json::Value,
   207→    },
   208→    /// Acknowledgment that teardown completed.
   209→    TeardownAck,
   210→    /// Result of invoking a custom generator.
   211→    Generate {
   212→        /// The generated value.
   213→        value: serde_json::Value,
   214→        /// Human-readable label for this generated value.
   215→        generator_id: String,
   216→        /// Serializable recipe for replaying this generation.
   217→        #[serde(default, skip_serializing_if = "Option::is_none")]
   218→        recipe: Option<serde_json::Value>,
   219→    },
   220→    /// Acknowledgment of shutdown request.
   221→    ShutdownAck,
   222→    /// Error response for any command.
   223→    Error {
   224→        /// Machine-readable error code.
   225→        code: ErrorCode,
   226→        /// Human-readable error message.
   227→        message: String,
   228→        /// Additional error details (e.g., stack trace, source location).
   229→        details: Option<serde_json::Value>,
   230→    },
   231→}

   233→/// A literal constant value extracted from source code during static analysis.
   234→///
   235→/// Used to seed the candidate input pool with values the function itself
   236→/// compares against, improving branch coverage on first pass.
   237→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   238→#[serde(tag = "type", rename_all = "snake_case")]
   239→pub enum LiteralValue {
   240→    Int { value: i64 },
   241→    Float { value: f64 },
   242→    Str { value: String },
   243→    Bool { value: bool },
   244→    /// Regex pattern string (source text, no delimiters or flags).
   245→    Regex { pattern: String },
   246→}

   248→/// Analysis result for a single function.
   249→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   250→pub struct FunctionAnalysis {
   251→    /// Fully qualified function name.
   252→    pub name: String,
   253→    /// Whether the function is exported (public) from its module.
   254→    #[serde(default)]
   255→    pub exported: bool,
   256→    /// Function parameters with type information.
   257→    pub params: Vec<ParamInfo>,
   258→    /// Branch points found in the function.
   259→    pub branches: Vec<BranchInfo>,
   260→    /// External dependencies detected.
   261→    pub dependencies: Vec<ExternalDependency>,
   262→    /// Return type of the function.
   263→    pub return_type: TypeInfo,
   264→    /// Source location.
   265→    pub start_line: u32,
   266→    pub end_line: u32,
   267→    /// Literal constants extracted from the function body for use as candidate inputs.
   268→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   269→    pub literals: Vec<LiteralValue>,
   270→    /// Cryptographic API boundaries detected by matching dependencies against the crypto registry.
   271→    /// Populated by core after analysis; frontends leave this empty.
   272→    #[serde(default, skip_serializing_if = "Vec::is_empty")]
   273→    pub crypto_boundaries: Vec<CryptoBoundary>,
   274→}

   276→/// A branch point found during static analysis.
   277→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   278→pub struct BranchInfo {
   279→    /// Unique identifier for this branch within the function.
   280→    pub id: u32,
   281→    /// Source line number.
   282→    pub line: u32,
   283→    /// Source text of the condition (for display).
   284→    pub condition_text: String,
   285→    /// Symbolic representation of the condition, if extractable.
   286→    pub condition: Option<SymExpr>,
   287→    /// Type of branch construct.
   288→    pub branch_type: BranchType,
   289→}

   291→/// The kind of branch construct in source code.
   292→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   293→#[serde(rename_all = "snake_case")]
   294→pub enum BranchType {
   295→    If,
   296→    ElseIf,
   297→    Switch,
   298→    Ternary,
   299→    LogicalAnd,
   300→    LogicalOr,
   301→    While,
   302→    For,
   303→    Select,
   304→}

   306→/// An external dependency detected during analysis.
   307→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   308→pub struct ExternalDependency {
   309→    /// The kind of dependency.
   310→    pub kind: DependencyKind,
   311→    /// Fully qualified symbol name.
   312→    pub symbol: String,
   313→    /// Module or package the symbol is imported from.
   314→    pub source_module: String,
   315→    /// Return type for generating mock values.
   316→    pub return_type: TypeInfo,
   317→    /// Parameter types for validating mock calls.
   318→    pub param_types: Vec<TypeInfo>,
   319→    /// Line numbers where this dependency is called.
   320→    pub call_sites: Vec<u32>,
   321→}
   322→
   323→/// A detected cryptographic API boundary within a function.
   324→///
   325→/// Produced by matching `ExternalDependency` entries against the crypto registry.
   326→/// Carries the crypto-specific metadata (direction, param roles, output semantics)
   327→/// alongside the call site information from the dependency.
   328→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   329→pub struct CryptoBoundary {
   330→    /// The symbol name from the dependency (e.g. "createDecipheriv").
   331→    pub symbol: String,
   332→    /// The source module (e.g. "crypto", "crypto/cipher").
   333→    pub source_module: String,
   334→    /// Whether this is an encrypt, decrypt, or both operation.
   335→    pub direction: CryptoDirection,
   336→    /// What the output represents (ciphertext, plaintext, key, etc.).
   337→    pub output: OutputSemantics,
   338→    /// Maps parameter positions to their cryptographic roles.
   339→    #[serde(default, skip_serializing_if = "HashMap::is_empty")]
   340→    pub param_roles: HashMap<String, ParamRole>,
   341→    /// Line numbers where this crypto API is called.
   342→    pub call_sites: Vec<u32>,
   343→}

   350→    MethodCall,
   351→    PropertyAccess,

   355→/// Result of executing an instrumented function.
   356→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   357→pub struct ExecuteResult {
   358→    /// Return value from the function, if it returned normally.
   359→    pub return_value: Option<serde_json::Value>,
   360→    /// Error thrown during execution, if any.
   361→    pub thrown_error: Option<ErrorInfo>,
   362→    /// Branch decisions recorded during execution.
   363→    #[serde(default)]
   364→    pub branch_path: Vec<BranchDecision>,
   365→    /// Source lines executed.
   366→    #[serde(default)]
   367→    pub lines_executed: Vec<u32>,
   368→    /// Calls to external dependencies observed.
   369→    #[serde(default)]
   370→    pub calls_to_external: Vec<ExternalCall>,
   371→    /// Symbolic path constraints collected.
   372→    #[serde(default)]
   373→    pub path_constraints: Vec<SymConstraint>,
   374→    /// Scope-annotated execution trace (branches + loop/call markers).
   375→    /// When non-empty, enables scope-aware path collapsing in `path_hash`.
   376→    #[serde(default)]
   377→    pub scope_events: Vec<TraceEvent>,
   378→    /// Side effects observed during execution.
   379→    #[serde(default)]
   380→    pub side_effects: Vec<SideEffect>,
   381→    /// Performance metrics.
   382→    pub performance: PerformanceMetrics,
   383→    /// Truncation metadata for captured side effects.
   384→    #[serde(default, skip_serializing_if = "Option::is_none")]
   385→    pub capture_truncation: Option<TruncationInfo>,
   386→}

   401→/// Machine-readable error codes for protocol errors.
   402→#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
   403→#[serde(rename_all = "snake_case")]
   404→pub enum ErrorCode {
   405→    /// File not found or not readable.
   406→    FileNotFound,
   407→    /// Function not found in the specified file.
   408→    FunctionNotFound,
   409→    /// Syntax or parse error in source file.
   410→    ParseError,
   411→    /// Instrumentation failed.
   412→    InstrumentationFailed,
   413→    /// Execution timed out.
   414→    ExecutionTimeout,
   415→    /// Execution crashed (segfault, OOM, etc.).
   416→    ExecutionCrash,
   417→    /// Protocol version mismatch.
   418→    VersionMismatch,
   419→    /// Invalid or malformed request.
   420→    InvalidRequest,
   421→    /// Frontend internal error.
   422→    InternalError,
   423→}

   456→    fn round_trip<T: Serialize + for<'de> Deserialize<'de> + PartialEq + std::fmt::Debug>(
   457→        value: &T,
   458→    ) {
   459→        let json = serde_json::to_string(value).expect("serialize");
   460→        let deserialized: T = serde_json::from_str(&json).expect("deserialize");
   461→        assert_eq!(*value, deserialized, "round-trip failed for json: {json}");
   462→    }
