     1‚Üí//! User-defined mock fixtures with scoped resolution and type validation.
     2‚Üí//!
     3‚Üí//! Mock fixtures let users declare exactly how external dependencies should
     4‚Üí//! behave during exploration. Fixtures are defined in `.shatter/config.yaml`
     5‚Üí//! at three scope levels (global ‚Üí file ‚Üí function), with innermost scope
     6‚Üí//! winning on conflicts.

    22‚Üí/// Strategy for selecting mock return values during exploration.
    23‚Üí#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
    24‚Üí#[serde(rename_all = "snake_case", tag = "kind")]
    25‚Üípub enum MockValueSpace {
    26‚Üí    /// Try the real service first; fall back to synthetic mocks on connection
    27‚Üí    /// failure. This is the default for network/database dependencies.
    28‚Üí    LiveFirst,
    29‚Üí    /// Use a fixed set of user-provided return values (round-robin).
    30‚Üí    FixedSet { values: Vec<Value> },
    31‚Üí    /// Engine generates values autonomously based on type information.
    32‚Üí    Autonomous,
    33‚Üí    /// Replay a recorded seed corpus (e.g. from a prior live session).
    34‚Üí    Seeded { seed_file: String },
    35‚Üí}

   112‚Üí/// Three-level scoped mock fixture configuration.
   113‚Üí///
   114‚Üí/// Resolution order (innermost wins):
   115‚Üí/// 1. `functions["file:func"].fixtures["symbol"]`
   116‚Üí/// 2. `files["file"].fixtures["symbol"]`
   117‚Üí/// 3. `global.fixtures["symbol"]`
   118‚Üí#[derive(Debug, Clone, Default, PartialEq, Serialize, Deserialize)]
   119‚Üípub struct MockFixtureConfig {
   120‚Üí    /// Global fixtures applied to all functions unless overridden.
   121‚Üí    #[serde(default)]
   122‚Üí    pub global: MockFixtureScope,
   123‚Üí
   124‚Üí    /// Per-file fixtures, keyed by file path or glob pattern.
   125‚Üí    #[serde(default, skip_serializing_if = "HashMap::is_empty")]
   126‚Üí    pub files: HashMap<String, MockFixtureScope>,
   127‚Üí
   128‚Üí    /// Per-function fixtures, keyed by `"file:function"` identifier.
   129‚Üí    #[serde(default, skip_serializing_if = "HashMap::is_empty")]
   130‚Üí    pub functions: HashMap<String, MockFixtureScope>,
   131‚Üí}

   133‚Üí/// A collection of fixtures at a single scope level.
   134‚Üí#[derive(Debug, Clone, Default, PartialEq, Serialize, Deserialize)]
   135‚Üípub struct MockFixtureScope {
   136‚Üí    /// Fixtures keyed by dependency symbol.
   137‚Üí    #[serde(default, skip_serializing_if = "HashMap::is_empty")]
   138‚Üí    pub fixtures: HashMap<String, MockFixture>,
   139‚Üí}

   145‚Üí/// Resolve the effective fixture for a given symbol, applying innermost-wins
   146‚Üí/// precedence: function ‚Üí file ‚Üí global.
   147‚Üípub fn resolve_fixture<'a>(
   148‚Üí    config: &'a MockFixtureConfig,
   149‚Üí    file: &str,
   150‚Üí    function: &str,
   151‚Üí    symbol: &str,
   152‚Üí) -> Option<&'a MockFixture> {
   153‚Üí    let func_key = format!("{file}:{function}");
   154‚Üí
   155‚Üí    // 1. Function-level (most specific)
   156‚Üí    if let Some(fixture) = config
   157‚Üí        .functions
   158‚Üí        .get(&func_key)
   159‚Üí        .and_then(|scope| scope.fixtures.get(symbol))
   160‚Üí    {
   161‚Üí        return Some(fixture);
   162‚Üí    }
   163‚Üí
   164‚Üí    // 2. File-level
   165‚Üí    if let Some(fixture) = config
   166‚Üí        .files
   167‚Üí        .get(file)
   168‚Üí        .and_then(|scope| scope.fixtures.get(symbol))
   169‚Üí    {
   170‚Üí        return Some(fixture);
   171‚Üí    }
   172‚Üí
   173‚Üí    // 3. Global (least specific)
   174‚Üí    config.global.fixtures.get(symbol)
   175‚Üí}

   181‚Üí/// Errors from fixture validation.
   182‚Üí#[derive(Debug, Clone, PartialEq)]
   183‚Üípub enum FixtureValidationError {
   184‚Üí    /// A return value doesn't match the declared return type.
   185‚Üí    TypeMismatch {
   186‚Üí        symbol: String,
   187‚Üí        index: usize,
   188‚Üí        expected: TypeInfo,
   189‚Üí        actual_value: Value,
   190‚Üí    },
   191‚Üí    /// FixedSet value space requires `return_values` to be non-empty.
   192‚Üí    EmptyFixedSet { symbol: String },
   193‚Üí}
   194‚Üí
   195‚Üíimpl std::fmt::Display for FixtureValidationError {
   196‚Üí    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   197‚Üí        match self {
   198‚Üí            Self::TypeMismatch {
   199‚Üí                symbol,
   200‚Üí                index,
   201‚Üí                expected,
   202‚Üí                actual_value,
   203‚Üí            } => write!(
   204‚Üí                f,
   205‚Üí                "fixture '{symbol}': return_values[{index}] doesn't match declared type {expected:?}: got {actual_value}"
   206‚Üí            ),
   207‚Üí            Self::EmptyFixedSet { symbol } => write!(
   208‚Üí                f,
   209‚Üí                "fixture '{symbol}': value_space is fixed_set but return_values is empty"
   210‚Üí            ),
   211‚Üí        }
   212‚Üí    }
   213‚Üí}

   215‚Üíimpl std::error::Error for FixtureValidationError {}

   217‚Üí/// Check that a JSON value is compatible with a TypeInfo.
   218‚Üífn json_matches_type(value: &Value, ty: &TypeInfo) -> bool {
   219‚Üí    match (value, ty) {
   220‚Üí        (Value::Null, TypeInfo::Nullable { .. }) => true,
   221‚Üí        (_, TypeInfo::Nullable { inner }) => json_matches_type(value, inner),
   222‚Üí        (Value::Number(n), TypeInfo::Int) => n.as_i64().is_some(),
   223‚Üí        (Value::Number(n), TypeInfo::Float) => n.as_f64().is_some(),
   224‚Üí        (Value::String(_), TypeInfo::Str) => true,
   225‚Üí        (Value::Bool(_), TypeInfo::Bool) => true,
   226‚Üí        (Value::Array(items), TypeInfo::Array { element }) => {
   227‚Üí            items.iter().all(|item| json_matches_type(item, element))
   228‚Üí        }
   229‚Üí        (Value::Object(map), TypeInfo::Object { fields }) => {
   230‚Üí            fields.iter().all(|(name, field_ty)| {
   231‚Üí                map.get(name)
   232‚Üí                    .is_some_and(|v| json_matches_type(v, field_ty))
   233‚Üí            })
   234‚Üí        }
   235‚Üí        (_, TypeInfo::Union { variants }) => {
   236‚Üí            variants.iter().any(|v| json_matches_type(value, v))
   237‚Üí        }
   238‚Üí        // Unknown/Complex/Opaque ‚Äî can't validate statically, accept anything.
   239‚Üí        (_, TypeInfo::Unknown | TypeInfo::Complex { .. } | TypeInfo::Opaque { .. }) => true,
   240‚Üí        _ => false,
   241‚Üí    }
   242‚Üí}

   244‚Üí/// Validate all fixtures in a config at load time.
   245‚Üí///
   246‚Üí/// Checks:
   247‚Üí/// - `FixedSet` value spaces have non-empty `return_values`
   248‚Üí/// - Each return value matches the declared `return_type` (when both are present)
   249‚Üípub fn validate_fixture_types(config: &MockFixtureConfig) -> Vec<FixtureValidationError> {
   250‚Üí    let mut errors = Vec::new();
   251‚Üí    let all_scopes = std::iter::once(&config.global)
   252‚Üí        .chain(config.files.values())
   253‚Üí        .chain(config.functions.values());
   254‚Üí
   255‚Üí    for scope in all_scopes {
   256‚Üí        for fixture in scope.fixtures.values() {
   257‚Üí            validate_single_fixture(fixture, &mut errors);
   258‚Üí        }
   259‚Üí    }
   260‚Üí    errors
   261‚Üí}
   262‚Üí
   263‚Üífn validate_single_fixture(fixture: &MockFixture, errors: &mut Vec<FixtureValidationError>) {
   264‚Üí    // FixedSet requires non-empty return_values.
   265‚Üí    if matches!(fixture.value_space, MockValueSpace::FixedSet { .. }) {
   266‚Üí        let has_values = fixture
   267‚Üí            .return_values
   268‚Üí            .as_ref()
   269‚Üí            .is_some_and(|v| !v.is_empty());
   270‚Üí        if !has_values {
   271‚Üí            errors.push(FixtureValidationError::EmptyFixedSet {
   272‚Üí                symbol: fixture.symbol.clone(),
   273‚Üí            });
   274‚Üí        }
   275‚Üí    }
   276‚Üí
   277‚Üí    // Type-check return_values against return_type when both are present.
   278‚Üí    if let (Some(values), Some(ty)) = (&fixture.return_values, &fixture.return_type) {
   279‚Üí        for (i, val) in values.iter().enumerate() {
   280‚Üí            if !json_matches_type(val, ty) {
   281‚Üí                errors.push(FixtureValidationError::TypeMismatch {
   282‚Üí                    symbol: fixture.symbol.clone(),
   283‚Üí                    index: i,
   284‚Üí                    expected: ty.clone(),
   285‚Üí                    actual_value: val.clone(),
   286‚Üí                });
   287‚Üí            }
   288‚Üí        }
   289‚Üí    }
   290‚Üí}

   303‚Üí/// Errors from post-execution expectation validation.
   304‚Üí#[derive(Debug, Clone, PartialEq)]
   305‚Üípub enum ExpectationError {
   306‚Üí    /// Call count didn't match the expectation.
   307‚Üí    CallCountMismatch {
   308‚Üí        symbol: String,
   309‚Üí        expected: CallCountExpectation,
   310‚Üí        actual: u32,
   311‚Üí    },
   312‚Üí    /// A call's arguments didn't match any `called_with` pattern.
   313‚Üí    ArgMismatch {
   314‚Üí        symbol: String,
   315‚Üí        call_index: usize,
   316‚Üí        args: Vec<Value>,
   317‚Üí    },
   318‚Üí}
   319‚Üí
   320‚Üíimpl std::fmt::Display for ExpectationError {
   321‚Üí    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
   322‚Üí        match self {
   323‚Üí            Self::CallCountMismatch {
   324‚Üí                symbol,
   325‚Üí                expected,
   326‚Üí                actual,
   327‚Üí            } => write!(
   328‚Üí                f,
   329‚Üí                "fixture '{symbol}': expected {expected:?} calls, got {actual}"
   330‚Üí            ),
   331‚Üí            Self::ArgMismatch {
   332‚Üí                symbol,
   333‚Üí                call_index,
   334‚Üí                args,
   335‚Üí            } => write!(
   336‚Üí                f,
   337‚Üí                "fixture '{symbol}': call #{call_index} args {args:?} didn't match any called_with pattern"
   338‚Üí            ),
   339‚Üí        }
   340‚Üí    }
   341‚Üí}

   345‚Üí/// Validate post-execution expectations against observed call records.
   346‚Üípub fn validate_expectations(
   347‚Üí    fixture: &MockFixture,
   348‚Üí    calls: &[MockCallRecord],
   349‚Üí) -> Vec<ExpectationError> {
   350‚Üí    let mut errors = Vec::new();
   351‚Üí    let Some(expectations) = &fixture.expectations else {
   352‚Üí        return errors;
   353‚Üí    };
   354‚Üí
   355‚Üí    // Check call count.
   356‚Üí    if let Some(count_exp) = &expectations.call_count {
   357‚Üí        let actual = calls.len() as u32;
   358‚Üí        let ok = match count_exp {
   359‚Üí            CallCountExpectation::Exact { n } => actual == *n,
   360‚Üí            CallCountExpectation::AtLeast { n } => actual >= *n,
   361‚Üí            CallCountExpectation::AtMost { n } => actual <= *n,
   362‚Üí            CallCountExpectation::Between { min, max } => actual >= *min && actual <= *max,
   363‚Üí        };
   364‚Üí        if !ok {
   365‚Üí            errors.push(ExpectationError::CallCountMismatch {
   366‚Üí                symbol: fixture.symbol.clone(),
   367‚Üí                expected: count_exp.clone(),
   368‚Üí                actual,
   369‚Üí            });
   370‚Üí        }
   371‚Üí    }
   372‚Üí
   373‚Üí    // Check called_with patterns.
   374‚Üí    if let Some(patterns) = &expectations.called_with {
   375‚Üí        for (i, call) in calls.iter().enumerate() {
   376‚Üí            let matches_any = patterns.iter().any(|pattern| args_match(pattern, &call.args));
   377‚Üí            if !matches_any {
   378‚Üí                errors.push(ExpectationError::ArgMismatch {
   379‚Üí                    symbol: fixture.symbol.clone(),
   380‚Üí                    call_index: i,
   381‚Üí                    args: call.args.clone(),
   382‚Üí                });
   383‚Üí            }
   384‚Üí        }
   385‚Üí    }
   386‚Üí
   387‚Üí    errors
   388‚Üí}
   389‚Üí
   390‚Üí/// Check whether a call's arguments match a pattern of ArgMatchers.
   391‚Üífn args_match(pattern: &[ArgMatcher], args: &[Value]) -> bool {
   392‚Üí    if pattern.len() != args.len() {
   393‚Üí        return false;
   394‚Üí    }
   395‚Üí    pattern.iter().zip(args.iter()).all(|(matcher, arg)| match matcher {
   396‚Üí        ArgMatcher::Exact { value } => arg == value,
   397‚Üí        ArgMatcher::Any => true,
   398‚Üí        ArgMatcher::TypeOf { expected } => json_matches_type(arg, expected),
   399‚Üí    })
   400‚Üí}
