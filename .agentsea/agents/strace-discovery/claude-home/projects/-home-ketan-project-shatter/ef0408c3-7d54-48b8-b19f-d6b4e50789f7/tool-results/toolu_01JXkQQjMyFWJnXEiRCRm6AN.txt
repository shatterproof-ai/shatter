     1→//! Input triage: predict whether a candidate input will produce a novel path
     2→//! without executing it.
     3→//!
     4→//! Contains a concrete evaluator for symbolic expressions ([`evaluate_constraint`])
     5→//! and the [`TriageState`] that accumulates observed branch traces and predicts
     6→//! verdicts for candidate inputs.
     7→
     8→use std::collections::HashSet;
     9→
    10→use serde_json::Value;
    11→
    12→use crate::execution_record::{BranchDecision, SymConstraint};
    13→use crate::orchestrator::hash_branch_path;
    14→use crate::sym_expr::{BinOpKind, ConstValue, SymExpr, UnOpKind};
    15→
    16→/// Maximum number of traces stored in [`TriageState`].
    17→pub const MAX_TRACES: usize = 64;
    18→
    19→/// Fraction of indeterminate branches above which a trace is too uncertain.
    20→const INDETERMINATE_THRESHOLD: f64 = 0.5;
    21→
    22→/// Every Nth skip verdict, execute anyway to validate the prediction.
    23→pub const TRIAGE_SAMPLE_INTERVAL: usize = 20;
    24→
    25→/// Minimum verdicts before evaluating whether triage is useful.
    26→pub const MIN_VERDICTS_FOR_EVAL: usize = 20;
    27→
    28→/// Minimum fraction of Skip verdicts for triage to remain active.
    29→pub const MIN_SKIP_RATE: f64 = 0.10;
    30→
    31→/// Maximum fraction of sampled skips that turned out wrong before disabling.
    32→pub const MAX_MISPREDICTION_RATE: f64 = 0.25;
    33→
    34→/// Predicted direction for a single branch.
    35→#[derive(Debug, Clone, Copy, PartialEq, Eq)]
    36→pub enum BranchPrediction {
    37→    Taken,
    38→    NotTaken,
    39→    Indeterminate,
    40→}
    41→
    42→/// Verdict from triaging a candidate input against observed traces.
    43→#[derive(Debug, Clone, PartialEq, Eq)]
    44→pub enum TriageVerdict {
    45→    /// Predicted path already covered — skip execution.
    46→    Skip,
    47→    /// At least one branch predicted to take a novel direction.
    48→    Execute {
    49→        novel_count: usize,
    50→        first_novel_depth: usize,
    51→    },
    52→    /// Too many unknowns or no matching traces to predict.
    53→    Indeterminate,
    54→}
    55→
    56→/// Why triage was disabled.
    57→#[derive(Debug, Clone, Copy, PartialEq, Eq)]
    58→pub enum TriageDisableReason {
    59→    /// Fewer than [`MIN_SKIP_RATE`] of verdicts were Skip.
    60→    LowSkipRate,
    61→    /// More than [`MAX_MISPREDICTION_RATE`] of sampled skips were wrong.
    62→    HighMisprediction,
    63→}
    64→
    65→/// Accumulates branch traces from observed executions and predicts whether
    66→/// candidate inputs will produce novel paths.
    67→pub struct TriageState {
    68→    /// Observed branch traces, deduped by branch-ID sequence.
    69→    traces: Vec<Vec<BranchDecision>>,
    70→    /// Dedup keys: branch-ID sequences already stored.
    71→    trace_keys: HashSet<Vec<u32>>,
    72→    /// Per-branch observed (branch_id, taken) pairs.
    73→    observed_directions: HashSet<(u32, bool)>,
    74→    /// Parameter names for constraint evaluation.
    75→    param_names: Vec<String>,
    76→    /// Total verdicts issued.
    77→    total_verdicts: usize,
    78→    /// Number of Skip verdicts issued.
    79→    skip_verdicts: usize,
    80→    /// Number of sampled skip verdicts validated by actual execution.
    81→    samples_taken: usize,
    82→    /// Number of samples where predicted path hash differed from actual.
    83→    mispredictions: usize,
    84→    /// Set when triage disables itself.
    85→    disable_reason: Option<TriageDisableReason>,
    86→}
    87→
    88→impl TriageState {
    89→    pub fn new(param_names: Vec<String>) -> Self {
    90→        Self {
    91→            traces: Vec::new(),
    92→            trace_keys: HashSet::new(),
    93→            observed_directions: HashSet::new(),
    94→            param_names,
    95→            total_verdicts: 0,
    96→            skip_verdicts: 0,
    97→            samples_taken: 0,
    98→            mispredictions: 0,
    99→            disable_reason: None,
   100→        }
   101→    }
   102→
   103→    /// Record a new execution trace. Deduplicates by branch-ID sequence and
   104→    /// caps storage at [`MAX_TRACES`] (drops oldest on overflow).
   105→    pub fn update(&mut self, branch_path: &[BranchDecision]) {
   106→        // Record all observed directions.
   107→        for d in branch_path {
   108→            self.observed_directions.insert((d.branch_id, d.taken));
   109→        }
   110→
   111→        // Dedup by branch-ID sequence.
   112→        let key: Vec<u32> = branch_path.iter().map(|d| d.branch_id).collect();
   113→        if !self.trace_keys.insert(key) {
   114→            return;
   115→        }
   116→
   117→        // Cap at MAX_TRACES — drop oldest.
   118→        if self.traces.len() >= MAX_TRACES {
   119→            let removed = self.traces.remove(0);
   120→            let removed_key: Vec<u32> = removed.iter().map(|d| d.branch_id).collect();
   121→            self.trace_keys.remove(&removed_key);
   122→        }
   123→
   124→        self.traces.push(branch_path.to_vec());
   125→    }
   126→
   127→    /// Number of stored traces.
   128→    pub fn trace_count(&self) -> usize {
   129→        self.traces.len()
   130→    }
   131→
   132→    /// Number of observed (branch_id, taken) pairs.
   133→    pub fn observed_direction_count(&self) -> usize {
   134→        self.observed_directions.len()
   135→    }
   136→
   137→    /// Whether triage has been disabled.
   138→    pub fn is_disabled(&self) -> bool {
   139→        self.disable_reason.is_some()
   140→    }
   141→
   142→    /// Why triage was disabled, if at all.
   143→    pub fn disable_reason(&self) -> Option<TriageDisableReason> {
   144→        self.disable_reason
   145→    }
   146→
   147→    /// Record a verdict and check whether triage should disable itself
   148→    /// due to low skip rate. Call this after every `triage_candidate` call.
   149→    pub fn record_verdict(&mut self, verdict: &TriageVerdict) {
   150→        if self.is_disabled() {
   151→            return;
   152→        }
   153→        self.total_verdicts += 1;
   154→        if *verdict == TriageVerdict::Skip {
   155→            self.skip_verdicts += 1;
   156→        }
   157→        if self.total_verdicts >= MIN_VERDICTS_FOR_EVAL {
   158→            let skip_rate = self.skip_verdicts as f64 / self.total_verdicts as f64;
   159→            if skip_rate < MIN_SKIP_RATE {
   160→                self.disable_reason = Some(TriageDisableReason::LowSkipRate);
   161→            }
   162→        }
   163→    }
   164→
   165→    /// Whether the current skip verdict should be sampled (executed anyway)
   166→    /// to validate prediction accuracy. Returns true every
   167→    /// [`TRIAGE_SAMPLE_INTERVAL`]th skip verdict.
   168→    pub fn should_sample(&self) -> bool {
   169→        self.skip_verdicts > 0 && self.skip_verdicts.is_multiple_of(TRIAGE_SAMPLE_INTERVAL)
   170→    }
   171→
   172→    /// Record the result of a sampled skip execution. If the actual path hash
   173→    /// differs from the predicted one, it counts as a misprediction.
   174→    /// Disables triage when the misprediction rate exceeds
   175→    /// [`MAX_MISPREDICTION_RATE`].
   176→    pub fn record_sample(&mut self, predicted_path_hash: u64, actual_path_hash: u64) {
   177→        if self.is_disabled() {
   178→            return;
   179→        }
   180→        self.samples_taken += 1;
   181→        if predicted_path_hash != actual_path_hash {
   182→            self.mispredictions += 1;
   183→        }
   184→        let misprediction_rate = self.mispredictions as f64 / self.samples_taken as f64;
   185→        if misprediction_rate > MAX_MISPREDICTION_RATE {
   186→            self.disable_reason = Some(TriageDisableReason::HighMisprediction);
   187→        }
   188→    }
   189→
   190→    /// Predict whether executing `inputs` would produce a novel path.
   191→    /// Returns `Indeterminate` immediately if triage has been disabled.
   192→    pub fn triage_candidate(
   193→        &self,
   194→        inputs: &[Value],
   195→        covered_paths: &HashSet<u64>,
   196→    ) -> TriageVerdict {
   197→        if self.is_disabled() {
   198→            return TriageVerdict::Indeterminate;
   199→        }
   200→        if self.traces.is_empty() {
   201→            return TriageVerdict::Indeterminate;
   202→        }
   203→
   204→        let mut any_skip = false;
   205→
   206→        for trace in &self.traces {
   207→            if trace.is_empty() {
   208→                continue;
   209→            }
   210→
   211→            let mut predicted_decisions = Vec::with_capacity(trace.len());
   212→            let mut indeterminate_count = 0usize;
   213→            let mut novel_count = 0usize;
   214→            let mut first_novel_depth = None;
   215→
   216→            for (depth, decision) in trace.iter().enumerate() {
   217→                let prediction = predict_branch(decision, inputs, &self.param_names);
   218→
   219→                let predicted_taken = match prediction {
   220→                    BranchPrediction::Taken => true,
   221→                    BranchPrediction::NotTaken => false,
   222→                    BranchPrediction::Indeterminate => {
   223→                        indeterminate_count += 1;
   224→                        // Use original direction as fallback for path hash.
   225→                        decision.taken
   226→                    }
   227→                };
   228→
   229→                // Check if this predicted direction is novel.
   230→                if prediction != BranchPrediction::Indeterminate
   231→                    && !self
   232→                        .observed_directions
   233→                        .contains(&(decision.branch_id, predicted_taken))
   234→                {
   235→                    novel_count += 1;
   236→                    if first_novel_depth.is_none() {
   237→                        first_novel_depth = Some(depth);
   238→                    }
   239→                }
   240→
   241→                predicted_decisions.push(BranchDecision {
   242→                    branch_id: decision.branch_id,
   243→                    line: decision.line,
   244→                    taken: predicted_taken,
   245→                    constraint: decision.constraint.clone(),
   246→                });
   247→            }
   248→
   249→            // Too many unknowns — skip this trace.
   250→            let total = trace.len();
   251→            if total > 0
   252→                && (indeterminate_count as f64 / total as f64) > INDETERMINATE_THRESHOLD
   253→            {
   254→                continue;
   255→            }
   256→
   257→            // Novel direction detected.
   258→            if novel_count > 0 {
   259→                return TriageVerdict::Execute {
   260→                    novel_count,
   261→                    first_novel_depth: first_novel_depth.unwrap_or(0),
   262→                };
   263→            }
   264→
   265→            // Check if predicted path is already covered.
   266→            let path_hash = hash_branch_path(&predicted_decisions);
   267→            if covered_paths.contains(&path_hash) {
   268→                any_skip = true;
   269→            } else {
   270→                // Predicted path is not covered and no novel directions detected
   271→                // from observed_directions perspective — but it's a new path hash,
   272→                // so it's worth executing.
   273→                return TriageVerdict::Execute {
   274→                    novel_count: 0,
   275→                    first_novel_depth: 0,
   276→                };
   277→            }
   278→        }
   279→
   280→        if any_skip {
   281→            TriageVerdict::Skip
   282→        } else {
   283→            TriageVerdict::Indeterminate
   284→        }
   285→    }
   286→}
   287→
   288→/// Predict which direction a branch will take for the given inputs.
   289→pub fn predict_branch(
   290→    decision: &BranchDecision,
   291→    inputs: &[Value],
   292→    param_names: &[String],
   293→) -> BranchPrediction {
   294→    let expr = match &decision.constraint {
   295→        SymConstraint::Expr { expr } => expr,
   296→        SymConstraint::Unknown { .. } => return BranchPrediction::Indeterminate,
   297→    };
   298→
   299→    match evaluate_constraint(expr, inputs, param_names) {
   300→        Some(val) => {
   301→            if is_truthy(&val) {
   302→                BranchPrediction::Taken
   303→            } else {
   304→                BranchPrediction::NotTaken
   305→            }
   306→        }
   307→        None => BranchPrediction::Indeterminate,
   308→    }
   309→}

   311→/// Evaluate a symbolic expression against concrete parameter values.
   312→///
   313→/// `params` are positional JSON values for each parameter.
   314→/// `param_names` maps parameter names to their index in `params`.
   315→/// Returns `None` when the expression contains `Unknown` nodes, references
   316→/// unresolvable parameters, or uses unsupported operations.
   317→pub fn evaluate_constraint(
   318→    expr: &SymExpr,
   319→    params: &[Value],
   320→    param_names: &[String],
   321→) -> Option<Value> {
   322→    match expr {
   323→        SymExpr::Param { name, path } => resolve_param(name, path, params, param_names),
   324→        SymExpr::Const(c) => Some(const_to_json(c)),
   325→        SymExpr::BinOp { op, left, right } => eval_binop(*op, left, right, params, param_names),
   326→        SymExpr::UnOp { op, operand } => eval_unop(*op, operand, params, param_names),
   327→        SymExpr::Call {
   328→            name,
   329→            receiver,
   330→            args,
   331→        } => eval_call(name, receiver.as_deref(), args, params, param_names),
   332→        SymExpr::Unknown => None,
   333→    }
   334→}
   335→
   336→/// Resolve a parameter reference to a concrete JSON value.
   337→fn resolve_param(
   338→    name: &str,
   339→    path: &[String],
   340→    params: &[Value],
   341→    param_names: &[String],
   342→) -> Option<Value> {
   343→    let idx = param_names.iter().position(|n| n == name)?;
   344→    let mut val = params.get(idx)?;
   345→    for segment in path {
   346→        val = val.get(segment.as_str())?;
   347→    }
   348→    Some(val.clone())
   349→}
   350→
   351→/// Convert a `ConstValue` to a `serde_json::Value`.
   352→fn const_to_json(c: &ConstValue) -> Value {
   353→    match c {
   354→        ConstValue::Int(i) => Value::from(*i),
   355→        ConstValue::Float(f) => Value::from(*f),
   356→        ConstValue::Str(s) => Value::from(s.as_str()),
   357→        ConstValue::Bool(b) => Value::from(*b),
   358→        ConstValue::Null | ConstValue::Undefined => Value::Null,
   359→        ConstValue::Complex { repr, .. } => const_to_json(repr),
   360→    }
   361→}
   362→
   363→/// JS-like truthiness: null, false, 0, 0.0, "" are falsy; everything else truthy.
   364→fn is_truthy(v: &Value) -> bool {
   365→    match v {
   366→        Value::Null => false,
   367→        Value::Bool(b) => *b,
   368→        Value::Number(n) => {
   369→            if let Some(i) = n.as_i64() {
   370→                i != 0
   371→            } else if let Some(f) = n.as_f64() {
   372→                f != 0.0
   373→            } else {
   374→                true
   375→            }
   376→        }
   377→        Value::String(s) => !s.is_empty(),
   378→        // Arrays and objects are always truthy in JS
   379→        _ => true,
   380→    }
   381→}
   382→
   383→/// Extract an f64 from a JSON number value.
   384→fn as_f64(v: &Value) -> Option<f64> {
   385→    v.as_f64()
   386→}

   388→/// Evaluate a binary operation.
   389→fn eval_binop(
   390→    op: BinOpKind,
   391→    left: &SymExpr,
   392→    right: &SymExpr,
   393→    params: &[Value],
   394→    param_names: &[String],
   395→) -> Option<Value> {
   396→    // Short-circuit logical ops
   397→    match op {
   398→        BinOpKind::And => {
   399→            let lv = evaluate_constraint(left, params, param_names)?;
   400→            if !is_truthy(&lv) {
   401→                return Some(lv);
   402→            }
   403→            return evaluate_constraint(right, params, param_names);
   404→        }
   405→        BinOpKind::Or => {
   406→            let lv = evaluate_constraint(left, params, param_names)?;
   407→            if is_truthy(&lv) {
   408→                return Some(lv);
   409→            }
   410→            return evaluate_constraint(right, params, param_names);
   411→        }
   412→        _ => {}
   413→    }
   414→
   415→    let lv = evaluate_constraint(left, params, param_names)?;
   416→    let rv = evaluate_constraint(right, params, param_names)?;
   417→
   418→    match op {
   419→        // Comparisons
   420→        BinOpKind::Eq => eval_eq(&lv, &rv).map(Value::from),
   421→        BinOpKind::Ne => eval_eq(&lv, &rv).map(|eq| Value::from(!eq)),
   422→        BinOpKind::Lt => eval_order(&lv, &rv).map(|ord| Value::from(ord.is_lt())),
   423→        BinOpKind::Le => eval_order(&lv, &rv).map(|ord| Value::from(ord.is_le())),
   424→        BinOpKind::Gt => eval_order(&lv, &rv).map(|ord| Value::from(ord.is_gt())),
   425→        BinOpKind::Ge => eval_order(&lv, &rv).map(|ord| Value::from(ord.is_ge())),
   426→
   427→        // Arithmetic
   428→        BinOpKind::Add => eval_add(&lv, &rv),
   429→        BinOpKind::Sub => eval_arith(&lv, &rv, |a, b| a - b),
   430→        BinOpKind::Mul => eval_arith(&lv, &rv, |a, b| a * b),
   431→        BinOpKind::Div => {
   432→            let b = as_f64(&rv)?;
   433→            if b == 0.0 {
   434→                return None;
   435→            }
   436→            eval_arith(&lv, &rv, |a, b| a / b)
   437→        }
   438→        BinOpKind::Mod => {
   439→            let b = as_f64(&rv)?;
   440→            if b == 0.0 {
   441→                return None;
   442→            }
   443→            eval_arith(&lv, &rv, |a, b| a % b)
   444→        }
   445→
   446→        // Unsupported for triage
   447→        BinOpKind::And | BinOpKind::Or => unreachable!(),
   448→        BinOpKind::BitwiseAnd
   449→        | BinOpKind::BitwiseOr
   450→        | BinOpKind::BitwiseXor
   451→        | BinOpKind::Shl
   452→        | BinOpKind::Shr
   453→        | BinOpKind::BitClear
   454→        | BinOpKind::In
   455→        | BinOpKind::InstanceOf => None,
   456→    }
   457→}

   459→/// Equality comparison with JS-like semantics.
   460→fn eval_eq(a: &Value, b: &Value) -> Option<bool> {
   461→    match (a, b) {
   462→        (Value::Null, Value::Null) => Some(true),
   463→        (Value::Bool(x), Value::Bool(y)) => Some(x == y),
   464→        (Value::Number(_), Value::Number(_)) => {
   465→            let fa = as_f64(a)?;
   466→            let fb = as_f64(b)?;
   467→            Some(fa == fb)
   468→        }
   469→        (Value::String(x), Value::String(y)) => Some(x == y),
   470→        _ => None,
   471→    }
   472→}
   473→
   474→/// Ordering comparison for numbers and strings.
   475→fn eval_order(a: &Value, b: &Value) -> Option<std::cmp::Ordering> {
   476→    match (a, b) {
   477→        (Value::Number(_), Value::Number(_)) => {
   478→            let fa = as_f64(a)?;
   479→            let fb = as_f64(b)?;
   480→            fa.partial_cmp(&fb)
   481→        }
   482→        (Value::String(x), Value::String(y)) => Some(x.cmp(y)),
   483→        _ => None,
   484→    }
   485→}
   486→
   487→/// Add: numeric addition or string concatenation.
   488→fn eval_add(a: &Value, b: &Value) -> Option<Value> {
   489→    match (a, b) {
   490→        (Value::String(x), Value::String(y)) => {
   491→            let mut result = x.clone();
   492→            result.push_str(y);
   493→            Some(Value::from(result))
   494→        }
   495→        _ => eval_arith(a, b, |x, y| x + y),
   496→    }
   497→}
   498→
   499→/// Arithmetic on numeric JSON values.
   500→fn eval_arith(a: &Value, b: &Value, f: fn(f64, f64) -> f64) -> Option<Value> {
   501→    let fa = as_f64(a)?;
   502→    let fb = as_f64(b)?;
   503→    let result = f(fa, fb);
   504→    // Preserve integer results when both inputs are integers and the result is exact
   505→    if a.is_i64() && b.is_i64() && let Some(i) = i64_from_f64(result) {
   506→        return Some(Value::from(i));
   507→    }
   508→    Some(Value::from(result))
   509→}
   510→
   511→/// Safely convert f64 to i64 when the value is an exact integer.
   512→fn i64_from_f64(f: f64) -> Option<i64> {
   513→    if f.fract() == 0.0 && f >= i64::MIN as f64 && f <= i64::MAX as f64 {
   514→        Some(f as i64)
   515→    } else {
   516→        None
   517→    }
   518→}

   520→/// Evaluate a unary operation.
   521→fn eval_unop(
   522→    op: UnOpKind,
   523→    operand: &SymExpr,
   524→    params: &[Value],
   525→    param_names: &[String],
   526→) -> Option<Value> {
   527→    let val = evaluate_constraint(operand, params, param_names)?;
   528→    match op {
   529→        UnOpKind::Not => Some(Value::from(!is_truthy(&val))),
   530→        UnOpKind::Neg => {
   531→            let f = as_f64(&val)?;
   532→            if val.is_i64() && let Some(i) = i64_from_f64(-f) {
   533→                return Some(Value::from(i));
   534→            }
   535→            Some(Value::from(-f))
   536→        }
   537→        UnOpKind::BitwiseNot | UnOpKind::TypeOf => None,
   538→    }
   539→}

   541→/// Evaluate a string method/function call against concrete values.
   542→///
   543→/// Supports the 8 canonical string operations defined in `data/string-ops.yaml`,
   544→/// matching all cross-language aliases.
   545→fn eval_call(
   546→    name: &str,
   547→    receiver: Option<&SymExpr>,
   548→    args: &[SymExpr],
   549→    params: &[Value],
   550→    param_names: &[String],
   551→) -> Option<Value> {
   552→    match classify_string_op(name) {
   553→        Some(StringOp::Contains) => {
   554→            let (haystack, needle) = eval_string_pair(name, receiver, args, params, param_names)?;
   555→            Some(Value::from(haystack.contains(needle.as_str())))
   556→        }
   557→        Some(StringOp::Prefix) => {
   558→            let (haystack, needle) = eval_string_pair(name, receiver, args, params, param_names)?;
   559→            Some(Value::from(haystack.starts_with(needle.as_str())))
   560→        }
   561→        Some(StringOp::Suffix) => {
   562→            let (haystack, needle) = eval_string_pair(name, receiver, args, params, param_names)?;
   563→            Some(Value::from(haystack.ends_with(needle.as_str())))
   564→        }
   565→        Some(StringOp::IndexOf) => {
   566→            let (haystack, needle) = eval_string_pair(name, receiver, args, params, param_names)?;
   567→            let idx = haystack
   568→                .find(needle.as_str())
   569→                .map_or(-1i64, |i| i as i64);
   570→            Some(Value::from(idx))
   571→        }
   572→        Some(StringOp::Length) => {
   573→            // receiver-style: "str".length  OR  free-style: len("str")
   574→            let s = if let Some(recv) = receiver {
   575→                eval_receiver_str(recv, params, param_names)?
   576→            } else {
   577→                eval_arg_str(args.first()?, params, param_names)?
   578→            };
   579→            Some(Value::from(s.len() as i64))
   580→        }
   581→        Some(StringOp::CharAt) => {
   582→            let recv = eval_receiver_str(receiver?, params, param_names)?;
   583→            let idx_val = evaluate_constraint(args.first()?, params, param_names)?;
   584→            let idx = idx_val.as_i64()? as usize;
   585→            let ch = recv.chars().nth(idx)?;
   586→            Some(Value::from(ch.to_string()))
   587→        }
   588→        Some(StringOp::Substr) => {
   589→            let recv = eval_receiver_str(receiver?, params, param_names)?;
   590→            let start_val = evaluate_constraint(args.first()?, params, param_names)?;
   591→            let start = start_val.as_i64()?.max(0) as usize;
   592→            if start >= recv.len() {
   593→                return Some(Value::from(""));
   594→            }
   595→            if let Some(end_expr) = args.get(1) {
   596→                let end_val = evaluate_constraint(end_expr, params, param_names)?;
   597→                let end = (end_val.as_i64()?.max(0) as usize).min(recv.len());
   598→                if end <= start {
   599→                    return Some(Value::from(""));
   600→                }
   601→                Some(Value::from(&recv[start..end]))
   602→            } else {
   603→                Some(Value::from(&recv[start..]))
   604→            }
   605→        }
   606→        Some(StringOp::Concat) => {
   607→            let recv = eval_receiver_str(receiver?, params, param_names)?;
   608→            let arg = eval_arg_str(args.first()?, params, param_names)?;
   609→            let mut result = recv;
   610→            result.push_str(&arg);
   611→            Some(Value::from(result))
   612→        }
   613→        None => None,
   614→    }
   615→}
   616→
   617→/// Canonical string operations matching `data/string-ops.yaml`.
   618→enum StringOp {
   619→    Contains,
   620→    Prefix,
   621→    Suffix,
   622→    IndexOf,
   623→    Length,
   624→    CharAt,
   625→    Substr,
   626→    Concat,
   627→}
   628→
   629→/// Map a call name to its canonical string operation.
   630→fn classify_string_op(name: &str) -> Option<StringOp> {
   631→    match name {
   632→        "includes" | "Contains" | "strings.Contains" | "contains" => Some(StringOp::Contains),
   633→        "startsWith" | "HasPrefix" | "strings.HasPrefix" | "starts_with" => Some(StringOp::Prefix),
   634→        "endsWith" | "HasSuffix" | "strings.HasSuffix" | "ends_with" => Some(StringOp::Suffix),
   635→        "indexOf" | "Index" | "strings.Index" | "find" | "index_of" => Some(StringOp::IndexOf),
   636→        "length" | "len" => Some(StringOp::Length),
   637→        "charAt" | "char_at" => Some(StringOp::CharAt),
   638→        "slice" | "substring" | "substr" => Some(StringOp::Substr),
   639→        "concat" => Some(StringOp::Concat),
   640→        _ => None,
   641→    }
   642→}
   643→
   644→/// Evaluate a receiver expression as a string.
   645→fn eval_receiver_str(
   646→    receiver: &SymExpr,
   647→    params: &[Value],
   648→    param_names: &[String],
   649→) -> Option<String> {
   650→    let val = evaluate_constraint(receiver, params, param_names)?;
   651→    val.as_str().map(String::from)
   652→}
   653→
   654→/// Evaluate an argument expression as a string.
   655→fn eval_arg_str(arg: &SymExpr, params: &[Value], param_names: &[String]) -> Option<String> {
   656→    let val = evaluate_constraint(arg, params, param_names)?;
   657→    val.as_str().map(String::from)
   658→}
   659→
   660→/// For operations that take receiver + first arg as strings (or first two args for free-style).
   661→fn eval_string_pair(
   662→    _name: &str,
   663→    receiver: Option<&SymExpr>,
   664→    args: &[SymExpr],
   665→    params: &[Value],
   666→    param_names: &[String],
   667→) -> Option<(String, String)> {
   668→    if let Some(recv) = receiver {
   669→        // receiver-style: recv.method(arg)
   670→        let haystack = eval_receiver_str(recv, params, param_names)?;
   671→        let needle = eval_arg_str(args.first()?, params, param_names)?;
   672→        Some((haystack, needle))
   673→    } else {
   674→        // free-style: method(haystack, needle)
   675→        let haystack = eval_arg_str(args.first()?, params, param_names)?;
   676→        let needle = eval_arg_str(args.get(1)?, params, param_names)?;
   677→        Some((haystack, needle))
   678→    }
   679→}
